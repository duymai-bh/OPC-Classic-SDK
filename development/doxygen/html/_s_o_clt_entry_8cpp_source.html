244HRESULT SOCltEntry::doStateTransition(IN BOOL deep, BYTE stateDeep)
245{
246 HRESULT res = S_OK;
247 BYTE targetState = getTargetObjectState();
248
249 if (disconnected == targetState)
250 {
251 doRecurseChildren(getBranchList(), targetState);
252 doRecurseChildren(getLeafList(), targetState);
253 doRecurseChildren(getAdditionalList(), targetState);
254 }
255
256 setObjectState(getTargetObjectState());
257
258 if (deep && disconnected != targetState)
259 {
260 // Recurse children
261 if (S_OK != doRecurseChildren(getBranchList(), stateDeep))
262 {
263 res = S_FALSE;
264 }
265
266 if (S_OK != doRecurseChildren(getLeafList(), stateDeep))
267 {
268 res = S_FALSE;
269 }
270
271 if (S_OK != doRecurseChildren(getAdditionalList(), stateDeep))
272 {
273 res = S_FALSE;
274 }
275 }
276
277 return res;

280HRESULT SOCltEntry::doRecurseChildren(SOCmnList<SOCmnElement>* plist, IN BYTE state)
281{
282 BOOL ok = TRUE;
283
284 if (NULL == plist)
285 {
286 return S_OK;
287 }
288
289 SOCmnList<SOCmnElement> list(plist);
290 SOCltElement* element = NULL;
291 DWORD max_handles = list.getCount();
292
293 if (max_handles == 0)
294 {
295 return S_OK;
296 }
297
298 //SOCmnEvents handles;
299 SOCmnEvent* handles = new SOCmnEvent[max_handles];
300 DWORD cnt = 0;
301 SOCmnListPosition pos = list.getStartPosition();
302
303 while (pos)
304 {
305 element = (SOCltElement*)list.getNext(pos);
306 SOCmnEventPointer wait;
307 SOCltWorkerThread* thread = element->getWorkerThread();
308 // Each server must have its own working thread.
309 // It does make sense to accomplish the tasks
310 // on them in paralell, no matter how this
311 // task is executed, either synch or asynch.
312 wait.set(SOCmnEventPointer(&handles[cnt]));
313
314 if (thread && !thread->isOwnThread())
315 {
316 handles[cnt++].create();
317 }
318
319 // Calling start(), connect(), disconnect()
320 // instead of doStateTransition() will create
321 // a separate task
322 switch (state)
323 {
324 case SOCltElement::started:
325 {
326 ok &= SUCCEEDED(element->start(TRUE, SOCLT_EXEC_ASYNC, wait));
327 }
328 break;
329
330 case SOCltElement::connected:
331 {
332 ok &= SUCCEEDED(element->connect(TRUE, SOCLT_EXEC_ASYNC, wait));
333 }
334 break;
335
336 case SOCltElement::disconnected:
337 {
338 ok &= SUCCEEDED(element->disconnect(SOCLT_EXEC_ASYNC, wait));
339 }
340 break;
341 }// end switch
342 }
343
344 list.autoUnlock();
345
346 // wait for completion
347 if (cnt > 0)
348 {
349 DWORD i;
350
351 for (i = 0 ; i < cnt ; i++)
352 {
353 handles[i].waitFor();
354 }
355
356 delete [] handles;
357
358 if (ok)
359 {
360 pos = list.getStartPosition();
361
362 while (pos)
363 {
364 element = (SOCltElement*)list.getNext(pos);
365
366 if (element->getObjectState() != state)
367 {
368 ok = FALSE;
369 break;
370 }
371 }
372 }
373 }
374
375 return ok ? S_OK : S_FALSE;
376}

415void SOCltEntry::dispatchCallbacks()
416{
417 while (1)
418 {
419 SOCltCallback cb;
420 g_callbackQueueSync.lock();
421 SOCltCallback* pcb = g_callbackQueue.getHead();
422
423 if (pcb)
424 {
425 cb = *pcb; // save the data, since removeHead() will call cdor
426 cb.e->addRef();
427 }
428
429 g_callbackQueue.removeHead();
430 g_callbackQueueSync.unlock();
431
432 // DON'T hold the queue locked while
433 // executing the callback!
434
435 if (pcb)
436 {
437 cb.execute();
438 }
439 else
440 {
441 break;
442 }
443 }

532//-----------------------------------------------------------------------------
533//
534// SOCltInvokeCallback
535void SOCltInvokeCallback(const SOCltCallback& clb)
536{
537 // Execute immideately, if we get called
538 // in the context of the main thread or no callback
539 // thread or window is specified
540 if (g_mainThreadID == ::GetCurrentThreadId() || NULL == g_hwndCallbackNotify)
541 {
542 clb.execute();
543 return;
544 }
545
546 g_callbackQueueSync.lock();
547
548 // No callbacks after the entry has been destroyed.
549
550 if (g_callbackQueue.isInit())
551 {
552 SOCltCallback* pdata = NULL;
553 SOCMN_ALLOCATE_OBJECT(pdata, SOCltCallback);
554
555 if (pdata)
556 {
557 *pdata = clb;
558 pdata->e->addRef();
559 g_callbackQueue.add(pdata);
560 }
561
562 ::PostMessage(g_hwndCallbackNotify, SOCltEntry::s_messageCallbackNotify, (WPARAM)0, (LPARAM)0);
563 }
564
565 g_callbackQueueSync.unlock();
566}

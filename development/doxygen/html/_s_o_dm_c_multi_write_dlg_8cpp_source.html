207void SODmCMultiWriteDlg::addCaptionRowToGridCtrl(CGridCtrl& gridCtrl)
208{
209 gridCtrl.SetColumnCount(4);
210 // set column widths
211 CRect rect;
212 gridCtrl.GetClientRect(&rect);
213 gridCtrl.SetColumnWidth(CFG_COL_ITEM, CFG_COLWIDTH_0);
214 gridCtrl.SetColumnWidth(CFG_COL_DATATYPE, CFG_COLWIDTH_1);
215 gridCtrl.SetColumnWidth(CFG_COL_VALUE, CFG_COLWIDTH_2);
216 gridCtrl.SetColumnWidth(CFG_COL_INDEX, CFG_COLWIDTH_3);
217 gridCtrl.ExpandLastColumn();
218 //Set format
219 gridCtrl.SetItemFormat(0, CFG_COL_ITEM , DT_CENTER | DT_WORDBREAK);
220 gridCtrl.SetItemFormat(0, CFG_COL_DATATYPE, DT_CENTER | DT_WORDBREAK);
221 gridCtrl.SetItemFormat(0, CFG_COL_VALUE , DT_CENTER | DT_WORDBREAK);
222 gridCtrl.SetItemFormat(0, CFG_COL_INDEX , DT_CENTER | DT_WORDBREAK);
223 // add caption row
224 gridCtrl.SetItemText(0, CFG_COL_ITEM, CFG_SZHEAD_ITEM);
225 gridCtrl.SetItemText(0, CFG_COL_DATATYPE, CFG_SZHEAD_DATATYPE);
226 gridCtrl.SetItemText(0, CFG_COL_VALUE, CFG_SZHEAD_VALUE);
227 gridCtrl.SetItemText(0, CFG_COL_INDEX, CFG_SZHEAD_INDEX);
228 SODmCMainView* pMainView = getMainView();
229 CListCtrl* pDA_List = pMainView->getListDA();
230 UINT uiCnt = pDA_List->GetSelectedCount();
231 gridCtrl.SetRowCount(uiCnt + 1);
232 POSITION pos = pDA_List->GetFirstSelectedItemPosition();
233 int iRow = 0;
234 int iGroupCount = 0;
235
236 if (pos != NULL)
237 {
238 while (pos)
239 {
240 int nItem = pDA_List->GetNextSelectedItem(pos);
241 iRow++;
242 SODaCItem* pItem = (SODaCItem*)pDA_List->GetItemData(nItem);
243 // Insert ItemID of actual tag
244 gridCtrl.SetItemText(iRow, CFG_COL_ITEM, pItem->getItemID());
245 gridCtrl.SetItemState(iRow, CFG_COL_ITEM, GVIS_READONLY);
246 gridCtrl.SetItemData(iRow, CFG_COL_ITEM, (LPARAM)(SODmCItem*)pItem);
247
248 // combo box with desired Datatype defaulted to native
249 if (gridCtrl.SetCellType(iRow, CFG_COL_DATATYPE, RUNTIME_CLASS(CGridCellCombo)))
250 {
251 CStringArray options;
252 getComboOptions(options);
253 CGridCellCombo* pCell = (CGridCellCombo*) gridCtrl.GetCell(iRow, CFG_COL_DATATYPE);
254 pCell->SetOptions(options);
255 pCell->SetStyle(CBS_DROPDOWNLIST); //CBS_DROPDOWN, CBS_DROPDOWNLIST, CBS_SIMPLE
256 pCell->SetText(VARTYPE2STR(pItem->getNativeDatatype()));
257 }
258
259 // Insert Value of actual tag
260 SOCmnVariant varItemValue;
261 SOCmnString szBuffer;
262 pItem->getReadValue(varItemValue);
263 VARIANT2STRING(&szBuffer, varItemValue, FALSE);
264 gridCtrl.SetItemText(iRow, CFG_COL_VALUE, szBuffer);
265
266 if (((pItem->getAccessRights() & OPC_WRITEABLE) == 0) || (!pItem->isConnected()))
267 {
268 gridCtrl.SetItemState(iRow, CFG_COL_VALUE, GVIS_READONLY);
269 gridCtrl.SetItemBkColour(iRow, CFG_COL_VALUE, COLORREADONLY);
270 }
271
272 if ((varItemValue.vt & VT_ARRAY) == VT_ARRAY)
273 {
274 LONG lBound, uBound;
275 LONG i;
276 SOCmnString valStr;
277 SafeArrayGetLBound(varItemValue.parray, 1, &lBound);
278 SafeArrayGetUBound(varItemValue.parray, 1, &uBound);
279
280 if (gridCtrl.SetCellType(iRow, CFG_COL_INDEX, RUNTIME_CLASS(CGridCellCombo)))
281 {
282 CStringArray options;
283 CGridCellCombo* pCell = (CGridCellCombo*) gridCtrl.GetCell(iRow, CFG_COL_INDEX);
284
285 for (i = lBound; i <= uBound; i++)
286 {
287 valStr.format(_T("%d"), i);
288 options.Add(valStr);
289 }
290
291 valStr.format(_T("%d"), lBound);
292 pCell->SetText(valStr);
293 pCell->SetOptions(options);
294 pCell->SetStyle(CBS_SIMPLE); //CBS_DROPDOWN, CBS_DROPDOWNLIST, CBS_SIMPLE
295 }
296 }
297 else
298 {
299 gridCtrl.SetItemText(iRow, CFG_COL_INDEX, _T(""));
300 gridCtrl.SetItemState(iRow, CFG_COL_INDEX, GVIS_READONLY);
301 gridCtrl.SetItemBkColour(iRow, CFG_COL_INDEX, COLORREADONLY);
302 }
303
304 if ((pItem->getAccessRights() & OPC_WRITEABLE) && (pItem->isConnected()))
305 {
306 SOCmnPointer<SODaCGroup> grp = (SODaCGroup*)pItem->getParentBranch();
307 SODmCItemList* pList = (SODmCItemList*) m_lstGroupList.findKey(grp->getOpcHandle());
308
309 if (pList != NULL)
310 {
311 pList->m_itemList.add(pItem);
312 }
313 else
314 {
315 pList = new SODmCItemList();
316
317 if (pList != NULL)
318 {
319 pList->m_itemList.add(pItem);
320 pList->m_group = grp;
321 }
322
323 m_lstGroupList.add(pList, grp->getOpcHandle());
324 iGroupCount++;
325 }
326
327 gridCtrl.SetItemData(iRow, CFG_COL_VALUE, (LPARAM)(SODaCGroup*)grp);
328 }
329 }
330 }
331}

342void SODmCMultiWriteDlg::OnBnClickedOk()
343{
344 if (m_pGridCtrl != NULL)
345 {
346 int iCnt = m_pGridCtrl->GetRowCount();
347 UpdateData(TRUE); // Update Values
348
349 for (int iRow = 1 ; iRow < iCnt ; iRow++)
350 {
351 SODmCItem* pItem = (SODmCItem*) m_pGridCtrl->GetItemData(iRow, CFG_COL_ITEM);
352 VARTYPE datatype = string2vartype(m_pGridCtrl->GetItemText(iRow, CFG_COL_DATATYPE));
353 SOCmnString valueString(m_pGridCtrl->GetItemText(iRow, CFG_COL_VALUE));
354 SOCmnString idxString(m_pGridCtrl->GetItemText(iRow, CFG_COL_INDEX));
355 DWORD startIdx = 0;
356
357 if (idxString.isEmpty())
358 {
359 startIdx = 0;
360 }
361 else
362 {
363 startIdx = _ttol(idxString);
364 }
365
366 SOCmnVariant value;
367
368 if (datatype == VT_EMPTY)
369 {
370 datatype = pItem->getNativeDatatype();
371 }
372
373 if (convertWriteStringToVariant(pItem, valueString, datatype, startIdx, value))
374 {
375 pItem->setWriteValue(value);
376 }
377 }
378
379 SOCmnListPosition pos = m_lstGroupList.getStartPosition();
380
381 while (pos) // loop for each group
382 {
383 SODmCItemList* pList = m_lstGroupList.GetNext(pos);
384 SODaCGroup* pGrp = pList->m_group;
385
386 if (pGrp->getDataRetrieval() == SODaCGroup::drAsyncAdvise)
387 {
388 pGrp->asyncWrite(&pList->m_itemList);
389 }
390 else //synched
391 {
392 pGrp->syncWrite(&pList->m_itemList);
393 }
394 }
395 }
396}

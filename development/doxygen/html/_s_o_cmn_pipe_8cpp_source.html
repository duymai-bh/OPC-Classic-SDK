69BOOL
70SOCmnPipe::create(
71 void)
72{
73 if (m_name.isEmpty() == TRUE)
74 {
75 _TRACE(SOCMNTRACE_L_WRN, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::create - please set the pipe name")));
76 return FALSE;
77 } // end if
78
79 SOCmnString pipeName;
80 pipeName.format(_T("%s%s"), _T("\\\\.\\pipe\\"), m_name);
81
82 if (!InitializeSecurityDescriptor(m_pSecurityDescriptor, SECURITY_DESCRIPTOR_REVISION))
83 {
84 _TRACE(SOCMNTRACE_L_WRN, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::create - could not initialize security descriptor for: %s"), m_name));
85 return FALSE;
86 } // end if
87
88 if (!SetSecurityDescriptorDacl(m_pSecurityDescriptor, TRUE, NULL, FALSE))
89 {
90 _TRACE(SOCMNTRACE_L_WRN, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::create - could not set security descriptor on pipe: %s"), m_name));
91 return FALSE;
92 } // end i
93
94 m_pSecurityAttributes->lpSecurityDescriptor = m_pSecurityDescriptor;
95
96 while (TRUE)
97 {
98 m_pipe = CreateFile(pipeName, GENERIC_WRITE, 0, m_pSecurityAttributes, OPEN_EXISTING, 0, NULL);
99
100 if (m_pipe != INVALID_HANDLE_VALUE)
101 {
102 // succesfully created
103 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::create - succesfully created pipe: %s"), m_name));
104 break;
105 } // end if
106
107 if (getLastError() != ERROR_PIPE_BUSY)
108 {
109 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::create - creating pipe: %s - failed with error 0x%x"), m_name, getLastError()));
110 return FALSE;
111 } // end if
112
113 // all pipe instances are busy wait 2000 ms
114 if (!WaitNamedPipe(pipeName, 2000))
115 {
116 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::create - waited for pipe: %s"), m_name));
117 return FALSE;
118 } // end if
119 } // end while
120
121 return TRUE;
122} // end create

178BOOL
179SOCmnPipe::handleWriteRequest(
180 void)
181{
182 if (m_bufferSize == 0)
183 {
184 _TRACE(SOCMNTRACE_L_WRN, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::handleWriteRequest - writting to: %s failed - empty message buffer"), m_name));
185 return FALSE;
186 } // end if
187
188 m_bufferSync.lock();
189 if (create() == FALSE)
190 {
191 // failed to create the pipe
192// _TRACEC(SOCMNTRACE_L_WRN, SOCMNTRACE_G_ALL, _T("SOCmnPipe::handleWriteRequest"), (_T("Named pipe: %s could not be created !"), m_name));
193 m_bufferSync.unlock();
194 return FALSE;
195 } // end if
196
197 if (connect() == FALSE)
198 {
199 // failed to connect to named pipe
200 _TRACEC(SOCMNTRACE_L_WRN, SOCMNTRACE_G_ALL, _T("SOCmnPipe::handleWriteRequest"), (_T("Failed to connect to named named pipe: %s!"), m_name));
201 m_bufferSync.unlock();
202 return FALSE;
203 } // end if
204
205 DWORD written;
206
207 if (WriteFile(m_pipe, m_buffer, m_bufferSize, &written, NULL) == TRUE)
208 {
209 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::handleWriteRequest - writting to pipe: %s - buffer: %s"), m_name, m_buffer));
210 m_bufferSync.unlock();
211 return TRUE;
212 } // end if
213
214 _TRACE(SOCMNTRACE_L_WRN, SOCMNTRACE_G_SOCKET, (_T("SOCmnPipe::handleWriteRequest - writting to pipe: %s failed"), m_name));
215
216 m_bufferSync.unlock();
217
218 return FALSE;
219} // end if

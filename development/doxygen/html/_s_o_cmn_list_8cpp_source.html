191{
192 SOCmnListEntry* entry;
193 DWORD entrySize;
194 DWORD size;
195 SOCmnListBlock* block;
196 DWORD i;
197
198 if (blockElements == 0)
199 {
200 return FALSE;
201 }
202
203 if ((m_type & SOCMNLIST_TYPE_BLOCKED) == 0)
204 {
205 return FALSE; // no blocked list
206 }
207
208 // allocate block
209 if ((m_type & SOCMNLIST_TYPE_KEY_MASK) == SOCMNLIST_TYPE_KEY_NO)
210 {
211 entrySize = sizeof(SOCmnListEntry);
212 }
213 else
214 {
215 entrySize = sizeof(SOCmnListEntryWithKey);
216 }
217
218 size = sizeof(SOCmnListBlock) + (entrySize * blockElements);
219 block = (SOCmnListBlock*)malloc(size);
220 memset(block, 0, size);
221 block->m_count = blockElements;
222
223 // create sequential list in block
224 if ((m_type & SOCMNLIST_TYPE_KEY_MASK) == SOCMNLIST_TYPE_KEY_NO)
225 {
226 entry = (SOCmnListEntry*)(block + 1);
227
228 for (i = 0; i < blockElements - 1; i++)
229 {
230 entry->m_next = entry + 1;
231 entry++;
232 }
233 }
234 else
235 {
236 SOCmnListEntryWithKey* entryWK = (SOCmnListEntryWithKey*)(block + 1);
237
238 for (i = 0; i < blockElements - 1; i++)
239 {
240 entryWK->m_next = entryWK + 1;
241 entryWK++;
242 }
243
244 entry = (SOCmnListEntry*)entryWK;
245 }
246
247 if (isList())
248 {
249 // full list
250 SOCmnListInfoFullList* infoF = (SOCmnListInfoFullList*)m_info;
251 SOCmnListEntry* firstFreeEntry = (SOCmnListEntry*)(block + 1);
252 // add block to the head of the block list
253 block->m_next = infoF->m_blockFirst;
254 infoF->m_blockFirst = block;
255 // add the block entries to the head of the free list
256 entry->m_next = (SOCmnListEntry*)infoF->m_free;
257 infoF->m_free = firstFreeEntry;
258 }
259 else
260 {
261 // map
262 SOCmnListInfoMap* infoM = (SOCmnListInfoMap*)m_info;
263 SOCmnListEntry* firstFreeEntry = (SOCmnListEntry*)(block + 1);
264 // add block to the head of the block list
265 block->m_next = infoM->m_blockFirst;
266 infoM->m_blockFirst = block;
267 // add the block entries to the head of the free list
268 entry->m_next = (SOCmnListEntry*)infoM->m_free;
269 infoM->m_free = firstFreeEntry;
270 }
271
272 return TRUE;

692BOOL SOCmnListHeader::create(
693 IN OPTIONAL SOCmnListConfig* creationConfig)
694{
695 if (isInit())
696 {
697 return FALSE;
698 }
699
700 if (isList())
701 {
702 if ((m_type & SOCMNLIST_TYPE_INFO_FULL_LIST) != 0)
703 {
704 SOCMN_FREE_OBJECT_C(m_info, (SOCmnListInfoFullList*));
705 }
706 else
707 {
708 SOCMN_FREE_OBJECT_C(m_info, (SOCmnListInfoSimpleList*));
709 }
710 }
711 else if (isMap())
712 {
713 SOCMN_FREE_OBJECT_C(m_info, (SOCmnListInfoMap*));
714 }
715 else
716 {
717 // something went wrong and the old type can not determined
718 SOCMN_FREE_OBJECT_C(m_info, (SOCmnListInfoFullList*));
719 SOCMN_FREE_OBJECT_C(m_info, (SOCmnListInfoSimpleList*));
720 SOCMN_FREE_OBJECT_C(m_info, (SOCmnListInfoSimpleList*));
721 }
722
723 WORD hashTableSize = 17;
724 BYTE blockElements = 10;
725 BOOL ok = TRUE;
726
727 if (creationConfig)
728 {
729 m_type = creationConfig->m_type;
730
731 if (creationConfig->m_hashTableSize > 0)
732 {
733 hashTableSize = creationConfig->m_hashTableSize;
734 }
735
736 if (creationConfig->m_entriesPerBlock > 0)
737 {
738 blockElements = creationConfig->m_entriesPerBlock;
739 }
740 }
741
742 // create info
743 if (isList())
744 {
745 SOCmnListInfoSimpleList* infoS;
746
747 if ((m_type & SOCMNLIST_TYPE_INFO_FULL_LIST) != 0)
748 {
749 SOCmnListInfoFullList* infoF;
750 SOCMN_ALLOCATE_OBJECT_C(m_info, SOCmnListInfoFullList, (void*))
751 infoF = (SOCmnListInfoFullList*)m_info;
752 infoF->m_count = 0;
753
754 if ((m_type & SOCMNLIST_TYPE_SYNCED) == SOCMNLIST_TYPE_SYNCED)
755 {
756 // synchronisation
757 SOCMN_ALLOCATE_OBJECT(infoF->m_sync, SOCmnSync())
758 ok &= (infoF->m_sync != NULL);
759 }
760
761 infoF->m_defaultBlockElements = blockElements;
762 }
763 else
764 {
765 SOCMN_ALLOCATE_OBJECT_C(m_info, SOCmnListInfoSimpleList, (void*))
766 }
767
768 infoS = (SOCmnListInfoSimpleList*)m_info;
769 infoS->m_start = NULL;
770 infoS->m_end = NULL;
771 }
772
773 if (isMap())
774 {
775 SOCmnListInfoMap* infoM;
776 SOCMN_ALLOCATE_OBJECT(infoM, SOCmnListInfoMap)
777 infoM->m_count = 0;
778
779 if ((m_type & SOCMNLIST_TYPE_SYNCED) == SOCMNLIST_TYPE_SYNCED)
780 {
781 // synchronisation
782 SOCMN_ALLOCATE_OBJECT(infoM->m_sync, SOCmnSync())
783 ok &= (infoM->m_sync != NULL);
784 }
785
786 m_type |= SOCMNLIST_TYPE_BLOCKED;
787 infoM->m_defaultBlockElements = blockElements;
788 infoM->m_hashTableSize = hashTableSize;
789 // create hash table start array
790 DWORD size;
791 size = sizeof(SOCmnListInfoSimpleList) * infoM->m_hashTableSize;
792 infoM->m_hashTable = malloc(size);
793 memset(infoM->m_hashTable, 0, size);
794 m_info = (void*)infoM;
795 }
796
797 m_state = (m_state & ~SOCMNLIST_STATE_AUTO_MASK) | SOCMNLIST_STATE_AUTO_NOTLOCK;
798 m_state |= SOCMNLIST_STATE_CREATED;
799 return ok;
800}

818BOOL SOCmnListHeader::clone(
819 IN SOCmnListHeader& list)
820{
821 SOCmnListConfig cfg;
822
823 if (!list.isInit())
824 {
825 return FALSE;
826 }
827
828 // no cloning of data lists
829 if ((m_type & SOCMNLIST_TYPE_OBJ_MASK) == SOCMNLIST_TYPE_OBJ_DATA)
830 {
831 return FALSE;
832 }
833
834 list.autoLock();
835 cfg.m_type = list.m_type;
836
837 if ((list.isList()) && ((list.m_type & SOCMNLIST_TYPE_INFO_FULL_LIST) != 0))
838 {
839 SOCmnListInfoFullList* infoF = (SOCmnListInfoFullList*)list.m_info;
840 cfg.m_entriesPerBlock = infoF->m_defaultBlockElements;
841 }
842
843 if (list.isMap())
844 {
845 SOCmnListInfoMap* infoM = (SOCmnListInfoMap*)list.m_info;
846 cfg.m_hashTableSize = infoM->m_hashTableSize;
847 cfg.m_entriesPerBlock = infoM->m_defaultBlockElements;
848 }
849
850 destroy();
851 create(&cfg);
852 SOCmnListInfoSimpleList* infoSD;
853 SOCmnListInfoSimpleList* infoSS;
854 SOCmnListEntry* curS;
855 SOCmnListEntry* curD;
856 BOOL loop = list.isMap();
857 DWORD hash = 0;
858 WORD cnt = 0;
859
860 do
861 {
862 if (loop)
863 {
864 SOCmnListInfoMap* infoMS = (SOCmnListInfoMap*)list.m_info;
865 infoSS = &(((SOCmnListInfoSimpleList*)infoMS->m_hashTable)[hash]);
866 SOCmnListInfoMap* infoMD = (SOCmnListInfoMap*)m_info;
867 infoSD = &(((SOCmnListInfoSimpleList*)infoMD->m_hashTable)[hash]);
868 hash++;
869
870 if (hash == infoMD->m_hashTableSize)
871 {
872 loop = FALSE;
873 }
874 }
875 else
876 {
877 infoSS = (SOCmnListInfoSimpleList*)list.m_info;
878 infoSD = (SOCmnListInfoSimpleList*)m_info;
879 }
880
881 curS = (SOCmnListEntry*)infoSS->m_start;
882 curD = NULL;
883
884 // first entry
885 if (curS)
886 {
887 infoSD->m_start = cloneEntry(curS);
888 curS = (SOCmnListEntry*)curS->m_next;
889 curD = (SOCmnListEntry*)infoSD->m_start;
890 cnt++;
891 }
892
893 // other entries
894 while (curS)
895 {
896 curD->m_next = cloneEntry(curS);
897 curS = (SOCmnListEntry*)curS->m_next;
898 curD = (SOCmnListEntry*)curD->m_next;
899 cnt++;
900 }
901
902 if (curD)
903 {
904 infoSD->m_end = (void*)curD;
905 }
906 }
907 while (loop);
908
909 if ((isList()) && ((m_type & SOCMNLIST_TYPE_INFO_FULL_LIST) != 0))
910 {
911 SOCmnListInfoFullList* infoF = (SOCmnListInfoFullList*)m_info;
912
913 if (cnt < 0xFFFF)
914 {
915 infoF->m_count = cnt;
916 }
917 else
918 {
919 infoF->m_count = 0xFFFF;
920 }
921 }
922
923 if (isMap())
924 {
925 SOCmnListInfoMap* infoM = (SOCmnListInfoMap*)m_info;
926
927 if (cnt < 0xFFFF)
928 {
929 infoM->m_count = cnt;
930 }
931 else
932 {
933 infoM->m_count = 0xFFFF;
934 }
935 }
936
937 setDirty(TRUE);
938 return TRUE;
939}

1209 }
1210 else
1211 {
1212 locKey = (void**)&objDWKey;

1216
1217 if (keyType == SOCMNLIST_TYPE_KEY_STRING)
1218 {

1220 }
1221 else
1222 {
1223 key = (void*)objDWKey;

1235 hash = hash % infoM->m_hashTableSize;

1296 ((IUnknown*)cur->m_object)->AddRef();
1297 }
1298
1299#endif // SOOS_WINDOWS
1300 break;
1301 }
1302 else
1303 {
1304 cur = cur->m_next;
1305 }
1306 }
1307
1308 if (!pos)
1309 {
1310 pos = add(obj, key, keyType);
1311 }
1312
1313 setDirty(TRUE);
1314 return pos;
1315}

2505 {

2512 case 53:

2515 case 151:

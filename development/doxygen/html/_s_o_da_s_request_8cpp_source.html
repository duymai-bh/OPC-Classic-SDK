50SODaSRequest::SODaSRequest(
51 IN BYTE operation, // request operation
52 IN WORD type) // request type
53 : SOCmnObject(SOCMNOBJECT_TYPE_REQUEST)
54{
55 m_operation = operation;
56 m_type = type;
57 m_item = NULL;
58 m_flag = 0;
59 setObjectState(stateInitialized);
60 m_startTime = 0;
61 m_result = S_OK;
62
63 switch (m_type)
64 {
65 case SODaSRequest::syncDevice:
66 case SODaSRequest::syncCache:
67 case SODaSRequest::syncMaxAge:
68 case SODaSRequest::serverDevice:
69 case SODaSRequest::serverMaxAge:
70 if (operation == SODaSRequest::read)
71 SOCMN_ALLOCATE_OBJECT(m_depSyncReadData, SODaSRequestSyncReadData())
72 else
73 SOCMN_ALLOCATE_OBJECT(m_depSyncWriteData, SODaSRequestSyncWriteData())
74 break;
75
76 case SODaSRequest::asyncMaxAge:
77 SOCMN_ALLOCATE_OBJECT(m_depAsyncReadData, SODaSRequestAsyncReadData())
78 case SODaSRequest::asyncDevice:
79 if (operation == SODaSRequest::write)
80 SOCMN_ALLOCATE_OBJECT(m_depAsyncWriteData, SODaSRequestAsyncWriteData())
81 break;
82
83 case SODaSRequest::report:
84 SOCMN_ALLOCATE_OBJECT(m_depReportData, SODaSRequestReportData())
85 setObjectState(stateCompleted);
86#ifdef SOFEATURE_WATCH
87 SODaSWatchHandler::doRequestStatistics(this, 0);
88#endif
89 break;
90
91 case SODaSRequest::delegatedDevice:
92 SOCMN_ALLOCATE_OBJECT(m_depDelegateData, SODaSRequestDelegateData)
93 break;
94 }
95}

504//-----------------------------------------------------------------------------
505// start
506//
507// - start request, set started flag
508// - add to queue
509// - signal start event
510//
511// returns:
512// TRUE - request started
513// FALSE - can't start request
514//
515BOOL SODaSRequest::start(
516 IN OPTIONAL BOOL signalQueue)
517{
518 SOCmnPointer<SODaSItemTag> tag = getItemTag();
519
520 if (!tag)
521 {
522 // item tag was released during starting of the request
523 complete(OPC_E_INVALIDITEMID, FALSE, OPC_QUALITY_UNCERTAIN, TRUE);
524 return FALSE; // -> error
525 }
526
527 SOCmnPointer<SODaSCache> cache = tag->getCache();
528
529 if (!cache)
530 {
531 complete(OPC_E_INVALIDITEMID, FALSE, OPC_QUALITY_UNCERTAIN, TRUE);
532 return FALSE; // cache not exists/ just report error
533 }
534
535 {
536 SOCmnSingleObjectLock<SOCmnSync> objectLock(this);
537
538 if (isStarted())
539 {
540 return TRUE;
541 }
542
543 setObjectState(stateStarted);
544 }
545
546 m_startTime = ::GetTickCount();
547
548 bool cacheExpired = false;
549 if (cache->isSet())
550 {
551 SOCmnDateTime timeNow, timeCache;
552 timeNow.now();
553 cache->getTimeStamp(timeCache);
554 timeCache.addMicroseconds(getMaxAge() * 1000);
555 if (timeNow.getMicroseconds() > timeCache.getMicroseconds())
556 {
557 cacheExpired = true;
558 }
559 }
560
561 if ((isDeviceRequest()) || (!cache->isSet()) || (cacheExpired))
562 {
563 // check cache age
564 if ((m_type == SODaSRequest::cyclic) || (m_type & SODASREQUEST_TYPE_MAXAGE))
565 {
566 SOCmnDateTime nowFT;
567 nowFT.now();
568 ULONGLONG nowMS = nowFT.getMicroseconds();
569 m_flag &= ~SODASREQUEST_FLAG_CACHE_AGE;
570
571 if (cache.isNotNull())
572 {
573 SOCmnDateTime cacheTime;
574 cache->getTimeStamp(this, cacheTime);
575 ULONGLONG cacheTimeMS = cacheTime.getMicroseconds();
576
577 if (m_type == SODaSRequest::cyclic)
578 {
579 // global cache age for cyclic requests
580 DWORD maxCacheAge = tag->getMaxAgeCyclicReq();
581
582 if (maxCacheAge != 0)
583 {
584 if (maxCacheAge == 0xFFFFFFFF)
585 {
586 // must be updated within the last group update rate period
587 SOCmnPointer<SODaSItem> item = getItem();
588
589 if (item.isNotNull())
590 {
591 SOCmnPointer<SODaSGroup> group = (SODaSGroup*)item->getParentBranch();
592
593 if (group.isNotNull())
594 {
595 SOCmnPointer<SODaSTransaction> trans = group->getCyclicTransaction();
596 SOCmnDateTime lastTransTime;
597 trans->getLastCompletionTime(lastTransTime);
598 ULONGLONG lastTransTimeMS = lastTransTime.getMicroseconds();
599
600 if ((cacheTimeMS > lastTransTimeMS) && (lastTransTimeMS != 0))
601 {
602 // newer cache
603 m_flag |= SODASREQUEST_FLAG_CACHE_AGE;
604 complete();
605 return TRUE;
606 }
607 }
608 }
609 }
610 else
611 {
612 if (((nowMS - cacheTimeMS) / 1000) < (ULONGLONG)maxCacheAge)
613 {
614 m_flag |= SODASREQUEST_FLAG_CACHE_AGE;
615 complete();
616 return TRUE;
617 }
618 }
619 }
620 }
621 else if (m_type & SODASREQUEST_TYPE_MAXAGE)
622 {
623 if (getMaxAge() > 0)
624 {
625 if (((nowMS - cacheTimeMS) / 1000) <= (ULONGLONG)getMaxAge())
626 {
627 m_flag |= SODASREQUEST_FLAG_CACHE_AGE;
628 complete();
629 return TRUE;
630 }
631 }
632 }
633 }
634 }
635
636 // device request
637 switch (tag->getDeviceIOMode(this))
638 {
639 case SODaSItemTag::direct:
640 {
641 tag->handleRequest(this);
642 }
643 break;
644
645 case SODaSItemTag::queue:
646 {
647 // insert request into item queue and set start event
648 // get the right queue and event
649 tag->addToQueue(this, m_operation == SODaSRequest::read);
650
651 if (signalQueue)
652 {
653 tag->signalQueueEvent(m_operation == SODaSRequest::read);
654 }
655 }
656 break;
657
658 case SODaSItemTag::cache:
659 case SODaSItemTag::constant:
660 {
661 complete();
662 }
663 break;
664
665 case SODaSItemTag::delegate:
666 {
667 startDelegateRequest();
668 }
669 break;
670
671 case SODaSItemTag::transaction:
672 break;
673 }
674 }
675 else
676 {
677 // cache request
678 complete();
679 }
680
681 return TRUE;
682} // start

685//-----------------------------------------------------------------------------
686// complete
687//
688// - complete request, set completed flag
689// - signal completed event if set
690//
691// returns:
692// TRUE - request completed
693// FALSE - can't complete request
694//
695BOOL SODaSRequest::complete(
696 IN OPTIONAL HRESULT result,
697 IN OPTIONAL BOOL validWriteQuality,
698 IN OPTIONAL WORD writeQuality,
699 IN OPTIONAL BOOL errorBeforeStart,
700 IN OPTIONAL BOOL writeChangeCache)
701{
702 DWORD completionSpan;
703 {
704 SOCmnSingleObjectLock<SOCmnSync> objectLock(this);
705
706 if (isCompleted())
707 {
708 return TRUE;
709 }
710
711 if ((!isStarted()) && (!errorBeforeStart))
712 {
713 return FALSE;
714 }
715
716 setResult(result); // request completed -> request handled
717 }
718
719 if (!errorBeforeStart)
720 {
721 SOCmnPointer<SODaSItemTag> tag = getItemTag();
722
723 if (tag.isNotNull())
724 {
725 SOCmnPointer<SODaSCache> cache = tag->getCache();
726
727 if (cache.isNotNull())
728 {
729 SOCmnPointer<SODaSItem> item = getItem();
730
731 if (m_operation == SODaSRequest::write)
732 {
733 // write operation
734 if (writeChangeCache)
735 {
736 // update cache
737 WORD cacheQuality, quality;
738 cacheQuality = cache->getQuality(this);
739
740 if (SUCCEEDED(getResult()))
741 {
742 SOCmnVariant cacheValue, writeValue;
743 cache->get(cacheValue, NULL, NULL, this);
744 getWriteValue(&writeValue);
745 cache->updateValue(writeValue, this);
746
747 if (!(getValidWriteQT() & SODASREQUEST_FLAG_QUALITY))
748 {
749 if (!validWriteQuality)
750 {
751 quality = OPC_QUALITY_GOOD;
752 }
753 else
754 {
755 quality = writeQuality;
756 }
757 }
758 else
759 {
760 quality = getWriteQuality();
761 }
762
763 cache->setQuality(quality, this);
764
765 if (getValidWriteQT() & SODASREQUEST_FLAG_TIMESTAMP)
766 {
767 SOCmnDateTime ts;
768 getWriteTimeStamp(ts);
769 cache->setTimeStamp(ts, this);
770 }
771
772 if (tag->getUpdateMode() == SODaSItemTag::report)
773 {
774 if ((quality != cacheQuality) || (writeValue != cacheValue))
775 {
776 tag->notifyAllConnectedObjects();
777 }
778 }
779 }
780 else
781 {
782 // write error
783 if (!validWriteQuality)
784 {
785 SOCmnDateTime timeStamp;
786 cache->getTimeStamp(this, timeStamp);
787
788 if (!timeStamp.isSet())
789 {
790 quality = OPC_QUALITY_LAST_KNOWN;
791 }
792 else
793 {
794 quality = OPC_QUALITY_COMM_FAILURE;
795 }
796 }
797 else
798 {
799 quality = writeQuality;
800 }
801
802 cache->setQuality(quality, this);
803
804 if (tag->getUpdateMode() == SODaSItemTag::report)
805 {
806 if (quality != cacheQuality)
807 {
808 tag->notifyAllConnectedObjects();
809 }
810 }
811 }
812 }
813
814 if (item.isNotNull())
815 {
816 item->onWrite();
817 }
818 }
819 else
820 {
821 // read operation
822 if ((tag->getDeviceIOMode(this) == SODaSItemTag::delegate) && SUCCEEDED(result))
823 {
824 completeDelegateRequest();
825 }
826
827 if (item.isNotNull())
828 {
829 item->onRead(isDeviceRequest());
830 }
831
832 if ((m_type != SODaSRequest::report) && (tag->getUpdateMode() == SODaSItemTag::report))
833 {
834 // values reported event driven must have a current timestamp
835 SOCmnDateTime now;
836 now.now();
837 if ((m_type & SODaSRequest::syncMaxAge) == SODaSRequest::syncMaxAge)
838 {
839 cache->setTimeStamp(now);
840 }
841 else
842 {
843 this->setReportTimeStamp(now);
844 }
845 }
846 if ((m_type == SODaSRequest::polledRefresh) && (item.isNotNull()))
847 {
848 WORD cacheQuality = cache->getQuality(this);
849
850 if ((cacheQuality & OPC_STATUS_MASK) < OPC_QUALITY_GOOD)
851 {
852 // check result of cyclic request
853 SOCmnPointer<SODaSRequest> cycReq = item->getCyclicRequest();
854
855 if (cycReq.isNotNull())
856 {
857 HRESULT cycRes = cycReq->getResult();
858
859 if (cycRes != S_OK)
860 {
861 setResult(cycRes);
862 }
863 }
864 }
865 }
866
867 if ((m_type & SODASREQUEST_TYPE_SYNC) == SODASREQUEST_TYPE_SYNC)
868 {
869 switch (m_type)
870 {
871 case SODaSRequest::syncDevice:
872 case SODaSRequest::syncCache:
873 {
874#ifdef SOFEATURE_DCOM
875 SOCmnVariant vV;
876 SOCmnDateTime tV;
877 OPCITEMSTATE* value;
878 value = getItemStateAddress();
879
880 if (SUCCEEDED(getResult()))
881 {
882 item = getItem();
883
884 if (item.isNotNull())
885 {
886 HRESULT convRes = S_OK;
887 value->hClient = item->getClientHandle();
888
889 if (cache->get(vV, &value->wQuality, &tV, this, item->getReqDatatype(), getLCID(), &convRes))
890 {
891 vV.copyTo(&value->vDataValue);
892 tV.get(&value->ftTimeStamp);
893 }
894 else
895 {
896 if (SUCCEEDED(getResult()))
897 {
898 setResult(convRes);
899 }
900 }
901
902 if (m_type == SODaSRequest::syncCache)
903 {
904 if (!item->isActive())
905 {
906 value->wQuality = OPC_QUALITY_OUT_OF_SERVICE;
907 }
908 }
909 }
910 else
911 {
912 setResult(E_FAIL);
913 }
914 }
915
916 if (FAILED(getResult()))
917 {
918 value->vDataValue.vt = VT_EMPTY;
919 item = getItem();
920
921 if (item.isNotNull())
922 {
923 value->hClient = item->getClientHandle();
924 }
925 }
926
927#endif
928 }
929 break;
930
931 case SODaSRequest::syncMaxAge:
932 case SODaSRequest::serverDevice:
933 case SODaSRequest::serverMaxAge:
934 {
935 SOCmnVariant vV;
936 SOCmnDateTime tV;
937#ifdef SOFEATURE_DCOM
938 LPVARIANT value;
939 value = getReadValueAddress();
940#endif
941#ifdef SOFEATURE_SOAP
942 SOCmnVariant* pV;
943 getReadValueAddress(pV);
944#endif
945
946 if (SUCCEEDED(getResult()))
947 {
948 SOCmnPointer<SODaSServer> server = getServer();
949 WORD* quality = getReadQualityAddress();
950#ifdef SOFEATURE_DCOM
951 FILETIME* timeStamp = getReadTimeStampAddress();
952#endif
953#ifdef SOFEATURE_SOAP
954 SOCmnDateTime* pT;
955 getReadTimeStampAddress(pT);
956#endif
957 LCID lcid = MAKELCID(LANG_SYSTEM_DEFAULT, SORT_DEFAULT);
958 HRESULT convRes = S_OK;
959
960 if (server.isNotNull())
961 {
962 lcid = server->getLCID();
963 }
964
965 VARTYPE reqVarType = getReqDatatype();
966
967 if (item.isNotNull() == TRUE)
968 {
969 reqVarType = item->getReqDatatype();
970 }
971
972 if (cache->get(vV, quality, &tV, this, reqVarType, lcid, &convRes))
973 {

1195//-----------------------------------------------------------------------------
1196// hasChanged
1197//
1198// - check if the value of the requests cache has changed since last advise
1199//
1200// returns:
1201// TRUE - changed
1202// FALSE - same
1203//
1204BOOL SODaSRequest::hasChanged(
1205 IN size_t connection, // connection id
1206 IN SOCmnVariant& value,
1207 IN WORD quality)
1208{
1209 if ((m_type != SODaSRequest::asyncInitNotify) && (m_type != SODaSRequest::cyclic) &&
1210 (m_type != SODaSRequest::report) && (m_type != SODaSRequest::polledRefresh))
1211 {
1212 return TRUE;
1213 }
1214
1215 SOCmnPointer<SODaSItem> item = getItem();
1216
1217 if (!item)
1218 {
1219 return TRUE;
1220 }
1221
1222 if (!item->isActive())
1223 {
1224 return FALSE;
1225 }
1226
1227 //-- cyclic requests --
1228 SOCmnVariant prevSendValue;
1229 WORD prevSendQuality;
1230 BOOL valueChanged;
1231
1232 //-- get previous send value for the connection --
1233 if (!item->getSendValue(connection, prevSendValue, &prevSendQuality))
1234 {
1235 if ((m_type == SODaSRequest::cyclic) || (m_type == SODaSRequest::report))
1236 {
1237 SOCmnPointer<SODaSGroup> group = (SODaSGroup*)item->getParentBranch();
1238
1239 if (group.isNotNull())
1240 {
1241 if (group->rejectCyclicDataBeforeInit())
1242 {
1243 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE, (_T("cyclic/report before asyncInitNotify")));
1244 return FALSE;
1245 }
1246 }
1247 }
1248
1249 return TRUE;
1250 }
1251
1252 //-- compare quality --
1253 if (prevSendQuality != quality)
1254 {
1255 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE, (_T("changed quality")));
1256 SOCmnVariant::variantClear(&prevSendValue);
1257 return TRUE;
1258 }
1259
1260 //-- compare value --
1261 valueChanged = !(prevSendValue == (LPCVARIANT)value);
1262
1263 //-- percent deadband --
1264 if (valueChanged)
1265 {
1266 // check deadband only if value changed.
1267 SOCmnPointer<SODaSGroup> group = (SODaSGroup*)item->getParentBranch();
1268
1269 if (group.isNotNull())
1270 {
1271 SOCmnPointer<SODaSItemTag> tag = getItemTag();
1272
1273 if (tag.isNotNull())
1274 {
1275 SOCmnPointer<SODaSCache> cache = getCache();
1276
1277 if (tag->getEUType() == OPC_ANALOG)
1278 {
1279 // percent deadband only counts for analog tags
1280 SOCmnVariant limits;
1281 SOCmnVariant locvalue;
1282 double* limit;
1283 double deadband = 0.0;
1284 FLOAT percDB = group->getDeadBand(); // group deadband (default 0.0)
1285 item->getDeadBand(percDB); // item deadband (if available)
1286 // calculate deadband
1287 tag->getEUInfo(limits);
1288
1289 if (limits.vt == (VT_R8 | VT_ARRAY))
1290 {
1291 if (limits.parray->rgsabound[0].cElements == 2)
1292 {
1293 SafeArrayAccessData(limits.parray, (void**)&limit);
1294 deadband = (limit[1] - limit[0]) * ((double)percDB / 100.0);
1295 SafeArrayUnaccessData(limits.parray);
1296 }
1297 }
1298
1299 // convert values to double
1300 if ((cache->changeDatatype(prevSendValue, VT_R8, prevSendValue)) &&
1301 (cache->changeDatatype(value, VT_R8, locvalue)))
1302 {
1303 valueChanged = (fabs(prevSendValue.dblVal - locvalue.dblVal) > fabs(deadband));
1304 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE, (_T("hasChanged %s deadband %lf difference %lf"), BOOL2STR(valueChanged), fabs(deadband), fabs(prevSendValue.dblVal - locvalue.dblVal)));
1305 }
1306 }
1307 }
1308 }
1309 }
1310
1311 SOCmnVariant::variantClear(&prevSendValue);
1312
1313 if (valueChanged)
1314 {
1315 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE, (_T("changed value")));
1316 return TRUE;
1317 }
1318
1319 return FALSE;
1320} // hasChanged

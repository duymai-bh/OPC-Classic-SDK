158 [SecurityPermission(SecurityAction.LinkDemand, Flags = SecurityPermissionFlag.UnmanagedCode)]
159 protected override void WndProc(ref Message m)
160 {
161 if (m.Msg == (int)Win32.Msgs.WM_NCLBUTTONDOWN)
162 {
163 if (IsDisposed)
164 return;
165
166 uint result = NativeMethods.SendMessage(this.Handle, (int)Win32.Msgs.WM_NCHITTEST, 0, (uint)m.LParam);
167 if (result == 2 && DockPanel.AllowEndUserDocking && this.AllowEndUserDocking) // HITTEST_CAPTION
168 {
169 Activate();
170 m_dockPanel.BeginDrag(this);
171 }
172 else
173 base.WndProc(ref m);
174
175 return;
176 }
177 else if (m.Msg == (int)Win32.Msgs.WM_NCRBUTTONDOWN)
178 {
179 uint result = NativeMethods.SendMessage(this.Handle, (int)Win32.Msgs.WM_NCHITTEST, 0, (uint)m.LParam);
180 if (result == 2) // HITTEST_CAPTION
181 {
182 DockPane theOnlyPane = (VisibleNestedPanes.Count == 1) ? VisibleNestedPanes[0] : null;
183 if (theOnlyPane != null && theOnlyPane.ActiveContent != null)
184 {
185 theOnlyPane.ShowTabPageContextMenu(this, PointToClient(Control.MousePosition));
186 return;
187 }
188 }
189
190 base.WndProc(ref m);
191 return;
192 }
193 else if (m.Msg == (int)Win32.Msgs.WM_CLOSE)
194 {
195 if (NestedPanes.Count == 0)
196 {
197 base.WndProc(ref m);
198 return;
199 }
200
201 for (int i = NestedPanes.Count - 1; i >= 0; i--)
202 {
203 DockContentCollection contents = NestedPanes[i].Contents;
204 for (int j = contents.Count - 1; j >= 0; j--)
205 {
206 IDockContent content = contents[j];
207 if (content.DockHandler.DockState != DockState.Float)
208 continue;
209
210 if (!content.DockHandler.CloseButton)
211 continue;
212
213 if (content.DockHandler.HideOnClose)
214 content.DockHandler.Hide();
215 else
216 content.DockHandler.Close();
217 }
218 }
219
220 return;
221 }
222 else if (m.Msg == (int)Win32.Msgs.WM_NCLBUTTONDBLCLK)
223 {
224 uint result = NativeMethods.SendMessage(this.Handle, (int)Win32.Msgs.WM_NCHITTEST, 0, (uint)m.LParam);
225 if (result != 2) // HITTEST_CAPTION
226 {
227 base.WndProc(ref m);
228 return;
229 }
230
231 DockPanel.SuspendLayout(true);
232
233 // Restore to panel
234 foreach (DockPane pane in NestedPanes)
235 {
236 if (pane.DockState != DockState.Float)
237 continue;
238 pane.RestoreToPanel();
239 }
240
241
242 DockPanel.ResumeLayout(true, true);
243 return;
244 }
245 else if (m.Msg == WM_CHECKDISPOSE)
246 {
247 if (NestedPanes.Count == 0)
248 Dispose();
249
250 return;
251 }
252
253 base.WndProc(ref m);
254 }

343 public void DockTo(DockPane pane, DockStyle dockStyle, int contentIndex)
344 {
345 if (dockStyle == DockStyle.Fill)
346 {
347 for (int i = NestedPanes.Count - 1; i >= 0; i--)
348 {
349 DockPane paneFrom = NestedPanes[i];
350 for (int j = paneFrom.Contents.Count - 1; j >= 0; j--)
351 {
352 IDockContent c = paneFrom.Contents[j];
353 c.DockHandler.Pane = pane;
354 if (contentIndex != -1)
355 pane.SetContentIndex(c, contentIndex);
356 }
357 }
358 }
359 else
360 {
361 DockAlignment alignment = DockAlignment.Left;
362 if (dockStyle == DockStyle.Left)
363 alignment = DockAlignment.Left;
364 else if (dockStyle == DockStyle.Right)
365 alignment = DockAlignment.Right;
366 else if (dockStyle == DockStyle.Top)
367 alignment = DockAlignment.Top;
368 else if (dockStyle == DockStyle.Bottom)
369 alignment = DockAlignment.Bottom;
370
371 MergeNestedPanes(VisibleNestedPanes, pane.NestedPanesContainer.NestedPanes, pane, alignment, 0.5);
372 }
373 }

375 public void DockTo(DockPanel panel, DockStyle dockStyle)
376 {
377 if (panel != DockPanel)
378 throw new ArgumentException(Strings.IDockDragSource_DockTo_InvalidPanel, "panel");
379
380 NestedPaneCollection nestedPanesTo = null;
381
382 if (dockStyle == DockStyle.Top)
383 nestedPanesTo = DockPanel.DockWindows[DockState.DockTop].NestedPanes;
384 else if (dockStyle == DockStyle.Bottom)
385 nestedPanesTo = DockPanel.DockWindows[DockState.DockBottom].NestedPanes;
386 else if (dockStyle == DockStyle.Left)
387 nestedPanesTo = DockPanel.DockWindows[DockState.DockLeft].NestedPanes;
388 else if (dockStyle == DockStyle.Right)
389 nestedPanesTo = DockPanel.DockWindows[DockState.DockRight].NestedPanes;
390 else if (dockStyle == DockStyle.Fill)
391 nestedPanesTo = DockPanel.DockWindows[DockState.Document].NestedPanes;
392
393 DockPane prevPane = null;
394 for (int i = nestedPanesTo.Count - 1; i >= 0; i--)
395 if (nestedPanesTo[i] != VisibleNestedPanes[0])
396 prevPane = nestedPanesTo[i];
397 MergeNestedPanes(VisibleNestedPanes, nestedPanesTo, prevPane, DockAlignment.Left, 0.5);
398 }

400 private static void MergeNestedPanes(VisibleNestedPaneCollection nestedPanesFrom, NestedPaneCollection nestedPanesTo, DockPane prevPane, DockAlignment alignment, double proportion)
401 {
402 if (nestedPanesFrom.Count == 0)
403 return;
404
405 int count = nestedPanesFrom.Count;
406 DockPane[] panes = new DockPane[count];
407 DockPane[] prevPanes = new DockPane[count];
408 DockAlignment[] alignments = new DockAlignment[count];
409 double[] proportions = new double[count];
410
411 for (int i = 0; i < count; i++)
412 {
413 panes[i] = nestedPanesFrom[i];
414 prevPanes[i] = nestedPanesFrom[i].NestedDockingStatus.PreviousPane;
415 alignments[i] = nestedPanesFrom[i].NestedDockingStatus.Alignment;
416 proportions[i] = nestedPanesFrom[i].NestedDockingStatus.Proportion;
417 }
418
419 DockPane pane = panes[0].DockTo(nestedPanesTo.Container, prevPane, alignment, proportion);
420 panes[0].DockState = nestedPanesTo.DockState;
421
422 for (int i = 1; i < count; i++)
423 {
424 for (int j = i; j < count; j++)
425 {
426 if (prevPanes[j] == panes[i - 1])
427 prevPanes[j] = pane;
428 }
429 pane = panes[i].DockTo(nestedPanesTo.Container, prevPanes[i], alignments[i], proportions[i]);
430 panes[i].DockState = nestedPanesTo.DockState;
431 }
432 }

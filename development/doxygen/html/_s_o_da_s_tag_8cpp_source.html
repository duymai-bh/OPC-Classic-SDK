106//-----------------------------------------------------------------------------
107// release
108//
109// - check if to delete dynamic created tags
110//
111// returns:
112// reference count
113//
114LONG SODaSTag::release(void)
115{
116 if (getDynamic())
117 {
118 SOCmnPointer<SOCmnElement> parent = getParentBranch();
119 SOCmnList<SODaSProperty> *pList = (SOCmnList<SODaSProperty> *)getAdditionalList();
120 SODaSProperty* prop;
121 SOCmnListPosition pos;
122 int nRefs = 2;
123#ifdef SOFEATURE_WATCH
124
125 if (getCollectChanges())
126 {
127 nRefs++;
128 }
129
130#endif
131
132 if (pList != NULL)
133 {
134 SOCmnList<SODaSProperty> propList(pList);
135
136 if (!!propList)
137 {
138 pos = propList.getStartPosition();
139
140 while (pos)
141 {
142 prop = propList.getNext(pos);
143
144 if (prop->getRefCount() == 1)
145 {
146 nRefs ++;
147 }
148 }
149
150 propList.autoUnlock();
151 }
152 }
153
154 if (getRefCount() == nRefs)
155 {
156 if (!(!parent))
157 {
158 SOCmnList<SOCmnElement> leafList(parent->getLeafList());
159 leafList.lock();
160
161 if (getRefCount() == nRefs)
162 {
163 setDynamic(FALSE); // prevent recursion
164 removeAllAdditionalElements();
165 parent->removeLeaf(this);
166 }
167
168 leafList.unlock();
169 }
170 }
171 }
172
173 return SOCmnObject::release();
174} // release

177//-----------------------------------------------------------------------------
178// setParentBranch
179//
180// - set parent branch of this object
181// - check if the elements name is unique in parent
182//
183// returns:
184// TRUE - branch set
185// FALSE - no unique name
186//
187BOOL SODaSTag::setParentBranch(
188 IN SOCmnElement* parent) // parent branch
189{
190 BOOL ret;
191#ifdef SOFEATURE_WATCH
192 BOOL doCollect = FALSE;
193 SOCmnPointer<SOCmnElement> prevParent = getParentBranch();
194
195 if (prevParent.isNotNull())
196 {
197 doCollect = prevParent->getCollectChanges();
198 }
199 else if (parent)
200 {
201 doCollect = parent->getCollectChanges();
202 }
203
204#endif
205 ret = SOCmnElement::setParentBranch(parent);
206#ifdef SOFEATURE_WATCH
207
208 if ((doCollect || (getCollectChanges())) && (ret))
209 {
210 SOCmnPointer<SOSrvEntry> entry = ::getSOSrvEntry();
211
212 if (parent != NULL)
213 {
214 entry->watchAddObject(parent, this);
215 }
216 else if ((parent == NULL) && (prevParent.isNotNull()))
217 {
218 entry->watchRemoveObject(prevParent, this);
219 prevParent.release();
220 }
221 }
222
223#endif
224 return ret;
225} // setParentBranch

228//-----------------------------------------------------------------------------
229// getDeviceIOMode
230//
231// - get the device I/O mode of the tag
232//
233// returns:
234// device I/O mode
235//
236BYTE SODaSTag::getDeviceIOMode(
237 IN SODaSRequest* req) // request
238{
239 if (req)
240 if ((getUpdateMode() == SODaSItemTag::report) && (req->getOperation() == SODaSRequest::read))
241 {
242 SOCmnPointer<SODaSCache> cache = getCache();
243
244 if (cache.isNotNull())
245 if (cache->isSet())
246 {
247 return SODaSItemTag::cache; // use the cache value for event driven read operations
248 }
249 }
250
251 if (!(m_deviceIOAndUpdateMode & SODASITEMTAG_MASK_DEVICEIOMODE)) // default I/O mode for tags is queued
252 {
253 return SODaSItemTag::queue;
254 }
255 else
256 {
257 return m_deviceIOAndUpdateMode & SODASITEMTAG_MASK_DEVICEIOMODE;
258 }
259} // getDeviceIOMode

263//-----------------------------------------------------------------------------
264// getObjWatchData
265//
266// - get the watch data of the object
267//
268void SODaSTag::getObjWatchData(
269 IN SOCmnStringEx& szXML, // XML data
270 IN BOOL withAttributes, // add attributes to watch data
271 IN LPTSTR objNodeName) // name of XML object node
272{
273 szXML.getBuffer(256);
274 SOSrvWatch::addStartElementText(szXML, objNodeName);
275 // object parameter elements
276 SOSrvWatch::createElementText(szXML, SOWATCH_ELEMENT_SPACE_T, SOWATCH_SPACE_DA_NAMESPACE);
277 SOSrvWatch::createElementText(szXML, SOWATCH_ELEMENT_TYPE_T, SOWATCH_TYPE_DA_TAG);
278 SOSrvWatch::createElementText(szXML, SOWATCH_ELEMENT_SERVERHANDLE_T, (DWORD)this->getOpcHandle());
279
280 // object attributes
281 if (withAttributes)
282 {
283 SOCmnString buffer;
284 SOCmnPointer<SOSrvConverter> conv = ::getSOSrvConverter();
285 SOCmnPointer<SODaSCache> cache = getCache();
286 SOCmnDateTime timeStamp;
287 WORD quality;
288 SOCmnStringEx szRawAttr;
289 SOSrvWatch::addStartElementText(szXML, SOWATCH_ELEMENT_ATTRIBUTES_T);
290 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_NAME_T, getName());
291 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_PATH_T, getPath());
292 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_DEVSTATE_T, getDeviceState());
293 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_DEVSTATEINFO_T, getDeviceStateInfo());
294 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_ITEMID_T, getItemID());
295 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_VALUE_T, cache->getValueString(buffer));
296 quality = cache->getQuality();
297 szRawAttr.format(_T("%s = \"%u\""), SOWATCH_ATTR_RAW_T, quality);
298 SOSrvWatch::createElementTextWithAttributes(szXML, SOWATCH_ATTRIBUTE_QUALITY_T, conv->quality2string(buffer, quality), szRawAttr);
299 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_DATATYPE_T, conv->vartype2string(buffer, getNativeDatatype()));
300 cache->getTimeStamp(timeStamp);
301 szRawAttr.format(_T("%s = \"%lu\""), SOWATCH_ATTR_RAW_T, DWORD(timeStamp.getMicroseconds() / 1000));
302 SOSrvWatch::createElementTextWithAttributes(szXML, SOWATCH_ATTRIBUTE_TIMESTAMP_T, conv->datetime2string(buffer, timeStamp), szRawAttr);
303 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_ACCESSRIGHTS_T, conv->accessRights2string(buffer, getAccessRights()));
304
305 if (getEUType() != OPC_NOENUM)
306 {
307 SOCmnVariant euInfo;
308 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_EUTYPE_T, conv->euType2string(buffer, getEUType()));
309 getEUInfo(euInfo);
310 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_EUINFO_T, conv->variant2string(buffer, euInfo));
311 }
312
313 SOCmnList<SOCmnElement> *pList = getAdditionalList();
314 BOOL hasProps = FALSE;
315
316 if (pList != NULL)
317 if (pList->isInit())
318 {
319 SOCmnList<SODaSProperty> propList((SOCmnList<SODaSProperty> *)pList);
320 hasProps = !propList.isEmpty();
321 }
322
323 SOSrvWatch::createElementText(szXML, SOWATCH_ATTRIBUTE_HASPROPERTIES_T, conv->bool2string(buffer, hasProps));
324 SOSrvWatch::addEndElementText(szXML, SOWATCH_ELEMENT_ATTRIBUTES_T);
325 }
326
327 SOSrvWatch::addEndElementText(szXML, objNodeName);
328} // getObjWatchData

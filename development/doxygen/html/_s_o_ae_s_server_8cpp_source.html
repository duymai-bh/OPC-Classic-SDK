129//-----------------------------------------------------------------------------
130// sendEventNotifications
131//
132// - send event notifications for all subscriptions
133//
134// returns:
135// time in milliseconds until next call is needed
136//
137DWORD SOAeSServer::sendEventNotifications(
138 IN DWORD now, // current time
139 IN DWORD minSpan) // minimal time till next call
140{
141 SOCmnList<SOAeSSubscription> list((SOCmnList<SOAeSSubscription> *)getLeafList());
142 SOCmnPointer<SOAeSSubscription> subscr;
143 SOCmnListPosition pos;
144 DWORD actSpan;
145 BOOL end;
146 DWORD listCnt;
147 // for every subscription
148 list.dontAutoLock();
149 list.lock();
150 list.setDirty(FALSE);
151 pos = list.getStartPosition();
152 list.unlock();
153
154 while (pos)
155 {
156 list.lock();
157
158 if (list.isDirty())
159 {
160 // check current position!
161 SOCmnListPosition checkPos;
162 BOOL found = FALSE;
163 checkPos = list.getStartPosition();
164
165 while (checkPos)
166 {
167 if (checkPos == pos)
168 {
169 found = TRUE;
170 break;
171 }
172
173 list.getNext(checkPos);
174 }
175
176 if (!found)
177 {
178 pos = list.getStartPosition();
179 }
180
181 list.setDirty(FALSE);
182
183 if (pos == NULL)
184 {
185 list.unlock();
186 break;
187 }
188 }
189
190 subscr = list.getNext(pos);
191 subscr.addRef();
192 list.unlock();
193 end = FALSE;
194 actSpan = 0;
195
196 do
197 {
198 // send events
199 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SRV_THREAD, (_T("send events %s"), subscr->getObjTraceId()))
200
201 if ((listCnt = subscr->doSendEvents(FALSE)) > 0)
202 {
203 // not all events send
204 // check max size:
205 // if count is greater than maxSize send the next events
206 // if not check the send time
207 if ((subscr->getMaxSize() == 0) || (listCnt <= subscr->getMaxSize()))
208 {
209 // get time till next send is allowed
210 actSpan = subscr->checkSendTime(now);
211
212 if (actSpan > 0)
213 {
214 end = TRUE; // not now
215 }
216 }
217 }
218 else
219 {
220 end = TRUE; // all events of the list send
221 }
222 }
223 while (!end);
224
225 if ((actSpan > 0) && (actSpan < minSpan))
226 {
227 minSpan = actSpan;
228 }
229
230 // check for the end event of the work thread
231
232 if (getEndEvent().waitFor(0))
233 {
234 return 0;
235 }
236 }
237
238 return minSpan;
239} // sendEventNotifications

242//-----------------------------------------------------------------------------
243// handleRefresh
244//
245// - send event notifications for all subscriptions
246//
247// returns:
248// time in milliseconds until next call is needed
249//
250DWORD SOAeSServer::handleRefresh(
251 IN DWORD /* now */, // current time
252 IN DWORD minSpan) // minimal time till next call
253{
254 SOCmnList<SOAeSSubscription> list((SOCmnList<SOAeSSubscription> *)getLeafList());
255 SOCmnPointer<SOAeSSubscription> subscr;
256 SOCmnListPosition pos;
257 DWORD actSpan;
258 SOCmnPointer<SOAeSRefresh> refresh;
259 // for every subscription
260 list.dontAutoLock();
261 list.lock();
262 list.setDirty(FALSE);
263 pos = list.getStartPosition();
264 list.unlock();
265
266 while (pos)
267 {
268 list.lock();
269
270 if (list.isDirty())
271 {
272 // check current position!
273 SOCmnListPosition checkPos;
274 BOOL found = FALSE;
275 checkPos = list.getStartPosition();
276
277 while (checkPos)
278 {
279 if (checkPos == pos)
280 {
281 found = TRUE;
282 break;
283 }
284
285 list.getNext(checkPos);
286 }
287
288 if (!found)
289 {
290 pos = list.getStartPosition();
291 }
292
293 list.setDirty(FALSE);
294
295 if (pos == NULL)
296 {
297 list.unlock();
298 break;
299 }
300 }
301
302 subscr = list.getNext(pos);
303 subscr.addRef();
304 list.unlock();
305 refresh = subscr->getRefresh();
306 actSpan = 0;
307
308 if (!!refresh)
309 {
310 // refresh object set
311 BYTE state = refresh->getObjectState();
312 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SRV_THREAD, (_T("handle refresh %s; state 0x%X"), subscr->getObjTraceId(), state))
313
314 // check state of refresh
315 if (state & (BYTE)SOAeSRefresh::created)
316 {
317 // collect all events from the condition objects
318 SOCmnPointer<SOAeSEntry> entry = ::getSOAeSEntry();
319 SOCmnPointer<SOAeSAreaSpaceRoot> areaRoot = entry->getAreaSpaceRoot();
320 SOCmnList<SOAeSEvent> eventList = refresh->getEventList();
321 // for all conditions ...
322 aggregateConditions(areaRoot, &eventList);
323 // set aggregated state
324 state &= ~SOAeSRefresh::created;
325 state |= SOAeSRefresh::aggregated;
326 refresh->setObjectState(state);
327 }
328
329 if ((state & (BYTE)SOAeSRefresh::collected) &&
330 (state & (BYTE)SOAeSRefresh::aggregated))
331 {
332 // all events collected
333 SOCmnListPosition refPos;
334 SOAeSEvent* event;
335 SOCmnList<SOAeSEvent> eventList = refresh->getEventList();
336 // filter the refresh events
337 refPos = eventList.getStartPosition();
338
339 while (refPos)
340 {
341 event = eventList.getNext(refPos);
342
343 if (!subscr->filterEvent(event))
344 {
345 // event filtered out
346 eventList.removeObject(event);
347 }
348 }
349
350 // switch to send state
351 state = SOAeSRefresh::send;
352 refresh->setObjectState(state);
353 }
354
355 if (state & (BYTE)SOAeSRefresh::canceled)
356 {
357 // refresh cancled
358 // send last refresh notification for cancel
359 subscr->doSendEvents(TRUE);
360 // release refresh object
361 subscr->releaseRefresh();
362 }
363 else if (state & (BYTE)SOAeSRefresh::send)
364 {
365 // send refresh events
366 DWORD remainingCnt;
367
368 do
369 {
370 // send events
371 remainingCnt = subscr->doSendEvents(TRUE);
372 }
373 while (remainingCnt > 0);
374
375 // release refresh object
376 subscr->releaseRefresh();
377 }
378 }
379
380 if ((actSpan > 0) && (actSpan < minSpan))
381 {
382 minSpan = actSpan;
383 }
384 }
385
386 return minSpan;
387} // handleRefresh

390//-----------------------------------------------------------------------------
391// aggregateConditions
392//
393// - collect events for all active or inactive unacked conditions
394//
395void SOAeSServer::aggregateConditions(
396 IN SOCmnElement* parent, // parent element
397 OUT SOCmnList<SOAeSEvent> *eventList) // event list
398{
399 SOCmnList<SOAeSArea> areaList;
400 SOAeSArea* area;
401 SOCmnListPosition posA;
402 SOCmnList<SOAeSSource> sourceList;
403 SOAeSSource* source;
404 SOCmnListPosition posS;
405 SOCmnList<SOAeSCondition> condList;
406 SOAeSCondition* cond;
407 SOCmnListPosition posC;
408 condList = (SOCmnList<SOAeSCondition> *)parent->getAdditionalList();
409
410 if (!!condList)
411 {
412 SOCmnPointer<SOAeSEvent> event;
413 WORD condState;
414 posC = condList.getStartPosition();
415
416 while (posC)
417 {
418 cond = condList.getNext(posC);
419 condState = cond->getObjectState();
420
421 if ((condState & OPC_CONDITION_ACTIVE) ||
422 ((!(condState & OPC_CONDITION_ACTIVE)) && (!(condState & OPC_CONDITION_ACKED))))
423 {
424 // conditin active or (inactive and unacknowledged)
425 event = cond->createEvent(0);
426 eventList->add((SOAeSEvent*)event);
427 }
428 }
429 }
430
431 sourceList = (SOCmnList<SOAeSSource> *)parent->getLeafList();
432
433 if (!!sourceList)
434 {
435 posS = sourceList.getStartPosition();
436
437 while (posS)
438 {
439 source = sourceList.getNext(posS);
440 aggregateConditions(source, eventList);
441 }
442 }
443
444 areaList = (SOCmnList<SOAeSArea> *)parent->getBranchList();
445
446 if (!!areaList)
447 {
448 posA = areaList.getStartPosition();
449
450 while (posA)
451 {
452 area = areaList.getNext(posA);
453 aggregateConditions(area, eventList);
454 }
455 }
456} // aggregateConditions

1052//-----------------------------------------------------------------------------
1053// checkClientConnection
1054//
1055// - check if the connection to the client still exitsts
1056//
1057// returns:
1058// TRUE - client is running
1059// FALSE - client does not react
1060//
1061BOOL SOAeSServer::checkClientConnection(OPTIONAL IN BOOL checkAllIFs)
1062{
1063 BOOL conOK = TRUE;
1064 BOOL checked = FALSE;
1065 SOSrvComObject<SOAeSComServer> *srv = (SOSrvComObject<SOAeSComServer> *)getMe();
1066
1067 if (!srv)
1068 {
1069 return FALSE;
1070 }
1071
1072 if (srv->connectedShutdownInterface())
1073 {
1074 conOK = srv->checkClientConnection();
1075
1076 if (!checkAllIFs)
1077 {
1078 checked = TRUE;
1079 }
1080 }
1081
1082 if (!checked)
1083 {
1084 int nTmpSubs;
1085 SOAeSSubscription** pTmpSubList = allocSubscriptionsAsArray(&nTmpSubs);
1086 SOCmnList<IOPCEventSink> esList;
1087 IOPCEventSink* es;
1088 SOCmnListPosition esPos;
1089 IUnknown* unk;
1090 HRESULT res;
1091
1092 for (int i = 0; !checked && i < nTmpSubs; i++)
1093 {
1094 SOAeSSubscription* pSub = pTmpSubList[i];
1095
1096 if (pSub->createConnectedEventSinkInterfaceList(&esList))
1097 {
1098 esPos = esList.getStartPosition();
1099
1100 while ((esPos) && (!checked))
1101 {
1102 if (!checkAllIFs)
1103 {
1104 checked = TRUE;
1105 }
1106
1107 es = esList.getNext(esPos);
1108 res = es->QueryInterface(IID_IOPCEventServer, (void**)&unk);
1109
1110 if (FAILED(res))
1111 {
1112 if (SOSRV_IS_RPC_ERROR(res))
1113 {
1114 conOK = FALSE;
1115 esPos = NULL;
1116 break;
1117 }
1118 }
1119 else
1120 {
1121 unk->Release();
1122 }
1123 }
1124 }
1125
1126 esList.destroy();
1127 }
1128
1129 freeSubscriptionsAsArray(pTmpSubList);
1130 }
1131
1132 releaseMe(srv);
1133 return conOK;
1134} // checkClientConnection

1181//-----------------------------------------------------------------------------
1182// disconnectClient
1183//
1184// - disconnect the OPC client (forced shutdown)
1185//
1186void SOAeSServer::disconnectClient(void)
1187{
1188 SOSrvComObject<SOAeSComServer> *srv = (SOSrvComObject<SOAeSComServer> *)getMe();
1189
1190 if (!srv)
1191 {
1192 return;
1193 }
1194
1195 DWORD extRefs;
1196 _TRACE(SOCMNTRACE_L_WRN, SOCMNTRACE_G_OBJ_REF, (_T("disconnect client %s"), (LPCTSTR)getName()))
1197 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OBJ_REF, (_T("disconnecting refCnt:%u intRefCnt:%u"), srv->m_dwRef, (ULONG)srv->m_internalCnt))
1198 SOSrvComObject<SOAeSComSubscription> *subscr;
1199 SOSrvComObject<SOAeSComAreaBrowser> *areaBrowser;
1200 SOCmnList<SOCmnElement> susList;
1201 SOCmnList<SOCmnElement> abList;
1202 SOAeSSubscription* sus;
1203 SOAeSAreaBrowser* ab;
1204 SOCmnListPosition pos;
1205 // disconnect the subscriptions
1206 susList = getLeafList();
1207 pos = susList.getStartPosition();
1208
1209 while (pos)
1210 {
1211 sus = (SOAeSSubscription*)susList.getNext(pos);
1212 subscr = (SOSrvComObject<SOAeSComSubscription> *)sus->getMe();
1213
1214 if (subscr)
1215 {
1216 extRefs = subscr->m_dwRef - (subscr->m_internalCnt + subscr->m_tempCnt);
1217 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OBJ_REF, (_T("disconnecting subscription refCnt:%u intRefCnt:%u"), subscr->m_dwRef, (ULONG)subscr->m_internalCnt));
1218 subscr->releaseComServer();
1219
1220 if (subscr->m_internalCnt > 0)
1221 while (subscr->internalRelease())
1222 ;
1223
1224 sus->releaseMe(subscr);
1225
1226 if (extRefs)
1227 {
1228 CoDisconnectObject((IOPCEventSubscriptionMgt*)subscr, 0);
1229 }
1230 }
1231 }
1232
1233 susList.removeAll();
1234 susList.autoUnlock();
1235 // disconnect area browser
1236 abList = getAdditionalList();
1237 pos = abList.getStartPosition();
1238
1239 while (pos)
1240 {
1241 ab = (SOAeSAreaBrowser*)abList.getNext(pos);
1242 areaBrowser = (SOSrvComObject<SOAeSComAreaBrowser> *)ab->getMe();
1243
1244 if (areaBrowser)
1245 {
1246 extRefs = areaBrowser->m_dwRef - (areaBrowser->m_internalCnt + areaBrowser->m_tempCnt);
1247 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OBJ_REF, (_T("disconnecting area browser refCnt:%u intRefCnt:%u"), areaBrowser->m_dwRef, (ULONG)areaBrowser->m_internalCnt));
1248 areaBrowser->releaseComServer();
1249
1250 if (areaBrowser->m_internalCnt > 0)
1251 while (areaBrowser->internalRelease())
1252 ;
1253
1254 ab->releaseMe(areaBrowser);
1255
1256 if (extRefs)
1257 {
1258 CoDisconnectObject((IOPCEventAreaBrowser*)areaBrowser, 0);
1259 }
1260 }
1261 }
1262
1263 abList.removeAll();
1264 abList.autoUnlock();
1265 srv->releaseAllAdvises();
1266 // disconnect server
1267 extRefs = srv->m_dwRef - (srv->m_internalCnt + srv->m_tempCnt);
1268
1269 while (srv->internalRelease())
1270 ;
1271
1272 releaseMe(srv); // use after release is ok in this case!
1273
1274 if (extRefs)
1275 {
1276 CoDisconnectObject((IOPCServer*)srv, 0);
1277 }
1278} // disconnectClient

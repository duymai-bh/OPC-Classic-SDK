51// onEventReceived
52//
53// - called after received an event
54//
55void SODmCEvent::onEventReceived(void)
56{
57 SODmCMainView* mainview = getMainView();
58 SOCmnPointer<SOAeCServer> server = (SOAeCServer*)m_subs->getParentBranch();
59 int img;
60 SOCmnString conv;
61
62 if (m_eventType != OPC_CONDITION_EVENT)
63 {
64 // simple and tracking related events -> log the event in the list
65 CListCtrl* list = mainview->getListAE();
66
67 if (m_eventType == OPC_SIMPLE_EVENT)
68 {
69 img = SODMCMAINVIEW_IMG_SIMPLEEVENT;
70 }
71 else
72 {
73 img = SODMCMAINVIEW_IMG_TRACKINGEVENT;
74 }
75
76 addRef();
77 int idx = SODmCListPush(list, m_source, (void*)(SODmCEvent*)this, img);
78 conv.format(_T("%lu"), m_severity);
79 list->SetItemText(idx, 1, conv);
80 list->SetItemText(idx, 2, m_message);
81 FILETIME2STRING(&conv, &m_ftTime);
82 list->SetItemText(idx, 3, conv);
83 list->SetItemText(idx, 4, m_actorID);
84 list->SetItemText(idx, 5, server->getURL());
85#ifdef SODMC_DCOM
86
87 if (((SOAeCSubscription*)m_subs)->getProtocol() == SOCLT_PROTOCOL_DCOM)
88 {
89 list->SetItemText(idx, 6, ((SODmCComSubscription*)(SOAeCSubscription*)m_subs)->getLabel());
90 }
91
92#endif
93#ifdef SOFEATURE_TUNNEL
94
95 if (((SOAeCSubscription*)m_subs)->getProtocol() == SOCLT_PROTOCOL_TUNNEL)
96 {
97 list->SetItemText(idx, 6, ((SODmCTPSubscription*)(SOAeCSubscription*)m_subs)->getLabel());
98 }
99
100#endif
101 DWORD itmCnt = list->GetItemCount();
102
103 if (itmCnt > 70)
104 {
105 DWORD i;
106
107 for (i = itmCnt - 1; i >= 70; i--)
108 {
109 list->DeleteItem(i);
110 }
111 }
112 }
113 else
114 {
115 // condition related events
116 CListCtrl* list = mainview->getListAEC();
117 SODmCEvent* event;
118 int idx = -1;
119
120 // try to find an entry with the source and condition
121 for (idx = 0; idx < list->GetItemCount(); idx++)
122 {
123 event = (SODmCEvent*)(SOCltElement*)list->GetItemData(idx);
124
125 if (event)
126 {
127 if (((SOAeCSubscription*)event->m_subs == (SOAeCSubscription*)m_subs) && (event->m_source == m_source) && (event->m_conditionName == m_conditionName))
128 {
129 // found the entry
130 list->DeleteItem(idx);
131 break;
132 }
133 }
134 }
135
136 if ((m_newState & (OPC_CONDITION_ACTIVE | OPC_CONDITION_ACKED)) != OPC_CONDITION_ACKED)
137 {
138 // not active and acknowledged
139 addRef();
140
141 // find the right index for the new condition
142 for (idx = 0; idx < list->GetItemCount(); idx++)
143 {
144 event = (SODmCEvent*)(SOCltElement*)list->GetItemData(idx);
145
146 if (event)
147 {
148 if ((event->m_newState & (OPC_CONDITION_ACTIVE | OPC_CONDITION_ACKED)) == (m_newState & (OPC_CONDITION_ACTIVE | OPC_CONDITION_ACKED)))
149 {
150 // same state check priority
151 if (event->m_severity < m_severity)
152 {
153 break;
154 }
155 }
156
157 if ((event->m_newState & OPC_CONDITION_ACKED) && (!(m_newState & OPC_CONDITION_ACKED)))
158 {
159 break;
160 }
161
162 if ((!(event->m_newState & OPC_CONDITION_ACKED)) && (!(m_newState & OPC_CONDITION_ACKED)))
163 if ((!(event->m_newState & OPC_CONDITION_ACTIVE)) && ((m_newState & OPC_CONDITION_ACTIVE)))
164 {
165 break;
166 }
167 }
168 }
169
170 LV_ITEM lv;
171 lv.mask = LVIF_TEXT | LVIF_PARAM | LVIF_IMAGE;
172 lv.iItem = idx;
173 lv.iSubItem = 0;
174 lv.pszText = (LPTSTR)m_source;
175 lv.lParam = (LPARAM)(void*)(SODmCEvent*)this;
176 lv.iImage = SODMCMAINVIEW_IMG_CONDITIONEVENT + m_newState;
177 list->InsertItem(&lv);
178 conv.format(_T("%lu"), m_severity);
179 list->SetItemText(idx, 2, conv);
180 list->SetItemText(idx, 3, m_message);
181 FILETIME2STRING(&conv, &m_ftTime);
182 list->SetItemText(idx, 4, conv);
183 list->SetItemText(idx, 1, m_conditionName);
184 list->SetItemText(idx, 5, m_actorID);
185 list->SetItemText(idx, 6, m_subconditionName);
186 list->SetItemText(idx, 7, server->getURL());
187#ifdef SODMC_DCOM
188
189 if (((SOAeCSubscription*)m_subs)->getProtocol() == SOCLT_PROTOCOL_DCOM)
190 {
191 list->SetItemText(idx, 8, ((SODmCComSubscription*)(SOAeCSubscription*)m_subs)->getLabel());
192 }
193
194#endif
195#ifdef SOFEATURE_TUNNEL
196
197 if (((SOAeCSubscription*)m_subs)->getProtocol() == SOCLT_PROTOCOL_TUNNEL)
198 {
199 list->SetItemText(idx, 8, ((SODmCTPSubscription*)(SOAeCSubscription*)m_subs)->getLabel());
200 }
201
202#endif
203 }
204 }
205} // onEventReceived

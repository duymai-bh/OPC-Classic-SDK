187BOOL CGridCell::Edit(int nRow, int nCol, CRect rect, CPoint /* point */, UINT nID, UINT nChar)
188{
189 if (m_bEditing)
190 {
191 if (m_pEditWnd)
192 {
193 m_pEditWnd->SendMessage(WM_CHAR, nChar);
194 }
195 }
196 else
197 {
198 DWORD dwStyle = ES_LEFT;
199
200 if (GetFormat() & DT_RIGHT)
201 {
202 dwStyle = ES_RIGHT;
203 }
204 else if (GetFormat() & DT_CENTER)
205 {
206 dwStyle = ES_CENTER;
207 }
208
209 m_bEditing = TRUE;
210 // InPlaceEdit auto-deletes itself
211 CGridCtrl* pGrid = GetGrid();
212 m_pEditWnd = new CInPlaceEdit(pGrid, rect, dwStyle, nID, nRow, nCol, GetText(), nChar);
213 }
214
215 return TRUE;
216}

235CGridDefaultCell::CGridDefaultCell()
236{
237#ifdef _WIN32_WCE
238 m_nFormat = DT_LEFT | DT_VCENTER | DT_SINGLELINE | DT_NOPREFIX;
239#else
240 m_nFormat = DT_LEFT | DT_VCENTER | DT_SINGLELINE | DT_NOPREFIX | DT_END_ELLIPSIS;
241#endif
242 m_crFgClr = CLR_DEFAULT;
243 m_crBkClr = CLR_DEFAULT;
244 m_Size = CSize(30, 10);
245 m_dwStyle = 0;
246#ifdef _WIN32_WCE
247 LOGFONT lf;
248 GetObject(GetStockObject(SYSTEM_FONT), sizeof(LOGFONT), &lf);
249 SetFont(&lf);
250#else // not CE
251 NONCLIENTMETRICS ncm = {0};
252 ncm.cbSize = sizeof(NONCLIENTMETRICS);
253 // check if windows Vista or higher
254 OSVERSIONINFO osvi;
255 ZeroMemory(&osvi, sizeof(OSVERSIONINFO));
256 osvi.dwOSVersionInfoSize = sizeof(OSVERSIONINFO);
257 ::GetVersionEx(&osvi);
258
259 if (osvi.dwMajorVersion < 6)
260 {
261 // the ncm.iPaddedBorderWidth is not part of the ncm for Windows XP or lower
262 ncm.cbSize -= sizeof(ncm.iPaddedBorderWidth);
263 }
264
265 VERIFY(SystemParametersInfo(SPI_GETNONCLIENTMETRICS, sizeof(NONCLIENTMETRICS), &ncm, 0));
266 SetFont(&(ncm.lfMessageFont));
267#endif
268}

275void CGridDefaultCell::SetFont(const LOGFONT* plf)
276{
277 ASSERT(plf);
278
279 if (!plf)
280 {
281 return;
282 }
283
284 m_Font.DeleteObject();
285 m_Font.CreateFontIndirect(plf);
286 CGridCell::SetFont(plf);
287 // Get the font size and hence the default cell size
288 CDC* pDC = CDC::FromHandle(::GetDC(NULL));
289
290 if (pDC)
291 {
292 CFont* pOldFont = pDC->SelectObject(&m_Font);
293 SetMargin(pDC->GetTextExtent(_T(" "), 1).cx);
294 m_Size = pDC->GetTextExtent(_T(" XXXXXXXXXXXX "), 14);
295 m_Size.cy = (m_Size.cy * 3) / 2;
296 pDC->SelectObject(pOldFont);
297 ReleaseDC(NULL, pDC->GetSafeHdc());
298 }
299 else
300 {
301 SetMargin(3);
302 m_Size = CSize(40, 16);
303 }
304}

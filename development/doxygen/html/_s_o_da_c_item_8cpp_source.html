55//-----------------------------------------------------------------------------
56// doExecute
57
58void SODaCItemStateTransitionTask::execute()
59{
60 _ASSERT_PTR((SOCltElement*)m_element);
61 SOCmnPointer<SODaCGroup> parent = (SODaCGroup*)m_element->getParentBranch();
62 _ASSERT_PTR((SODaCGroup*)parent);
63
64 if (!m_element->getConnectionMonitorStatus())
65 {
66 if (m_targetState != SOCLTTASKSTATETRANSITION_NO_STATE_CHANGE)
67 {
68 m_element->setTargetObjectState(m_targetState);
69 }
70 parent->setSingleItemTransactionHandle(m_element->getOpcHandle());
71 m_res = parent->doStateTransition(TRUE, SOCLTTASKSTATETRANSITION_NO_STATE_CHANGE);
72 parent->setSingleItemTransactionHandle(0);
73 }
74} // doExecute

90
91void SODaCItemTaskUpdate::execute()
92{
93 _ASSERT_PTR((SOCltElement*)m_element);
94 HRESULT res = S_OK;
95 SOCmnPointer<SODaCGroup> parent = (SODaCGroup*)m_element->getParentBranch();
96 _ASSERT_PTR((SODaCGroup*)parent);
97 SOCmnList<SODaCItem> itemList;
98 itemList.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_OBJECT);
99 itemList.add((SODaCItem*)(SOCltElement*)m_element);
100 res = parent->updateItems(m_fromServer, &itemList, m_whatParameters);
101 itemList.destroy();
102 m_res = res;
103}

209//-----------------------------------------------------------------------------
210// read
211//
212// - read item value
213//
214// returns:
215// S_OK - operation succeeded
216// S_FALSE - operation succeeded, but specific item error set
217// E_FAIL - operation failed
218// E_INVALIDARG - invalid argument
219// E_OUTOFMEMORY - not enough memory
220// E_NOINTERFACE - interface not supported
221// CONNECT_E_NOCONNECTION - no read advise connection
222//
223HRESULT SODaCItem::read(
224 IN DWORD maxAge) // max cache age
225{
226 SOCmnPointer<SODaCGroup> grp = (SODaCGroup*)getParentBranch();
227 HRESULT res;
228
229 if ((SODaCGroup*) grp)
230 {
231 SODaCItem* pThis = this;
232 HRESULT error;
233
234 // call the read method
235 if (grp->getDataRetrieval() == SODaCGroup::drAsyncAdvise)
236 {
237 res = grp->asyncRead(1, &pThis, maxAge, 0, NULL, &error);
238 }
239 else
240 {
241 res = grp->syncRead(1, &pThis, maxAge, NULL, NULL, NULL, &error);
242 }
243
244 if (res == S_FALSE)
245 {
246 res = error;
247 }
248 }
249 else
250 {
251 res = E_FAIL;
252 }
253
254 return res;
255} // read

309//-----------------------------------------------------------------------------
310// update
311//
312// - update item parameters with the OPC Item object in the OPC Server
313//
314// returns:
315// S_OK - operation succeeded
316// E_FAIL - operation failed
317// E_INVALIDARG - invalid argument
318// E_OUTOFMEMORY - not enough memory
319// E_NOINTERFACE - interface not supported
320//
321HRESULT SODaCItem::update(
322 IN BOOL fromServer, // update direction: write to server or read from
323 IN OPTIONAL DWORD whatParameters) // update what parameters
324{

334 itemList.destroy();

346//-----------------------------------------------------------------------------
347// serialization
348//
349
350BOOL SODaCItem::serialize(IN OUT SOCltStorage& ar, IN BOOL isStoring)
351{
352 BOOL ok = TRUE;
353
354 if (isStoring)
355 {
356 if (!getPersistence())
357 {
358 return TRUE;
359 }
360
361 // store own data
362 ar << getAccessPath();
363 ar << getItemID();
364 ar << (DWORD)getReqDatatype();
365 ar << getDeadBand();
366 }
367 else
368 {
369 // load own data
370 SOCmnPointer<SODaCEntry> entry = getSODaCEntry();
371 SOCmnString str;
372 DWORD dw = 0;
373 FLOAT f = 0.0;
374 ar >> str;
375 setAccessPath(str);
376 ar >> str;
377 setItemID(str);
378 ar >> dw;
379 setReqDatatype((VARTYPE)dw);
380
381 if (entry->getFileVersion() >= MAKELONG(20, 3))
382 {
383 ar >> f;
384 setDeadBand(f);
385 }
386 }
387
388 return ok;
389}

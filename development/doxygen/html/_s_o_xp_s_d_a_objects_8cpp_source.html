41void fillRequestData(
42 BYTE ioOperation,
43 SODaSRequest *req,
44 SODaSItemTag *itemTag,
45 SOXpSDARequestData *requestData,
46 SOCmnString *itemPath)
47{
48 req->addRef();
49 req->setUserData(SOXPS_REQUEST_STATE_NOT_HANDLED);
50 // set request pending
51 req->pending();
52
53 // fill request data struct
54 requestData->m_itemHandle = (SOXpSItemHandle)(SODaSItemTag *)itemTag;
55 *itemPath = itemTag->getItemID();
56 requestData->m_itemPath = (LPCTSTR)(*itemPath);
57 requestData->m_requestHandle = (SOXpSRequestHandle)req;
58 requestData->m_requestState = SOXPS_REQUEST_STATE_NOT_HANDLED;
59 requestData->m_result = S_OK;
60
61 if (itemTag->is(SOCMNOBJECT_TYPE_TAG))
62 requestData->m_itemUserData = ((SOXpSTag *)(SODaSItemTag *)itemTag)->getXpSUserData();
63 else
64 {
65 if (itemTag->is(SOCMNOBJECT_TYPE_PROPERTY))
66 requestData->m_itemUserData = ((SOXpSProperty *)(SODaSItemTag *)itemTag)->getXpSUserData();
67 else
68 requestData->m_itemUserData = ((SOXpSNode *)(SODaSItemTag *)itemTag)->getXpSUserData();
69 }
70
71 if (ioOperation == SOXPS_REQUEST_OPERATION_READ)
72 {
73 if (g_initData->m_provideCacheValueOnRead)
74 {
75 SOCmnPointer<SODaSCache> cache;
76 SOCmnVariant cVal;
77 SOCmnDateTime cTS;
78 cache = req->getCache();
79 cache->getValue(cVal);
80 cVal.copyTo(&(requestData->m_value));
81 requestData->m_quality = cache->getQuality();
82 cache->getTimeStamp(cTS);
83 cTS.get(&requestData->m_timeStamp);
84 }
85 else
86 {
87 ::VariantInit(&(requestData->m_value));
88 requestData->m_quality = 0;
89 requestData->m_timeStamp.dwHighDateTime = 0;
90 requestData->m_timeStamp.dwLowDateTime = 0;
91 }
92 }
93 else
94 {
95 req->getWriteValue(&(requestData->m_value));
96 requestData->m_quality = SOXPS_QUALITY_GOOD;
97 memset(&requestData->m_timeStamp, 0, sizeof(FILETIME));
98 }
99}

101void fillRequestData(
102 BYTE ioOperation,
103 DWORD *numRequests,
104 SOXpSDARequestData *requestData,
105 SOCmnString *itemPath,
106 SOCmnList<SODaSRequest> *requestQueue,
107 BYTE deviceIOModeFilter)
108{
109 SOCmnListPosition pos;
110 SODaSRequest *req;
111
112 // get started requests of queue
113 requestQueue->lock();
114 *numRequests = 0;
115 pos = requestQueue->getStartPosition();
116 while ((pos) && (*numRequests < SOXPS_REQUEST_MAX_REQUESTS))
117 { // while request list is not empty
118 req = requestQueue->getNext(pos);
119
120 if (req->getObjectState() == SODASREQUEST_FLAG_STARTED)
121 {
122 SOCmnPointer<SODaSItemTag> itemTag;
123 itemTag = req->getItemTag();
124 if (deviceIOModeFilter)
125 {
126 if (itemTag->getDeviceIOMode(req) != deviceIOModeFilter)
127 continue;
128 }
129
130 fillRequestData(ioOperation, req, itemTag, &requestData[*numRequests], &itemPath[*numRequests]);
131 (*numRequests)++;
132 }
133 }
134 requestQueue->unlock();
135}

138void removeHandledRequests(
139 SOCmnList<SODaSRequest> *requestQueue)
140{
141 SOCmnListPosition pos, curPos;
142 SODaSRequest *req;
143
144 requestQueue->lock();
145 pos = requestQueue->getStartPosition();
146 while (pos)
147 {
148 curPos = pos;
149 req = requestQueue->getNext(pos);
150
151 if ((req->isCompleted()) &&
152 (req->getUserData() != SOXPS_REQUEST_STATE_NOT_HANDLED) )
153 {
154 requestQueue->removeAt(curPos);
155 }
156 }
157 requestQueue->unlock();
158}

161BOOL processRequestData(
162 IN DWORD numRequests,
163 IN SOXpSDARequestData *requestData)
164{
165 SOCmnPointer<SODaSEntry> entry = getSODaSEntry();
166 SOCmnPointer<SODaSNameSpaceRoot> root = entry->getNameSpaceRoot();
167 SOCmnPointer<SODaSCache> cache;
168 SODaSItemTag *itemTag;
169 SOCmnPointer<SODaSTag> tag;
170 SOCmnPointer<SODaSProperty> prop;
171 SOCmnPointer<SODaSNode> node;
172 SOCmnElement *propParent = NULL;
173 SODaSRequest *req;
174 SOCmnString itemId;
175 DWORD i;
176 BOOL ret = TRUE;
177 SOCmnVariant value;
178 SOCmnDateTime timeStamp;
179
180 for (i = 0; i < numRequests; i++)
181 {
182 if (requestData[i].m_requestState == SOXPS_REQUEST_STATE_HANDLED)
183 {
184 if (requestData[i].m_requestHandle)
185 {
186 req = (SODaSRequest *)requestData[i].m_requestHandle;
187 req->setUserData(SOXPS_REQUEST_STATE_HANDLED);
188 if (req->getOperation() == SODaSRequest::read)
189 {
190 cache = req->getCache();
191 if (cache.isNotNull())
192 {
193 value = &requestData[i].m_value;
194 cache->setValue(value);
195 cache->setQuality(requestData[i].m_quality);
196 if ((requestData[i].m_timeStamp.dwLowDateTime != 0) ||
197 (requestData[i].m_timeStamp.dwHighDateTime != 0))
198 {
199 timeStamp = &requestData[i].m_timeStamp;
200 cache->setTimeStamp(timeStamp);
201 }
202 }
203
204 // only with 'TimestampChangeSendsValue != 0'
205 if (g_tsChangeSendsValue) {
206 SOCmnPointer<SOXpSTag> tag = (SOXpSTag*)req->getItemTag();
207
208 SOCmnList<SOCmnObject> items;
209
210 if ((requestData[i].m_timeStamp.dwLowDateTime != 0) ||
211 (requestData[i].m_timeStamp.dwHighDateTime != 0))
212 {
213 items.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_OBJ_OBJECT | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_SYNCED);
214 ((SOXpSTag*)tag)->getAllConnectedObjects(SOCMNOBJECT_TYPE_ITEM, items);
215
216 SOCmnListPosition pos = items.getStartPosition();
217 while(pos) {
218 SOXpSItem* pItem = (SOXpSItem*)items.getNext(pos);
219 pItem->clearAllSendTimeStamps();
220 }
221 }
222 }
223
224
225 req->complete(requestData[i].m_result);
226 }
227 else
228 {
229 BOOL updateCacheWithWrite = (requestData[i].m_value.vt != VT_EMPTY);
230
231 req->complete(requestData[i].m_result, TRUE, requestData[i].m_quality, FALSE, updateCacheWithWrite);
232 if ((requestData[i].m_timeStamp.dwLowDateTime != 0) ||
233 (requestData[i].m_timeStamp.dwHighDateTime != 0))
234 {
235 cache = req->getCache();
236 if (cache.isNotNull())
237 {
238 timeStamp = &requestData[i].m_timeStamp;
239 cache->setTimeStamp(timeStamp);
240 }
241 }
242 }
243 req->release();
244 }
245 else
246 { // no request
247 if (requestData[i].m_itemHandle == NULL)
248 {
249 SOCmnPointer<SODaSObjectRoot> oroot = (SODaSObjectRoot *)entry->getObjectRoot();
250 SOCmnList<SODaSServer> *srvList = oroot->getServerList();
251 SODaSServer *srv = srvList->getHead();;
252 if (srv)
253 srv->getItemTag(requestData[i].m_itemPath, &itemTag);
254 }
255 else
256 {
257 itemTag = (SODaSItemTag *)requestData[i].m_itemHandle;
258 itemTag->addRef();
259 }
260
261 if (itemTag)
262 {
263 if (itemTag->getUpdateMode() != SODaSItemTag::report)
264 {
265 cache = itemTag->getCache();
266 value = &requestData[i].m_value;
267 cache->setValue(value);
268 cache->setQuality(requestData[i].m_quality);
269 if ((requestData[i].m_timeStamp.dwLowDateTime != 0) ||
270 (requestData[i].m_timeStamp.dwHighDateTime != 0))
271 {
272 timeStamp = &requestData[i].m_timeStamp;
273 cache->setTimeStamp(timeStamp);
274 }
275 }
276 else
277 {
278 value = &requestData[i].m_value;
279 if ((requestData[i].m_timeStamp.dwLowDateTime != 0) ||
280 (requestData[i].m_timeStamp.dwHighDateTime != 0))
281 {
282 timeStamp = &requestData[i].m_timeStamp;
283 itemTag->notifyAllConnectedObjects(&value, requestData[i].m_quality, &timeStamp);
284 }
285 else
286 itemTag->notifyAllConnectedObjects(&value, requestData[i].m_quality);
287 }
288
289 // only with 'TimestampChangeSendsValue != 0'
290 if (g_tsChangeSendsValue) {
291 SOCmnList<SOCmnObject> items;
292
293 if ((requestData[i].m_timeStamp.dwLowDateTime != 0) ||
294 (requestData[i].m_timeStamp.dwHighDateTime != 0))
295 {
296 items.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_OBJ_OBJECT | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_SYNCED);
297 ((SOXpSTag*)itemTag)->getAllConnectedObjects(SOCMNOBJECT_TYPE_ITEM, items);
298
299 SOCmnListPosition pos = items.getStartPosition();
300 while(pos) {
301 SOXpSItem* pItem = (SOXpSItem*)items.getNext(pos);
302 pItem->clearAllSendTimeStamps();
303 }
304 }
305 }
306
307 itemTag->release();
308 }
309 else
310 {
311 ret = FALSE;
312 SetLastError(ERROR_PATH_NOT_FOUND);
313 }
314 }
315 }
316 else if (requestData[i].m_requestState == SOXPS_REQUEST_STATE_HANDLED_USE_CACHE)
317 {
318 SOCmnPointer<SODaSItemTag> itemTag;
319 if (requestData[i].m_requestHandle)
320 {
321 req = (SODaSRequest *)requestData[i].m_requestHandle;
322 req->setUserData(SOXPS_REQUEST_STATE_HANDLED);
323 if (req->getOperation() == SODaSRequest::read)
324 {
325 req->complete(requestData[i].m_result);
326 }
327 req->release();
328 }
329 else
330 {
331 ret = FALSE;
332 SetLastError(ERROR_INVALID_HANDLE);
333 }
334 }
335 }
336 return ret;
337}

339//-----------------------------------------------------------------------------
340// requestThreadMain
341//
342// - thread routine of request thread (queued I/O)
343//
344// returns:
345// thread end state
346//
347DWORD WINAPI requestThreadMain(
348 IN LPVOID context)
349{
350 HANDLE eventWait[2];
351 BOOL end = FALSE;
352 BOOL doWait = TRUE;
353 DWORD ret;
354 SOXpSDARequestData requestData[SOXPS_REQUEST_MAX_REQUESTS];
355 SOCmnString itemPath[SOXPS_REQUEST_MAX_REQUESTS];
356 DWORD numRequests;
357 DWORD i;
358
359 if (g_fpSOXpSHandleDARequests == NULL)
360 {
361 _TRACEC(SOCMNTRACE_L_ERR, SOCMNTRACE_G_USER1, _T("XPRESS"), (_T("No SOXpSHandleDARequests but started request thread")));
362 return -1;
363 }
364
365 // create events
366 SOCmnEventPool *eventPool = getEventPool();
367 g_requestThreadEndEvent = eventPool->popEvent();
368 g_requestThreadWorkEvent = eventPool->popEvent();
369 g_requestThreadWorkEventObj = new SOCmnEvent();
370 g_requestThreadWorkEventObj->setHandle(g_requestThreadWorkEvent);
371 eventWait[SOCMN_EVENT_WORK] = g_requestThreadWorkEvent;
372 eventWait[SOCMN_EVENT_END] = g_requestThreadEndEvent;
373
374 // create queues
375 g_requestThreadReadQueue.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_OBJECT | SOCMNLIST_TYPE_SYNCED);
376 g_requestThreadWriteQueue.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_OBJECT | SOCMNLIST_TYPE_SYNCED);
377
378 while(!end)
379 {
380 SOCMN_TRY
381 {
382 if (doWait)
383 ret = ::WaitForMultipleObjectsEx(2, eventWait, FALSE, 300000, TRUE);
384 else
385 ret = WAIT_OBJECT_0 + SOCMN_EVENT_WORK;
386
387 doWait = TRUE;
388 switch (ret - WAIT_OBJECT_0)
389 {
390 case SOCMN_EVENT_END:
391 end = TRUE;
392 break;
393 default:
394 {
395 // fill request data struct
396 fillRequestData(SOXPS_REQUEST_OPERATION_WRITE, &numRequests, requestData, itemPath, &g_requestThreadWriteQueue, 0);
397 if (numRequests == SOXPS_REQUEST_MAX_REQUESTS)
398 doWait = FALSE;
399
400 // call DLL
401 if (numRequests > 0)
402 {
403 g_fpSOXpSHandleDARequests(SOXPS_REQUEST_OPERATION_WRITE, numRequests, requestData);
404 processRequestData(numRequests, requestData);
405 for (i = 0; i < numRequests; i++)
406 ::VariantClear(&requestData[i].m_value);
407 }
408
409 // remove all completed request
410 removeHandledRequests(&g_requestThreadWriteQueue);
411
412
413 // fill request data struct
414 fillRequestData(SOXPS_REQUEST_OPERATION_READ, &numRequests, requestData, itemPath, &g_requestThreadReadQueue, 0);
415 if (numRequests == SOXPS_REQUEST_MAX_REQUESTS)
416 doWait = FALSE;
417
418 // call DLL
419 if (numRequests > 0)
420 {
421 g_fpSOXpSHandleDARequests(SOXPS_REQUEST_OPERATION_READ, numRequests, requestData);
422 processRequestData(numRequests, requestData);
423 for (i = 0; i < numRequests; i++)
424 ::VariantClear(&requestData[i].m_value);
425 }
426
427 // remove all completed request
428 removeHandledRequests(&g_requestThreadReadQueue);
429 }
430 break;
431 }
432 }
433 SOCMN_CATCH(...)
434 {
435 }
436 }
437
438 g_requestThreadReadQueue.destroy();
439 g_requestThreadWriteQueue.destroy();
440
441 ::CloseHandle(g_requestThreadEndEvent);
442 delete g_requestThreadWorkEventObj;
443
444 return 0;
445} // requestThreadMain

448//-----------------------------------------------------------------------------
449// handleRequest
450//
451// - handle request for direct I/O
452//
453void handleRequest(
454 IN SODaSItemTag *pItemTag, // item tag
455 IN SODaSRequest *pReq) // request
456{
457 BYTE ioOp = (pReq->getOperation() == SODaSRequest::read) ? SOXPS_REQUEST_OPERATION_READ : SOXPS_REQUEST_OPERATION_WRITE;
458 SOXpSDARequestData requestData;
459 SOCmnString itemPath;
460
461 // fill request data struct
462 fillRequestData(ioOp, pReq, pItemTag, &requestData, &itemPath);
463
464 // call DLL
465 g_fpSOXpSHandleDARequests(ioOp, 1, &requestData);
466 processRequestData(1, &requestData);
467 ::VariantClear(&requestData.m_value);
468} // handleRequest

737//-----------------------------------------------------------------------------
738// activateConnectedObject
739//
740// - activate connected object
741//
742void SODaSItemTagConnectQueueOneItemsStore::activateConnectedObject(
743 IN SOCmnObject *obj) // connected object
744{
745 if (obj)
746 {
747 if (obj->is(SOCMNOBJECT_TYPE_ITEM))
748 {
749 if (m_activeItems == 0)
750 if (g_fpSOXpSActivateDAItems)
751 {
752 SOXpSDAItemData itemData;
753 SODaSItemTag *pObj = (SODaSItemTag *)getObjectDaSItemTagIConnect();
754 SOCmnString itemID = pObj->getItemID();
755
756 itemData.m_itemHandle = (SOXpSItemHandle)pObj;
757 itemData.m_itemPath = (LPCTSTR)itemID;
758 if (pObj->is(SOCMNOBJECT_TYPE_TAG))
759 itemData.m_itemUserData = ((SOXpSTag *)pObj)->getXpSUserData();
760 else
761 {
762 if (pObj->is(SOCMNOBJECT_TYPE_PROPERTY))
763 itemData.m_itemUserData = ((SOXpSProperty *)pObj)->getXpSUserData();
764 else
765 itemData.m_itemUserData = ((SOXpSNode *)pObj)->getXpSUserData();
766 }
767
768 g_fpSOXpSActivateDAItems(TRUE, 1, &itemData);
769 }
770 m_activeItems++;
771 }
772 }

781void SODaSItemTagConnectQueueOneItemsStore::deactivateConnectedObject(
782 IN SOCmnObject *obj) // connected object
783{
784 if (obj)
785 {
786 if (obj->is(SOCMNOBJECT_TYPE_ITEM))

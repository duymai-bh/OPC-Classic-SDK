226 // A private static handler function.
227 static public Boolean Handler(MyWin32.CtrlTypes CtrlType)
228 {
229 string message = string.Empty;
230
231 // A switch to handle the event type.
232 switch (CtrlType)
233 {
234 case MyWin32.CtrlTypes.CTRL_C_EVENT:
235 case MyWin32.CtrlTypes.CTRL_BREAK_EVENT:
236 case MyWin32.CtrlTypes.CTRL_CLOSE_EVENT:
237 case MyWin32.CtrlTypes.CTRL_LOGOFF_EVENT:
238 case MyWin32.CtrlTypes.CTRL_SHUTDOWN_EVENT:
239 message = "Stop execution, since CTRL command!";
240 Console.EndEvent.Set();
241 Console.End = true;
242 break;
243 } // end switch
244 // Use interop to display a message for the type of event.
245 System.Console.WriteLine(message);
246 return true;
247 } // end Handler

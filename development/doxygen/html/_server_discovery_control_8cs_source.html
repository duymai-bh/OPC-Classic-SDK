174 private static ArrayList GetServerList(SV_101_TYPES ServerType, string domain)
175 {
176 int entriesRead = 0, totalEntries = 0;
177 ArrayList allServers = new ArrayList();
178
179 do
180 {
181 // Buffer to store the available servers
182 // Filled by the NetServerEnum function
183 IntPtr buf = new IntPtr();
184
185 SERVER_INFO_101 server;
186 int ret = NetServerEnum(null, 101, out buf, -1,
187 ref entriesRead, ref totalEntries,
188 ServerType, domain, 0);
189
190 // if the function returned any data, fill the tree view
191 if (ret == ERROR_SUCCESS ||
192 ret == ERROR_MORE_DATA ||
193 entriesRead > 0)
194 {
195 Int32 ptr = buf.ToInt32();
196
197 for (int i = 0; i < entriesRead; i++)
198 {
199 // cast pointer to a SERVER_INFO_101 structure
200 server = (SERVER_INFO_101)Marshal.PtrToStructure(new IntPtr(ptr), typeof(SERVER_INFO_101));
201
202 ptr += Marshal.SizeOf(server);
203
204 // add the machine name and comment to the arrayList.
205 //You could return the entire structure here if desired
206 allServers.Add(server);
207 }
208 }
209
210 // free the buffer
211 NetApiBufferFree(buf);
212
213 }
214 while
215 (
216 entriesRead < totalEntries &&
217 entriesRead != 0
218 );
219
220 return allServers;
221 }

231 private void ExpandTreeNode(TreeNode currentNode)
232 {
233 string ipAddress = null;
234
235 if (currentNode.Text.Equals(REMOTE))
236 {
237 //add the domains tree nodes
238 currentNode.Nodes.Clear();
239 ArrayList domains = GetServerList(SV_101_TYPES.SV_TYPE_DOMAIN_ENUM, null);
240 for (int i = 0; i < domains.Count; i++)
241 {
242 string domain = "";
243 if (!string.IsNullOrEmpty(((SERVER_INFO_101)domains[i]).sv101_name))

245 domain = ((SERVER_INFO_101)domains[i]).sv101_name.Substring(0, 1).ToUpper() + ((SERVER_INFO_101)domains[i]).sv101_name.Substring(1, ((SERVER_INFO_101)domains[i]).sv101_name.Length - 1).ToLower();
246 }

248 currentNode.Nodes.Add(domainNode);
249 domainNode.Nodes.Add(new TreeNode("", REMOTEPC_INDEX, REMOTEPC_INDEX));
250 }

252 }
253
254 if (currentNode.Parent != null)
255 {
256 //add data nodes for localhost

332 private void CreateTreeBrowse(string ipAddress, OPCSpecification specification, TreeNode currentNode)
333 {
334 currentNode.Nodes.Clear();
335
336 if (specification == OPCSpecification.AlarmEvent)
337 {
338 //add alarm and event servers
339 ServerBrowserData[] serverData = GetServerBrowseData(ipAddress, Softing.OPCToolbox.EnumOPCSpecification.AE10);
340 if (serverData == null) return;
341 List<ServerBrowserData> orderData = new List<ServerBrowserData>();
342 orderData.AddRange(serverData);
343 orderData.Sort(new ServerBrowseDataComparer());
344 AddBrowserData(orderData.ToArray(), currentNode);
345 }
346 else
347 {
348 //combine the 3 data access
349 Dictionary<string, ServerBrowserData> serverData = new Dictionary<string, ServerBrowserData>();
350 List<ServerBrowserData> orderData = new List<ServerBrowserData>();
351
352 ServerBrowserData[] serverDataDA10 = GetServerBrowseData(ipAddress, Softing.OPCToolbox.EnumOPCSpecification.DA10);
353 if (serverDataDA10 == null) return;
354
355 ServerBrowserData[] serverDataDA20 = GetServerBrowseData(ipAddress, Softing.OPCToolbox.EnumOPCSpecification.DA20);
356 if (serverDataDA20 == null) return;
357
358
359 ServerBrowserData[] serverDataDA30 = GetServerBrowseData(ipAddress, Softing.OPCToolbox.EnumOPCSpecification.DA30);
360 if (serverDataDA30 == null) return;
361
362 //build server data list
363 for (int i = 0; i < serverDataDA10.Length; i++)
364 {
365 if (!serverData.ContainsKey(serverDataDA10[i].Url))
366 {
367 serverData.Add(serverDataDA10[i].Url, serverDataDA10[i]);
368 }
369
370 }
371 for (int i = 0; i < serverDataDA20.Length; i++)
372 {
373 if (!serverData.ContainsKey(serverDataDA20[i].Url))
374 {
375 serverData.Add(serverDataDA20[i].Url, serverDataDA20[i]);
376 }
377 }
378 for (int i = 0; i < serverDataDA30.Length; i++)
379 {
380 if (!serverData.ContainsKey(serverDataDA30[i].Url))
381 {
382 serverData.Add(serverDataDA30[i].Url, serverDataDA30[i]);
383 }
384 }
385 orderData.AddRange(serverData.Values);
386 orderData.Sort(new ServerBrowseDataComparer());
387 AddBrowserData(orderData.ToArray(), currentNode);
388 }

397 private static ServerBrowserData[] GetServerBrowseData(string ipAddress, Softing.OPCToolbox.EnumOPCSpecification specification)
398 {
399 ServerBrowserData[] serverData = null;
400 if (string.IsNullOrEmpty(ipAddress)) return null;
401
402 Softing.OPCToolbox.Client.ServerBrowser serverBrowser = new ServerBrowser(ipAddress);
403 ExecutionOptions executionOptions = new ExecutionOptions();
404 try
405 {
406 if (ResultCode.SUCCEEDED(serverBrowser.Browse(
407 specification,
408 EnumServerBrowserData.ALL,
409 out serverData,
410 executionOptions)))
411 {
412 return serverData;
413 }
414 }
415 catch (Exception ex)
416 {
417 MessageBox.Show(ex.Message, DemoClient.Properties.Resources.DEMO_CLIENT, MessageBoxButtons.OK, MessageBoxIcon.Error);
418 }
419 return null;
420 }

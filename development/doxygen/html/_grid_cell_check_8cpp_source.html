56CSize CGridCellCheck::GetCellExtent(CDC* pDC)
57{
58 // Using SM_CXHSCROLL as a guide to the size of the checkbox
59 int nWidth = GetSystemMetrics(SM_CXHSCROLL) + 2 * GetMargin();
60 CSize cellSize = CGridCell::GetCellExtent(pDC);
61 cellSize.cx += nWidth;
62 cellSize.cy = max(cellSize.cy, nWidth);

104void CGridCellCheck::OnClick(CPoint PointCellRelative)
105{
106 // PointCellRelative is relative to the topleft of the cell. Convert to client coords
107 PointCellRelative += m_Rect.TopLeft();
108 // Bail if cell is read-only
109 CCellID cell = GetGrid()->GetCellFromPt(PointCellRelative);
110
111 if (!GetGrid()->IsCellEditable(cell))
112 {
113 return;
114 }
115
116 // GetCheckPlacement returns the checkbox dimensions in client coords. Only check/
117 // uncheck if the user clicked in the box
118 if (GetCheckPlacement().PtInRect(PointCellRelative))
119 {
120 m_bChecked = !m_bChecked;
121 GetGrid()->InvalidateRect(m_Rect);
122 }
123}

151// Returns the dimensions and placement of the checkbox in client coords.
152CRect CGridCellCheck::GetCheckPlacement()
153{
154 int nWidth = GetSystemMetrics(SM_CXHSCROLL);
155 CRect place = m_Rect + CSize(GetMargin(), GetMargin());
156 place.right = place.left + nWidth;
157 place.bottom = place.top + nWidth;
158
159 /* for centering
160 int nDiff = (place.Width() - nWidth)/2;
161 if (nDiff > 0)
162 {
163 place.left += nDiff;
164 place.right = place.left + nWidth;
165 }
166 nDiff = (place.Height() - nWidth)/2;
167 if (nDiff > 0)
168 {
169 place.top += nDiff;
170 place.bottom = place.top + nWidth;
171 }
172 */
173
174 if (m_Rect.Height() < nWidth + 2 * static_cast<int>(GetMargin()))
175 {
176 place.top = m_Rect.top + (m_Rect.Height() - nWidth) / 2;
177 place.bottom = place.top + nWidth;
178 }
179
180 return place;
181}

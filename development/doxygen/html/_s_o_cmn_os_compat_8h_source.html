555//
556// A language ID is a 16 bit value which is the combination of a
557// primary language ID and a secondary language ID. The bits are
558// allocated as follows:
559//
560// +-----------------------+-------------------------+
561// | Sublanguage ID | Primary Language ID |
562// +-----------------------+-------------------------+
563// 15 10 9 0 bit
564//
565//
566// Language ID creation/extraction macros:
567//
568// MAKELANGID - construct language id from a primary language id and
569// a sublanguage id.
570// PRIMARYLANGID - extract primary language id from a language id.
571// SUBLANGID - extract sublanguage id from a language id.
572//
573
574#define MAKELANGID(p, s) ((((WORD )(s)) << 10) | (WORD )(p))
575#define PRIMARYLANGID(lgid) ((WORD )(lgid) & 0x3ff)
576#define SUBLANGID(lgid) ((WORD )(lgid) >> 10)

579//
580// A locale ID is a 32 bit value which is the combination of a
581// language ID, a sort ID, and a reserved area. The bits are
582// allocated as follows:
583//
584// +-------------+---------+-------------------------+
585// | Reserved | Sort ID | Language ID |
586// +-------------+---------+-------------------------+
587// 31 20 19 16 15 0 bit
588//
589//
590// Locale ID creation/extraction macros:
591//
592// MAKELCID - construct the locale id from a language id and a sort id.
593// MAKESORTLCID - construct the locale id from a language id, sort id, and sort version.
594// LANGIDFROMLCID - extract the language id from a locale id.
595// SORTIDFROMLCID - extract the sort id from a locale id.
596// SORTVERSIONFROMLCID - extract the sort version from a locale id.
597//
598
599#define NLS_VALID_LOCALE_MASK 0x000fffff
600
601#define MAKELCID(lgid, srtid) ((DWORD)((((DWORD)((WORD )(srtid))) << 16) | \
602 ((DWORD)((WORD )(lgid)))))
603#define MAKESORTLCID(lgid, srtid, ver) \
604 ((DWORD)((MAKELCID(lgid, srtid)) | \

557DWORD SOSrvWeb::expandTemplates(SOCmnStringEx& data, SOSrvWebExpandData* exData)
558{
559 // Start of template token: <%#
560 // End of template token: #%>
561 // Parameter token: ##
562 LONG startIdxT;
563 LONG endIdxT;
564 DWORD tCnt;
565 BOOL needSecondPass;
566 BOOL expanded;
567
568 do
569 {
570 exData->m_pass++;
571 startIdxT = 0;
572 endIdxT = -1;
573 tCnt = 0;
574 needSecondPass = FALSE;
575
576 do
577 {
578 startIdxT = data.find(_T("<%#"), startIdxT);
579
580 if (startIdxT != -1)
581 {
582 endIdxT = data.find(_T("#%>"), startIdxT + 3);
583
584 if (endIdxT != -1)
585 {
586 SOCmnStringEx templ = data.mid(startIdxT + 3, (endIdxT) - (startIdxT + 3));
587 SOCmnStringEx result;
588 expanded = expandTemplate(templ, exData, result);
589
590 if (expanded)
591 {
592 data.replace(startIdxT, endIdxT + 3, result, result.getLength());
593 startIdxT = startIdxT + result.getLength();
594 }
595 else
596 {
597 startIdxT = endIdxT + 3;
598 }
599
600 needSecondPass |= (!expanded);
601 tCnt++;
602 }
603 }
604 }
605 while ((startIdxT != -1) && (endIdxT != -1));
606
607 if ((exData->m_pass == 1) && (needSecondPass))
608 {
609 exData->execute();
610 }
611 }
612 while ((exData->m_pass < 2) && (needSecondPass));
613
614 return tCnt;
615}

617BOOL SOSrvWeb::expandTemplate(SOCmnStringEx& templ, SOSrvWebExpandData* pExData, SOCmnStringEx& result)

622 argsList.create();

624 int argStartIdx;
625 int argEndIdx;

628

630 {
631 argStartIdx += 2;
632 argEndIdx = templ.find(_T("##"), argStartIdx);
633
634 if (argEndIdx == -1)
635 {
636 arg = templ.right(templ.getLength() - argStartIdx);
637 argStartIdx = -1;
638 }

641 arg = templ.mid(argStartIdx, argEndIdx - argStartIdx);

645 argsList.add(arg);

647

649 if (_tcsncmp((LPCTSTR)templ, _T("ITEMVALUE##"), 11) == 0)
650 {
651 SOCmnString itemID = argsList.pop();
652
653 if (pExData->m_pass == 1)
654 {

656 pExData->addReadRequest(itemID);
657 expanded = FALSE;
658 }

660 {
661 // set the item value
662 pExData->getReadValue(itemID, &argsList, result);

664 }
665 else if (_tcsncmp((LPCTSTR)templ, _T("ITEMQUALITY##"), 13) == 0)

678 pExData->getReadQuality(itemID, &argsList, result);

682 {
683 result = pExData->m_message;
684 }
685 else if (_tcsncmp((LPCTSTR)templ, _T("TRACE##"), 7) == 0)
686 {
687 SOCmnString cmd = argsList.pop();
688 handleTRACETemplate(cmd, &argsList, pExData, result);
689 }
690 else if (_tcsncmp((LPCTSTR)templ, _T("ITEMLISTVALUES##"), 16) == 0)
691 {
692 handleITEMLISTVALUESTemplate(&argsList, pExData, result);
693 }
694 else if (_tcsncmp((LPCTSTR)templ, _T("POSTARG##"), 9) == 0)
695 {
696 SOCmnListPosition pos;
697 SOCmnString par;
698 SOCmnString def;
699 SOCmnString key;
700 SOCmnString keyPart;
701 SOCmnString val;
702 pos = argsList.getStartPosition();
703
704 while (pos)
705 {
706 par = argsList.getNext(pos);
707
708 if (par != _T("DEFAULT"))
709 {
710 if (!key.isEmpty())
711 {
712 keyPart.format(_T("##%s"), (LPCTSTR)par);
713 }
714 else

716 keyPart.format(_T("%s"), (LPCTSTR)par);
717 }
718
719 key += keyPart;
720 }
721 else
722 {
723 if (pos)
724 {
725 def = argsList.getNext(pos);
726 }
727
728 break;
729 }
730 }
731
732 if (pExData->m_postArgsList.findKey(key, &val))
733 {
734 SOCmnString::doXMLReplacements(val);
735 result = val;
736 }
737 else
738 {
739 result = def;
740 }
741 }
742 else
743 {
744 expanded = handleTemplate((LPCTSTR)templ, &argsList, pExData, result);

1093WORD SOSrvWeb::post(IN SOSrvHTTPRequest* pReq)
1094{
1095 SOSrvHTTPHandlerData* pHTTPData = pReq->m_httpData;
1096 DWORD lenBody = pReq->m_lenBody;
1097 BYTE* pBody = pReq->m_pBody;
1098 WORD ret = SOCMNHTTP_CODEI_NOT_HANDLED;
1099 DWORD bodyLength = 0;
1100 BYTE* pBodyData = NULL;
1101 SOCmnStringEx resHeader;
1102 SOCmnDateTime now;
1103 SOCmnStringEx szDefResAttrs;
1104 SOCmnString bodyType;
1105 SOCmnString filePath;
1106 SOCmnStringList attributeList(pReq->getAttributeList());
1107 SOCmnString httpAttributes;
1108 now.now();
1109
1110 // check URL and load file
1111 // the requested file must be located in or below the static or dynamic
1112 // directory of the web server
1113 if ((ret = getFile(pReq, pHTTPData, NULL, &bodyLength, &pBodyData, &bodyType, &filePath, &httpAttributes)) == SOCMNHTTP_CODE_OK)
1114 {
1115 if (_tcsncmp((LPCTSTR)bodyType, _T("text"), 4) == 0)
1116 {
1117 // text file
1118 SOCmnPointer<SOSrvWebExpandData> exData;
1119 SOCmnStringEx body;
1120 SOCmnStringEx resBody;
1121 SOCmnString cacheControl;
1122 SOCmnString postCommand;
1123 DWORD templCnt;
1124 SOCmnString contentType;
1125 exData = createExpandData();
1126 exData->init(pReq, pHTTPData, &httpAttributes);
1127
1128 if (attributeList.findKey(_T("content-type"), &contentType))
1129 {
1130 if (_tcscmp(contentType, _T("application/x-www-form-urlencoded")) == 0)
1131 {
1132 body.setCHAR((CHAR*)pBody, lenBody);
1133
1134 if (!body.isEmpty())
1135 {
1136 // parse POST arguments format: id1=val1&id2=val2&...
1137 int argStartIdx;
1138 int argEndIdx;
1139 SOCmnStringEx arg;
1140 LPTSTR name, val;
1141 int perRepIdx;
1142 SOCmnString repChars;
1143 TCHAR repSign;
1144 body.replace(_T("+"), _T(" "));
1145
1146 while ((perRepIdx = body.find(_T('%'))) != -1)
1147 {
1148 repChars = body.mid(perRepIdx, 3);
1149 repSign = (TCHAR)_tcstoul((LPCTSTR)repChars + 1, NULL, 16);
1150 body.replace(perRepIdx, perRepIdx + 3, (TCHAR*)&repSign, 1);
1151 /*
1152 body.replace(_T("%23"), _T("#"));
1153 body.replace(_T("%2F"), _T("/"));
1154 body.replace(_T("%20"), _T(" "));
1155 body.replace(_T("%3A"), _T(":"));
1156 body.replace(_T("%5C"), _T("\\"));
1157 body.replace(_T("%2B"), _T("+"));
1158 */
1159 }
1160
1161 argStartIdx = 0;
1162
1163 while (argStartIdx != -1)
1164 {
1165 argEndIdx = body.find(_T("&"), argStartIdx);
1166
1167 if (argEndIdx == -1)
1168 {
1169 arg = body.right(body.getLength() - argStartIdx);
1170 argStartIdx = -1;
1171 }
1172 else
1173 {
1174 arg = body.mid(argStartIdx, argEndIdx - argStartIdx);
1175 argStartIdx = argEndIdx + 1;
1176 }
1177
1178 if (_tcsncmp(arg, _T("CMD##"), 5) != 0)
1179 {
1180 SOCmnElement::reducePath(arg, name, val, TRUE, _T("="));
1181
1182 if (val == NULL)
1183 {
1184 val = _T("");
1185 }
1186
1187 exData->m_postArgsList.add(val, name);
1188 }
1189 else
1190 {
1191 if (postCommand.isEmpty())
1192 {
1193 postCommand = arg.right(arg.getLength() - 5);
1194 }
1195 }
1196 }
1197 }
1198 }
1199 }
1200
1201 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_WEB_SRV, (_T("POST %s - %s ; %s"), (LPCTSTR)pReq->getURL(), (LPCTSTR)bodyType, (LPCTSTR)body));
1202
1203 if (!postCommand.isEmpty())
1204 {
1205 handlePOSTCommand(postCommand, pReq, pHTTPData, &exData->m_postArgsList, exData, resBody, &bodyType);
1206 }
1207
1208 if (resBody.isEmpty())
1209 {
1210 // no result data from handlePOSTArguments
1211#ifndef UNICODE
1212 resBody.attachTo((LPTSTR)pBodyData, bodyLength);
1213#else
1214 resBody.setCHAR((LPCSTR)pBodyData, bodyLength);
1215#endif
1216 }
1217 else
1218 {
1219 // received results -> free read file data
1220 if (pBodyData)
1221 {
1222 free(pBodyData);
1223 pBodyData = NULL;
1224 }
1225 }
1226
1227 templCnt = expandTemplates(resBody, exData);
1228 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_WEB_SRV, (_T("POST %s - %s ; length: %lu"), (LPCTSTR)pReq->getURL(), (LPCTSTR)bodyType, resBody.getLength()));
1229
1230 if (exData->m_httpResponse.isEmpty())
1231 {
1232 DWORD sizeB = 0;
1233 BYTE* pByteBody = resBody.cloneToUTF8(&sizeB);
1234
1235 if (pByteBody != NULL)
1236 {
1237 free(pByteBody);
1238 }
1239
1240 getResponseAttributes(pReq->m_flags, szDefResAttrs);
1241
1242 if (templCnt > 0)
1243 {
1244 cacheControl = _T("Cache-Control: must-revalidate\r\n");
1245 }
1246
1247 resHeader.format(_T("HTTP/1.1 200 OK\r\nServer: %s\r\nDate: %s\r\n%s%s%sContent-Type: %s; charset=utf-8\r\nContent-Length: %lu\r\n\r\n"), (LPCTSTR)m_httpServerDescription, (LPCTSTR)now.getHTTPDate(), (LPCTSTR)szDefResAttrs, (LPCTSTR)httpAttributes, (LPCTSTR)cacheControl, (LPCTSTR)bodyType, sizeB);
1248 // send result file
1249 pReq->m_httpHandlerForResponse->queueSndData(pHTTPData, pReq->m_flags, resHeader, resBody);
1250 }
1251 else
1252 {
1253 pReq->m_httpHandlerForResponse->queueSndData(pHTTPData, pReq->m_flags, exData->m_httpResponse, 0, NULL);
1254 }
1255
1256#ifdef UNICODE
1257
1258 if (pBodyData)
1259 {
1260 free(pBodyData);
1261 }
1262
1263#endif
1264 }
1265 }
1266 else
1267 {
1268 if (ret != 0)
1269 {
1270 pReq->m_httpHandlerForResponse->sndHTTPErrorResponse(pHTTPData, pReq->m_flags, ret, httpAttributes);
1271 }
1272 }
1273
1274 return ret;

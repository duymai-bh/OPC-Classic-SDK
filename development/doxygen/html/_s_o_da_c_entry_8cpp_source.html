362SODaCServer* SODaCEntry::parseURL(IN LPCTSTR url, IN SODaCServer* pServerIn)
363{
364 SOCmnPointer<SODaCServer> server;
365 SOCmnString i1(url);
366 SOCmnString i2;
367 SOCmnString protocol;
368 SOCmnString host;
369 SOCmnString path;
370 int cnt1 = i1.getLength();
371 int cnt2 = 0;
372 int ppos;
373 int hpos = 0;
374 ppos = i1.find(_T("://"));
375
376 if (ppos != -1)
377 {
378 protocol = (LPCTSTR)i1.left(ppos); // the left of "://"
379 i2 = (LPCTSTR)i1.right(cnt1 - ppos - 3); // the right of "://"
380 cnt2 = i2.getLength();
381 hpos = i2.find(_T("/"));
382
383 if (hpos != -1)
384 {
385 host = (LPCTSTR)i2.left(hpos); // the left of "/"
386 path = (LPCTSTR)i2.right(cnt2 - hpos - 1); // the right of "/"
387 }
388 else
389 {
390 // no / as path seperator -> i2 is only the host name
391 host = i2;
392 }
393 }
394
395#ifdef SOFEATURE_DCOM
396
397 if (protocol == _T("opcda"))
398 {
399 if (pServerIn)
400 if (pServerIn->getProtocol() != SOCLT_PROTOCOL_DCOM)
401 {
402 return NULL;
403 }
404
405 USES_CONVERSION;
406 CLSID clsid;
407 SOCmnString progID, classID;
408 int pos3, cnt3;
409 BOOL clsidOK = FALSE;
410
411 if (host == _T("localhost"))
412 {
413 host.empty();
414 }
415
416 // get CLSID
417 pos3 = path.find(_T("/"));
418
419 if (pos3 != -1)
420 {
421 cnt3 = path.getLength();
422 progID = (LPCTSTR)path.left(pos3);
423 classID = (LPCTSTR)path.right(cnt3 - pos3 - 1);
424 }
425 else
426 {
427 progID = path;
428 }
429
430 if (classID.isEmpty())
431 {
432 // no clsid specified -> use progID to the the clsid
433 if (host.isEmpty())
434 {
435 // local connection -> use the registry
436 if (SUCCEEDED(::CLSIDFromProgID(T2W(progID), &clsid)))
437 {
438 clsidOK = TRUE;
439 }
440 }
441 else
442 {
443 // remote connection -> use OPCEnum
444 SOCmnPointer<SOCltServerBrowser> sb;
445 sb = new SOCltServerBrowser();
446 sb->setNodeName(host);
447
448 if (sb->getCLSIDFromProgID(progID, &clsid))
449 {
450 clsidOK = TRUE;
451 }
452 }
453 }
454 else
455 {
456 if (SUCCEEDED(::CLSIDFromString(T2W(classID), &clsid)))
457 {
458 clsidOK = TRUE;
459 }
460 }
461
462 if (clsidOK)
463 {
464 // create Server
465 if (!pServerIn)
466 {
467 server = m_creator->createServer(SOCLT_PROTOCOL_DCOM, this);
468 }
469 else
470 {
471 server = pServerIn;
472 server.addRef();
473 }
474
475 server->initClt();
476 server->init();
477 server->setProtocol(SOCLT_PROTOCOL_DCOM);
478 server->setURL(url);
479 server->setNodeName(host);
480 server->setClassID(clsid);
481 }
482 }
483
484#endif // SOFEATURE_DCOM
485#ifdef SOFEATURE_SOAP
486
487 if (protocol == _T("http"))
488 {
489 if (pServerIn)
490 if (pServerIn->getProtocol() != SOCLT_PROTOCOL_SOAP)
491 {
492 return NULL;
493 }
494
495 WORD port = 80;
496 int pos3, cnt3;
497 // get port
498 pos3 = host.find(_T(":"));
499
500 if (pos3 != -1)
501 {
502 cnt3 = host.getLength();
503 port = (WORD)_tcstoul((LPCTSTR)host + pos3 + 1, NULL, 10);
504 host = (LPCTSTR)host.left(pos3);
505 }
506
507 if (host == _T("localhost"))
508 {
509 host.empty();
510 }
511
512 path = (LPCTSTR)i2.right(cnt2 - hpos); // path with leading /
513
514 if (!pServerIn)
515 {
516 server = m_creator->createServer(SOCLT_PROTOCOL_SOAP, this);
517 }
518 else
519 {
520 server = pServerIn;
521 server.addRef();
522 }
523
524 server->initClt();
525 server->init();
526 server->setProtocol(SOCLT_PROTOCOL_SOAP);
527 server->setURL(url);
528 server->setNodeName(host);
529 server->setIPPort(port);
530 server->setHTTPPath(path);
531 }
532
533#endif // SOFEATURE_SOAP
534#ifdef SOFEATURE_TUNNEL
535
536 if (protocol == _T("tpda"))
537 {
538 if (pServerIn)
539 if (pServerIn->getProtocol() != SOCLT_PROTOCOL_TUNNEL)
540 {
541 return NULL;
542 }
543
544 WORD port = TP_DEFAULT_PORT;
545 int pos3, cnt3;
546 // get port
547 pos3 = host.find(_T(":"));
548
549 if (pos3 != -1)
550 {
551 cnt3 = host.getLength();
552 port = (WORD)_tcstoul((LPCTSTR)host + pos3 + 1, NULL, 10);
553 host = (LPCTSTR)host.left(pos3);
554 }
555
556 if (host == _T("localhost"))
557 {
558 host.empty();
559 }
560
561 if (!pServerIn)
562 {
563 server = m_creator->createServer(SOCLT_PROTOCOL_TUNNEL, this);
564 }
565 else
566 {
567 server = pServerIn;
568 server.addRef();
569 }
570
571 server->initClt();
572 server->init();
573 server->setProtocol(SOCLT_PROTOCOL_TUNNEL);
574 server->setURL(url);
575 server->setNodeName(host);
576 server->setIPPort(port);
577 }
578
579#endif // SOFEATURE_TUNNEL
580
581 if (!pServerIn)
582 {
583 if (server.isNotNull())
584 {
585 // add server to session
586 addBranch((SODaCServer*)server);
587 }
588
589 server.addRef();
590 }
591
592 return server;
593} // addServer

603//
604// returns:
605// pointer to server object

607SODaCServer* SODaCEntry::addServer(
608 IN LPCTSTR serverName, // server name
609 IN LPCTSTR nodeName, // node name
610 IN REFCLSID clsid, // class Id
611 IN DWORD clsctx) // execution context
612{
613 // create Server
614 SOCmnPointer<SODaCServer> server = m_creator->createServer(SOCLT_PROTOCOL_DCOM, this);
615 server->initClt();
616 server->init();
617 // initialize server
618 server->setProtocol(SOCLT_PROTOCOL_DCOM);
619 server->setServerName(serverName);
620 server->setNodeName(nodeName);
621 server->setClassID(clsid);
622 server->setClsctx(clsctx);
623 SOCmnString classID;
624 SOCmnString url;
625 classID.format(_T("{%8.8X-%4.4X-%4.4X-%2.2X%2.2X-%2.2X%2.2X%2.2X%2.2X%2.2X%2.2X}"), clsid.Data1, clsid.Data2, clsid.Data3, clsid.Data4[0], clsid.Data4[1], clsid.Data4[2], clsid.Data4[3], clsid.Data4[4], clsid.Data4[5], clsid.Data4[6], clsid.Data4[7]);
626 url.format(_T("opcda://%s//%s"), nodeName, (LPCTSTR)classID);
627 server->setURL(url);
628 // add server to session
629 addBranch((SODaCServer*)server);
630 server.addRef();
631 return server;
632} // addServer

638{
639 HRESULT res = S_OK;
640
641 if (getTargetObjectState() != disconnected)
642 {
643#ifdef SOFEATURE_SOAP
644
645 if (!m_httpClient->start())
646 {
647 res = E_FAIL;
648 }
649
650#endif // SOFEATURE_SOAP
651#ifdef SOFEATURE_TUNNEL

654 {
655 res = E_FAIL;

657
658#endif // SOFEATURE_TUNNEL
659 }
660

662 {
663 res = SOCltEntry::doStateTransition(deep, stateDeep);

665
666 if (getTargetObjectState() == disconnected)

669 m_tpClient->stop();
670#endif // SOFEATURE_TUNNEL
671#ifdef SOFEATURE_SOAP
672 m_httpClient->stop();
673#endif // SOFEATURE_SOAP
674 }
675
676 return res;
677}

722//-----------------------------------------------------------------------------
723// serialize
724//
725BOOL SODaCEntry::serialize(IN OUT SOCltStorage& ar, IN BOOL isStoring)
726{
727 BOOL ok = TRUE;
728 SOCmnList<SOCmnElement> list(getBranchList());
729 SODaCServer* server = NULL;
730
731 if (isStoring)
732 {
733 DWORD fileVersion;
734 fileVersion = MAKELONG(MINOR_VERSION_NUMBER, MAJOR_VERSION_NUMBER);
735 ar << fileVersion;
736 DWORD dwCount = 0;
737 SOCmnListPosition pos = list.getStartPosition();
738
739 while (pos)
740 {
741 server = (SODaCServer*)list.getNext(pos);
742
743 if (server->getPersistence())
744 {
745 dwCount++;
746 }
747 }
748
749 ar << dwCount;
750 pos = list.getStartPosition();
751
752 while (pos && ok)
753 {
754 server = (SODaCServer*)list.getNext(pos);
755 ar << server->getProtocol();
756 ar << server->getURL();
757
758 if (server->getPersistence())
759 {
760 ok = server->serialize(ar, isStoring);
761 }
762 }
763 }
764 else
765 {
766 // stop communication with the servers
767 disconnect();
768 // destroy object tree
769 removeAllLeafs(TRUE);
770 removeAllBranches(TRUE);
771 DWORD dwCount = 0;
772 ar >> m_fileVersion;
773
774 if ((HIWORD(m_fileVersion) < 3) || (HIWORD(m_fileVersion) > 5))
775 {
776 return FALSE;
777 }
778 else
779 {
780 ar >> dwCount;
781 }
782
783 if (dwCount > 0)
784 {
785 while (dwCount-- && ok)
786 {
787 BYTE protocol;
788
789 if (m_fileVersion < MAKELONG(03, 4))
790 {
791 protocol = SOCLT_PROTOCOL_DCOM;
792 // create server
793 server = m_creator->createServer(protocol, this);
794
795 if (server)
796 {
797 server->initClt();
798 server->init();
799 server->setProtocol(protocol);
800 addBranch((SODaCServer*)server);
801 }
802 }
803 else
804 {
805 SOCmnString url;
806 ar >> protocol;
807 ar >> url;
808 server = addServer(url);
809
810 if (server)
811 {
812 server->setProtocol(protocol);
813 }
814 }
815
816 if (server)
817 {
818 ok = server->serialize(ar, isStoring);
819 server->release();
820 }
821 }
822 }
823 }
824
825 return ok;
826}

107// getReturnedAttributes
108//
109// - get the returned attributes list for a event category
110//
111// returns:
112// returned attributes list
113//
114SOCmnDWORDList* SOAeCSubscription::getReturnedAttributes(
115 IN DWORD category) // category
116{
117 BOOL ok = TRUE;
118 // find existing list entry
119 SOAeCReturnedAttrListList retAttrList(m_retAttrList);
120 SOCmnDWORDList* retAttr = retAttrList.findKey(category);
121
122 if (!retAttr)
123 {
124 SOCMN_ALLOCATE_OBJECT(retAttr, SOCmnDWORDList());
125
126 if (ok)
127 {
128 ok &= retAttr->create(SOCMNLIST_TYPE_FLAGS | SOCMNLIST_TYPE_ORG_LIST |
129 SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_DWORD);
130 }
131
132 if (ok)
133 {
134 ok &= retAttrList.add(retAttr, category);
135 }
136
137 if (!ok && retAttr)
138 {
139 delete retAttr;
140 retAttr = NULL;
141 }
142 }
143
144 return retAttr ;
145} // getReturnedAttributes

317BOOL SOAeCSubscription::serialize(IN OUT SOCltStorage& ar, IN BOOL isStoring)
318{
319 BOOL ok = TRUE;
320
321 if (!getPersistence())
322 {
323 return TRUE;
324 }
325
326 if (isStoring)
327 {
328 ar << m_dwBufferTime;
329 ar << m_dwMaxSize;
330 ar << m_dwLowSeverity;
331 ar << m_dwHighSeverity;
332 ar << m_dwEventTypes;
333 }
334 else
335 {
336 SOCmnPointer<SOAeCEntry> entryAE = ::getSOAeCEntry();
337
338 if (entryAE->getFileVersion() >= MAKELONG(4, 3)) // Version 3.04
339 {
340 ar >> m_dwBufferTime;
341 ar >> m_dwMaxSize;
342 ar >> m_dwLowSeverity;
343
344 if (m_dwLowSeverity == 0)
345 {
346 m_dwLowSeverity = 1; // OPC AE 1.03 change !
347 }
348
349 ar >> m_dwHighSeverity;
350 ar >> m_dwEventTypes;
351 }
352 }
353
354 SOCmnStringList listSrc(getSourceList());
355 SOCmnStringList listArea(getAreaList());
356 SOCmnDWORDList listCategory(getCategoryList());
357 serialize_list(ar, isStoring, listSrc);
358 serialize_list(ar, isStoring, listArea);
359 serialize_list(ar, isStoring, listCategory);
360 // Returned attribute list
361 SOAeCReturnedAttrListList retAttrList(m_retAttrList);
362 SOCmnListPosition pos = retAttrList.getStartPosition();
363
364 if (isStoring)
365 {
366 size_t category = 0;
367 ar << retAttrList.getCount();
368
369 while (pos)
370 {
371 SOCmnDWORDList list(retAttrList.getNext(pos, &category));
372 ar << (DWORD)category;
373 serialize_list(ar, TRUE, list);
374 }
375 }
376 else
377 {
378 DWORD dwCount = 0;
379 DWORD category = 0;
380 ar >> dwCount;
381
382 while (dwCount--)
383 {
384 category = 0;
385 ar >> category;
386 SOCmnDWORDList list(getReturnedAttributes(category));
387 serialize_list(ar, FALSE, list);
388 }
389 }
390
391 return ok;
392}

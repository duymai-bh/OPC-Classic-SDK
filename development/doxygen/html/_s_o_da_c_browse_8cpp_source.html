126BOOL SODaCNameSpaceBrowser::expand(IN DWORD type, OUT SOCmnList<SOCltBrowseObject> &objList)
127{
128 BYTE os = getObjectState();
129 BOOL result = FALSE;
130 switch (os)
131 {
132 case SODACNAMESPACEBROWSER_STATE_USE_IOPCBROWSESERVERADDRESSSPACE_AND_IOPCITEMPROPERTIES:
133 case SODACNAMESPACEBROWSER_STATE_USE_IOPCBROWSESERVERADDRESSSPACE:
134 result = expand20(this , _T(""), type, objList);
135 break;
136
137 case SODACNAMESPACEBROWSER_STATE_USE_IOPCBROWSE:
138 {
139 SOCmnString contPoint = getContinuationPoint();
140 result = expand30(this, _T(""), _T(""), type, contPoint, objList);
141 setContinuationPoint(contPoint);
142 }
143 break;
144 default:
145 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_CLIENT, (_T("There is no interface to perform the browse operation.")));
146 break;
147 }
148
149 return result;
150}

610void SODaCProperty::convertPropertyValueFromSOAPtoDCOM(IN DWORD propID, IN OUT SOCmnVariant& value)
611{
612 switch (propID)
613 {
614 case 1: // dataType
615 {
616 WORD datatype = VT_EMPTY;
617 SOCmnString dtSZ;
618 value.getBSTR(dtSZ);
619 SOCmnVariant::getVARTYPE(dtSZ, datatype);
620 value.setUI2(datatype);
621 break;
622 }
623
624 case 5: // accessRights
625 {
626 DWORD accessRights = OPC_READABLE | OPC_WRITEABLE;
627 SOCmnString arSZ;
628 value.getBSTR(arSZ);
629
630 if (arSZ.find(_T("ead")) == -1)
631 {
632 accessRights &= ~OPC_READABLE;
633 }
634
635 if (arSZ.find(_T("rit")) == -1)
636 {
637 accessRights &= ~OPC_WRITEABLE;
638 }
639
640 value.setUI4(accessRights);
641 break;
642 }
643
644 case 7: //euType
645 {
646 DWORD euType = OPC_NOENUM;
647 SOCmnString euSZ;
648 value.getBSTR(euSZ);
649
650 if (euSZ == _T("analog"))
651 {
652 euType = OPC_ANALOG;
653 }
654 else if (euSZ == _T("enumerated"))
655 {
656 euType = OPC_ENUMERATED;
657 }
658
659 value.setI4(euType);
660 break;
661 }
662 }
663}

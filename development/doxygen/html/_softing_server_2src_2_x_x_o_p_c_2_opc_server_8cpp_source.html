540//********************************************************************
541// Create -- All the real work of creating the DACL gets done here
542// rather than in the constructor, so we can return a
543// success/failure status
544//
545bool NotQuiteNullDacl::Create()
546{
547 bool bCreateOk = false;
548 m_dwErr = 0;
549 // need a SID for the Everyone group
550 bool bEveryoneSID = CreateEveryoneSid();
551
552 if (bEveryoneSID)
553 {
554 // calculate a size for the DACL we're going to use
555 ULONG lACLSize = CalculateAclSize();
556
557 if (lACLSize > 0)
558 {
559 // Allocate memory for the ACL
560 m_pDACL = (PACL)::HeapAlloc(GetProcessHeap(), 0, lACLSize);
561
562 if (m_pDACL != NULL)
563 {
564 // initialize the actual ACL
565 BOOL bInitACL = ::InitializeAcl(m_pDACL, lACLSize, ACL_REVISION);
566
567 if (bInitACL)
568 {
569 // add the ACEs that deny/grant the accesses we're after
570 bCreateOk = AddNotQuiteNullAces();
571 }
572 else
573 {
574 m_dwErr = ::GetLastError();
575 }
576 }
577 else
578 {
579 m_dwErr = ::GetLastError();
580 }
581 }
582 }
583
584 return bCreateOk;
585}

620//********************************************************************
621// CalculateAclSize -- This method calculates the size of the ACL we
622// are about to create.
623// NB: This is just a special-purpose version based on
624// "Ask Dr. Gui #49". A more general method is
625// outlined in Richter/Clark, Chapter 10, page 455. This is
626// sufficient for the purpose of creating our NotQuiteNullDacl but
627// not for creating more general ACLs.
628//
629ULONG NotQuiteNullDacl::CalculateAclSize()
630{
631 ULONG lACLSize = 0;
632 m_dwErr = 0;
633 // validate that SID we created, just to be sure
634 BOOL bSIDOk = ::IsValidSid(m_pEveryoneSid);
635
636 if (bSIDOk)
637 {
638 // we're using that SID twice
639 lACLSize += (::GetLengthSid(m_pEveryoneSid) * 2);
640 // we're using one access allowed ace and one access denied
641 // we substract the length of the sidstart member which
642 // is already counted in the sid lengths above
643 // (See the Initialize ACL documentation for more info.)
644 lACLSize += (sizeof(ACCESS_ALLOWED_ACE) - sizeof(((ACCESS_ALLOWED_ACE*)0)->SidStart));
645 lACLSize += (sizeof(ACCESS_DENIED_ACE) - sizeof(((ACCESS_DENIED_ACE*)0)->SidStart));
646 // and we need to add in the size of the acl itself
647 lACLSize += sizeof(ACL);
648 }
649
650 // else -- there's no extended error info to get
651 return lACLSize;
652}

655//********************************************************************
656// AddNotQuiteNullAces -- Add the two ACEs to the DACL
657// NB: Note that the order of execution is critical here.
658//
659bool NotQuiteNullDacl::AddNotQuiteNullAces()
660{
661 bool bAddedOk = false;
662 m_dwErr = 0;
663 BOOL bAddAce;
664 // Add an ace that denies WRITE_OWNER and WRITE_DAC to 'everyone'
665 // which denies everyone but the owner of the associated object
666 // access to the object's security.
667 bAddAce = ::AddAccessDeniedAce(m_pDACL, // the acl to add the ace to
668 ACL_REVISION, // revision, must be ACL_REVISION
669 WRITE_OWNER | WRITE_DAC, // accesses to deny
670 m_pEveryoneSid); // SID to be denied (everyone)
671
672 if (bAddAce)
673 {
674 // Add an ace that gives 'everyone' all accesses to the object.
675 // By itself, the bit of code below would be the moral equivalent
676 // of a NULL DACL -- it gives all rights to everyone. But, because
677 // accesses are evaluating in order of their placement in the DACL
678 // the DeniedACE above acts as a filter, before this can be evaluated.
679 bAddAce = ::AddAccessAllowedAce(m_pDACL, // the acl to add the ace to
680 ACL_REVISION, // revision, must be ACL_REVISION
681 GENERIC_ALL, // accesses to allow
682 m_pEveryoneSid); // SID to be allowed
683
684 if (bAddAce)
685 {
686 bAddedOk = true;
687 }
688 }
689
690 if (!bAddedOk)
691 {
692 m_dwErr = ::GetLastError();
693 }
694
695 return bAddedOk;

256 case MyWin32.CtrlTypes.CTRL_BREAK_EVENT:
257 case MyWin32.CtrlTypes.CTRL_CLOSE_EVENT:
258 case MyWin32.CtrlTypes.CTRL_LOGOFF_EVENT:
259 case MyWin32.CtrlTypes.CTRL_SHUTDOWN_EVENT:
260 message = "Stop execution, since CTRL command!";
261 Console.EndEvent.Set();
262 Console.End = true;
263 break;
264 } // end switch
265 // Use interop to display a message for the type of event.

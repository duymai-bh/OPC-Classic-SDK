296 OTFreeMemory(conditionState.m_actSubConditionName);

332 }

334 break;

336 case _T('F'):

339 break;
340
341 case _T('f'):

362 }
363

366 case _T('?'):

369 usage();

374 case _T('Q'):

452void ChangeCondition(IN DWORD idx, IN OTCEventData* pEvent)
453{
454 if (g_conditionList[idx].m_eventType == 0)
455 {
456 if ((!(pEvent->m_newState & OT_CONDITIONSTATE_ACTIVE)) && // not active and
457 (pEvent->m_newState & OT_CONDITIONSTATE_ACKED)) // acknowledged
458 {
459 return;
460 }
461
462 // new condition
463 g_conditionList[idx].m_eventType = pEvent->m_eventType;
464 g_conditionList[idx].m_newState = pEvent->m_newState;
465 g_conditionList[idx].m_sourcePath = _tcsdup(pEvent->m_sourcePath);
466 g_conditionList[idx].m_conditionName = _tcsdup(pEvent->m_conditionName);
467 g_conditionList[idx].m_subConditionName = _tcsdup(pEvent->m_subConditionName);
468 g_conditionList[idx].m_occurenceTime = pEvent->m_occurenceTime;
469 g_conditionList[idx].m_severity = pEvent->m_severity;
470 g_conditionList[idx].m_message = _tcsdup(pEvent->m_message);
471 g_conditionList[idx].m_activeTime = pEvent->m_activeTime;
472 g_conditionList[idx].m_cookie = pEvent->m_cookie;
473 }
474 else
475 {
476 if ((!(pEvent->m_newState & OT_CONDITIONSTATE_ACTIVE)) && // not active and
477 (pEvent->m_newState & OT_CONDITIONSTATE_ACKED)) // acknowledged
478 {
479 ClearCondition(idx); // remove
480 }
481 else
482 {
483 // set the changed values
484 free(g_conditionList[idx].m_subConditionName);
485 free(g_conditionList[idx].m_message);
486 g_conditionList[idx].m_newState = pEvent->m_newState;
487 g_conditionList[idx].m_subConditionName = _tcsdup(pEvent->m_subConditionName);
488 g_conditionList[idx].m_occurenceTime = pEvent->m_occurenceTime;
489 g_conditionList[idx].m_severity = pEvent->m_severity;
490 g_conditionList[idx].m_message = _tcsdup(pEvent->m_message);
491 g_conditionList[idx].m_activeTime = pEvent->m_activeTime;
492 g_conditionList[idx].m_cookie = pEvent->m_cookie;
493 }
494 }

497void OTCAPI_CALL OnReceivedEvents(IN OTCObjectContext objectContext, IN DWORD count, IN OTCEventData* pEvents, BYTE refresh, BYTE lastRefresh)
498{
499 ClearConsole();
500 DWORD i, j;
501 DWORD freeCondListIdx;
502
503 for (i = 0; i < count; i++)
504 {
505 freeCondListIdx = MAX_CONDITIONS;
506
507 // find condition in condition list
508 for (j = 0; j < MAX_CONDITIONS; j++)
509 {
510 if (g_conditionList[j].m_eventType != 0)
511 {
512 if ((_tcscmp(g_conditionList[j].m_conditionName, pEvents[i].m_conditionName) == 0) &&
513 (_tcscmp(g_conditionList[j].m_sourcePath, pEvents[i].m_sourcePath) == 0))
514 {
515 // found condition
516 ChangeCondition(j, &pEvents[i]);
517 break;
518 }
519 }
520 else
521 {
522 if (freeCondListIdx == MAX_CONDITIONS)
523 {
524 freeCondListIdx = j;
525 }
526 }
527 }
528
529 if ((j == MAX_CONDITIONS) && (freeCondListIdx != MAX_CONDITIONS))
530 {
531 // not found condition -> add it
532 ChangeCondition(freeCondListIdx, &pEvents[i]);
533 }
534 }
535
536 TCHAR state[12];
537 _tprintf(_T("\nACTIVE and NOT ACKNOWLEDGED CONDITIONS\n\n"));
538
539 for (j = 0; j < MAX_CONDITIONS; j++)
540 {
541 if (g_conditionList[j].m_eventType != 0)
542 {
543 _tcscpy(state, _T(" "));
544
545 if (g_conditionList[j].m_newState & OT_CONDITIONSTATE_ACTIVE)
546 {
547 _tcsncpy(state + 0, _T("ACT"), 3);
548 }
549
550 if (g_conditionList[j].m_newState & OT_CONDITIONSTATE_ACKED)
551 {
552 _tcsncpy(state + 4, _T("ACK"), 3);
553 }
554
555 if (g_conditionList[j].m_newState & OT_CONDITIONSTATE_ENABLED)
556 {
557 _tcsncpy(state + 8, _T("ENA"), 3);
558 }
559
560 _tprintf(_T("%s # %-30.30s # %-15.15s # %-15.15s # %s # %-4.4u # %s\n"), state, g_conditionList[j].m_sourcePath, g_conditionList[j].m_conditionName, g_conditionList[j].m_subConditionName, GetDateTimeSZ(g_conditionList[j].m_occurenceTime), g_conditionList[j].m_severity, g_conditionList[j].m_message);
561 }
562 }
563
564 _tprintf(_T("\n\n"));
565 usage();
566}

122 if( RunResult != null )

124 throw new ApplicationException( "Clear the results before running again" );
125 }
126
127 if( CheckBoxes && !TreeNode.Checked )
128 return;
129
130 bool bMultipleRuns = false;
131 int repeatCount = int.MaxValue;
132
133 if( RepeatCount != null )
134 {
135 int tmp;
136 if( int.TryParse( RepeatCount, out tmp ) )
137 {
138 repeatCount = tmp;

142
143 double repeatDuration = double.MaxValue;
144
145 if( RepeatDuration != null )
146 {
147 double tmp;
148 if( double.TryParse( RepeatDuration, out tmp ) )
149 {
150 repeatDuration = tmp;
151 bMultipleRuns = true;
152 }
153 }
154
155 TestResult result = null;
156
157 if( bMultipleRuns )
158 {
159 int counter = 0;
160 int errors = 0, warnings = 0;
161 List<TestResult> innerResults = new List<TestResult>();
162 DateTime startTime = DateTime.Now;
163 DateTime maxTime = repeatDuration == double.MaxValue ? DateTime.MaxValue : startTime.AddSeconds( repeatDuration );
164
165 while( counter < repeatCount && DateTime.Now <= maxTime )
166 {
167 TestResult runResult;
168
169 runResult = RunOnce();
170 innerResults.Add( runResult );
171
172 switch( runResult.Type )
173 {
174 case TestResults.Failed: errors++;
175 break;
176 case TestResults.Warning: warnings++;
177 break;
178 }
179
180 counter++;
181 }
182
183 MultipleTestResult multipleResult;
184
185 if( errors > 0 )
186 multipleResult = new MultipleTestResult( TestResults.Failed );
187 else if( warnings > 0 )
188 multipleResult = new MultipleTestResult( TestResults.Warning );
189 else
190 multipleResult = new MultipleTestResult( TestResults.Success );
191
192 multipleResult.StartTime = startTime;
193 multipleResult.InnerResults = innerResults;
194
195 ResultsTreeNode.Expand();
196
197 result = multipleResult;
198 }
199 else
200 {
201 result = RunOnce();
202 }
203
204 RunResult = result;
205 ReportFinished();
206 }

209 WaitHandle.Set();

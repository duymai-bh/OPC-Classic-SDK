428// OnTabSC
429//
430// - selected tab changed
431//
432void SODmCMainView::OnTabSC(NMHDR* pNMHDR, LRESULT* pResult)
433{
434 m_listDA.ShowWindow(SW_HIDE);
435 m_listAE.ShowWindow(SW_HIDE);
436 m_listAEC.ShowWindow(SW_HIDE);
437 m_listErr.ShowWindow(SW_HIDE);
438 m_treeSrv.ShowWindow(SW_HIDE);
439 m_lbURLSrv.ShowWindow(SW_HIDE);
440 m_edURLSrv.ShowWindow(SW_HIDE);
441 m_buAddSrv.ShowWindow(SW_HIDE);
442 m_treeDA.ShowWindow(SW_HIDE);
443 m_treeAE.ShowWindow(SW_HIDE);
444
445 switch (m_tab.GetCurSel())
446 {
447 case SODMCMAINVIEW_TAB_SRV: // Installed OPC Servers
448#ifdef SODMC_DCOM
449 m_treeSrv.ShowWindow(SW_SHOW);
450#endif // SODMC_DCOM
451 m_lbURLSrv.ShowWindow(SW_SHOW);
452 m_edURLSrv.ShowWindow(SW_SHOW);
453 m_buAddSrv.ShowWindow(SW_SHOW);
454 break;
455#ifdef SODMC_DA
456
457 case SODMCMAINVIEW_TAB_TDA: // DA Browse
458 {
459 SODmCObjectView* objview = ::getObjectView();
460 SOCmnPointer<SODaCServer> curSelSrv = objview->getSelectedServerDA();;
461
462 if ((SODaCServer*)m_treeDAServer != (SODaCServer*)curSelSrv)
463 {
464 m_treeDA.DeleteAllItems();
465 m_treeDAServer = curSelSrv;
466
467 if (m_treeDAServer.isNotNull())
468 {
469 SODmCCmnServerDA* pCmnSrv = NULL;
470#ifdef SODMC_DCOM
471
472 if (m_treeDAServer->getProtocol() == SOCLT_PROTOCOL_DCOM)
473 {
474 pCmnSrv = (SODmCCmnServerDA*)(SODmCComServerDA*)(SODaCServer*)m_treeDAServer;
475 }
476
477#endif
478#ifdef SODMC_SOAP
479
480 if (m_treeDAServer->getProtocol() == SOCLT_PROTOCOL_SOAP)
481 {
482 pCmnSrv = (SODmCCmnServerDA*)(SODmCXmlServerDA*)(SODaCServer*)m_treeDAServer;
483 }
484
485#endif
486#ifdef SOFEATURE_TUNNEL
487

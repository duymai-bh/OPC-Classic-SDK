345 PropertyDescriptorCollection ICustomTypeDescriptor.GetProperties(Attribute[] attributes)
346 {
347 ArrayList props = new ArrayList();
348
349 foreach (CustomProperty property in properties)
350 {
351 ArrayList attrs = new ArrayList();
352
353 // If a category, description, editor, or type converter are specified
354 // in the PropertySpec, create attributes to define that relationship.
355 if (property.Category != null)
356 attrs.Add(new CategoryAttribute(property.Category));
357
358 if (property.Description != null)
359 attrs.Add(new DescriptionAttribute(property.Description));
360
361 if (property.EditorTypeName != null)
362 attrs.Add(new EditorAttribute(property.EditorTypeName, typeof(UITypeEditor)));
363
364 if (property.ConverterTypeName != null)
365 attrs.Add(new TypeConverterAttribute(property.ConverterTypeName));
366
367 // Additionally, append the custom attributes associated with the
368 // PropertySpec, if any.
369 if (property.Attributes != null)
370 attrs.AddRange(property.Attributes);
371
372 Attribute[] attrArray = (Attribute[])attrs.ToArray(typeof(Attribute));
373
374 // Create a new property descriptor for the property item, and add
375 // it to the list.
376 CustomPropertyDescriptor pd = new CustomPropertyDescriptor(property,
377 this, property.Name, attrArray);
378 props.Add(pd);
379 }
380
381 // Convert the list of PropertyDescriptors to a collection that the
382 // ICustomTypeDescriptor can use, and return it.
383 PropertyDescriptor[] propArray = (PropertyDescriptor[])props.ToArray(
384 typeof(PropertyDescriptor));
385 return new PropertyDescriptorCollection(propArray);
386

656 internal AddressSpaceElement GetParent(uint aHandle)
657 {
658 if (aHandle == 0)
659 {
660 return m_root;
661 } // end if
662
663 OTObjectData parent = new OTObjectData();
664 IntPtr pParent = Marshal.AllocCoTaskMem(Marshal.SizeOf(parent));
665 Marshal.StructureToPtr(parent, pParent, false);
666
667 if ((uint) EnumResultCode.S_OK != OTBFunctions.OTSGetParent(aHandle, pParent))
668 {
669 return null;
670 } // end if
671
672 parent = (OTObjectData) Marshal.PtrToStructure(pParent, typeof (OTObjectData));
673 Marshal.FreeCoTaskMem(pParent);
674
675 AddressSpaceElement elementParent = GetElementFromArray(parent.m_userData);
676 if (elementParent == null)
677 {
678 return m_root;
679 } // end if
680
681 return elementParent;
682 } // end getParent

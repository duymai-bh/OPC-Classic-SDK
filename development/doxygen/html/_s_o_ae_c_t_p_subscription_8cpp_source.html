52//-----------------------------------------------------------------------------
53// doStateTransitionImpl
54//
55// - do the state transition for the Subscription if the target state differs from the
56// actual state of the Subscription
57//
58// returns:
59// S_OK - transit to new state
60// E_FAIL - error during state change
61//
62HRESULT SOAeCTPSubscriptionImpl::doStateTransitionImpl(void)
63{
64 SOAeCSubscription* pAeCSubscription = getObjectAeCSubscription();
65 BYTE targetState = pAeCSubscription->getTargetObjectState();
66 SOCmnPointer<SOAeCServer> server = (SOAeCServer*)pAeCSubscription->getParentBranch();
67 HRESULT res = S_OK;
68
69 // check server object
70 if (!server)
71 {
72 pAeCSubscription->setLastError(E_FAIL, SOCLT_ERR_NO_PARENT_OBJECT);
73 return E_FAIL;
74 }
75
76 if (pAeCSubscription->getObjectState() != targetState)
77 {
78 SOCmnPointer<SOCltTPConnection> tpCon = (SOCltTPConnection*)server->getTPConnection();
79 SOCmnPointer<SOAeCTPCall> tpCall;
80 SOCMN_ALLOCATE_OBJECT_C(tpCall, SOAeCTPCall(tpCon), (SOAeCTPCall*));
81
82 switch (pAeCSubscription->getObjectState())
83 {
84 case SOCltElement::disconnected:
85 {
86 //-- connect or start Subscription --
87 if (!server->isConnected())
88 {
89 pAeCSubscription->setLastError(E_FAIL, SOCLT_ERR_PARENT_NOT_CONNECTED);
90 return E_FAIL;
91 }
92
93 pAeCSubscription->beforeInterfaceCall();
94 res = tpCall->addSubscription(pAeCSubscription, m_serverHandle);
95 pAeCSubscription->afterInterfaceCall();
96 _TRACEO_LEVEL_FROM_RES(res, SOCMNTRACE_G_OPC_CALL_C, pAeCSubscription, (_T("TPAE::Server::AddSubscription [0x%X] "), res));
97 pAeCSubscription->setLastError(res, SOAEC_ERR_ADD_SUBSCRIPTION);
98
99 if (SUCCEEDED(res))
100 {
101 res = pAeCSubscription->doUpdate(FALSE, SOAEC_SUBSPAR_FILTER | SOAEC_SUBSPAR_RETURNED_ATTRIBUTES);
102 }
103
104 if (SUCCEEDED(res))
105 {
106 pAeCSubscription->setObjectState(targetState); // set object state to target state
107 }
108 else
109 {
110 // undo all if failed
111 res = pAeCSubscription->doDisconnect();
112 }
113 }
114 break;
115
116 default:
117 if (targetState == SOCltElement::disconnected)
118 {
119 res = pAeCSubscription->doDisconnect();
120 }
121 else
122 {
123 //-- start or connect Subscription --
124 pAeCSubscription->m_bActive = (targetState == SOCltElement::started) ? TRUE : FALSE;
125 res = pAeCSubscription->doUpdate(FALSE, SOAEC_SUBSPAR_ACTIVE);
126
127 if (SUCCEEDED(res))
128 {
129 pAeCSubscription->setObjectState(targetState);
130 }
131 }
132
133 break;
134 }
135 }
136
137 return res;
138} // doStateTransitionImpl

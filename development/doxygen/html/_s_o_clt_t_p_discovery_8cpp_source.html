160BOOL SOCltTPDiscovery::initCommunication(
161 const SOCmnString& address,
162 USHORT port,
163 USHORT localPort,
164 DWORD timeout)
165{
166 if (port < 0 || port >= USHRT_MAX)
167 {
168 _TRACEC(SOCMNTRACE_L_ERR, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("initCommunication: invalid tunnel discovery port number")));
169 return FALSE;
170 } // end if
171
172 if (localPort < 0 || localPort >= USHRT_MAX)
173 {
174 _TRACEC(SOCMNTRACE_L_ERR, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("initCommunication: invalid local port number")));
175 return FALSE;
176 } // end if
177
178 if (timeout <= 0)
179 {
180 _TRACEC(SOCMNTRACE_L_WRN, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("initCommunication: invalid tunnel discovery timeout")));
181 return FALSE;
182 } // end if
183
184 if (address.isEmpty() != TRUE)
185 {
186 USES_CONVERSION;
187 m_udpCommunicationSender->setBroadcastAddress(inet_addr(T2CA((LPCTSTR)address)));
188 } // end if
189
190 // send on userDefined port + 1 !!! - needed if running on same machine
191 USHORT sendPort = port + 1;
192 m_udpCommunicationSender->setBroadCastPort(sendPort);
193 m_udpCommunicationSender->setLocalPort(localPort);
194 m_udpCommunicationSender->setBroadcastReplyTimeOut(timeout);
195 m_udpCommunicationReceiver->setBroadCastPort(port);
196 m_udpCommunicationReceiver->setLocalPort(localPort);
197 m_udpCommunicationReceiver->setBroadcastReplyTimeOut(timeout);
198
199 if (m_udpCommunicationSender->init(FALSE) &&
200 m_udpCommunicationReceiver->init(TRUE))
201 {
202 m_listenThread->start(listenThreadProc, this);
203 _TRACEC(SOCMNTRACE_L_INF, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("initCommunication: ok on - %s:%d with timeout %d"), address, port, timeout));
204 return TRUE;
205 } // end if
206
207 _TRACEC(SOCMNTRACE_L_ERR, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("initCommunication: not ok on - %s:%d with timeout %d"), address, port, timeout));
208 return FALSE;
209} // end initCommunication

322BOOL SOCltTPDiscovery::handleRequest(
323 UCHAR* pMessage,
324 ULONG size)
325{
326 if (pMessage == NULL || size == 0)
327 {
328 m_listenThread->getEvents()->signal(m_listenEvent);
329 _TRACEC(SOCMNTRACE_L_WRN, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("handleRequest: failed on argument checking %d %d"), pMessage, size));
330 return FALSE;
331 } // end if
332
333 // unmarshal the message
334 m_unMarshaler->SetMarshalData(pMessage, size);
335 SOCmnString requestIp;
336 m_unMarshaler->UnMarshalString(&requestIp);
337 T_TP_REQ_HEADER request;
338 m_unMarshaler->UnMarshalShort(&request.Length);
339 m_unMarshaler->UnMarshalShort(&request.ResponseType);
340 m_unMarshaler->UnMarshalShort(&request.Version);
341 m_unMarshaler->UnMarshalShort(&request.Service);
342 m_unMarshaler->UnMarshalShort(&request.DeviceType);
343
344 // check if really is discovery message
345 if ((request.DeviceType != TP_DISCOVERY_DEVICE_TYPE_ALL &&
346 request.DeviceType != TP_DISCOVERY_DEVICE_TYPE_OPC_TUNNEL &&
347 request.DeviceType != TP_DISCOVERY_DEVICE_TYPE_OPC_XMLDA) ||
348 request.Service != TP_DISCOVERY_SERVICE)
349 {
350 m_listenThread->getEvents()->signal(m_listenEvent);
351 _TRACEC(SOCMNTRACE_L_WRN, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("handleRequest: not a discovery message %d %d"), pMessage, size));
352 return FALSE;
353 } // end if
354
355 // if multi-interface setup determine IP address
356 setIpAddressOnDevices(m_udpCommunicationReceiver->getBroadcastReceiverInterface());
357 m_listenThread->getEvents()->reset(m_listenEvent);
358 // always answer only to asker, do not broadcast response
359 requestIp = m_udpCommunicationReceiver->getSenderAddress();
360
361 if (requestIp)
362 {
363 sendUnicastResponse(request.DeviceType);
364 }
365 else
366 {
367 // just wait for another request
368 m_listenThread->getEvents()->signal(m_listenEvent);
369 } // end if .. else
370
371 _TRACEC(SOCMNTRACE_L_INF, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("handleRequest: succesfully handled request %d %d"), pMessage, size));
372 return TRUE;
373} // end if

403BOOL SOCltTPDiscovery::sendResponse(
404 USHORT deviceType,
405 bool sendBroadcast)
406{
407 UCHAR* message = NULL;
408 ULONG size = 0;
409 SOCmnListPosition pos = m_deviceIdentityList.getStartPosition();
410
411 while (pos)
412 {
413 DeviceIdentity* pDevice = m_deviceIdentityList.getNext(pos);
414
415 if (pDevice != NULL)
416 {
417 message = NULL;
418 size = 0;
419
420 // send only response if device is matching request type
421 if ((deviceType == TP_DISCOVERY_DEVICE_TYPE_ALL) || // all
422 (deviceType == TP_DISCOVERY_DEVICE_TYPE_OPC_TUNNEL && pDevice->isTunnel() == TRUE) || // tunnel
423 (deviceType == TP_DISCOVERY_DEVICE_TYPE_OPC_XMLDA && pDevice->isXmlDa() == TRUE))
424 {
425 if (marshalResponse(pDevice, &message, size, deviceType) == TRUE)
426 {
427 if (sendBroadcast)
428 {
429 for (UINT i = 0L, n = m_udpCommunicationSender->getNoOfIfaces(); i < n; ++i)
430 {
431 m_udpCommunicationSender->setBroadcastAddress(m_udpCommunicationSender->getIfaceBcastAddr(i));
432 m_udpCommunicationSender->sendData((char*)message, size);
433 _TRACEC(SOCMNTRACE_L_INF, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("sendResponse: sent response %s of size %d"), message, size));
434 }
435 }
436 else
437 {
438 USES_CONVERSION;
439 m_udpCommunicationSender->setBroadcastAddress(inet_addr(T2CA((LPCTSTR)m_udpCommunicationReceiver->getSenderAddress())));
440 m_udpCommunicationSender->sendData((char*)message, size);
441 _TRACEC(SOCMNTRACE_L_INF, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("sendResponse: sent response %s of size %d"), message, size));
442 }
443 } // end if
444 } // end if
445 } // end if
446 } // end while
447
448 m_listenThread->getEvents()->signal(m_listenEvent);
449 return TRUE;
450} // end sendResponse

484BOOL SOCltTPDiscovery::marshalResponse(
485 DeviceIdentity* pDevice,
486 UCHAR** message,
487 ULONG& size,
488 USHORT deviceType)
489{
490 // check response type
491 if (deviceType == TP_DISCOVERY_DEVICE_TYPE_OPC_TUNNEL &&
492 !pDevice->isTunnel())
493 {
494 // no tunnel IP is set
495 _TRACEC(SOCMNTRACE_L_DEB, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("marshalResponse: tunnel response requested but no tunnel ip is defined")));
496 return FALSE;
497 } // end if
498
499 if (deviceType == TP_DISCOVERY_DEVICE_TYPE_OPC_XMLDA &&
500 !pDevice->isXmlDa())
501 {
502 // no http port or XMLDA url is set
503 _TRACEC(SOCMNTRACE_L_DEB, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("marshalResponse: xmlda response requested but no xmlda url is defined")));
504 return FALSE;
505 } // end if
506
507 T_TP_DEVICE_RES response;
508 response.Header.Length = TP_DISCOVERY_RESPONSE_HEADER_LENGTH;
509 response.Header.Version = TP_DISCOVERY_RESPONSE_VERSION_NUMBER;
510 response.Header.Service = TP_DISCOVERY_RESPONSE_SERVICE;
511 response.Header.DeviceType = deviceType;
512 response.Header.Status = pDevice->getStatus();
513 memcpy(response.Url, pDevice->getUrlWithIp(), sizeof(response.Url));
514 memcpy(response.ServerName, pDevice->getServerName(), sizeof(response.ServerName));
515 memcpy(response.VendorInfo, pDevice->getVendorInfo(), sizeof(response.VendorInfo));
516 memcpy(response.HostName, pDevice->getHostName(), sizeof(response.HostName));
517 response.MajorVersion = pDevice->getMajorVersion();
518 response.MinorVersion = pDevice->getMinorVersion();
519 response.PatchVersion = pDevice->getPatchVersion();
520 response.BuildNumber = pDevice->getBuildNumber();
521 m_marshaler->BeginMarshal(sizeof(T_TP_DEVICE_RES));
522 getMarshaler()->MarshalShort(response.Header.Length);
523 getMarshaler()->MarshalShort(response.Header.Version);
524 getMarshaler()->MarshalShort(response.Header.Service);
525 getMarshaler()->MarshalShort(response.Header.Status);
526 getMarshaler()->MarshalShort(response.Header.DeviceType);
527 getMarshaler()->MarshalString(pDevice->getUrlWithIp());
528 getMarshaler()->MarshalString(pDevice->getServerName());
529 getMarshaler()->MarshalString(pDevice->getVendorInfo());
530 getMarshaler()->MarshalShort(response.MajorVersion);
531 getMarshaler()->MarshalShort(response.MinorVersion);
532 getMarshaler()->MarshalShort(response.PatchVersion);
533 getMarshaler()->MarshalLong(response.BuildNumber);
534 getMarshaler()->MarshalString(pDevice->getHostName());
535
536 if (getMarshaler()->GetMarshalDataLen() <= 0 ||
537 getMarshaler()->GetMarshalData() == 0)
538 {
539 _TRACEC(SOCMNTRACE_L_WRN, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("marshalResponse: invalid marshaled data for: %s"), pDevice->getUrl()));
540 return FALSE;
541 } // end if
542
543 // else allocate and copy the message
544 *message = (UCHAR*)malloc(getMarshaler()->GetMarshalDataLen());
545 memcpy(*message, getMarshaler()->GetMarshalData(), getMarshaler()->GetMarshalDataLen());
546 size = getMarshaler()->GetMarshalDataLen();
547 _TRACEC(SOCMNTRACE_L_DEB, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("marshalResponse: succesfully marshaled response for: %s"), pDevice->getUrl()));
548 return TRUE;
549} // end marshalResponse

553DWORD _stdcall SOCltTPDiscovery::listenThreadProc(
554 void* pContext)
555{
556 if (pContext == NULL)
557 {
558 _TRACEC(SOCMNTRACE_L_ERR, SOCMNTRACE_G_DISCOVERY_CLT, _T("SOCltTPDiscovery"), (_T("listenThreadProc: invalid context")));
559 return S_FALSE;
560 } // end if
561
562 SOCltTPDiscovery* pT = reinterpret_cast<SOCltTPDiscovery*>(pContext);
563 pT->addRef();
564 SOCmnThread* pMe = pT->getListenThread();
565 DWORD waitR;
566 BOOL end = FALSE;
567 SOCmnUDPSocket* pCommunicationHandler = pT->getUdpCommunicationReceiver();
568 DWORD dwWait = pCommunicationHandler->getBroadcastReplyTimeOut();
569
570 while (!end)
571 {
572 waitR = pMe->waitForEvents(dwWait);
573
574 if (waitR == pMe->getStopEventId())
575 {
576 end = TRUE;
577 }
578 else if (waitR == pT->getListenEvent())
579 {
580 UCHAR* pMessage = NULL;
581 ULONG size = 0;
582
583 while (TRUE)
584 {
585 if (pCommunicationHandler->hasPendingData() == TRUE)
586 {
587 if (pCommunicationHandler->receiveData((char**)&pMessage, (ULONG)size) == S_OK)
588 {
589 pT->handleRequest(pMessage, size);
590 }
591 else
592 {
593 //error = SOCmnUDPSocket::WSAHandler::instance()->getLastError();
594 pMe->getEvents()->signal(pT->getListenEvent());
595 break;
596 } // end if .. else
597 }
598 else
599 {
600 //error = SOCmnUDPSocket::WSAHandler::instance()->getLastError();
601 pMe->getEvents()->signal(pT->getListenEvent());
602 break;
603 } // end if .. else
604 } // end while
605 } // end else .. if
606 } // end while
607
608 pT->release();
609 return S_OK;
610} // end listenThreadProc

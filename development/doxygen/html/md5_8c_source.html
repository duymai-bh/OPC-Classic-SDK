50/* F, G and H are basic MD5 functions: selection, majority, parity */
51#define F(x, y, z) (((x) & (y)) | ((~x) & (z)))
52#define G(x, y, z) (((x) & (z)) | ((y) & (~z)))
53#define H(x, y, z) ((x) ^ (y) ^ (z))
54#define I(x, y, z) ((y) ^ ((x) | (~z)))
55
56/* ROTATE_LEFT rotates x left n bits */
57#define ROTATE_LEFT(x, n) (((x) << (n)) | ((x) >> (32-(n))))
58
59/* FF, GG, HH, and II transformations for rounds 1, 2, 3, and 4 */
60/* Rotation is separate from addition to prevent recomputation */
61#define FF(a, b, c, d, x, s, ac) \
62 {(a) += F ((b), (c), (d)) + (x) + (UINT4)(ac); \
63 (a) = ROTATE_LEFT ((a), (s)); \
64 (a) += (b); \
65 }
66#define GG(a, b, c, d, x, s, ac) \
67 {(a) += G ((b), (c), (d)) + (x) + (UINT4)(ac); \
68 (a) = ROTATE_LEFT ((a), (s)); \
69 (a) += (b); \
70 }
71#define HH(a, b, c, d, x, s, ac) \
72 {(a) += H ((b), (c), (d)) + (x) + (UINT4)(ac); \
73 (a) = ROTATE_LEFT ((a), (s)); \
74 (a) += (b); \
75 }
76#define II(a, b, c, d, x, s, ac) \
77 {(a) += I ((b), (c), (d)) + (x) + (UINT4)(ac); \
78 (a) = ROTATE_LEFT ((a), (s)); \
79 (a) += (b); \
80 }

94void MD5Update (
95MD5_CTX *mdContext,
96unsigned char *inBuf,
97unsigned int inLen)
98{
99 UINT4 in[16];
100 int mdi;
101 unsigned int i, ii;
102
103 /* compute number of bytes mod 64 */
104 mdi = (int)((mdContext->i[0] >> 3) & 0x3F);
105
106 /* update number of bits */
107 if ((mdContext->i[0] + ((UINT4)inLen << 3)) < mdContext->i[0])
108 mdContext->i[1]++;
109 mdContext->i[0] += ((UINT4)inLen << 3);
110 mdContext->i[1] += ((UINT4)inLen >> 29);
111
112 while (inLen--) {
113 /* add new character to buffer, increment mdi */
114 mdContext->in[mdi++] = *inBuf++;
115
116 /* transform if necessary */
117 if (mdi == 0x40) {
118 for (i = 0, ii = 0; i < 16; i++, ii += 4)
119 in[i] = (((UINT4)mdContext->in[ii+3]) << 24) |
120 (((UINT4)mdContext->in[ii+2]) << 16) |
121 (((UINT4)mdContext->in[ii+1]) << 8) |
122 ((UINT4)mdContext->in[ii]);
123 Transform (mdContext->buf, in);
124 mdi = 0;
125 }
126 }

129void MD5Final (MD5_CTX *mdContext)
130{
131 UINT4 in[16];
132 int mdi;
133 unsigned int i, ii;
134 unsigned int padLen;
135
136 /* save number of bits */
137 in[14] = mdContext->i[0];
138 in[15] = mdContext->i[1];
139
140 /* compute number of bytes mod 64 */
141 mdi = (int)((mdContext->i[0] >> 3) & 0x3F);
142
143 /* pad out to 56 mod 64 */
144 padLen = (mdi < 56) ? (56 - mdi) : (120 - mdi);
145 MD5Update (mdContext, PADDING, padLen);
146
147 /* append length in bits and transform */
148 for (i = 0, ii = 0; i < 14; i++, ii += 4)
149 in[i] = (((UINT4)mdContext->in[ii+3]) << 24) |
150 (((UINT4)mdContext->in[ii+2]) << 16) |
151 (((UINT4)mdContext->in[ii+1]) << 8) |
152 ((UINT4)mdContext->in[ii]);
153 Transform (mdContext->buf, in);
154
155 /* store buffer in digest */
156 for (i = 0, ii = 0; i < 4; i++, ii += 4) {
157 mdContext->digest[ii] = (unsigned char)(mdContext->buf[i] & 0xFF);
158 mdContext->digest[ii+1] =
159 (unsigned char)((mdContext->buf[i] >> 8) & 0xFF);
160 mdContext->digest[ii+2] =
161 (unsigned char)((mdContext->buf[i] >> 16) & 0xFF);
162 mdContext->digest[ii+3] =
163 (unsigned char)((mdContext->buf[i] >> 24) & 0xFF);
164 }
165}

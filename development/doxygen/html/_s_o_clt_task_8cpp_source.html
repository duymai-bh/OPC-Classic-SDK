192//-----------------------------------------------------------------------------
193// execute
194//
195void SOCltTaskConnectionMonitor::execute(void)
196{
197 _ASSERT_PTR((SOCltElement*)m_element);
198
199 if (SOCltElement::disconnected != m_element->getObjectState())
200 {
201 if (m_element->is(SOCMNOBJECT_TYPE_SERVER))
202 {
203#ifdef SOFEATURE_DCOM
204 {
205 SOCltServer* pServer = (SOCltServer*)(SOCltElement*)m_element;
206
207 if (pServer->getProtocol() == SOCLT_PROTOCOL_DCOM)
208 {
209 // Ping the real COM object, to see if it is still alive.
210 // We simply ask for an non-existing interface.
211 // Asking for an existing one would not work since COM cashes
212 // interface pointers.
213 CComPtr<IUnknown> pUnknown;
214 HRESULT res = m_element->queryInterface(IID_IUnknown, (void**)&pUnknown);
215
216 if (!!pUnknown)
217 {
218 void* pv = NULL;
219 m_element->beforeInterfaceCall();
220 res = pUnknown->QueryInterface(_iid_not_there_, &pv);
221 m_element->afterInterfaceCall();
222 }
223
224 // Since we know the interface does not exist, the only valid
225 // return code is E_NOINTERFACE. Anything else means connection
226 // failure or server crash.
227
228 if (res == E_NOINTERFACE)
229 {
230 m_res = S_OK;
231 m_state = is_idle;
232 m_lastCall = ::GetTickCount();
233 return; // everything okay
234 }
235
236 m_res = res;
237 m_state = reconnect1;
238 m_cnt_attempts = 0;
239 // Set element and it's children status to 'disconnected'
240 doDisconnectElement(m_element);
241 }
242 }
243#endif // SOFEATURE_DCOM
244#ifdef SOFEATURE_SOAP
245 {
246 SOCltServer* pServer = (SOCltServer*)(SOCltElement*)m_element;
247
248 if (pServer->getProtocol() == SOCLT_PROTOCOL_SOAP)
249 {
250 SODaCServer* pDaCServer = (SODaCServer*)(SOCltElement*)m_element;
251 SOCmnPointer<SOCltHTTPConnection> pHttpCon = pDaCServer->getHTTPConnection();
252
253 if (pHttpCon.isNotNull() == TRUE)
254 {
255 if (pHttpCon->isConnected() != TRUE)
256 {
257 doDisconnectElement(m_element);
258 } // end if
259 } // end if
260 } // end if
261 }
262#endif // SOFEATURE_SOAP
263#ifdef SOFEATURE_TUNNEL
264 {
265 SOCltServer* pServer = (SOCltServer*)(SOCltElement*)m_element;
266
267 if (pServer->getProtocol() == SOCLT_PROTOCOL_TUNNEL)
268 {
269 if (pServer->is(SOCMNOBJECT_TYPE_SERVER_DA))
270 {
271 SODaCServer* pDaCServer = (SODaCServer*)(SOCltElement*)m_element;
272 SOCmnPointer<SOCltTPConnection> tpCon = pDaCServer->getTPConnection();
273
274 if (tpCon.isNotNull())
275 {
276 SOCmnPointer<SOCltServer> tpServer;
277 tpServer = tpCon->getServer();
278
279 if (!tpServer)
280 {
281 doDisconnectElement(m_element);
282 }
283 }
284 }
285
286#ifdef SOAEC
287 else if (pServer->is(SOCMNOBJECT_TYPE_SERVER_AE))
288 {
289#ifdef SOOS_WINDOWS
290 SOAeCServer* pAeCServer = (SOAeCServer*)(SOCltElement*)m_element;
291 SOCmnPointer<SOCltTPConnection> tpCon = pAeCServer->getTPConnection();
292
293 if (tpCon.isNotNull())
294 {
295 SOCmnPointer<SOCltServer> tpServer;
296 tpServer = tpCon->getServer();
297
298 if (!tpServer)
299 {
300 doDisconnectElement(m_element);
301 }
302 }
303
304#endif // SOOS_WINDOWS
305#ifdef SOOS_LINUX
306 //TODO Verify AE tunnel
307#endif // SOOS_LINUX
308 }
309
310#endif // SOAEC
311 }
312 }
313#endif // SOFEATURE_TUNNEL
314 }
315 }
316
317 doReconnectElement(m_element);
318
319 if (reconnect1 == m_state)
320 {
321 m_cnt_attempts++;
322
323 if (m_cnt_attempts > m_attempts)
324 {
325 m_state = reconnect2;
326 m_cnt_attempts = 0;
327 }
328 }
329
330 m_lastCall = ::GetTickCount();
331} // execute

440//-----------------------------------------------------------------------------
441// doReconnectElement
442//
443void SOCltTaskConnectionMonitor::doReconnectElement(IN SOCltElement* element)
444{
445 _ASSERT_PTR(element);
446
447 if (element->getObjectState() != element->getTargetObjectState())
448 {
449 element->setConnectionMonitorStatus(FALSE);
450 HRESULT res = element->doStateTransition(TRUE, SOCLTTASKSTATETRANSITION_NO_STATE_CHANGE);
451
452 if (SUCCEEDED(res))
453 {
454 doReconnectElementsInList(element->getBranchList());
455 doReconnectElementsInList(element->getLeafList());
456 doReconnectElementsInList(element->getAdditionalList());
457 }
458 else
459 {
460 if (SOSRV_IS_RPC_ERROR(res)) // The RPC server is unavailable.
461 {
462 element->setConnectionMonitorStatus(TRUE);
463 }
464 }
465 }

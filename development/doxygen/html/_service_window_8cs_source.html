345 private void ExecuteWriteItem()
346 {
347 //Get the new values
348 Helper.DataTypes dataTypeHelper = new DataTypes();
349 Helper.CustomProperties writeServiceColl = propertyGrid.SelectedObject as Helper.CustomProperties;
350 if (writeServiceColl == null) return;
351 string strItemId = "";
352 string strData = "";
353 string strDataType = "";
354 string strStartIndex = "";
355 foreach (Helper.CustomProperty customProperty in writeServiceColl.Properties)
356 {
357 if (customProperty == null) return;
358 if (customProperty.Name.Equals("Item Id"))
359 {
360 strItemId = customProperty.DefaultValue.ToString();
361 }
362 if (customProperty.Name.Equals("Value"))
363 {
364 strData = customProperty.DefaultValue.ToString();
365 }
366 else if (customProperty.Name.Equals("Data Type"))
367 {
368 strDataType = customProperty.DefaultValue.ToString();
369 }
370 else if (customProperty.Name.Equals("Start Index"))
371 {
372 strStartIndex = customProperty.DefaultValue.ToString();
373 }
374 }
375
376 //Create the new ValueQT
377 ValueQT itemValue = new ValueQT();
378 Type dataType = null;
379 if (dataTypeHelper.RequiredDataTypes.ContainsValue(strDataType))
380 {
381 foreach (Type type in dataTypeHelper.RequiredDataTypes.Keys)
382 {
383 if (dataTypeHelper.RequiredDataTypes[type] == strDataType)
384 {
385 dataType = type;
386 break;
387 }
388 }
389 }
390 if (dataType == null) return;
391 if (!dataType.IsArray)
392 {
393 //data type is not array
394 object objData = null;
395 ConvertToType(dataType, out objData, strData);
396 itemValue.SetData(objData, EnumQuality.GOOD, DateTime.Now);
397 if (item != null)
398 {
399 SetWriteItemToDaItem(itemValue, dataType);
400 }
401 else if (item == null )
402 {
403 SetWriteItemToAddSpaceElem(strItemId, itemValue, dataType);
404 }
405 }
406 else
407 {
408 //data type is array
409 string typeStr = dataType.FullName;
410 Type arrayType = System.Type.GetType(typeStr.Substring(0, typeStr.IndexOf('[')));
411 int startIndex = 0;
412 if (int.TryParse(strStartIndex, out startIndex))
413 {
414 string[] objDatas = strData.Trim().Split(';');
415 if (objDatas == null) return;
416 Array array = null;
417 if (startIndex > 0)
418 {
419 //non-zero bound array
420 array = Array.CreateInstance(arrayType, new int[] { objDatas.Length }, new int[] { startIndex });
421 }
422 else
423 {
424 //zero boud array
425 array = Array.CreateInstance(arrayType, objDatas.Length);
426 }
427 int strIndex = 0;
428 for (int index = array.GetLowerBound(0);
429 index <= array.GetUpperBound(0);
430 index++)
431 {
432 object objData = null;
433 ConvertToType(arrayType, out objData, objDatas[strIndex]);
434 array.SetValue(objData, index);
435 strIndex++;
436 }
437 itemValue.SetData(array, EnumQuality.GOOD, DateTime.Now);
438 if (item != null)
439 {
440 SetWriteItemToDaItem(itemValue, dataType);
441 }
442 else if (item == null)
443 {
444 SetWriteItemToAddSpaceElem(strItemId,itemValue, dataType);
445 }
446 }
447 }

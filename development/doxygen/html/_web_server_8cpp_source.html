24BOOL CWebServer::handleTemplate(
25 LPCTSTR templ,
26 SOCmnStringList* pArgsList,
27 SOSrvWebExpandData* /* pExData */,
28 SOCmnStringEx& result)
29{
30 BOOL ret = TRUE;
31
32 if (_tcsncmp(templ, _T("GETVERSION"), 10) == 0)
33 {
34#ifdef SOOS_WINDOWS
35 result.format(_T("%u.%2.2u.%u.%4.4u"), MAJOR_VERSION_NUMBER, MINOR_VERSION_NUMBER, PATCH_VERSION_NUMBER, BUILD_NUMBER);
36#else
37 result.format(_T("%u.%2.2u.%u.%4.4u"), MAJOR_VERSION_NUMBER, MINOR_VERSION_NUMBER, PATCH_VERSION_NUMBER, BUILD_NUMBER);
38#endif
39 }
40 else if (_tcsncmp(templ, _T("CLIENTCONNECTIONS##"), 19) == 0)
41 {
42 SOCmnString type = pArgsList->pop();
43 SOCmnString language = pArgsList->pop();
44 getClientConnections(type, language, result);
45 }
46 else
47 {
48 ServerApp* pApp = ::getServerApp();
49
50 if (pApp->m_callbackFunctions.m_OTSWebHandleTemplate)
51 {
52 DWORD numArgs = pArgsList->getCount();
53 LPTSTR* pArgs = NULL;
54 LPTSTR pResult = NULL;
55 SOCmnListPosition posArgs;
56 DWORD i = 0;
57 SOCmnString arg;
58
59 if (numArgs > 0)
60 {
61 pArgs = (LPTSTR*)OTAllocateMemory(sizeof(LPTSTR) * numArgs);
62 }
63
64 posArgs = pArgsList->getStartPosition();
65
66 while (posArgs)
67 {
68 arg = pArgsList->getNext(posArgs);
69 pArgs[i] = (LPTSTR)OTAllocateMemory(sizeof(TCHAR) * (DWORD)(_tcslen(arg) + 1));
70 _tcscpy(pArgs[i], arg);
71 i++;
72 }
73
74 SOCMN_TRY
75 {
76 ret = SUCCEEDED(pApp->m_callbackFunctions.m_OTSWebHandleTemplate((LPTSTR)(LPCTSTR)templ, numArgs, pArgs, &pResult));
77 }
78 SOCMN_CATCH(...)
79 {
80 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_SERVER, (_T("CWebServer::handleTemplate - Exception occurred in vendor application callback: OTSWebHandleTemplate")));
81 }
82
83 if (ret)
84 {
85 result = pResult;
86 OTFreeMemory(pResult);
87 }
88
89 for (i = 0; i < numArgs; i++)
90 {
91 OTFreeMemory(pArgs[i]);
92 }
93
94 if (numArgs > 0)
95 {
96 OTFreeMemory(pArgs);
97 }
98 }
99 else
100 {
101 ret = FALSE;
102 }
103 }
104
105 return ret;

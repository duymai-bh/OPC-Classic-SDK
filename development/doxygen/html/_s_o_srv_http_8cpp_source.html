235{
236 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SOCKET, (_T("Add client %lu 0x%8.8lX"), clientId, (DWORD)pSocket));

239 BOOL ret;

244 ret = srvList.add(srvData, pSocket->getOpcHandle());
245 srvList.autoUnlock();
246 m_serverHTTP->setClientHandle(pSocket, srvData);
247 SOCmnList<SOSrvHTTPReceiver> receiverList(getReceiverList());
248 SOCmnListPosition posRec;
249 SOSrvHTTPReceiver* pRec;
250 posRec = receiverList.getStartPosition();
251
252 while (posRec)
253 {
254 pRec = receiverList.getNext(posRec);
255 pRec->addClient(srvData);
256 }
257
258 receiverList.autoUnlock();
259 return ret;

263{
264 SOCmnList<SOSrvHTTPHandlerData> srvList(getServerList());
265 SOCmnPointer<SOSrvHTTPHandlerData> srvData = srvList.findKey(pSocket->getOpcHandle());
266 srvList.removeKey(pSocket->getOpcHandle());

268 m_serverHTTP->setClientHandle(pSocket, NULL);
269
270 if (srvData.isNotNull())
271 {
272 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SOCKET, (_T("Remove client %lu 0x%8.8lX (reason=%ld)"), srvData->m_clientId, (DWORD)(SOCmnObject*)srvData->m_socket, reason));
273 SOCmnList<SOSrvHTTPReceiver> receiverList(getReceiverList());
274 SOCmnListPosition posRec;
275 SOSrvHTTPReceiver* pRec;
276 posRec = receiverList.getStartPosition();
277
278 while (posRec)
279 {
280 pRec = receiverList.getNext(posRec);
281 pRec->removeClient(srvData);
282 }
283
284 srvData->m_clientId = 0;
285 srvData.release(); // should free srvData
286 return TRUE;
287 }
288
289 return FALSE;

292void SOSrvHTTPHandler::checkClients(void)
293{
294 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_SOCKET, (_T("Checking (each %lums) for clients inactive more than %lums."), m_checkClientInterval, m_clientInactiveTime));
295 SOCmnList<SOSrvHTTPHandlerData> srvList(getServerList());
296 SOSrvHTTPHandlerData* srvData;
297 SOCmnListPosition pos;
298 SOCmnDateTime now;
299 ULONGLONG nowMS, rcvMS;
300 SOCmnList<SOSrvHTTPHandlerData> removeList;
301 now.now();
302 nowMS = now.getMicroseconds();
303 pos = srvList.getStartPosition();
304
305 while (pos)
306 {
307 srvData = srvList.getNext(pos);
308 rcvMS = srvData->m_rcvTime.getMicroseconds();
309
310 if (nowMS >= rcvMS)
311 {
312 DWORD inactiveTime = (DWORD)((nowMS - rcvMS) / 1000);
313
314 if ((inactiveTime > m_clientInactiveTime) &&
315 (srvData->m_resPending == FALSE)) // no response pending
316 {
317 // add entry to remove list
318 if (!removeList.isInit())
319 {
320 removeList.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_OBJECT);
321 }
322
323 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_SOCKET, (_T(" Marking socket %lu (inactive for %lums) for disconnection."), srvData->m_socket->m_socket, inactiveTime));
324 removeList.add(srvData);
325 }
326 }
327 else if (rcvMS - nowMS > 1000000)
328 {
329 // last rcv time more than a second in the future
330 srvData->m_rcvTime.now();
331 }
332 }
333
334 srvList.autoUnlock();
335
336 if (removeList.isInit())
337 {
338 // close the sockets
339 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_SOCKET, (_T(" Now closing %lu sockets."), removeList.GetCount()));
340 pos = removeList.getStartPosition();
341
342 while (pos)
343 {
344 srvData = removeList.getNext(pos);
345 m_serverHTTP->removeClient(srvData->m_socket);
346 }
347
348 removeList.destroy();
349 }
350}

353void SOSrvHTTPHandler::rcvData(IN SOSrvHTTPHandlerData* srvData, IN LONG size, IN BYTE* pData)
354{
355 SOCmnSingleLock<SOCmnSync> lock(&srvData->m_lock);
356 srvData->m_rcvTime.now();
357
358 // check buffer size
359 if (srvData->m_rcvDataSize + size + 1 > srvData->m_rcvBufferSize)
360 {
361 srvData->m_rcvBufferSize = __max(srvData->m_rcvBufferSize + 4096, srvData->m_rcvDataSize + size + 1);
362 srvData->m_rcvBuffer = (BYTE*)realloc(srvData->m_rcvBuffer, srvData->m_rcvBufferSize);
363
364 if (!srvData->m_rcvBuffer)
365 {
366 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_SOCKET, (_T("Out of memory - alloc:%lu"), srvData->m_rcvBufferSize));
367 return;
368 }
369 }
370
371 // copy new data
372 if (srvData->m_rcvDataSize == 0)
373 {
374 memcpy(srvData->m_rcvBuffer, pData, size);
375 srvData->m_rcvDataSize = size + 1;
376 }
377 else
378 {
379 memcpy(srvData->m_rcvBuffer - 1 + srvData->m_rcvDataSize, pData, size);
380 srvData->m_rcvDataSize += size;
381 }
382
383 srvData->m_rcvBuffer[srvData->m_rcvDataSize - 1] = '\0';
384 lock.unlock();
385 // signal decode thread
386 SOCmnList<SOSrvHTTPHandlerData> decodeList(getDecodeList());
387 decodeList.add(srvData);
388 decodeList.autoUnlock();
389 m_workThread.getEvents()->signal(m_decodeEventId);
390}

512DWORD SOSrvHTTPHandler::workThread(void)
513{
514 BYTE* pBodyS;
515 DWORD bodyLen;
516 DWORD discardTo;
517 WORD resCode = SOCMNHTTP_CODEI_NOT_HANDLED;
518 DWORD waitRes;
519 DWORD nowTC;
520 DWORD checkClientsTC = GetTickCount();
521 DWORD waitTime = m_checkClientInterval;
522
523 while (TRUE)
524 {
525 waitRes = m_workThread.waitForEvents(waitTime);
526
527 if (waitRes == m_decodeEventId)
528 {
529 // decode
530 SOCmnList<SOSrvHTTPHandlerData> decodeList(getDecodeList());
531 BOOL end = FALSE;
532 SOCmnList<SOSrvHTTPReceiver> receiverList(getReceiverList());
533 SOCmnListPosition posRec;
534 SOSrvHTTPReceiver* pRec;
535 SOCmnString recURL;
536 SOCmnString expect;
537 SOCmnString connection;
538 SOCmnString reqURL;
539
540 for (;;)
541 {
542 SOCmnPointer<SOSrvHTTPHandlerData> srvData;
543 srvData = decodeList.pop();
544 decodeList.autoUnlock();
545
546 if (!srvData)
547 {
548 break;
549 }
550
551 {
552 SOCmnSingleLock<SOCmnSync> lock(&srvData->m_lock);
553
554 if (srvData->m_rcvDataSize <= 0)
555 {
556 continue;
557 }
558
559 end = FALSE;
560
561 do
562 {
563 // check if header is complete
564 // note: if a NUL char is in the data stream the strstr
565 // functions cannot find the end of header mark
566 // so stop waiting for more header if a NUL byte was received.
567 // NUL bytes in the content section are okay.
568 if ((strstr((CHAR*)srvData->m_rcvBuffer, "\n\n") == NULL) &&
569 (strstr((CHAR*)srvData->m_rcvBuffer, "\n\r\n") == NULL) &&
570 (strlen((CHAR*)srvData->m_rcvBuffer) == (DWORD)srvData->m_rcvDataSize - 1))
571 {
572 break; // do while(!end)
573 }
574
575 // decode message
576 pBodyS = NULL;
577 discardTo = decodeHTTPRequestHeader(srvData->m_rcvDataSize, srvData->m_rcvBuffer, &pBodyS);
578
579 // the pBodyS is set only if the HTTP-header contents has been completely parsed
580 // if the header is still incomplete and the size exceeds 16k than close connection and reject the request
581 if ( (!pBodyS) && (srvData->m_rcvDataSize >= MAX_ACCEPTED_HTTP_HEADER_SIZE_BYTES) )
582 {
583 srvData->m_flags |= SOSRVHTTP_FLAG_CONNECTION_CLOSE;
584 sndHTTPErrorResponse(srvData, srvData->m_flags, SOCMNHTTP_CODE_BAD_REQUEST);
585 break; // do while(!end)
586 }
587
588 // check for Expect: 100-continue
589 if ((m_attributeList.findKey(_T("expect"), &expect)) && (!m_sendContinue))
590 {
591 expect.makeLower();
592
593 if (expect == _T("100-continue"))
594 {
595 sndHTTPErrorResponse(srvData, srvData->m_flags, SOCMNHTTP_CODE_CONTINUE);
596 m_sendContinue = TRUE;
597 }
598 }
599
600 // check for Connection: close
601 if (m_attributeList.findKey(_T("connection"), &connection))
602 {
603 connection.makeLower();
604
605 if (connection == _T("close"))
606 {
607 srvData->m_flags |= SOSRVHTTP_FLAG_CONNECTION_CLOSE;
608 }
609 }
610
611 // check for HTTP 1.0 -> close connection
612 if (m_httpVersion == 0x0100)
613 {
614 srvData->m_flags |= SOSRVHTTP_FLAG_CONNECTION_CLOSE;
615 }
616
617 if ((bodyLen = getContentsLength()) != 0)
618 {
619 // check if complete contents arrived
620 if ((DWORD)srvData->m_rcvDataSize < discardTo + bodyLen + 1) // m_rcvDataSize is always terminated with '\0' -> + 1!
621 {
622 break;
623 }
624
625 discardTo += bodyLen + 1;
626 }
627
628 // start handle message
629 srvData->m_resPending = TRUE;
630 resCode = SOCMNHTTP_CODEI_NOT_HANDLED;
631 // remove the HTTP query string ? from the URL for handler check
632 reqURL = m_httpURL;
633 int queryStartIdx;
634
635 if ((queryStartIdx = reqURL.find(_T('?'))) != -1)
636 {
637 reqURL = reqURL.left(queryStartIdx);
638 }
639
640 posRec = receiverList.getStartPosition();
641 SOSRV_HTTP_CHECK_DEMO(resCode);
642
643 while ((posRec) && (resCode == SOCMNHTTP_CODEI_NOT_HANDLED))
644 {
645 pRec = receiverList.getNext(posRec);
646 recURL = pRec->getURL();
647
648 if (((pRec->getHTTPMethod() & m_httpMethod) == m_httpMethod) && // receiver for this method
649 ((recURL.isEmpty()) || (_tcsicmp(recURL, reqURL) == 0))) // URL case insensitive
650 {
651 // handleHTTPRequest works asynchronous!
652 // it only start the request handling if this is more than building the result string
653 if ((resCode = pRec->handleHTTPRequest(this, srvData, bodyLen, pBodyS)) != SOCMNHTTP_CODEI_NOT_HANDLED)
654 {
655 break;
656 }
657 }
658 }
659
660 receiverList.autoUnlock();
661
662 if (resCode == SOCMNHTTP_CODEI_NOT_HANDLED)
663 {
664 // request not handled
665 if ((m_httpMethod == 0) || (!((m_httpVersion == MAKEWORD(1, 1)) || (m_httpVersion == MAKEWORD(0, 1)))))
666 {
667 resCode = SOCMNHTTP_CODE_BAD_REQUEST; // no HTTP request
668 }
669 else
670 {
671 resCode = SOCMNHTTP_CODE_NOT_FOUND; // in most reasons URL not found
672 }
673
674 // send error response
675 sndHTTPErrorResponse(srvData, srvData->m_flags, resCode);
676 }
677
678 // discard message
679 m_sendContinue = FALSE;
680 srvData->m_flags = 0;
681
682 if (discardTo >= (DWORD)srvData->m_rcvDataSize - 1)
683 {
684 srvData->m_rcvDataSize = 0;
685 end = TRUE;
686 }
687 else
688 {
689 if (discardTo > 0)
690 {
691 memmove(srvData->m_rcvBuffer, srvData->m_rcvBuffer + discardTo, srvData->m_rcvDataSize - discardTo);
692 srvData->m_rcvDataSize -= discardTo;
693 }
694 }
695 }
696 while (!end);
697 } // srvData lock
698 } // for(;;)
699 }
700 else if (waitRes == m_sndEventId)
701 {
702 // send
703 SOCmnList<SOSrvHTTPHandlerData> sndList(getSndList());
704
705 for (;;)
706 {
707 SOCmnPointer<SOSrvHTTPHandlerData> srvData;
708 srvData = sndList.pop();
709 sndList.autoUnlock();
710
711 if (!srvData)
712 {
713 break;
714 }
715
716 {
717 SOCmnSingleLock<SOCmnSync> lock(&srvData->m_lock);
718 sndData(srvData->m_socket, srvData->m_sndDataSize, srvData->m_sndBuffer);
719 srvData->m_sndDataSize = 0;
720 srvData->m_resPending = FALSE;
721
722 if (srvData->m_sndFlags & SOSRVHTTP_FLAG_CONNECTION_CLOSE)
723 {
724 m_serverHTTP->shutdownClient(srvData->m_socket);
725 }
726 }
727 }
728 }
729 else if (waitRes == m_workThread.getStopEventId())
730 {
731 return 0; // received end event
732 }
733
734 nowTC = GetTickCount();
735 waitTime = getTimeSpan(checkClientsTC, nowTC);
736
737 if (waitTime >= m_checkClientInterval)
738 {
739 // every 10 seconds
740 checkClientsTC = GetTickCount();
741 waitTime = m_checkClientInterval;
742 // check the socket connections from the clients and close the unused if neccessary
743 checkClients();
744 }
745 else
746 {
747 waitTime = m_checkClientInterval - waitTime;
748 }
749 }
750
751 return 0;
752}

852 *pSize = (int)strlen((CHAR*)*ppData);

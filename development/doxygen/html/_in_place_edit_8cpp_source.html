61CInPlaceEdit::CInPlaceEdit(CWnd* pParent, CRect& rect, DWORD dwStyle, UINT nID,
62 int nRow, int nColumn, CString sInitText,
63 UINT nFirstChar)
64{
65 m_sInitText = sInitText;
66 m_nRow = nRow;
67 m_nColumn = nColumn;
68 m_nLastChar = 0;
69 m_bExitOnArrows = (nFirstChar != VK_LBUTTON); // If mouse click brought us here,
70 // then no exit on arrows
71 m_Rect = rect; // For bizarre CE bug.
72 DWORD dwEditStyle = WS_BORDER | WS_CHILD | WS_VISIBLE | ES_AUTOHSCROLL //|ES_MULTILINE
73 | dwStyle;
74
75 if (!Create(dwEditStyle, rect, pParent, nID))
76 {
77 return;
78 }
79
80 SetFont(pParent->GetFont());
81 SetWindowText(sInitText);
82 SetFocus();
83
84 switch (nFirstChar)
85 {
86 case VK_LBUTTON:
87 case VK_RETURN:
88 SetSel((int)_tcslen(m_sInitText), -1);
89 return;
90
91 case VK_BACK:
92 SetSel((int)_tcslen(m_sInitText), -1);
93 break;
94
95 case VK_TAB:
96 case VK_DOWN:
97 case VK_UP:
98 case VK_RIGHT:
99 case VK_LEFT:
100 case VK_NEXT:
101 case VK_PRIOR:
102 case VK_HOME:
103 case VK_SPACE:
104 case VK_END:
105 SetSel(0, -1);
106 return;
107
108 default:
109 SetSel(0, -1);
110 }
111
112 // Added by KiteFly. When entering DBCS chars into cells the first char was being lost
113 // SenMessage changed to PostMessage (John Lagerquist)
114 if (nFirstChar < 0x80)
115 {
116 PostMessage(WM_CHAR, nFirstChar);
117 }
118 else
119 {
120 PostMessage(WM_IME_CHAR, nFirstChar);
121 }
122}

141// If an arrow key (or associated) is pressed, then exit if
142// a) The Ctrl key was down, or
143// b) m_bExitOnArrows == TRUE
144void CInPlaceEdit::OnKeyDown(UINT nChar, UINT nRepCnt, UINT nFlags)
145{
146 if ((nChar == VK_PRIOR || nChar == VK_NEXT ||
147 nChar == VK_DOWN || nChar == VK_UP ||
148 nChar == VK_RIGHT || nChar == VK_LEFT) &&
149 (m_bExitOnArrows || GetKeyState(VK_CONTROL) < 0))
150 {
151 m_nLastChar = nChar;
152 GetParent()->SetFocus();
153 return;
154 }
155
156 CEdit::OnKeyDown(nChar, nRepCnt, nFlags);
157}

166void CInPlaceEdit::OnChar(UINT nChar, UINT nRepCnt, UINT nFlags)
167{
168 if (nChar == VK_TAB || nChar == VK_RETURN)
169 {
170 m_nLastChar = nChar;
171 GetParent()->SetFocus(); // This will destroy this window
172 return;
173 }
174
175 if (nChar == VK_ESCAPE)
176 {
177 SetWindowText(m_sInitText); // restore previous text
178 m_nLastChar = nChar;
179 GetParent()->SetFocus();
180 return;
181 }
182
183 CEdit::OnChar(nChar, nRepCnt, nFlags);
184 // Resize edit control if needed
185 // Get text extent
186 CString str;
187 GetWindowText(str);
188 // add some extra buffer
189 str += _T(" ");
190 CWindowDC dc(this);
191 CFont* pFontDC = dc.SelectObject(GetFont());
192 CSize size = dc.GetTextExtent(str);
193 dc.SelectObject(pFontDC);
194 // Get client rect
195 CRect ParentRect;
196 GetParent()->GetClientRect(&ParentRect);
197
198 // Check whether control needs to be resized
199 // and whether there is space to grow
200 if (size.cx > m_Rect.Width())
201 {
202 if (size.cx + m_Rect.left < ParentRect.right)
203 {
204 m_Rect.right = m_Rect.left + size.cx;
205 }
206 else
207 {
208 m_Rect.right = ParentRect.right;
209 }
210
211 MoveWindow(&m_Rect);
212 }
213}

245void CInPlaceEdit::EndEdit()
246{
247 CString str;
248 // EFW - BUG FIX - Clicking on a grid scroll bar in a derived class
249 // that validates input can cause this to get called multiple times
250 // causing assertions because the edit control goes away the first time.
251 static BOOL bAlreadyEnding = FALSE;
252
253 if (bAlreadyEnding)
254 {
255 return;
256 }
257
258 bAlreadyEnding = TRUE;
259 GetWindowText(str);
260 // Send Notification to parent
261 GV_DISPINFO dispinfo;
262 dispinfo.hdr.hwndFrom = GetSafeHwnd();
263 dispinfo.hdr.idFrom = GetDlgCtrlID();
264 dispinfo.hdr.code = GVN_ENDLABELEDIT;
265 dispinfo.item.mask = LVIF_TEXT | LVIF_PARAM;
266 dispinfo.item.row = m_nRow;
267 dispinfo.item.col = m_nColumn;
268 dispinfo.item.strText = str;
269 dispinfo.item.lParam = (LPARAM) m_nLastChar;
270 CWnd* pOwner = GetOwner();
271
272 if (pOwner)
273 {
274 pOwner->SendMessage(WM_NOTIFY, GetDlgCtrlID(), (LPARAM)&dispinfo);
275 }
276
277 // Close this window (PostNcDestroy will delete this)
278 if (IsWindow(GetSafeHwnd()))
279 {
280 SendMessage(WM_CLOSE, 0, 0);
281 }
282
283 bAlreadyEnding = FALSE;
284}

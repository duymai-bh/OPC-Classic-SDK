114// EFW - Various changes to make it draw cells better when using alternate
115// color schemes. Also removed printing references as that's now done
116// by PrintCell() and fixed the sort marker so that it doesn't draw out
117// of bounds.
118BOOL CGridCellBase::Draw(CDC* pDC, int nRow, int nCol, CRect rect, BOOL bEraseBkgnd /*=TRUE*/)
119{
120 // Note - all through this function we totally brutalise 'rect'. Do not
121 // depend on it's value being that which was passed in.
122 CGridCtrl* pGrid = GetGrid();
123 ASSERT(pGrid);
124
125 if (!pGrid || !pDC)
126 {
127 return FALSE;
128 }
129
130 if (rect.Width() <= 0 || rect.Height() <= 0) // prevents imagelist item from drawing even
131 {
132 return FALSE; // though cell is hidden
133 }
134
135 //TRACE3("Drawing %scell %d, %d\n", IsFixed()? _T("Fixed ") : _T(""), nRow, nCol);
136 int nSavedDC = pDC->SaveDC();
137 pDC->SetBkMode(TRANSPARENT);
138 // Get the default cell implementation for this kind of cell. We use it if this cell
139 // has anything marked as "default"
140 CGridDefaultCell* pDefaultCell = (CGridDefaultCell*) GetDefaultCell();
141
142 if (!pDefaultCell)
143 {
144 return FALSE;
145 }
146
147 // Set up text and background colours
148 COLORREF TextClr, TextBkClr;
149 TextClr = (GetTextClr() == CLR_DEFAULT) ? pDefaultCell->GetTextClr() : GetTextClr();
150
151 if (GetBackClr() == CLR_DEFAULT)
152 {
153 TextBkClr = pDefaultCell->GetBackClr();
154 }
155 else
156 {
157 bEraseBkgnd = TRUE;
158 TextBkClr = GetBackClr();
159 }
160
161 // Draw cell background and highlighting (if necessary)
162 if (IsFocused() || IsDropHighlighted())
163 {
164 // Always draw even in list mode so that we can tell where the
165 // cursor is at. Use the highlight colors though.
166 if (GetState() & GVIS_SELECTED)
167 {
168 TextBkClr = ::GetSysColor(COLOR_HIGHLIGHT);
169 TextClr = ::GetSysColor(COLOR_HIGHLIGHTTEXT);
170 bEraseBkgnd = TRUE;
171 }
172
173 rect.right++;
174 rect.bottom++; // FillRect doesn't draw RHS or bottom
175
176 if (bEraseBkgnd)
177 {
178 TRY
179 {
180 CBrush brush(TextBkClr);
181 pDC->FillRect(rect, &brush);
182 }
183 CATCH(CResourceException, e)
184 {
185 //e->ReportError();
186 }
187 END_CATCH
188 }
189
190 // Don't adjust frame rect if no grid lines so that the
191 // whole cell is enclosed.
192 if (pGrid->GetGridLines() != GVL_NONE)
193 {
194 rect.right--;
195 rect.bottom--;
196 }
197
198 if (pGrid->GetFrameFocusCell())
199 {
200 // Use same color as text to outline the cell so that it shows
201 // up if the background is black.
202 TRY
203 {
204 CBrush brush(TextClr);
205 pDC->FrameRect(rect, &brush);
206 }
207 CATCH(CResourceException, e)
208 {
209 //e->ReportError();
210 }
211 END_CATCH
212 }
213
214 pDC->SetTextColor(TextClr);
215
216 // Adjust rect after frame draw if no grid lines
217 if (pGrid->GetGridLines() == GVL_NONE)
218 {
219 rect.right--;
220 rect.bottom--;
221 }
222
223 //rect.DeflateRect(0,1,1,1); - Removed by Yogurt
224 }
225 else if ((GetState() & GVIS_SELECTED))
226 {
227 rect.right++;
228 rect.bottom++; // FillRect doesn't draw RHS or bottom
229 pDC->FillSolidRect(rect, ::GetSysColor(COLOR_HIGHLIGHT));
230 rect.right--;
231 rect.bottom--;
232 pDC->SetTextColor(::GetSysColor(COLOR_HIGHLIGHTTEXT));
233 }
234 else
235 {
236 if (bEraseBkgnd)
237 {
238 rect.right++;
239 rect.bottom++; // FillRect doesn't draw RHS or bottom
240 CBrush brush(TextBkClr);
241 pDC->FillRect(rect, &brush);
242 rect.right--;
243 rect.bottom--;
244 }
245
246 pDC->SetTextColor(TextClr);
247 }
248
249 // Draw lines only when wanted
250 if (IsFixed() && pGrid->GetGridLines() != GVL_NONE)
251 {
252 CCellID FocusCell = pGrid->GetFocusCell();
253 // As above, always show current location even in list mode so
254 // that we know where the cursor is at.
255 BOOL bHiliteFixed = pGrid->GetTrackFocusCell() && pGrid->IsValid(FocusCell) &&
256 (FocusCell.row == nRow || FocusCell.col == nCol);
257
258 // If this fixed cell is on the same row/col as the focus cell,
259 // highlight it.
260 if (bHiliteFixed)
261 {
262 rect.right++;
263 rect.bottom++;
264 pDC->DrawEdge(rect, BDR_SUNKENINNER /*EDGE_RAISED*/, BF_RECT);
265 rect.DeflateRect(1, 1);
266 }
267 else
268 {
269 CPen lightpen(PS_SOLID, 1, ::GetSysColor(COLOR_3DHIGHLIGHT)),
270 darkpen(PS_SOLID, 1, ::GetSysColor(COLOR_3DDKSHADOW)),
271 *pOldPen = pDC->GetCurrentPen();
272 pDC->SelectObject(&lightpen);
273 pDC->MoveTo(rect.right, rect.top);
274 pDC->LineTo(rect.left, rect.top);
275 pDC->LineTo(rect.left, rect.bottom);
276 pDC->SelectObject(&darkpen);
277 pDC->MoveTo(rect.right, rect.top);
278 pDC->LineTo(rect.right, rect.bottom);
279 pDC->LineTo(rect.left, rect.bottom);
280 pDC->SelectObject(pOldPen);
281 rect.DeflateRect(1, 1);
282 }
283 }
284
285 // Draw Text and image
286#if !defined(_WIN32_WCE_NO_PRINTING) && !defined(GRIDCONTROL_NO_PRINTING)
287
288 if (!pDC->m_bPrinting)
289#endif
290 {
291 CFont* pFont = GetFontObject();
292 ASSERT(pFont);
293
294 if (pFont)
295 {
296 pDC->SelectObject(pFont);
297 }
298 }
299
300 //rect.DeflateRect(GetMargin(), 0); - changed by Yogurt
301 rect.DeflateRect(GetMargin(), GetMargin());
302 rect.right++;
303 rect.bottom++;
304
305 if (pGrid->GetImageList() && GetImage() >= 0)
306 {
307 IMAGEINFO Info;
308
309 if (pGrid->GetImageList()->GetImageInfo(GetImage(), &Info))
310 {
311 // would like to use a clipping region but seems to have issue
312 // working with CMemDC directly. Instead, don't display image
313 // if any part of it cut-off
314 //
315 // CRgn rgn;
316 // rgn.CreateRectRgnIndirect(rect);
317 // pDC->SelectClipRgn(&rgn);
318 // rgn.DeleteObject();
319 /*
320 // removed by Yogurt
321 int nImageWidth = Info.rcImage.right-Info.rcImage.left+1;
322 int nImageHeight = Info.rcImage.bottom-Info.rcImage.top+1;
323 if( nImageWidth + rect.left <= rect.right + (int)(2*GetMargin())
324 && nImageHeight + rect.top <= rect.bottom + (int)(2*GetMargin()) )
325 {
326 pGrid->GetImageList()->Draw(pDC, GetImage(), rect.TopLeft(), ILD_NORMAL);
327 }
328 */
329 // Added by Yogurt
330 int nImageWidth = Info.rcImage.right - Info.rcImage.left;
331 int nImageHeight = Info.rcImage.bottom - Info.rcImage.top;
332
333 if ((nImageWidth + rect.left <= rect.right) && (nImageHeight + rect.top <= rect.bottom))
334 {
335 pGrid->GetImageList()->Draw(pDC, GetImage(), rect.TopLeft(), ILD_NORMAL);
336 }
337
338 //rect.left += nImageWidth+GetMargin();
339 }
340 }
341
342 // Draw sort arrow
343 if (pGrid->GetSortColumn() == nCol && nRow == 0)
344 {
345 CSize size = pDC->GetTextExtent(_T("M"));
346 int nOffset = 2;
347 // Base the size of the triangle on the smaller of the column
348 // height or text height with a slight offset top and bottom.
349 // Otherwise, it can get drawn outside the bounds of the cell.
350 size.cy -= (nOffset * 2);
351
352 if (size.cy >= rect.Height())
353 {
354 size.cy = rect.Height() - (nOffset * 2);
355 }
356
357 size.cx = size.cy; // Make the dimensions square
358 // Kludge for vertical text
359 BOOL bVertical = (GetFont()->lfEscapement == 900);
360
361 // Only draw if it'll fit!
362 //if (size.cx + rect.left < rect.right + (int)(2*GetMargin())) - changed / Yogurt
363 if (size.cx + rect.left < rect.right)
364 {
365 int nTriangleBase = rect.bottom - nOffset - size.cy; // Triangle bottom right
366 //int nTriangleBase = (rect.top + rect.bottom - size.cy)/2; // Triangle middle right
367 //int nTriangleBase = rect.top + nOffset; // Triangle top right
368 //int nTriangleLeft = rect.right - size.cx; // Triangle RHS
369 //int nTriangleLeft = (rect.right + rect.left - size.cx)/2; // Triangle middle
370 //int nTriangleLeft = rect.left; // Triangle LHS
371 int nTriangleLeft;
372
373 if (bVertical)
374 {
375 nTriangleLeft = (rect.right + rect.left - size.cx) / 2; // Triangle middle
376 }
377 else
378 {
379 nTriangleLeft = rect.right - size.cx; // Triangle RHS
380 }
381
382 CPen penShadow(PS_SOLID, 0, ::GetSysColor(COLOR_3DSHADOW));
383 CPen penLight(PS_SOLID, 0, ::GetSysColor(COLOR_3DHILIGHT));
384
385 if (pGrid->GetSortAscending())
386 {
387 // Draw triangle pointing upwards
388 CPen* pOldPen = (CPen*) pDC->SelectObject(&penLight);
389 pDC->MoveTo(nTriangleLeft + 1, nTriangleBase + size.cy + 1);
390 pDC->LineTo(nTriangleLeft + (size.cx / 2) + 1, nTriangleBase + 1);
391 pDC->LineTo(nTriangleLeft + size.cx + 1, nTriangleBase + size.cy + 1);
392 pDC->LineTo(nTriangleLeft + 1, nTriangleBase + size.cy + 1);
393 pDC->SelectObject(&penShadow);
394 pDC->MoveTo(nTriangleLeft, nTriangleBase + size.cy);
395 pDC->LineTo(nTriangleLeft + (size.cx / 2), nTriangleBase);
396 pDC->LineTo(nTriangleLeft + size.cx, nTriangleBase + size.cy);
397 pDC->LineTo(nTriangleLeft, nTriangleBase + size.cy);
398 pDC->SelectObject(pOldPen);
399 }
400 else
401 {
402 // Draw triangle pointing downwards
403 CPen* pOldPen = (CPen*) pDC->SelectObject(&penLight);
404 pDC->MoveTo(nTriangleLeft + 1, nTriangleBase + 1);
405 pDC->LineTo(nTriangleLeft + (size.cx / 2) + 1, nTriangleBase + size.cy + 1);
406 pDC->LineTo(nTriangleLeft + size.cx + 1, nTriangleBase + 1);
407 pDC->LineTo(nTriangleLeft + 1, nTriangleBase + 1);
408 pDC->SelectObject(&penShadow);
409 pDC->MoveTo(nTriangleLeft, nTriangleBase);
410 pDC->LineTo(nTriangleLeft + (size.cx / 2), nTriangleBase + size.cy);
411 pDC->LineTo(nTriangleLeft + size.cx, nTriangleBase);
412 pDC->LineTo(nTriangleLeft, nTriangleBase);
413 pDC->SelectObject(pOldPen);
414 }
415
416 if (!bVertical)
417 {
418 rect.right -= size.cy;
419 }
420 }
421 }
422
423 // We want to see '&' characters so use DT_NOPREFIX
424 GetTextRect(rect);
425 rect.right++;
426 rect.bottom++;
427 DrawText(pDC->m_hDC, GetText(), -1, rect, GetFormat() | DT_NOPREFIX);
428 pDC->RestoreDC(nSavedDC);
429 return TRUE;
430}

520// By default this uses the selected font (which is a bigger font)
521CSize CGridCellBase::GetTextExtent(LPCTSTR szText, CDC* pDC /*= NULL*/)
522{
523 CGridCtrl* pGrid = GetGrid();
524 ASSERT(pGrid);
525 BOOL bReleaseDC = FALSE;
526
527 if (pDC == NULL || szText == NULL)
528 {
529 if (szText)
530 {
531 pDC = pGrid->GetDC();
532 }
533
534 if (pDC == NULL || szText == NULL)
535 {
536 CGridDefaultCell* pDefCell = (CGridDefaultCell*) GetDefaultCell();
537 ASSERT(pDefCell);
538 return CSize(pDefCell->GetWidth(), pDefCell->GetHeight());
539 }
540
541 bReleaseDC = TRUE;
542 }
543
544 CFont* pOldFont = NULL,
545 *pFont = GetFontObject();
546
547 if (pFont)
548 {
549 pOldFont = pDC->SelectObject(pFont);
550 }
551
552 CSize size;
553 int nFormat = GetFormat();
554
555 // If the cell is a multiline cell, then use the width of the cell
556 // to get the height
557 if ((nFormat & DT_WORDBREAK) && !(nFormat & DT_SINGLELINE))
558 {
559 CString str = szText;
560 int nMaxWidth = 0;
561
562 while (TRUE)
563 {
564 int nPos = str.Find(_T('\n'));
565 CString TempStr = (nPos < 0) ? str : str.Left(nPos);
566 int nTempWidth = pDC->GetTextExtent(TempStr).cx;
567
568 if (nTempWidth > nMaxWidth)
569 {
570 nMaxWidth = nTempWidth;
571 }
572
573 if (nPos < 0)
574 {
575 break;
576 }
577
578 str = str.Mid(nPos + 1); // Bug fix by Thomas Steinborn
579 }
580
581 CRect rect;
582 rect.SetRect(0, 0, nMaxWidth + 1, 0);
583 pDC->DrawText(szText, -1, rect, nFormat | DT_CALCRECT);
584 size = rect.Size();
585 }
586 else
587 {
588 size = pDC->GetTextExtent(szText, _tcslen(szText));
589 }
590
591 // Removed by Yogurt
592 //TEXTMETRIC tm;
593 //pDC->GetTextMetrics(&tm);
594 //size.cx += (tm.tmOverhang);
595
596 if (pOldFont)
597 {
598 pDC->SelectObject(pOldFont);
599 }
600
601 size += CSize(2 * GetMargin(), 2 * GetMargin());
602 // Kludge for vertical text
603 LOGFONT* pLF = GetFont();
604
605 if (pLF->lfEscapement == 900 || pLF->lfEscapement == -900)
606 {
607 int nTemp = size.cx;
608 size.cx = size.cy;
609 size.cy = nTemp;
610 size += CSize(0, 4 * GetMargin());
611 }
612
613 if (bReleaseDC)
614 {
615 pGrid->ReleaseDC(pDC);
616 }
617
618 return size;
619}

622CSize CGridCellBase::GetCellExtent(CDC* pDC)
623{
624 CSize size = GetTextExtent(GetText(), pDC);
625 CSize ImageSize(0, 0);
626 int nImage = GetImage();
627
628 if (nImage >= 0)
629 {
630 CGridCtrl* pGrid = GetGrid();
631 ASSERT(pGrid);
632 IMAGEINFO Info;
633
634 if (pGrid->GetImageList() && pGrid->GetImageList()->GetImageInfo(nImage, &Info))
635 {
636 ImageSize = CSize(Info.rcImage.right - Info.rcImage.left,
637 Info.rcImage.bottom - Info.rcImage.top);
638
639 if (size.cx > 2 * (int)GetMargin())
640 {
641 ImageSize.cx += GetMargin();
642 }
643
644 ImageSize.cy += 2 * (int)GetMargin();
645 }
646 }
647
648 size.cx += ImageSize.cx + 1;
649 size.cy = max(size.cy, ImageSize.cy) + 1;
650
651 if (IsFixed())
652 {
653 size.cx++;
654 size.cy++;
655 }
656
657 return size;
658}

660// EFW - Added to print cells so that grids that use different colors are
661// printed correctly.
662BOOL CGridCellBase::PrintCell(CDC* pDC, int /*nRow*/, int /*nCol*/, CRect rect)
663{
664#if defined(_WIN32_WCE_NO_PRINTING) || defined(GRIDCONTROL_NO_PRINTING)
665 return FALSE;
666#else
667 COLORREF crFG, crBG;
668 GV_ITEM Item;
669 CGridCtrl* pGrid = GetGrid();
670
671 if (!pGrid || !pDC)
672 {
673 return FALSE;
674 }
675
676 if (rect.Width() <= 0
677 || rect.Height() <= 0) // prevents imagelist item from drawing even
678 {
679 return FALSE; // though cell is hidden
680 }
681
682 int nSavedDC = pDC->SaveDC();
683 pDC->SetBkMode(TRANSPARENT);
684
685 if (pGrid->GetShadedPrintOut())
686 {
687 // Get the default cell implementation for this kind of cell. We use it if this cell
688 // has anything marked as "default"
689 CGridDefaultCell* pDefaultCell = (CGridDefaultCell*) GetDefaultCell();
690
691 if (!pDefaultCell)
692 {
693 return FALSE;
694 }
695
696 // Use custom color if it doesn't match the default color and the
697 // default grid background color. If not, leave it alone.
698 if (IsFixed())
699 {
700 crBG = (GetBackClr() != CLR_DEFAULT) ? GetBackClr() : pDefaultCell->GetBackClr();
701 }
702 else
703 crBG = (GetBackClr() != CLR_DEFAULT && GetBackClr() != pDefaultCell->GetBackClr()) ?
704 GetBackClr() : CLR_DEFAULT;
705
706 // Use custom color if the background is different or if it doesn't
707 // match the default color and the default grid text color.
708 if (IsFixed())
709 {
710 crFG = (GetBackClr() != CLR_DEFAULT) ? GetTextClr() : pDefaultCell->GetTextClr();
711 }
712 else
713 {
714 crFG = (GetBackClr() != CLR_DEFAULT) ? GetTextClr() : pDefaultCell->GetTextClr();
715 }
716
717 // If not printing on a color printer, adjust the foreground color
718 // to a gray scale if the background color isn't used so that all
719 // colors will be visible. If not, some colors turn to solid black
720 // or white when printed and may not show up. This may be caused by
721 // coarse dithering by the printer driver too (see image note below).
722 if (pDC->GetDeviceCaps(NUMCOLORS) == 2 && crBG == CLR_DEFAULT)
723 crFG = RGB(GetRValue(crFG) * 0.30, GetGValue(crFG) * 0.59,
724 GetBValue(crFG) * 0.11);
725
726 // Only erase the background if the color is not the default
727 // grid background color.
728 if (crBG != CLR_DEFAULT)
729 {
730 CBrush brush(crBG);
731 rect.right++;
732 rect.bottom++;
733 pDC->FillRect(rect, &brush);
734 rect.right--;
735 rect.bottom--;
736 }
737 }
738 else
739 {
740 crBG = CLR_DEFAULT;
741 crFG = RGB(0, 0, 0);
742 }
743
744 pDC->SetTextColor(crFG);
745 CFont* pFont = GetFontObject();
746
747 if (pFont)
748 {
749 pDC->SelectObject(pFont);
750 }
751
752 /*
753 // ***************************************************
754 // Disabled - if you need this functionality then you'll need to rewrite.
755 // Create the appropriate font and select into DC.
756 CFont Font;
757 // Bold the fixed cells if not shading the print out. Use italic
758 // font it it is enabled.
759 const LOGFONT* plfFont = GetFont();
760 if(IsFixed() && !pGrid->GetShadedPrintOut())
761 {
762 Font.CreateFont(plfFont->lfHeight, 0, 0, 0, FW_BOLD, plfFont->lfItalic, 0, 0,

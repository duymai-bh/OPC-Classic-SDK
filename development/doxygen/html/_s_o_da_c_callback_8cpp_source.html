163//-----------------------------------------------------------------------------
164// OnDataChange
165//
166// - OPC advise callback
167//
168STDMETHODIMP_(void) SODaCAdviseSink::OnDataChange(
169 IN LPFORMATETC format, // data format
170 IN LPSTGMEDIUM medium) // transport medium
171{
172 SOCltComGuard gurad(this);
173
174#ifdef _WIN64
175 // NOTE: OPCDA1.0 uses a memory block for transferring data between client and server.
176 // The memory contains structures with platform-dependent members.
177 // Sizeof(VARIANT) is 16 for 32-bit systems and 24 for 64-bit systems.
178 // Since OPC DA v1.0a dates back in 1997, it is a common agreement that OPCDA1.0 servers are 32-bit only.
179 // Remove this at your own risk!
180 if (TRUE)
181 {
182 // avoid warnings
183 return;
184 }
185#endif
186
187 if ((!m_pComGroupImpl) || (!m_group))
188 {
189 return;
190 }
191
192 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_ADVISE_C, (_T("IDataObject::OnDataChange")));
193 SOCMN_TRY
194 {
195 if (SOCLT_COM_CHECK_DEMO)
196 {
197 return;
198 }
199
200 USES_CONVERSION;
201
202 // check format
203 if ((format->cfFormat != OPCSTMFORMATDATA_C) &&
204 (format->cfFormat != OPCSTMFORMATDATATIME_C) &&
205 (format->cfFormat != OPCSTMFORMATWRITECOMPLETE_C))
206 {
207 ReleaseStgMedium(medium);
208 return;
209 }
210
211 // check transport medium type
212 if (format->tymed != TYMED_HGLOBAL)
213 {
214 ReleaseStgMedium(medium);
215 return;
216 }
217
218 // check transport medium value
219 if (medium->hGlobal == NULL)
220 {
221 return;
222 }
223
224
225 // handle advise
226 handleAdvise(format->cfFormat, medium->hGlobal);
227
228 ReleaseStgMedium(medium);
229
230 }
231 SOCMN_CATCH(...)
232 {
233 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_ADVISE_C, (_T("EXCEPTION in IDataObject::OnDataChange")));
234 return;
235 }
236} // OnDataChange

239//-----------------------------------------------------------------------------
240// handleAdvise
241//
242// - handle OPC advise
243//
244void SODaCAdviseSink::handleAdvise(
245 IN CLIPFORMAT format, // format id
246 IN HGLOBAL memory) // memory
247{
248 SOCmnPointer<SODaCItem> item;
249 BYTE* mem = (BYTE*)GlobalLock(memory);
250 BYTE* actPtr = mem;
251 DWORD i;
252
253 if (mem == NULL)
254 {
255 return;
256 }
257
258 if (format != OPCSTMFORMATWRITECOMPLETE_C)
259 {
260 // read
261 m_pComGroupImpl->onDataChangeBegin();
262 OPCGROUPHEADER* groupHeader;
263 OPCITEMHEADER1* itemHeader;
264 DWORD valueOffset;
265 DWORD groupHeaderSize;
266 VARIANT* value;
267 SOCmnVariant val;
268 // for advise call
269 HRESULT masterQuality = S_OK;
270 HRESULT masterError = S_OK;
271 DWORD itemCnt = 0;
272 SODaCItem** ppItems = NULL;
273 SOCmnVariant* pValues = NULL;
274 WORD* pQualities = NULL;
275 SOCmnDateTime* pTimestamps = NULL;
276 HRESULT* pErrors = NULL;
277 // group header
278 groupHeader = (OPCGROUPHEADER*)actPtr;
279 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE_C, (_T(" IN : dwItemCount: %lu ; hClientGroup: 0x%8.8X ; hrStatus: 0x%X"), groupHeader->dwItemCount, groupHeader->hClientGroup, groupHeader->hrStatus));
280 actPtr += sizeof(OPCGROUPHEADER);
281 itemCnt = groupHeader->dwItemCount;
282 ppItems = (SODaCItem**) malloc(itemCnt * sizeof(SODaCItem*));
283 pValues = new SOCmnVariant[itemCnt];
284 pQualities = new WORD[itemCnt];
285 pTimestamps = new SOCmnDateTime[itemCnt];
286 pErrors = new HRESULT[itemCnt];
287 masterError = groupHeader->hrStatus;
288 itemCnt = 0;
289
290 if (format == OPCSTMFORMATDATA_C)
291 {
292 valueOffset = sizeof(OPCGROUPHEADER) + (sizeof(OPCITEMHEADER2) * groupHeader->dwItemCount);
293 groupHeaderSize = sizeof(OPCITEMHEADER2);
294 }
295 else
296 {
297 valueOffset = sizeof(OPCGROUPHEADER) + (sizeof(OPCITEMHEADER1) * groupHeader->dwItemCount);
298 groupHeaderSize = sizeof(OPCITEMHEADER1);
299 }
300
301 // item header and data
302 for (i = 0; i < groupHeader->dwItemCount; i++)
303 {
304 itemHeader = (OPCITEMHEADER1*)actPtr;
305 item = m_group->getItemByClientHandle(itemHeader->hClient);
306
307 if (item.isNotNull())
308 {
309 item->addRef();
310 ppItems[itemCnt] = (SODaCItem*)item;
311 value = (VARIANT*)(mem + itemHeader->dwValueOffset);
312
313 // correct BSTR
314 if (value->vt == VT_BSTR)
315 {
316 value->bstrVal = (BSTR)(((DWORD*)(value + 1)) + 1);
317 }
318
319 // correct ARRAY
320 if ((value->vt & VT_ARRAY) == VT_ARRAY)
321 {
322 value->parray = (SAFEARRAY*)(value + 1);
323 value->parray->pvData = (void*)(value->parray + 1);
324
325 if (value->vt == (VT_BSTR | VT_ARRAY))
326 {
327 BSTR* bstr = (BSTR*)value->parray->pvData;
328 DWORD* pBstrCount;
329 BYTE* pActBArray = ((BYTE*)value->parray->pvData) + (value->parray->cbElements * value->parray->rgsabound[0].cElements);
330 ULONG j;
331 BSTR pBstr;
332
333 for (j = 0; j < value->parray->rgsabound[0].cElements; ++j)
334 {
335 pBstrCount = (DWORD*)(pActBArray);
336 pBstr = (BSTR)(pBstrCount + 1);
337 bstr[j] = pBstr;
338 pActBArray += *pBstrCount + 6;
339 }
340 }
341 }
342
343 val = value;
344
345 if (format == OPCSTMFORMATDATATIME_C)
346 {
347 _TRACE_IF(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE_C)
348 {
349 SOCmnString v, q, t;
350 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE_C, (_T(" [%lu] %s ; %s %s %s"), i, (LPCTSTR)item->getItemID(), variantWithType2string(&v, &val), quality2string(&q, itemHeader->wQuality), filetime2string(&t, &itemHeader->ftTimeStampItem)));
351 }
352 SOCmnDateTime timestamp(&itemHeader->ftTimeStampItem);
353 item->setRead(S_OK, &val, itemHeader->wQuality, &timestamp, groupHeader->dwTransactionID);
354 pTimestamps[itemCnt] = timestamp;
355 }
356 else
357 {
358 _TRACE_IF(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE_C)
359 {
360 SOCmnString v, q, t;
361 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE_C, (_T(" [%lu] %s ; %s %s"), i, (LPCTSTR)item->getItemID(), variantWithType2string(&v, &val), quality2string(&q, itemHeader->wQuality)));
362 }
363 item->setRead(S_OK, &val, itemHeader->wQuality, NULL, groupHeader->dwTransactionID);
364 pTimestamps[itemCnt].clear();
365 }
366
367 pValues[itemCnt].initFrom(val);
368 pQualities[itemCnt] = itemHeader->wQuality;
369 pErrors[itemCnt] = S_OK;
370 itemCnt++;
371 val.clear();
372 }
373 else
374 {
375 _TRACE(SOCMNTRACE_L_WRN, SOCMNTRACE_G_OPC_ADVISE_C, (_T(" [%lu] Client handle 0x%8.8X: no item found"), i, (DWORD)itemHeader->hClient));
376 m_group->setLastError(OPC_E_INVALIDHANDLE, SODAC_ERR_ADVISE_DATACALLBACK);
377 ppItems[i] = NULL;
378 }
379
380 actPtr += groupHeaderSize;
381 }
382
383 if (m_group)
384 {
385 m_group->onReadAdvise((groupHeader->dwTransactionID == 0), groupHeader->dwTransactionID, masterQuality, masterError, itemCnt, ppItems, pValues, pQualities, pTimestamps, pErrors);
386 }
387
388 if (ppItems)
389 {
390 for (i = 0; i < itemCnt; i++)
391 {
392 ppItems[i]->release();
393 }
394
395 free(ppItems);
396 }
397
398 if (pValues)
399 {
400 delete[] pValues;
401 }
402
403 if (pQualities)
404 {
405 delete[] pQualities;
406 }
407
408 if (pTimestamps)
409 {
410 delete[] pTimestamps;
411 }
412
413 if (pErrors)
414 {
415 delete[] pErrors;
416 }
417
418 m_pComGroupImpl->onReadAdvise(groupHeader->dwTransactionID, mem);
419 m_pComGroupImpl->onDataChangeEnd();
420 }
421 else
422 {
423 // write
424 OPCGROUPHEADERWRITE* groupHeader;
425 OPCITEMHEADERWRITE* itemHeader;
426 // for advise call
427 DWORD itemCnt = 0;
428 SODaCItem** ppItems = NULL;
429 HRESULT* pErrors = NULL;
430 HRESULT masterError = S_OK;
431 // group header
432 groupHeader = (OPCGROUPHEADERWRITE*)actPtr;
433 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE_C, (_T(" IN : dwItemCount: %lu ; hClientGroup: 0x%8.8X ; hrStatus 0x%X"), groupHeader->dwItemCount, groupHeader->hClientGroup, groupHeader->hrStatus));
434 actPtr += sizeof(OPCGROUPHEADERWRITE);
435 itemCnt = groupHeader->dwItemCount;
436 ppItems = (SODaCItem**) malloc(itemCnt * sizeof(SODaCItem*));
437 pErrors = new HRESULT[itemCnt];
438 masterError = groupHeader->hrStatus;
439 itemCnt = 0;
440
441 // item header
442 for (i = 0; i < groupHeader->dwItemCount; i++)
443 {
444 itemHeader = (OPCITEMHEADERWRITE*)actPtr;
445 item = m_group->getItemByClientHandle(itemHeader->hClient);
446
447 if (item.isNotNull())
448 {
449 item->addRef();
450 ppItems[itemCnt] = (SODaCItem*)item;
451 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OPC_ADVISE_C, (_T(" [%lu] %s ; 0x%8.8X"), i, (LPCTSTR)item->getItemID(), itemHeader->dwError));
452 item->setWrite(itemHeader->dwError, groupHeader->dwTransactionID);
453 pErrors[itemCnt] = itemHeader->dwError;
454 itemCnt++;
455 }
456 else
457 {
458 _TRACE(SOCMNTRACE_L_WRN, SOCMNTRACE_G_OPC_ADVISE_C, (_T(" [%lu] Client handle 0x%8.8X: no item found"), i, (DWORD)itemHeader->hClient));
459 m_group->setLastError(OPC_E_INVALIDHANDLE, SODAC_ERR_ADVISE_DATACALLBACK);
460 ppItems[i] = NULL;
461 }
462
463 actPtr += sizeof(OPCITEMHEADERWRITE);
464 }
465
466 if (m_group)
467 {
468 m_group->onWriteAdvise(groupHeader->dwTransactionID, masterError, itemCnt, ppItems, pErrors);
469 }
470
471 if (ppItems)
472 {
473 for (i = 0; i < itemCnt; i++)
474 {
475 ppItems[i]->release();
476 }
477
478 free(ppItems);
479 }
480
481 if (pErrors)
482 {
483 delete[] pErrors;
484 }
485
486 m_pComGroupImpl->onWriteAdvise(groupHeader->dwTransactionID, mem);
487 }
488
489 GlobalUnlock(memory);
490} // handleAdvise

516//-----------------------------------------------------------------------------
517// advise
518//
519// - connect the advise sink of the client to the server
520//
521// returns:
522// S_OK - operation succeeded
523// E_FAIL - operation failed
524// E_INVALIDARG - invalid argument
525// E_OUTOFMEMORY - not enough memory
526// E_NOINTERFACE - interface not supported
527// CONNECT_E_ADVISELIMIT - reached maximal advise limit of server
528//
529HRESULT SODaCAdviseSink::advise(void)
530{
531 CComPtr<IDataObject> dataObject;
532 FORMATETC format;
533 HRESULT res = S_OK;
534 format.dwAspect = DVASPECT_CONTENT;
535 format.ptd = NULL;
536 format.tymed = TYMED_HGLOBAL;
537 format.lindex = -1;
538
539#ifdef _WIN64
540 // NOTE: OPCDA1.0 uses a memory block for transferring data between client and server.
541 // The memory contains structures with platform-dependent members.
542 // Sizeof(VARIANT) is 16 for 32-bit systems and 24 for 64-bit systems.
543 // Since OPC DA v1.0a dates back in 1997, it is a common agreement that OPCDA1.0 servers are 32-bit only.
544 // Remove this at your own risk!
545 if (TRUE)
546 {
547 // avoid warnings
548 return E_INVALIDARG;
549 }
550#endif
551
552 // already advised
553 if ((m_readConnection) && (m_writeConnection))
554 {
555 return S_OK;
556 }
557
558 if (!m_group)
559 {
560 return E_FAIL;
561 }
562
563 // get the COM interface
564 if (FAILED(res = m_group->queryInterface(IID_IDataObject, (void**)&dataObject)))
565 {
566 return res;
567 }
568
569 // call functions on OPC Server
570 if (!m_readConnection)
571 {
572 // advise read connection
573 format.cfFormat = (CLIPFORMAT)OPCSTMFORMATDATATIME_C;
574 m_group->beforeInterfaceCall();
575 res = dataObject->DAdvise(&format, ADVF_PRIMEFIRST, (IAdviseSink*)this, &m_readConnection);
576 m_group->afterInterfaceCall();
577 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, (_T("IDataObject::DAdvise (read connection) [0x%X] "), res));
578 m_group->setLastError(res, SODAC_ERR_ADVISE_DATAOBJECT_READ);
579
580 if (SUCCEEDED(res))
581 {
582 m_establishedRead = TRUE;
583 }
584 }
585
586 if (!m_writeConnection)
587 {
588 // advise write connection
589 format.cfFormat = (CLIPFORMAT)OPCSTMFORMATWRITECOMPLETE_C;
590 m_group->beforeInterfaceCall();
591 res = dataObject->DAdvise(&format, ADVF_PRIMEFIRST, (IAdviseSink*)this, &m_writeConnection);
592 m_group->afterInterfaceCall();
593 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, (_T("IDataObject::DAdvise (write connection) [0x%X] "), res));
594 m_group->setLastError(res, SODAC_ERR_ADVISE_DATAOBJECT_WRITE);
595
596 if (SUCCEEDED(res))
597 {
598 m_establishedWrite = TRUE;
599 }
600 }
601
602 return res;
603} // advise

606//-----------------------------------------------------------------------------
607// unadvise
608//
609// - disconnect the advise sink of the client to the server
610//
611// returns:
612// S_OK - operation succeeded
613// E_FAIL - operation failed
614// E_INVALIDARG - invalid argument
615// E_OUTOFMEMORY - not enough memory
616// E_NOINTERFACE - interface not supported
617//
618HRESULT SODaCAdviseSink::unadvise(void)
619{
620 CComPtr<IDataObject> dataObject;
621 HRESULT res = S_OK;
622
623 // already unadvised
624 if ((!m_establishedRead) && (!m_establishedWrite))
625 {
626 return S_OK;
627 }
628
629 if (!m_group)
630 {
631 return E_FAIL;
632 }
633
634 // get the COM interface
635 if (FAILED(res = m_group->queryInterface(IID_IDataObject, (void**)&dataObject)))
636 {
637 return res;
638 }
639
640 // call functions on OPC Server
641 if (m_establishedRead)
642 {
643 // unadvise read connection
644 m_group->beforeInterfaceCall();
645 res = dataObject->DUnadvise(m_readConnection);
646 m_group->afterInterfaceCall();
647 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, (_T("IDataObject::DUnadvise (read connection) [0x%X] "), res));
648 }
649
650 if (m_establishedWrite)
651 {
652 // unadvise write connection
653 m_group->beforeInterfaceCall();
654 res = dataObject->DUnadvise(m_writeConnection);
655 m_group->afterInterfaceCall();
656 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, (_T("IDataObject::DUnadvise (write connection) [0x%X] "), res));
657 }
658
659 if (FAILED(res))
660 {
661 res = ::CoDisconnectObject(static_cast<IUnknown*>(this), 0);
662 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, (_T("::CoDisconnectObject() [0x%X] "), res));
663 }
664
665 m_readConnection = 0;
666 m_establishedRead = FALSE;
667 m_writeConnection = 0;
668 m_establishedWrite = FALSE;
669 m_group->setLastError(res, SODAC_ERR_UNADVISE_DATAOBJECT);
670 return res;
671} // unadvise

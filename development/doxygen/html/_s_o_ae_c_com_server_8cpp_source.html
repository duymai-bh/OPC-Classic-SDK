52//-----------------------------------------------------------------------------
53// doStateTransition
54//
55// - do the state transition for the server if the target state differs from the
56// actual state of the server
57//
58// returns:
59// TRUE - transit to new state
60// FALSE - error during state change
61//
62HRESULT SOAeCComServerImpl::doStateTransitionImpl()
63{
64 SOCltComServerImpl* pCltComImpl = getObjectCltComServerImpl();
65 SOAeCServer* pAeCServer = getObjectAeCServer();
66 HRESULT res = S_OK;
67 BYTE targetState = pAeCServer->getTargetObjectState();
68
69 while ((SUCCEEDED(res)) && (pAeCServer->getObjectState() != targetState))
70 {
71 switch (pAeCServer->getObjectState())
72 {
73 case SOCltElement::disconnected:
74 {
75 // disconnected
76 //-- connect server --
77 MULTI_QI mqi[4];
78 COSERVERINFO sin, *sinptr;
79 SOCmnString errDescr;
80 USES_CONVERSION;
81 ::ZeroMemory(mqi, sizeof(mqi));
82
83 // create OPC server object via COM
84 if (!pAeCServer->m_nodeName.IsEmpty())
85 {
86 // remote server
87 sinptr = &sin;
88 sin.dwReserved1 = 0;
89 sin.dwReserved2 = 0;
90 sin.pwszName = T2W((LPTSTR)((LPCTSTR)pAeCServer->m_nodeName));
91 sin.pAuthInfo = 0;
92 }
93 else
94 {
95 // local server, inproc server or inproc handler
96 sinptr = 0;
97 }
98
99 mqi[0].pIID = &IID_IUnknown;
100 mqi[1].pIID = &IID_IOPCCommon;
101 mqi[2].pIID = &IID_IOPCEventServer;
102 mqi[3].pIID = &IID_IOPCSecurityPrivate;
103
104 if ((SUCCEEDED(res)) && (pAeCServer->getConnectionMonitorStatus()))
105 {
106 res = 0x800706ba; // The RPC server is unavailable.
107 _TRACEO(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, pAeCServer, (_T("::CoCreateInstanceEx skipped, ConnectionMonitor is marked disconnected: [0x%X] "), res));
108 }
109
110 if (SUCCEEDED(res))
111 {
112 pAeCServer->beforeInterfaceCall();
113 res = ::CoCreateInstanceEx(pCltComImpl->m_clsid, NULL, pCltComImpl->m_clsctx, sinptr, _COUNTOF(mqi), mqi);
114 pAeCServer->afterInterfaceCall();
115 _TRACEO(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, pAeCServer, (_T("::CoCreateInstanceEx() [0x%X] "), res));
116 }
117
118 if (SUCCEEDED(res))
119 {
120 // created object in OPC server
121 pCltComImpl->m_unknown = SUCCEEDED(mqi[0].hr) ? (IUnknown*)mqi[0].pItf : 0;
122 pCltComImpl->m_common = SUCCEEDED(mqi[1].hr) ? (IOPCCommon*)mqi[1].pItf : 0;
123 m_server = SUCCEEDED(mqi[2].hr) ? (IOPCEventServer*)mqi[2].pItf : 0;
124 pCltComImpl->m_secPrivate = SUCCEEDED(mqi[3].hr) ? (IOPCSecurityPrivate*)mqi[3].pItf : 0;
125
126 if (SUCCEEDED(mqi[0].hr))
127 {
128 mqi[0].pItf->Release();
129 }
130
131 if (SUCCEEDED(mqi[1].hr))
132 {
133 mqi[1].pItf->Release();
134 }
135
136 if (SUCCEEDED(mqi[2].hr))
137 {
138 mqi[2].pItf->Release();
139 }
140
141 if (SUCCEEDED(mqi[3].hr))
142 {
143 mqi[3].pItf->Release();
144 }
145 }
146
147 res = S_OK; // ignore not provided interfaces for return
148 // logon
149 pCltComImpl->logon();
150
151 if (SUCCEEDED(res) && pCltComImpl->m_shutdown)
152 {
153 // connect shutdown event sink
154 res = pCltComImpl->m_shutdown->advise();
155 }
156
157 // get/set server object attributes
158 if (SUCCEEDED(res))
159 {
160 // set valid object attributes
161 if (pAeCServer->m_validParameter)
162 {
163 res = pAeCServer->doUpdate(FALSE, pAeCServer->m_validParameter);
164 }
165 }
166 else
167 {
168 pAeCServer->setLastError(res, SOCLT_ERR_CREATE_SERVER);
169 }
170
171 if (SUCCEEDED(res))
172 {
173 pAeCServer->setObjectState(SOCltElement::connected);
174 pAeCServer->setLastError(S_OK, 0);
175 }
176 else
177 {
178 // something failed -> release actions
179 pAeCServer->doDisconnect();
180 }
181 }
182 break;
183
184 case SOCltElement::started:
185 case SOCltElement::connected:
186 {
187 if (targetState == SOCltElement::disconnected)
188 {
189 //-- disconnect server --
190 res = pAeCServer->doDisconnect();
191 pAeCServer->setLastError(res, SOCLT_ERR_DISCONNECT);
192 }
193 else
194 {
195 // start server
196 pAeCServer->setObjectState(targetState);
197 pAeCServer->setLastError(S_OK, 0);
198 }
199 }
200 break;
201 }
202 }
203
204 return res;
205} // doStateTransition

210{
211 SOCltComServerImpl* pCltComImpl = getObjectCltComServerImpl();
212 SOAeCServer* pAeCServer = getObjectAeCServer();

219

221

223 {
224 return res;
225 }

242 pAeCServer->setLCID(lcid);
243 pAeCServer->m_validParameter = validPars;
244 }
245 }
246 }
247 else
248 {
249 if (SUCCEEDED(res) && (whatParameters & SOCLT_SRVPAR_LCID))
250 {
251 // set locale id to the server

256 pAeCServer->setLastError(res, SOCLT_ERR_SET_LOCALE_ID);
257 }
258
259 if (SUCCEEDED(res) && (whatParameters & SOCLT_SRVPAR_CLIENT_NAME))
260 {
261 // set client name to the server
262 USES_CONVERSION;
263 SOCmnString cname;
264 cname = pAeCServer->getClientName();
265 pAeCServer->beforeInterfaceCall();
266 res = pIOPCCommon->SetClientName(T2CW((LPCTSTR)cname));
267 pAeCServer->afterInterfaceCall();
268 _TRACEO(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, pAeCServer, (_T("IOPCCommon::SetClientName [0x%X] "), res));
269 pAeCServer->setLastError(res, SOCLT_ERR_SET_CLIENT_NAME);
270 }
271 }

273 return res;
274}

281// - query for a COM interface of the OPC Server
282//
283// returns:
284// S_OK - interface returned
285// E_NOINTERFACE - interface not supported

287HRESULT SOAeCComServerImpl::queryInterfaceImpl(
288 IN REFIID riid, // interface id
289 IN void** interf) // interface pointer
290{
291 SOCltComServerImpl* pCltComImpl = getObjectCltComServerImpl();
292 SOAeCServer* pAeCServer = getObjectAeCServer();
293 HRESULT res = S_OK;
294
295 if (SOCLT_COM_CHECK_DEMO_AE)
296 {
297 return E_NOINTERFACE;
298 }
299
300 if (!pCltComImpl->m_unknown)
301 {
302 return E_NOINTERFACE;
303 }
304
305 if (IID_IUnknown == riid)
306 {
307 res = pCltComImpl->m_unknown.getInterface((IUnknown**)interf);
308 }
309 else if (IID_IOPCCommon == riid)
310 {
311 res = pCltComImpl->m_common.getInterface((IOPCCommon**)interf);
312 }
313 else if (IID_IOPCEventServer == riid)
314 {
315 res = m_server.getInterface((IOPCEventServer**)interf);
316 }
317 else
318 {
319 CComPtr<IUnknown> pUnk;
320 res = pCltComImpl->m_unknown.getInterface(&pUnk);
321
322 if (SUCCEEDED(res))
323 {
324 pAeCServer->beforeInterfaceCall();
325 res = pUnk->QueryInterface(riid, (void**)interf);
326 pAeCServer->afterInterfaceCall();
327#ifndef _DEBUG
328 _TRACEO(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, pAeCServer, (_T("IUnknown::QueryInterface [0x%X] "), res));
329#else
330 SOCmnString iidString;
331 IID2string(&iidString, riid);
332 _TRACEO(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, pAeCServer, (_T("IUnknown::QueryInterface %s [0x%X] "), (LPCTSTR)iidString, res));
333#endif //_DEBUG
334 }
335 }
336
337 pAeCServer->setLastError(res, SOCLT_ERR_COM_QI);
338 return res;
339} // queryInterface

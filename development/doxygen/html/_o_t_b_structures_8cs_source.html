26 public OTDateTime(DateTime aTimeStamp)
27 {
28 try
29 {
30 ulong ulTime = (ulong) aTimeStamp.ToFileTimeUtc();
31
32 this.dwLowDateTime = (uint) ((ulTime & 0x00000000ffffffff));
33 this.dwHighDateTime = (uint) ((ulTime >> 32) & 0x00000000ffffffff);
34 }
35 catch
36 {
37 this.dwHighDateTime = 0;
38 this.dwLowDateTime = 0;
39 } // end catch
40 } // end ctr
41
42
46 internal DateTime ToDateTime()
47 {
48 try
49 {
50 ulong lowNibble = this.dwLowDateTime;
51 ulong highNibble = this.dwHighDateTime;
52 highNibble = (highNibble << 32) & 0xFFFFFFFF00000000;
53 ulong complete = highNibble | (lowNibble & 0x00000000FFFFFFFF);
54 DateTime dateTime = DateTime.FromFileTimeUtc((long) complete);
55 return dateTime;
56 }
57 catch
58 {
59 return new DateTime(1601, 1, 1);
60 } // end try ... catch
61 } // end ToDateTime

63 } // end OTDateTime

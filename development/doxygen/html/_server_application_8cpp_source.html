771long Application::changeConditions(std::vector<AeCondition*>& aConditionList)
772{
773 long result = S_OK;
774 unsigned long index = 0;
775 AeCondition* condition = NULL;
776 unsigned long count = (unsigned long)aConditionList.size();
777
778 if (count == 0)
779 {
780 return S_FALSE;
781 } // end if
782
783 unsigned long* inHandles = new unsigned long[count];
784 OTSConditionData* conditionData = new OTSConditionData[count];
785 memset(conditionData, 0, count * sizeof(OTSConditionData));
786 unsigned long* outHandles = new unsigned long[count];
787
788 for (index = 0; index < count; index++)
789 {
790 condition = aConditionList[index];
791
792 if (condition != NULL)
793 {
794 result = OTSGlobals::fillOTSConditionData(*condition, conditionData[index]);
795
796 if (!SUCCEEDED(result))
797 {
798 break;
799 } //end if
800
801 inHandles[index] = condition->getHandle();
802 outHandles[index] = 0;
803 } // end if
804 } // end for
805
806 if (SUCCEEDED(result))
807 {
808 result = OTSConditionsChanged(count, inHandles, conditionData, outHandles);
809 } // end if
810
811 if (SUCCEEDED(result))
812 {
813 m_conditionListJanitor.lock();
814 for (index = 0; index < count; index++)
815 {
816 condition = aConditionList[index];
817
818 if (inHandles[index] == 0)
819 {
820 if (outHandles[index] == 0)
821 {
822 // strange situation, new condition to add, but not accepted
823 result = E_FAIL;
824 }
825 else
826 {
827 condition->setHandle(outHandles[index]);
828 m_conditionList.insert(std::pair<unsigned long, AeCondition*>(outHandles[index], condition));
829 } // end if ... else
830 }
831 else
832 {
833 if (outHandles[index] == 0)
834 {
835 // remove the condition
836 condition->setHandle(0);
837 m_conditionList.erase(inHandles[index]);
838 }
839 else
840 {
841 // normal situation for a update only action
842 } // end if ... else
843 } // end if ... else
844 } // end for
845
846 m_conditionListJanitor.unlock();
847 } // end if
848
849 for (index = 0; index < count; index++)
850 {
851 OTSGlobals::releaseOTSConditionData(conditionData[index]);
852 } // end for
853
854 delete [] conditionData;
855 delete [] inHandles;
856 delete [] outHandles;
857 return result;
858} // end changeConditions

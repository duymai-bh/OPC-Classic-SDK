193LONG OTSAPI_CALL HandleReadRequests(IN LONG count, IN OTSRequestData* paRequests)
194{
195 LONG i;
196 OTSValueData value;
197 VARIANT v;
198 LONG result = S_OK;
199 ::VariantInit(&v);
200 value.m_pValue = &v;
201
202 for (i = 0; i < count; i++)
203 {
204 if (paRequests[i].m_object.m_userData != OBJ_IDX_TAG_SESSION_SPECIFIC)
205 {
206 OTSGetCacheValue(paRequests[i].m_object.m_objectHandle, &value);
207 OTSCompleteRequests(1, &paRequests[i], &result, &value);
208 }
209 else
210 {
211 // client specific data
212 DWORD idx;
213
214 for (idx = 0; idx < g_curSessionCnt; idx++)
215 {
216 if (g_clientHandle[idx] == paRequests[i].m_clientHandle)
217 {
218 break;
219 }
220 }
221
222 value.m_timestamp.dwHighDateTime = 0;
223 value.m_timestamp.dwLowDateTime = 0;
224 value.m_quality = OT_QUALITY_GOOD;
225 value.m_pValue->vt = VT_UI4;
226 value.m_pValue->ulVal = (DWORD)idx;
227 OTSCompleteRequests(1, &paRequests[i], &result, &value);
228 }
229
230 ::VariantClear(&v);
231 }
232
233 return S_OK;
234}

280 g_curSessionCnt++;

298

303 }
304

308 }
309
310 g_curSessionCnt--;
311 }
312 else if (pSessionStateData->m_state == OTS_SESSIONSTATE_LOGON)

315 (_tcscmp(pSessionStateData->m_password, _T("ots")) == 0))
316 {
317 if ((g_clientInControl == NULL) || (g_clientInControl == hSession))
318 {
319 g_clientInControl = hSession;
320 }
321 else
322 {
323 res = E_ACCESSDENIED;
324 }
325 }
326 else
327 {
328 res = E_ACCESSDENIED;
329 }
330 }
331 else if (pSessionStateData->m_state == OTS_SESSIONSTATE_LOGOFF)
332 {
333 if (g_clientInControl == hSession)
334 {
335 g_clientInControl = NULL;
336 }
337 }
338
339 PrintSessionsAndItems();

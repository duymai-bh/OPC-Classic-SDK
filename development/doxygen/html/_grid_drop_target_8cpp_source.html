53// Overloaded Register() function performs the normal COleDropTarget::Register
54// but also serves to connect this COleDropTarget with the parent grid control,
55// where all drop messages will ultimately be forwarded.
56BOOL CGridDropTarget::Register(CGridCtrl* pGridCtrl)
57{
58 if (m_bRegistered)
59 {
60 return FALSE;
61 }
62
63 // Stop re-entry problems
64 static BOOL bInProcedure = FALSE;
65
66 if (bInProcedure)
67 {
68 return FALSE;
69 }
70
71 bInProcedure = TRUE;
72 ASSERT(pGridCtrl->IsKindOf(RUNTIME_CLASS(CGridCtrl)));
73 ASSERT(pGridCtrl);
74
75 if (!pGridCtrl || !pGridCtrl->IsKindOf(RUNTIME_CLASS(CGridCtrl)))
76 {
77 bInProcedure = FALSE;
78 return FALSE;
79 }
80
81 m_pGridCtrl = pGridCtrl;
82 m_bRegistered = COleDropTarget::Register(pGridCtrl);
83 bInProcedure = FALSE;
84 return m_bRegistered;
85}

101DROPEFFECT CGridDropTarget::OnDragScroll(CWnd* pWnd, DWORD dwKeyState, CPoint /*point*/)
102{
103// TRACE("In CGridDropTarget::OnDragScroll\n");
104 if (pWnd->GetSafeHwnd() == m_pGridCtrl->GetSafeHwnd())
105 {
106 if (dwKeyState & MK_CONTROL)
107 {
108 return DROPEFFECT_COPY;
109 }
110 else
111 {
112 return DROPEFFECT_MOVE;
113 }
114 }
115 else
116 {
117 return DROPEFFECT_NONE;
118 }
119}

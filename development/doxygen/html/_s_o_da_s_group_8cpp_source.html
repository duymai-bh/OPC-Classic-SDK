382//-----------------------------------------------------------------------------
383// validateItem
384//
385// - validate item parameters
386//
387// returns:
388// S_OK - validated item parameters
389// OPC_E_INVALIDITEMID - invalid item id
390// OPC_E_UNKNOWNITEMID - no namespace tag with the item id
391// OPC_E_BADTYPE - bad requested data type
392// OPC_E_UNKNOWNPATH - unknown access path
393//
394HRESULT SODaSGroup::validateItem(
395 IN LPCTSTR id, // item id
396 IN LPCTSTR accessPath, // access path
397 IN VARTYPE reqDataType, // requested data type
398 OUT VARTYPE* nativeDataType, // native data type
399 OUT DWORD* accessRights, // access rights
400 OUT OPTIONAL SODaSItemTag** ppTag)// tag object
401{
402 SOCmnPointer<SODaSServer> server = (SODaSServer*)getParentBranch();
403 SOCmnPointer<SODaSItemTag> itemTag;
404 HRESULT res = S_OK;
405
406 if (!server)
407 {
408 return E_FAIL;
409 }
410
411 if (nativeDataType)
412 {
413 *nativeDataType = 0;
414 }
415
416 if (accessRights)
417 {
418 *accessRights = 0;
419 }
420
421 if (!id)
422 {
423 return OPC_E_INVALIDITEMID;
424 }
425
426 res = server->getItemTag(id, &itemTag);
427
428 if (FAILED(res))
429 {
430 return res;
431 }
432
433 if (nativeDataType)
434 {
435 *nativeDataType = itemTag->getNativeDatatype();
436 }
437
438 if (accessRights)
439 {
440 *accessRights = itemTag->getAccessRights(server);
441 }
442
443 // check data type
444 if (FAILED(res = itemTag->supportReqDatatype(reqDataType)))
445 {
446 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL, (_T("can't validate item; bad requested datatype %u"), reqDataType))
447 return res;
448 }
449
450 // check access path
451 if (!itemTag->checkAccessPath(accessPath))
452 {
453 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL, (_T("can't validate item; bad access path %s"), accessPath))
454 return OPC_E_UNKNOWNPATH;
455 }
456
457 // return the found tag
458 if (ppTag)
459 {
460 *ppTag = itemTag;
461 (*ppTag)->addRef();
462 }
463
464 return S_OK;
465} // validateItem

468//-----------------------------------------------------------------------------
469// addItem
470//
471// - add item to group
472//
473// returns:
474// S_OK - validated item parameters
475// OPC_E_INVALIDITEMID - invalid item id
476// OPC_E_UNKNOWNITEMID - no namespace tag with the item id
477// OPC_E_BADTYPE - bad requested data type
478// OPC_E_UNKNOWNPATH - unknown access path
479//
480HRESULT SODaSGroup::addItem(
481 IN LPCTSTR id, // item id
482 IN LPCTSTR accessPath, // access path
483 IN BOOL active, // activation state
484 IN OPCHANDLE clientHandle, // client handle
485 IN VARTYPE reqDataType, // requested data type
486 OUT SODaSItem** item, // item object
487 OUT VARTYPE* nativeDataType, // native data type
488 OUT DWORD* accessRights) // access rights
489{
490 SOCmnPointer<SODaSItemTag> tag;
491 SODaSItem* itm = NULL;
492 HRESULT res;
493 res = validateItem(id, accessPath, reqDataType, nativeDataType, accessRights, (SODaSItemTag**)&tag);
494
495 if (SUCCEEDED(res))
496 {
497 SOCmnPointer<SODaSEntry> engine = ::getSODaSEntry();
498 SOCmnPointer<SODaSCreator> creator;
499 // create OPC tree object
500 creator = engine->getCreator();
501 itm = creator->createItem(this, tag);
502
503 if (itm == NULL)
504 {
505 return E_OUTOFMEMORY;
506 }
507
508 // add to tree
509 addLeaf(itm);
510 // set item parameters
511 itm->setItemTag(tag, ((active) && (m_active)));
512 itm->setReqDatatype(reqDataType);
513 itm->setAccessPath(accessPath);
514 itm->setClientHandle(clientHandle);
515
516 if (!itm->setActive(active))
517 {
518 tag->removeConnectedObject(itm);
519 removeLeaf(itm);
520 itm->release();
521 return E_FAIL;
522 }
523
524#ifdef SOFEATURE_WATCH
525
526 if (getCollectChanges())
527 {
528 SOCmnPointer<SOSrvEntry> entry = ::getSOSrvEntry();
529 entry->watchAddObject(this, itm);
530 }
531
532#endif
533 }
534
535 if (item != NULL)
536 {
537 *item = itm;
538 }
539 else
540 {
541 if (itm)
542 {
543 itm->release();
544 }
545 }
546
547 return res;
548} // addItem

551//-----------------------------------------------------------------------------
552// removeItem
553//
554// - remove item from the group
555//
556// returns:
557// S_OK - removed item
558// OPC_E_INVALIDHANDLE - invalid item handle
559//
560HRESULT SODaSGroup::removeItem(IN OPCHANDLE serverHandle) // server handle
561{
562 // get item
563 SOCMN_TRY
564 {
565 SOCmnPointer<SODaSItem> pItem = getItemByHandle(serverHandle);
566
567 if (!pItem)
568 {
569 return OPC_E_INVALIDHANDLE;
570 }
571
572 pItem->setActive(FALSE);
573
574 SOCmnPointer<SODaSItemTag> itemTag = pItem->getItemTag();
575
576 if (itemTag.isNotNull())
577 {
578 pItem->setItemTag(NULL, FALSE);
579 }
580
581 removeLeaf(pItem);
582
583 // check the servers transaction list for outstanding requests to the item
584 SOCmnPointer<SODaSServer> server = (SODaSServer*)getParentBranch();
585
586 if (server.isNotNull())
587 {
588 SOCmnList<SODaSTransaction> tacList(server->getTransactionList());
589 SODaSTransaction* pTac;
590 SOCmnListPosition posTac;
591 SOCmnList<SODaSRequest> reqList;
592 SOCmnListPosition posReq;
593 SODaSRequest* pReq;
594 SOCmnPointer<SODaSItem> reqItem;
595 SOCmnPointer<SODaSGroup> tacGrp;
596 posTac = tacList.getStartPosition();
597
598 while (posTac)
599 {
600 pTac = tacList.getNext(posTac);
601
602 if (pTac->getType() == SODaSTransaction::cyclic)
603 {
604 continue;
605 }
606
607 tacGrp = pTac->getGroup();
608
609 if ((SODaSGroup*)tacGrp == this)
610 {
611 reqList = pTac->getRequestList();
612 posReq = reqList.getStartPosition();
613
614 while (posReq)
615 {
616 pReq = reqList.getNext(posReq);
617 reqItem = pReq->getItem();
618
619 if ((SODaSItem*)pItem == (SODaSItem*)reqItem)
620 {
621 if (pReq->isStarted())
622 {
623 pTac->notiofyCompletedInternally(pReq);
624 pReq->complete(OPC_E_INVALIDHANDLE);
625 }
626 else
627 {
628 // complete request before it was started
629 pReq->complete(OPC_E_INVALIDHANDLE, FALSE, OPC_QUALITY_UNCERTAIN, TRUE);
630 }
631 }
632 }
633 }
634 }
635 }
636
637 return S_OK;
638 }
639 SOCMN_CATCH(...)
640 {
641 return OPC_E_INVALIDHANDLE;
642 }
643} // removeItem

646//-----------------------------------------------------------------------------
647// clone
648//
649// - creates a copy of the group
650//
651// returns:
652// S_OK - created new group
653// E_INVALIDARG - group name not valid
654// OPC_E_DUPLICATENAME - group name not unique
655//
656HRESULT SODaSGroup::clone(
657 IN SODaSServer* server, // server object of clone
658 IN LPCTSTR name, // group name
659 OUT SODaSGroup** clone) // created clone
660{
661 HRESULT res;
662 LONG timeBias;
663#ifdef SOFEATURE_SOAP
664 timeBias = 0;
665#endif
666#ifdef SOFEATURE_DCOM
667 timeBias = getTimeBias();
668#endif
669 FLOAT deadband = getDeadBand();
670 BOOL createPubTempl;
671 OPCHANDLE clientHandle = 0;
672
673 if (!server)
674 {
675 return E_FAIL;
676 }
677
678 // creating a public template by moveToPublic?
679 // -> set client handles to NULL
680 createPubTempl = server->isPublicGroups();
681
682 if (!createPubTempl)
683 {
684 clientHandle = getClientHandle();
685 }
686
687 // create tree group object
688 res = server->addGroup(name, FALSE, getUpdateRate(), clientHandle,
689 &timeBias, &deadband, getLCID(), clone, NULL, FALSE);
690
691 if (SUCCEEDED(res))
692 {
693 // cloning a public template?
694 if (isPublicTemplate())
695 {
696 // -> creates a public group
697 (*clone)->m_publicTemplate = this;
698 }
699
700 // clone all items
701 SOCmnList<SOCmnElement> list(getLeafList());
702 SOCmnPointer<SODaSItemTag> tag;
703 SODaSItem* item;
704 SODaSItem* clonedItem;
705 SOCmnListPosition pos;
706 VARTYPE canDT; // native data type
707 DWORD accessRights; // access rights
708 // for all items of the group
709 pos = list.getStartPosition();
710
711 while (pos)
712 {
713 item = (SODaSItem*)list.getNext(pos);
714 tag = item->getItemTag();
715
716 if (!createPubTempl)
717 {
718 clientHandle = item->getClientHandle();
719 }
720
721 if (SUCCEEDED((*clone)->addItem(tag->getItemID(), item->getAccessPath(), item->getActive(),
722 clientHandle, item->getReqDatatype(), &clonedItem,
723 &canDT, &accessRights)))
724 {
725 clonedItem->release();
726 }
727 }
728 }
729
730 return res;
731} // clone

824 SOCmnPointer<SODaSCreator> creator = engine->getCreator();

1449//-----------------------------------------------------------------------------
1450// initSpan
1451//
1452// - initialize time span till next update is needed
1453//
1454DWORD SODaSGroup::initSpan(
1455 IN DWORD toLate) // to late called
1456{
1457 if (m_updateRate < 200000)
1458 {
1459 // don't correct timer if update rate is bigger than 3 minutes
1460 DWORD initCnt = m_initCnt;
1461 // correct bad NT Timer (~11 ms interval)
1462 initCnt += ((m_updateRate * 150) >> 10);
1463
1464 if (initCnt < 1000)
1465 {
1466 m_correctNTtimer = 0;
1467 }
1468 else
1469 {
1470 DWORD correctNTtimer = m_correctNTtimer;
1471 correctNTtimer = 10 * (initCnt >> 10);
1472
1473 if (correctNTtimer < 0xFF)
1474 {
1475 m_correctNTtimer = (BYTE)correctNTtimer;
1476 }
1477 else
1478 {
1479 m_correctNTtimer = 0xFF;
1480 }
1481
1482 initCnt = initCnt & 0x3ff;
1483 }
1484
1485 m_initCnt = (WORD)initCnt;
1486 toLate += m_correctNTtimer;
1487 }
1488
1489 m_initTime = ::GetTickCount();
1490
1491 if (toLate < m_updateRate)
1492 {
1493 return m_updateRate - toLate;
1494 }
1495 else
1496 {
1497 return m_updateRate;
1498 }
1499} // initSpan

1503// updateSpan
1504//
1505// - update time span till next request start
1506//
1507// returns:
1508// TRUE - time span ellapsed
1509// FALSE - time span not ellapsed
1510//
1511BOOL SODaSGroup::updateSpan(
1512 IN DWORD now, // current time
1513 IN DWORD* span) // time till next group update
1514{
1515 DWORD newSpan;
1516
1517 if (!m_active)
1518 {
1519 if (span)
1520 {
1521 *span = INFINITE;
1522 }
1523
1524 return FALSE;
1525 }
1526
1527 // check for immediate update rate
1528 if (m_updateRate < 10)
1529 {
1530 if (span)
1531 {
1532 *span = INFINITE;
1533 }
1534
1535 if (isNotifyAllChanges())
1536 {
1537 return TRUE;
1538 }
1539 else
1540 {
1541 SOCmnPointer<SODaSTransaction> cycTrans;
1542 cycTrans = getCyclicTransaction();
1543 return !cycTrans->isStarted();
1544 }
1545 }
1546
1547 DWORD ellapsed = getTimeSpan(m_initTime, now);
1548 DWORD updateRate = m_updateRate - m_correctNTtimer;
1549 BOOL fire = FALSE;
1550
1551 if (ellapsed < updateRate)
1552 {
1553 newSpan = updateRate - ellapsed;
1554 }
1555 else // span ellapsed
1556 {
1557 DWORD toLate = ellapsed - updateRate;
1558 newSpan = initSpan(toLate);
1559 fire = TRUE;
1560 }
1561
1562 if (span)
1563 {
1564 *span = newSpan;
1565 }
1566

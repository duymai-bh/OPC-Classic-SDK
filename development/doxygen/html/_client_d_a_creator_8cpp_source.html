36SODaCServer* ClientDACreator::createServer(
37 IN BYTE protocolType,
38 IN SODaCEntry* /* parent */)
39{
40 SODaCServer* p = NULL;
41
42 switch (protocolType)
43 {
44#ifdef SOFEATURE_DCOM
45
46 case SOCLT_PROTOCOL_DCOM:
47 SOCMN_ALLOCATE_OBJECT_C(p, ClientDASessionCom(), (SODaCServer*));
48 break;
49#endif
50#ifdef SOFEATURE_SOAP
51
52 case SOCLT_PROTOCOL_SOAP:
53 SOCMN_ALLOCATE_OBJECT_C(p, ClientDASessionXml(), (SODaCServer*));
54 if (m_xmlGetStatusInterval > 0)
55 {
56 SOCmnPointer<SOCltTaskXmlGetStatus> getStatusCyclicTask = createTaskXmlGetStatus(p, m_xmlGetStatusInterval);
57 p->setXmlStatusTask(getStatusCyclicTask);
58 }
59 break;
60#endif
61#ifdef SOFEATURE_TUNNEL
62
63 case SOCLT_PROTOCOL_TUNNEL:
64 SOCMN_ALLOCATE_OBJECT_C(p, ClientDASessionTp(), (SODaCServer*));
65 break;
66#endif
67
68 default:
69 return NULL;
70 break;
71 }
72
73 return p;
74}

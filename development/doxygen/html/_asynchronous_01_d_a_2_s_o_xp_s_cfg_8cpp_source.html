433//
434// - handle a OPC DA I/O request
435//
436void handleRequest(DWORD ioOperation,
437 IN OUT SOXpSDARequestData* requestData)// request data
438{
439 if (ioOperation == SOXPS_REQUEST_OPERATION_READ)
440 {
441 // read
442 // do the differen simmulations
443 if (requestData->m_itemHandle == g_timeProperty)
444 {
445 // check the saved item handles (tag or property)
446 // saved the hanlde for the time property (seconds since 1970)
447 VariantClear(&requestData->m_value);
448 requestData->m_value.lVal = time(NULL);
449 requestData->m_value.vt = VT_I4;
450 requestData->m_quality = SOXPS_QUALITY_GOOD;
451 requestData->m_requestState = SOXPS_REQUEST_STATE_HANDLED;
452 memset(&requestData->m_timeStamp, 0, sizeof(FILETIME));
453 }
454 else if ((requestData->m_itemUserData >= 0x0100) && (requestData->m_itemUserData <= 0x0103))
455 {
456 // check the item user data
457 // increment branch
458 switch (requestData->m_itemUserData)
459 {
460 case 0x0100:
461 requestData->m_value.lVal += (LONG)10;
462 break;
463
464 case 0x0101:
465 requestData->m_value.iVal += (SHORT)10;
466 break;
467
468 case 0x0102:
469 requestData->m_value.lVal++;
470 break;
471
472 case 0x0103:
473 requestData->m_value.iVal++;
474 break;
475 }
476
477 // set the quality to good
478 requestData->m_quality = SOXPS_QUALITY_GOOD;
479 // set the request state to handled
480 requestData->m_requestState = SOXPS_REQUEST_STATE_HANDLED;
481 // let the XPress Server calculate the current time
482 memset(&requestData->m_timeStamp, 0, sizeof(FILETIME));
483 }
484 else if (requestData->m_itemPath[0] == _T('t'))
485 {
486 // check the item path
487 // time branch
488 SYSTEMTIME time;
489 TCHAR type = _T(' ');
490 WCHAR wideString[10];
491 // set the current time
492 CoFileTimeNow(&requestData->m_timeStamp);
493
494 if (_tcsncmp(requestData->m_itemPath, _T("time.local."), 11) == 0)
495 {
496 GetLocalTime(&time);
497 type = requestData->m_itemPath[11];
498 }
499 else if (_tcsncmp(requestData->m_itemPath, _T("time.GMT."), 9) == 0)

503 }
504
505 // allways clear the request value!!!
506 // it includes the current cache value
507 VariantClear(&requestData->m_value);

510

513 case _T('s'):

545//-----------------------------------------------------------------------------
546// requestThreadMain
547//
548// - thread for handling async requests
549//
550DWORD WINAPI requestThreadMain(
551 IN LPVOID context) // thread data
552{
553 BOOL end = FALSE;
554 DWORD waitR;
555 DWORD i, j;
556 list<RequestData>::iterator requestData;
557 DWORD size;
558 SOXpSDARequestData* requestArray;
559
560 while (!end)
561 {
562 // wait for request data or the end
563 waitR = WaitForMultipleObjects(2, g_requestThreadEvents, FALSE, INFINITE);
564
565 switch (waitR - WAIT_OBJECT_0)
566 {
567 case END_EVENT:
568 end = TRUE;
569 break;
570
571 default:
572 {
573 // request data in list
574 EnterCriticalSection(&g_requestListLock);
575
576 if ((size = g_requestList.size()) > 0)
577 {
578 // alloc array
579 requestArray = (SOXpSDARequestData*)malloc(sizeof(SOXpSDARequestData) * size);
580 i = 0;
581
582 for (requestData = g_requestList.begin(); requestData != g_requestList.end(); ++requestData)
583 {
584 (*requestData).copy(requestArray[i]);
585 // handle the request
586 handleRequest((*requestData).getIOOperation(), &requestArray[i]);
587 i++;
588 }
589
590 // call request completion function of XPress Tool
591 g_completedReq(i, requestArray);
592 // free all memory
593 g_requestList.clear();
594
595 for (j = 0; j < i; j++)
596 {
597 free((void*)requestArray[j].m_itemPath);
598 VariantClear(&requestArray[j].m_value);
599 }
600
601 free((void*)requestArray);
602 }
603
604 LeaveCriticalSection(&g_requestListLock);
605 }
606 break;
607 }
608 }
609
610 return 0;
611} // requestThreadMain

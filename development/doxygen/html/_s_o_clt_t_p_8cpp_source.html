156bool SOCltTPConnection::handleInitMessage()
157{
158 TP_INITIAL_MESSAGE_HEADER response;
159 // Unmarshal header of initial response message. Payload follows depending
160 // on the flags. Stop connecting if any of the following conditions apply:
161 // 1) The header cannot be unmarshaled.
162 // 2) The magic value is wrong.
163 // 3) The server is the prototype server.
164 // -----------------------------------------------------------------------
165 m_unmarshal->SetReadPos(0);
166
167 if (!unmarshalInitialMessageHeader(m_unmarshal, response))
168 {
169 return (false);
170 }
171
172 if ((response.magic_value != TP_MAGIC_VALUE_TPV1) && (response.magic_value != TP_MAGIC_VALUE_ADD_ACCEPT))
173 {
174 return (false);
175 }
176
177 if (response.protocol_version == 0)
178 {
179 return (false);
180 }
181
182 m_flags = response.flags & TP_INIT_FLAG_PROTOCOL_MASK;
183
184 if ( (m_flags & TP_INIT_FLAG_ACCEPT_FULL_DATE) != TP_INIT_FLAG_ACCEPT_FULL_DATE)
185 {
186 m_flags &= (~TP_INIT_FLAG_USE_FULL_DATE);
187 }
188
189 // If the server did not establish a session (indicated by a cleared
190 // TP_INIT_FLAG_NEW_SESSION_ID flag) we shutdown the connection.
191 // -----------------------------------------------------------------
192 m_init = true;
193 SOCmnPointer<SOCmnObject> sock = getSocket();
194
195 if (!(response.flags & TP_INIT_FLAG_NEW_SESSION_ID))
196 {
197 m_srv->shutdownClient(sock);
198 return (true);
199 };
200
201 // Signal we are connected to the server.
202 // --------------------------------------
203 m_srv->setTimeOut(sock, m_timeout);
204
205 m_connectEvent.signal();
206
207 return true;
208}

210bool SOCltTPConnection::sendInitMessage()
211{
212 SOCmnString strUserName;
213 SOCmnString strPassword;
214 SOCmnPointer <SOCmnTP_Marshaler> pMarshaler;
215 TP_INITIAL_MESSAGE_HEADER request;
216 SOCmnPointer<SOCmnTPCredentials> pCredentials;
217 u_char* pRequestData;
218 // Build initial request message and marshal it into a buffer. The size
219 // passed to BeginMarshal is an estimation of the maximum size required.
220 // The marshaler will reallocate if it turns out to be too small.
221 // ---------------------------------------------------------------------
222 request.message_len = sizeof(TP_INITIAL_MESSAGE_HEADER);
223 request.magic_value = TP_MAGIC_VALUE_USE_CLT;
224 request.protocol_version = TP_PROTOCOL_VERSION;
225 request.flags = TP_INIT_FLAG_NEW_SESSION_ID | TP_INIT_FLAG_CREDENTIALS_PROVIDED | TP_INIT_FLAG_USE_FULL_DATE;
226 request.reserved = 0;
227#ifdef SOOS_WINDOWS_CE
228 request.flags |= TP_INIT_FLAG_USE_UTF16_STRINGS;
229#endif
230#ifdef SOOS_LINUX
231 request.flags |= TP_INIT_FLAG_NOT_SUPPORT_UTF16_STRINGS;
232#endif
233 pMarshaler = new SOCmnTP_Marshaler(0); // not protocol flags allowed for the marshaling of the init message!
234 pMarshaler->BeginMarshal(sizeof(TP_INITIAL_MESSAGE_HEADER) + 200);
235 marshalInitialMessageHeader(pMarshaler, request);
236 pCredentials = getCredentials();
237 marshalCredentials(pMarshaler, pCredentials);
238 pMarshaler->SetWritePos(0);
239 request.message_len = pMarshaler->GetMarshalDataLen();
240 pMarshaler->MarshalLong(request.message_len);
241 // Send initial request message to server.
242 // ---------------------------------------
243 pRequestData = pMarshaler->GetMarshalData();
244
245 if (pRequestData == NULL)
246 {
247 return (false);
248 }
249
250 SOCmnPointer<SOCmnObject> sock = getSocket();
251
252 if (m_srv->sndData(sock, request.message_len, pRequestData) == SOCKET_ERROR)
253 {
254 return (false);
255 }
256
257 m_srv->setTimeOut(sock, m_timeout);
258 return (true);
259}

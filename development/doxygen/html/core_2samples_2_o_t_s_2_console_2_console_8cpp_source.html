665long OTSAPI_CALL HandleWriteRequests(IN long count, IN OTSRequestData* paRequests, IN OTSValueData* pValues)
666{
667 long i;
668 long result;
669
670 for (i = 0; i < count; i++)
671 {
672 result = S_OK;
673
674 if (paRequests[i].m_propertyID == 0)
675 {
676 // process value
677 OTSCompleteRequests(1, &paRequests[i], &result, NULL);
678
679 if (SUCCEEDED(result)) // write succeeded -> update cache
680 {
681 OTSValuesChanged(1, &paRequests[i].m_object.m_objectHandle, &pValues[i]);
682 }
683 }
684 else
685 {
686 // property
687 if (paRequests[i].m_propertyID == 6020)
688 {
689 if (pValues[i].m_pValue->iVal < 359)
690 {
691 g_angle = pValues[i].m_pValue->iVal;
692 double radianAngle;
693 static double pi = 3.1415926535;
694 OTSValueData value;
695 OTVariant var;
696 value.m_pValue = &var;
697 radianAngle = (2 * pi) * ((double)g_angle / 360.0);
698 value.m_pValue->vt = VT_R8;
699 value.m_pValue->dblVal = sin(radianAngle);
700 value.m_quality = OT_QUALITY_GOOD;
701#if defined(WIN32) || defined(WIN64)
702 value.m_timestamp.dwHighDateTime = 0;
703 value.m_timestamp.dwLowDateTime = 0;
704#else
705 value.m_timestamp.tv_sec = 0;
706 value.m_timestamp.tv_usec = 0;
707#endif
708 OTSValuesChanged(1, &g_objHandles[OBJ_IDX_TAG_MATHS_SIN], &value);
709 }
710 else
711 {
712 result = OT_E_OPC_RANGE;
713 }
714
715 OTSCompleteRequests(1, &paRequests[i], &result, NULL);
716 }
717 }
718 }
719
720 return S_OK;

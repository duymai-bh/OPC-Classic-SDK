165//-----------------------------------------------------------------------------
166// SOXpSStart
167//
168// - starts additional thread of the DLL
169// - this function is called after building all spaces (name, event, area)
170//
171// returns:
172// TRUE - DLL startup OK, continue starting the OPC server
173// FALSE - error during startup, end the OPC server
174//
175DWORD SOXPSCFG_CALL SOXpSStart(void)
176{
177 DWORD dwSectorsPerCluster = 0;
178 DWORD dwBytesPerSector = 0;
179 DWORD dwNumberOfFreeClusters = 0;
180 DWORD dwTotalNumberOfClusters = 0;
181// Use the flags FILE_FLAG_OVERLAPPED and FILE_FLAG_NO_BUFFERING
182// to instruct the operating system that asynchronous i/o with no buffering
183// should be used for this file
184 hFile = ::CreateFile(szFileName,
185 GENERIC_READ | GENERIC_WRITE,
186 FILE_SHARE_READ | FILE_SHARE_WRITE,
187 NULL,
188 OPEN_ALWAYS,
189 FILE_FLAG_NO_BUFFERING | FILE_FLAG_OVERLAPPED,
190 NULL);
191// An application must meet certain requirements when working with files opened
192// with FILE_FLAG_NO_BUFFERING:
193//
194// File access must begin at byte offsets within the file that are integer multiples
195// of the volume's sector size. To determine a volume's sector size, we call the
196// GetDiskFreeSpace function.
197//
198// File access must be for numbers of bytes that are integer multiples
199// of the volume's sector size. For example, if the sector size is 512 bytes,
200// an application can request reads and writes of 512, 1024, or 2048 bytes, but
201// not of 335, 981, or 7171 bytes.
202//
203// Buffer addresses for read and write operations must be sector
204// aligned (aligned on addresses in memory that are integer multiples of
205// the volume's sector size). One way to sector align buffers is to use
206// the VirtualAlloc function to allocate the buffers.
207// This function allocates memory that is aligned on addresses that are
208// integer multiples of the system's page size. Because both page and
209// volume sector sizes are powers of 2, memory aligned by multiples of the
210// system's page size is also aligned by multiples of the volume's sector size.
211 ::GetDiskFreeSpace(NULL, // use the root of the current directory
212 &dwSectorsPerCluster,
213 &dwBytesPerSector,
214 &dwNumberOfFreeClusters,
215 &dwTotalNumberOfClusters);
216 g_dwBytesPerSector = dwBytesPerSector;
217 return (hFile != INVALID_HANDLE_VALUE);
218} // SOXpSStart

258VOID CALLBACK fileIOCompletionRoutine(
259 DWORD dwErrorCode, // completion code
260 DWORD dwNumberOfBytesTransfered, // number of bytes transferred
261 LPOVERLAPPED lpOverlapped) // I/O information buffer
262{
263 // Calculate the pointer to the CDataRequest from
264 //the pointer to it's m_ov member
265 CDataRequest* pdrq = (CDataRequest*)(((LPBYTE)lpOverlapped) - offsetof(CDataRequest, m_ov));
266 pdrq->m_rq.m_requestState = SOXPS_REQUEST_STATE_HANDLED;
267
268 if (0 == dwErrorCode)
269 {
270 LPWSTR lpwszBuff = (LPWSTR)_alloca(sizeof(WCHAR) * (dwNumberOfBytesTransfered + 1));
271 pdrq->m_rq.m_result = S_OK;
272 pdrq->m_rq.m_quality = SOXPS_QUALITY_GOOD;
273 LPVARIANT pv = &pdrq->m_rq.m_value;
274 // Convert the buffer to UNICODE
275 DWORD dwChars = ::MultiByteToWideChar(CP_ACP, 0, (LPSTR)pdrq->m_lpBuffer,
276 (int)dwNumberOfBytesTransfered,
277 lpwszBuff, (int)dwNumberOfBytesTransfered);
278 lpwszBuff[dwChars] = 0;
279 V_VT(pv) = VT_BSTR;
280 V_BSTR(pv) = ::SysAllocString(lpwszBuff);
281 }
282 else
283 {
284 pdrq->m_rq.m_result = HRESULT_FROM_WIN32(dwErrorCode);
285 pdrq->m_rq.m_quality = SOXPS_QUALITY_BAD;
286 }
287
288 pfnRequestCompleted(1, &pdrq->m_rq);

293//-----------------------------------------------------------------------------
294// SOXpSHandleDARequests
295//
296// - store the OPC DA I/O requests in the list for async I/O
297// - this function is called for every device read or write of a OPC client
298//
299void SOXPSCFG_CALL SOXpSHandleDARequests(
300 IN DWORD ioOperation, // type of I/O operation
301 IN DWORD numRequests, // number of requests
302 IN OUT SOXpSDARequestData* requestArray)// request data
303{
304 for (DWORD i = 0; i < numRequests; i++) // handle all requests
305 {
306 CDataRequest* pdrq = NULL;
307
308 try
309 {
310 if (tag == requestArray[i].m_itemHandle)
311 {
312 _ASSERTE(IS_VALID_HANDLE(hFile));
313 _ASSERTE(SOXPS_REQUEST_OPERATION_READ == ioOperation);
314 // Read the first file sector
315 pdrq = new CDataRequest(g_dwBytesPerSector, requestArray[i]);
316
317 if (NULL == pdrq)
318 {
319 throw E_OUTOFMEMORY;
320 }
321
322 LPOVERLAPPED lpOverlapped = &pdrq->m_ov;
323 lpOverlapped.hEvent = CreateEvent(
324 NULL, // default security attribute
325 TRUE, // manual reset event
326 TRUE, // initial state = signaled
327 NULL); // unnamed event object
328 // Start overlapped i/o. The result of the operation will be reported
329 // in the fileIOCompletionRoutine. It will also free the corresponding
330 // CDataRequest instance
331 BOOL ret = ::ReadFileEx(hFile, pdrq->m_lpBuffer, pdrq->m_dwSize,
332 lpOverlapped,
333 (LPOVERLAPPED_COMPLETION_ROUTINE) fileIOCompletionRoutine);
334
335 if (!ret)
336 {
337 throw E_FAIL;
338 }
339 }
340 else
341 {
342 _ASSERTE(false); // this should never be reached
343 }
344 }
345 catch (HRESULT hr)
346 {
347 requestArray[i].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
348 requestArray[i].m_result = hr;
349
350 if (pdrq)
351 {
352 delete pdrq;
353 }
354
355 continue;
356 }
357 }
358} // SOXpSHandleDARequests

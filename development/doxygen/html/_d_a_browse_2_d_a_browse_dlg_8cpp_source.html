168// checks whether a given DaAddressSpaceElement has children (leaves or branches) or
169// vendor defined properties
170BOOL CDABrowseDlg::isExpandable(DaAddressSpaceElement& element)
171{
172 BOOL returnedResult = FALSE;
173 DaAddressSpaceElementBrowseOptions someBrowseOptions;
174 std::vector<DaAddressSpaceElement*>addressSpaceElements;
175 std::vector<DaAddressSpaceElement*>::iterator addressSpaceelementsIterator;
176 someBrowseOptions.setElementTypeFilter(EnumAddressSpaceElementType_ALL);
177 // gets all the element's children
178 m_session->browse(
179 element.getItemId(),
180 element.getItemPath(),
181 &someBrowseOptions,
182 addressSpaceElements,
183 NULL);
184 // if the element has children set the returned result to TRUE
185 if (addressSpaceElements.size() != 0)
186 {
187 returnedResult = TRUE;
188 }
189
190 for (addressSpaceelementsIterator = addressSpaceElements.begin(); addressSpaceelementsIterator != addressSpaceElements.end(); addressSpaceelementsIterator++)
191 {
192 delete(*addressSpaceelementsIterator);
193 }
194
195 // it is not necessary to go further because we already found that the element is expandable
196 if (returnedResult)
197 {
198 return TRUE;
199 }
200
201 std::vector<DaProperty*>::iterator propertiesIterator;
202 std::vector<DaProperty*>allProperties;
203 DaGetPropertiesOptions propertiesOptions;
204 propertiesOptions.setWhatPropertyData(EnumPropertyData_ALL);
205 m_session->getDaProperties(
206 element.getItemId(),
207 element.getItemPath(),
208 &propertiesOptions,
209 allProperties,
210 NULL);
211 size_t count = allProperties.size();
212 size_t i = 0;
213
214 while ((i < count) && (!returnedResult))
215 {
216 // only the vendor defined properties must be added as children of the current element
217 // so only if the element has such properties it gets expandable
218 if (allProperties[i]->getId() >= 100)
219 {
220 returnedResult = TRUE;
221 }
222
223 i++;
224 }
225
226 for (propertiesIterator = allProperties.begin(); propertiesIterator != allProperties.end(); propertiesIterator++)
227 {
228 delete(*propertiesIterator);
229 }
230
231 return returnedResult;
232}

234void CDABrowseDlg::browseNamespace()
235{
236 m_tree.DeleteAllItems();
237 // Insert the root item.
238 // The children will be inserted 'on demand' when a node
239 // gets expanded
240 tstring rootNodeText = m_session->getUrl();
241 DaAddressSpaceElement* rootNode = new DaAddressSpaceElement(
242 EnumAddressSpaceElementType_BRANCH,
243 rootNodeText, _T(""), 0, _T(""), m_session);
244 TVINSERTSTRUCT tvis;
245 tvis.hParent = TVI_ROOT;
246 tvis.hInsertAfter = TVI_FIRST;
247 tvis.item.mask = TVIF_TEXT | TVIF_PARAM | TVIF_CHILDREN | TVIF_IMAGE | TVIF_SELECTEDIMAGE;
248 tvis.item.pszText = (LPTSTR)(LPCTSTR) rootNodeText.c_str();
249 tvis.item.lParam = (LPARAM)rootNode;
250 tvis.item.cChildren = 1;
251 tvis.item.iImage = 0;
252 tvis.item.iSelectedImage = 0;
253 HTREEITEM hRootNode = m_tree.InsertItem(&tvis);
254
255 //get root's properties
256 if (m_session->isConnected())
257 {
258 std::vector<DaProperty*>::iterator propertiesIterator;
259 std::vector<DaProperty*>allProperties;
260 DaGetPropertiesOptions* propertiesOptions = new DaGetPropertiesOptions();
261 propertiesOptions->setWhatPropertyData(EnumPropertyData_ALL);
262 long result = m_session->getDaProperties(
263 rootNode->getItemId(),
264 rootNode->getItemPath(),
265 propertiesOptions,
266 allProperties,
267 NULL);
268 size_t count = allProperties.size();
269 TV_INSERTSTRUCT property;
270 property.hParent = hRootNode;
271 property.hInsertAfter = TVI_LAST;
272 property.item.mask = TVIF_TEXT | TVIF_PARAM | TVIF_CHILDREN | TVIF_IMAGE | TVIF_SELECTEDIMAGE;
273
274 for (unsigned long i = 0; i < count; i++)
275 {
276 if (allProperties[i]->getId() >= 100)
277 {
278 tstring name = allProperties[i]->getName();
279 property.item.pszText = (LPTSTR)(LPCTSTR) name.c_str() ;

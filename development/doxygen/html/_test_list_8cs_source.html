115 public override TestResult Execute()
116 {
117 int errors = 0, warnings = 0;
118
119 if( !m_bRunInParallel )
120 {
121 foreach( Element child in m_children )
122 {
123 if( IsCancelationPending() )
124 break;
125
126 if( m_bStopIfOneFailed && errors > 0 )
127 break;
128
129 child.Run();
130
131 if( child.RunResult != null ) //it may be null if the node is unchecked
132 {
133 switch( child.RunResult.Type )
134 {
135 case TestResults.Failed: errors++;
136 break;
137 case TestResults.Warning: warnings++;
138 break;
139 }
140
141 if( DontKeepResults.HasValue && DontKeepResults.Value )
142 {
143 child.RunResult = child.RunResult.Strip();
144 }
145 }
146 }
147 }
148 else
149 {
150 List<WaitHandle> waitHandles = new List<WaitHandle>( m_children.Count );
151
152 foreach( Element child in m_children )
153 {
154 waitHandles.Add( child.WaitHandle );
155
156 ThreadPool.QueueUserWorkItem( new WaitCallback( child.Run ) );
157 }
158
159 System.Threading.WaitHandle.WaitAll( waitHandles.ToArray() );
160
161 foreach( Element child in m_children )
162 {
163 if( child.RunResult != null )
164 {
165 switch( child.RunResult.Type )
166 {
167 case TestResults.Failed: errors++;
168 break;
169 case TestResults.Warning: warnings++;
170 break;
171 }
172
173 if( DontKeepResults.HasValue && DontKeepResults.Value )
174 {
175 child.RunResult = child.RunResult.Strip();
176 }
177 }
178 }
179 }
180
181 if( DontKeepResults.HasValue && DontKeepResults.Value )
182 {
183 this.Description = null;
184 }
185
186 //if( m_bGarbageCollect )
187 //{
188
189 //}
190
191 TestResult result;
192 if( errors > 0 )
193 result = new TestResult( TestResults.Failed );
194 else if( warnings > 0 )
195 result = new TestResult( TestResults.Warning );
196 else
197 result = new TestResult( TestResults.Success );
198
199 return result;
200 }

202 internal override void Run()
203 {
204 try
205 {
206 if( RunResult != null )
207 {
208 throw new ApplicationException( "Clear the results before running again" );
209 }
210
211 if( CheckBoxes && !TreeNode.Checked )
212 return;
213
214 bool bMultipleRuns = false;
215 int repeatCount = int.MaxValue;
216
217 if( RepeatCount != null )
218 {
219 int tmp;
220 if( int.TryParse( RepeatCount, out tmp ) )
221 {
222 repeatCount = tmp;
223 bMultipleRuns = true;
224 }
225 }
226
227 double repeatDuration = double.MaxValue;
228
229 if( RepeatDuration != null )
230 {
231 double tmp;
232 if( double.TryParse( RepeatDuration, out tmp ) )
233 {
234 repeatDuration = tmp;
235 bMultipleRuns = true;
236 }
237 }
238
239 TestResult result = null;
240 DateTime startTime = DateTime.Now;
241
242 if( bMultipleRuns )
243 {
244 int runCounter = 0;
245 int errors = 0, warnings = 0;
246 List<TestList> innerLists = new List<TestList>();
247 DateTime maxTime = repeatDuration == double.MaxValue ? DateTime.MaxValue : startTime.AddSeconds( repeatDuration );
248
249 XmlWriter externalRunFileWriter = null;
250
251 if( SaveAfterEachRun )
252 {
253 string timestamp = DateTime.Now.ToString( "yyyy-MM-dd-HH-mm-ss", System.Globalization.DateTimeFormatInfo.InvariantInfo );
254 ExternalRunFile = timestamp + "Results.xml";
255 ExternalTracesFile = timestamp + "Traces.xml";
256
257 externalRunFileWriter = XmlWriter.Create( ExternalRunFile );
258 externalRunFileWriter.WriteStartElement( "TestList", "http://softing-ia.com/TestClient.xsd" );
259
260 m_externalTracesFileWriter = XmlWriter.Create( ExternalTracesFile );
261 m_externalTracesFileWriter.WriteStartElement( "TestList", "http://softing-ia.com/TestClient.xsd" );
262 m_externalTracesFileWriter.WriteStartElement( "Traces" );
263 }
264
265 //set a temporary result to be able to refresh the view of partial results
266 RunResult = new TestListTestResult( TestResults.Executing );
267 RunResult.StartTime = startTime;
268 ( (TestListTestResult) RunResult ).TestLists = innerLists;
269
270 while( runCounter < repeatCount && DateTime.Now <= maxTime && !IsCancelationPending() )
271 {
272 TestResult runResult;
273
274 TestList clone = (TestList) Clone();
275
276 if( !SaveAfterEachRun )
277 {
278 lock( innerLists )
279 {
280 innerLists.Add( clone );
281 }
282 ReportFinished();
283 }
284
285 runResult = clone.RunOnce();
286 clone.RunResult = runResult;
287
288 if( SaveAfterEachRun )
289 {
290 lock( m_serializer )
291 {
292 Element.Serializing = true;
293 m_serializer.Serialize( externalRunFileWriter, clone );
294 Element.Serializing = false;
295 }
296 }
297
298 switch( runResult.Type )
299 {
300 case TestResults.Failed: errors++;
301 break;
302 case TestResults.Warning: warnings++;
303 break;
304 }
305
306 runCounter++;
307 }
308
309 if( SaveAfterEachRun )
310 {
311 externalRunFileWriter.WriteEndElement();
312 externalRunFileWriter.Close();
313
314 m_externalTracesFileWriter.WriteEndElement();
315 m_externalTracesFileWriter.WriteEndElement();
316 m_externalTracesFileWriter.Close();
317 }
318
319 TestListTestResult listResult = (TestListTestResult) RunResult;
320
321 if( errors > 0 )
322 listResult.Type = TestResults.Failed;
323 else if( warnings > 0 )
324 listResult.Type = TestResults.Warning;
325 else
326 listResult.Type = TestResults.Success;
327
328 listResult.EndTime = DateTime.Now;
329 listResult.StartTime = startTime;
330 result = listResult;
331 }
332 else
333 {
334 SaveAfterEachRun = false;
335 RunResult = new TestResult( TestResults.Executing );
336 RunResult.StartTime = startTime;
337 ReportFinished();
338
339 result = RunOnce();
340 }
341
342 RunResult = result;
343 ReportFinished();
344 }
345 finally
346 {
347 WaitHandle.Set();
348 }

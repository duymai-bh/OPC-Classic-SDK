59//-----------------------------------------------------------------------------
60// doStateTransitionImpl
61//
62// - do the state transition for the server if the target state differs from the
63// actual state of the server
64//
65// returns:
66// S_OK - transit to new state
67// error code of GetStatus - error during state change
68//
69HRESULT SODaCXmlServerImpl::doStateTransitionImpl(void)
70{
71 SODaCServer* pDaCServer = getObjectDaCServer();
72 BYTE targetState = pDaCServer->getTargetObjectState();
73 BYTE currentState = pDaCServer->getObjectState();
74 HRESULT res = S_OK;
75
76 if (currentState != targetState)
77 {
78 switch (targetState)
79 {
80 case SOCltElement::connected:
81 case SOCltElement::started:
82 {
83 if (currentState == SOCltElement::disconnected)
84 {
85 SOCmnString hostName = pDaCServer->getNodeName();
86
87 if (hostName.isEmpty())
88 {
89 hostName = _T("localhost");
90 }
91
92 m_httpCon->setHostName(hostName);
93 m_httpCon->setServer(pDaCServer);
94 SODaCServerStatus status;
95 res = getStatus(status);
96
97 if (SUCCEEDED(res))
98 {
99 pDaCServer->setObjectState(targetState);
100 }
101 else
102 {
103 _TRACEO(SOCMNTRACE_L_INF, SOCMNTRACE_G_SOCKET, m_httpCon->getHTTPClient(), (_T("connect failed")));
104 m_httpCon->setServer(NULL);
105 m_httpCon->disconnect();
106 }
107 }
108 else
109 {
110 pDaCServer->setObjectState(targetState);
111 }
112 }
113 break;
114
115 case SOCltElement::disconnected:
116 m_httpCon->setServer(NULL);
117 m_httpCon->disconnect();
118 pDaCServer->setObjectState(targetState);
119 break;
120 }
121 }
122
123 return res;
124} // doStateTransition

168HRESULT SODaCXmlServerImpl::getStatus(OUT SODaCServerStatus& status)
169{
170 if (SOCLT_XML_CHECK_DEMO)
171 {
172 return E_FAIL;
173 }
174
175 SODaCServer* pDaCServer = getObjectDaCServer();
176 SOCmnPointer<SODaCXmlCall> httpCall;
177 HRESULT res;
178 SOCMN_ALLOCATE_OBJECT_C(httpCall, SODaCXmlCall(m_httpCon), (SODaCXmlCall*));
179 SODaCXmlRequestOptions reqOptions;
180 httpCall->getStatus(fillRequestOptions(&reqOptions));
181 pDaCServer->beforeInterfaceCall();
182 res = m_httpCon->invoke(httpCall, TRUE);
183 pDaCServer->afterInterfaceCall();
184
185 if (SUCCEEDED(res))
186 {
187 SOCmnPointer<SODaCXmlResponse> response;
188 res = httpCall->getStatusResponse(&response);
189
190 if (SUCCEEDED(res))
191 {
192 SOCmnStringList xmlOptions(response->getResponseOptions());
193 SOCmnList<SODaCXmlResponseData> xmlData(response->getResponseData());
194 SODaCXmlResponseData* pResData;
195 SOCmnListPosition posD, posO;
196 SOCmnString key, val;
197 LCID lcid;
198 status.clear();
199 posO = xmlOptions.getStartPosition();
200
201 while (posO)
202 {
203 val = xmlOptions.getNext(posO, &key);
204
205 if (_tcsicmp(key, _T("ServerState")) == 0)
206 {
207 status.m_state = getServerStateFromString(val);
208 }
209 else if (_tcsicmp(key, _T("StatusInfo")) == 0)
210 {
211 status.m_statusInfo = val;
212 }
213 else if (_tcsicmp(key, _T("ProductVersion")) == 0)
214 {
215 status.m_productVersion = val;
216 }
217 else if (_tcsicmp(key, _T("StartTime")) == 0)
218 {
219 status.m_startTime.setXMLDateTime(val);
220 }
221 else if (_tcsicmp(key, _T("ReplyTime")) == 0)
222 {
223 status.m_currentTime.setXMLDateTime(val);
224 }
225 }
226
227 posD = xmlData.getStartPosition();
228
229 while (posD)
230 {
231 pResData = xmlData.getNext(posD);
232
233 switch (pResData->m_type)
234 {
235 case SODACXMLRESPONSEDATA_TYPE_VENDORINFO:
236 status.m_vendorInfo = pResData->m_dataSZ;
237 break;
238
239 case SODACXMLRESPONSEDATA_TYPE_LOCALEID:
240 if (pDaCServer->convertLocaleIDtoLCID(pResData->m_dataSZ, lcid))
241 {
242 status.m_supportedLCIDs.add(lcid);
243 }
244
245 break;
246 }
247 }
248
249 status.m_supportedInterfaces.add(SODAC_VERSION_OPC_XMLDAV10);
250 }
251 }
252
253 _TRACEO_LEVEL_FROM_RES(res, SOCMNTRACE_G_OPC_CALL_C, pDaCServer, (_T("GetStatus [0x%lX]"), res));
254 return res;
255}

263HRESULT SODaCXmlServerImpl::getPropertiesData(IN LPCTSTR itemID, IN LPCTSTR itemPath, IN WORD whatData, OUT SOCmnList<SODaCPropertyData> &propList)
264{
265 if (SOCLT_XML_CHECK_DEMO)
266 {
267 return E_FAIL;
268 }
269
270 SODaCServer* pDaCServer = getObjectDaCServer();
271 SOCmnPointer<SOCltHTTPConnection> httpCon = getHTTPConnection();
272 SOCmnPointer<SODaCXmlCall> httpCall;
273 HRESULT res;
274 BOOL getPropertyValues = ((whatData & SODAC_PROPDATA_VALUE) == SODAC_PROPDATA_VALUE);
275 BOOL getAllProperties;
276 SOCmnStringList propNameList;
277 SOCmnListPosition posP;
278 SODaCPropertyData* prop;
279 SODaCXmlRequestOptions reqOptions;
280 USES_CONVERSION;
281
282 if (!propList.isInit())
283 {
284 propList.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_DATA);
285 }
286
287 getAllProperties = propList.isEmpty();
288
289 if (!getAllProperties)
290 {
291 propNameList.create();
292 posP = propList.getStartPosition();
293 SOCmnString propertyName;
294
295 while (posP)
296 {
297 prop = propList.getNext(posP);
298 if (prop->m_name.getLength() > 0)
299 {
300 propNameList.add(prop->m_name);
301 }
302 else
303 {
304 SODaCProperty::getStandardPropertyNameByID(prop->m_propertyID, propertyName);
305 propNameList.add(propertyName);
306 }
307 }
308 }
309
310 SOCMN_ALLOCATE_OBJECT_C(httpCall, SODaCXmlCall(httpCon), (SODaCXmlCall*));
311 httpCall->getProperties(itemID, itemPath, NULL, NULL, getPropertyValues, getAllProperties, &propNameList, fillRequestOptions(&reqOptions));
312 pDaCServer->beforeInterfaceCall();
313 res = httpCon->invoke(httpCall, TRUE);
314 pDaCServer->afterInterfaceCall();
315
316 if (SUCCEEDED(res))
317 {
318 SOCmnPointer<SODaCXmlResponse> response;
319 res = httpCall->getGetPropertiesResponse(&response);
320
321 if (SUCCEEDED(res))
322 {
323 SOCmnStringList xmlOptions(response->getResponseOptions());
324 SOCmnList<SODaCXmlResponseData> xmlData(response->getResponseData());
325 SODaCXmlResponseData* pResData;
326 SOCmnListPosition pos;
327 pos = xmlData.getStartPosition();
328 posP = propList.getStartPosition();
329
330 while (pos)
331 {
332 pResData = xmlData.getNext(pos);
333
334 if (pResData->m_type == SODACXMLRESPONSEDATA_TYPE_PROPERTY)
335 {
336 SOCmnVariant propValue;
337
338 if (getAllProperties)
339 {
340 SOCMN_ALLOCATE_OBJECT(prop, SODaCPropertyData);
341 }
342 else
343 {
344 prop = propList.getNext(posP);
345
346 if (!prop)
347 {
348 return E_FAIL;
349 }
350 }
351
352 prop->m_name = pResData->m_name;
353 prop->m_propertyID = pDaCServer->getPropertyIDByName(prop->m_name);
354 prop->m_itemID = pResData->m_itemName;
355 prop->m_itemPath = pResData->m_itemPath;
356 prop->m_description = pResData->m_description;
357 SOCmnVariant::getVARTYPE(pResData->m_value.m_datatype, prop->m_datatype);
358
359 if (propValue.initFromXMLValue(&pResData->m_value))
360 {
361 if ((prop->m_propertyID == 1) && (propValue.vt == VT_BSTR)) // datatype
362 {
363 SOCmnString propDT;
364 propValue.getBSTR(propDT);
365
366 if (response->includesNamespacePrefix(propDT))
367 {
368 // property name with namespace
369 response->expandNamespace(propDT);
370 response->removeOPCXMLDANamespace(propDT); // remove default OPC XML DA NS
371 propValue.setBSTR(propDT);
372 }
373 }
374
375 prop->m_value = propValue;
376 }
377
378 if (getAllProperties)
379 {
380 propList.add(prop);
381 }
382 }
383 else if (pResData->m_type == SODACXMLRESPONSEDATA_TYPE_ELEMENT)
384 {
385 // only get error code
386 res = httpCall->getErrorCodeFromXML(pResData->m_resultID);
387 }
388 }
389 }
390 }
391
392 _TRACEO_LEVEL_FROM_RES(res, SOCMNTRACE_G_OPC_CALL_C, pDaCServer, (_T("GetProperties [0x%lX]"), res));
393 return res;
394}

396HRESULT SODaCXmlServerImpl::read(IN DWORD count, IN SOCmnString* pItemIDs, IN SOCmnString* pItemPaths, IN DWORD maxAge, OUT SOCmnVariant* pValues, OUT WORD* pQualities, OUT SOCmnDateTime* pTimestamps, OUT HRESULT* pErrors)
397{
398 if (SOCLT_XML_CHECK_DEMO)
399 {
400 return E_FAIL;
401 }
402
403 SODaCServer* pDaCServer = getObjectDaCServer();
404 SOCmnPointer<SOCltHTTPConnection> httpCon = getHTTPConnection();
405 SOCmnPointer<SODaCXmlCall> httpCall;
406 HRESULT res;
407 SODaCXmlRequestOptions reqOptions;
408 reqOptions.m_returnItemTime = TRUE;
409 SOCMN_ALLOCATE_OBJECT_C(httpCall, SODaCXmlCall(httpCon), (SODaCXmlCall*));
410 httpCall->read(count, pItemIDs, pItemPaths, maxAge, fillRequestOptions(&reqOptions));
411 pDaCServer->beforeInterfaceCall();
412 res = httpCon->invoke(httpCall, TRUE);
413 pDaCServer->afterInterfaceCall();
414
415 if (SUCCEEDED(res))
416 {
417 SOCmnPointer<SODaCXmlResponse> response;
418 res = httpCall->getReadResponse(&response);
419
420 if (SUCCEEDED(res))
421 {
422 SOCmnStringList xmlOptions(response->getResponseOptions());
423 SOCmnList<SODaCXmlResponseData> xmlData(response->getResponseData());
424 SODaCXmlResponseData* pResData;
425 SOCmnListPosition pos;
426 SOCmnVariant itemVal;
427 DWORD i = 0;
428 pos = xmlData.getStartPosition();
429
430 while (pos)
431 {
432 pResData = xmlData.getNext(pos);
433
434 if (pResData->m_type == SODACXMLRESPONSEDATA_TYPE_ITEM)
435 {
436 if (pResData->m_resultID.isEmpty())
437 {
438 if (pValues)
439 {
440 itemVal.clear();
441 itemVal.initFromXMLValue(&pResData->m_value);
442 itemVal.copyTo(pValues[i]);
443 }
444
445 if (pQualities)
446 {
447 pQualities[i] = httpCall->getDCOMQuality((LPCTSTR)pResData->m_quality, (LPCTSTR)pResData->m_qualityLimit, (LPCTSTR)pResData->m_qualityVendor);
448 }
449
450 if (pTimestamps)
451 {
452 pTimestamps[i].setXMLDateTime((LPCTSTR)pResData->m_timestamp);
453 }
454
455 if (pErrors)
456 {
457 pErrors[i] = S_OK;
458 }
459 }
460 else
461 {
462 if (pErrors)
463 {
464 pErrors[i] = httpCall->getErrorCodeFromXML(pResData->m_resultID);
465 }
466
467 res = S_FALSE;
468 }
469
470 i++;
471 }
472 }
473 }
474 }
475
476 _TRACEO_LEVEL_FROM_RES(res, SOCMNTRACE_G_OPC_CALL_C, pDaCServer, (_T("Read [0x%lX]"), res));
477 return res;
478}

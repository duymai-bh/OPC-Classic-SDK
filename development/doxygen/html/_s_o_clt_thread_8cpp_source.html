54// cleanUp

56void SOCltWorkerThread::cleanUp(void)
57{
58 // Task being currently in the queue will never get
59 // executeted. Call reportCompletion so that objects
60 // waiting on a task get signaled.
61 SOCmnList<SOCltTask> queue(m_taskQueue);
62 SOCmnListPosition pos = queue.getStartPosition();
63
64 while (pos)
65 {
66 SOCltTask* task = queue.getNext(pos);
67 task->setResult(E_ABORT);
68 task->reportCompletion();
69 }
70
71 queue.removeAll();
72 SOCmnList<SOCltCyclicTask> queue_cyc(m_cyclicQueue);
73 pos = queue_cyc.getStartPosition();
74
75 while (pos)
76 {
77 SOCltCyclicTask* task = queue_cyc.getNext(pos);
78 task->setResult(E_ABORT);
79 task->reportCompletion();
80 }
81
82 queue_cyc.removeAll();
83}

63BOOL CComboEdit::PreTranslateMessage(MSG* pMsg)
64{
65 // Make sure that the keystrokes continue to the appropriate handlers
66 if (pMsg->message == WM_KEYDOWN || pMsg->message == WM_KEYUP)
67 {
68 ::TranslateMessage(pMsg);
69 ::DispatchMessage(pMsg);
70 return TRUE;
71 }
72
73 // Catch the Alt key so we don't choke if focus is going to an owner drawn button
74 if (pMsg->message == WM_SYSCHAR)
75 {
76 return TRUE;
77 }
78
79 return CEdit::PreTranslateMessage(pMsg);
80}

157CInPlaceList::CInPlaceList(CWnd* pParent, CRect& rect, DWORD dwStyle, UINT nID,
158 int nRow, int nColumn,
159 COLORREF crFore, COLORREF crBack,
160 CStringArray& Items, CString sInitText,
161 UINT nFirstChar)
162{
163 m_crForeClr = crFore;
164 m_crBackClr = crBack;
165 m_nNumLines = 4;
166 m_sInitText = sInitText;
167 m_nRow = nRow;
168 m_nCol = nColumn;
169 m_nLastChar = 0;
170 m_bExitOnArrows = FALSE; //(nFirstChar != VK_LBUTTON); // If mouse click brought us here,
171 // Create the combobox
172 DWORD dwComboStyle = WS_BORDER | WS_CHILD | WS_VISIBLE | WS_VSCROLL |
173 CBS_AUTOHSCROLL | dwStyle;
174 int nHeight = rect.Height();
175 rect.bottom = rect.bottom + m_nNumLines * nHeight + ::GetSystemMetrics(SM_CYHSCROLL);
176
177 if (!Create(dwComboStyle, rect, pParent, nID))
178 {
179 return;
180 }
181
182 // Add the strings
183 for (int i = 0; i < Items.GetSize(); i++)
184 {
185 AddString(Items[i]);
186 }
187
188 SetFont(pParent->GetFont());
189 SetItemHeight(-1, nHeight);
190 int nMaxLength = GetCorrectDropWidth();
191 /*
192 if (nMaxLength > rect.Width())
193 rect.right = rect.left + nMaxLength;
194 // Resize the edit window and the drop down window
195 MoveWindow(rect);
196 */
197 SetDroppedWidth(nMaxLength);
198 SetHorizontalExtent(0); // no horz scrolling
199
200 // Set the initial text to m_sInitText
201 if (::IsWindow(m_hWnd) && SelectString(-1, m_sInitText) == CB_ERR)
202 {
203 SetWindowText(m_sInitText); // No text selected, so restore what was there before
204 }
205
206 ShowDropDown();
207
208 // Subclass the combobox edit control if style includes CBS_DROPDOWN
209 if ((dwStyle & CBS_DROPDOWNLIST) != CBS_DROPDOWNLIST)
210 {
211 m_edit.SubclassDlgItem(IDC_COMBOEDIT, this);
212 SetFocus();
213
214 switch (nFirstChar)
215 {
216 case VK_LBUTTON:
217 case VK_RETURN:
218 m_edit.SetSel((int)_tcslen(m_sInitText), -1);
219 return;
220
221 case VK_BACK:
222 m_edit.SetSel((int)_tcslen(m_sInitText), -1);
223 break;
224
225 case VK_DOWN:
226 case VK_UP:
227 case VK_RIGHT:
228 case VK_LEFT:
229 case VK_NEXT:
230 case VK_PRIOR:
231 case VK_HOME:
232 case VK_END:
233 m_edit.SetSel(0, -1);
234 return;
235
236 default:
237 m_edit.SetSel(0, -1);
238 }
239
240 SendMessage(WM_CHAR, nFirstChar);
241 }
242 else
243 {
244 SetFocus();
245 }
246}

252void CInPlaceList::EndEdit()
253{
254 CString str;
255
256 if (::IsWindow(m_hWnd))
257 {
258 GetWindowText(str);
259 }
260
261 // Send Notification to parent
262 GV_DISPINFO dispinfo;
263 dispinfo.hdr.hwndFrom = GetSafeHwnd();
264 dispinfo.hdr.idFrom = GetDlgCtrlID();
265 dispinfo.hdr.code = GVN_ENDLABELEDIT;
266 dispinfo.item.mask = LVIF_TEXT | LVIF_PARAM;
267 dispinfo.item.row = m_nRow;
268 dispinfo.item.col = m_nCol;
269 dispinfo.item.strText = str;
270 dispinfo.item.lParam = (LPARAM) m_nLastChar;
271 CWnd* pOwner = GetOwner();
272
273 if (IsWindow(pOwner->GetSafeHwnd()))
274 {
275 pOwner->SendMessage(WM_NOTIFY, GetDlgCtrlID(), (LPARAM)&dispinfo);
276 }
277
278 // Close this window (PostNcDestroy will delete this)
279 if (::IsWindow(m_hWnd))
280 {
281 PostMessage(WM_CLOSE, 0, 0);
282 }
283}

285int CInPlaceList::GetCorrectDropWidth()
286{
287 const int nMaxWidth = 200; // don't let the box be bigger than this
288 // Reset the dropped width
289 int nNumEntries = GetCount();
290 int nWidth = 0;
291 CString str;
292 CClientDC dc(this);
293 int nSave = dc.SaveDC();
294 dc.SelectObject(GetFont());
295 int nScrollWidth = ::GetSystemMetrics(SM_CXVSCROLL);
296
297 for (int i = 0; i < nNumEntries; i++)
298 {
299 GetLBText(i, str);
300 int nLength = dc.GetTextExtent(str).cx + nScrollWidth;
301 nWidth = max(nWidth, nLength);
302 }
303
304 // Add margin space to the calculations
305 nWidth += dc.GetTextExtent(_T("0")).cx;
306 dc.RestoreDC(nSave);
307 nWidth = min(nWidth, nMaxWidth);
308 return nWidth;
309 //SetDroppedWidth(nWidth);
310}

477// Override draw so that when the cell is selected, a drop arrow is shown in the RHS.
478BOOL CGridCellCombo::Draw(CDC* pDC, int nRow, int nCol, CRect rect, BOOL bEraseBkgnd /*=TRUE*/)
479{
480#ifdef _WIN32_WCE
481 return CGridCell::Draw(pDC, nRow, nCol, rect, bEraseBkgnd);
482#else
483
484 // Cell selected?
485 //if ( !IsFixed() && IsFocused())
486 if (GetGrid()->IsCellEditable(nRow, nCol) && !IsEditing())
487 {
488 // Get the size of the scroll box
489 CSize sizeScroll(GetSystemMetrics(SM_CXVSCROLL), GetSystemMetrics(SM_CYHSCROLL));
490
491 // enough room to draw?
492 if (sizeScroll.cy < rect.Width() && sizeScroll.cy < rect.Height())
493 {
494 // Draw control at RHS of cell
495 CRect ScrollRect = rect;
496 ScrollRect.left = rect.right - sizeScroll.cx;
497 ScrollRect.bottom = rect.top + sizeScroll.cy;
498 // Do the draw
499 pDC->DrawFrameControl(ScrollRect, DFC_SCROLL, DFCS_SCROLLDOWN);
500 // Adjust the remaining space in the cell
501 rect.right = ScrollRect.left;
502 }
503 }
504
505 CString strTempText = GetText();
506
507 if (IsEditing())
508 {
509 SetText(_T(""));
510 }
511
512 // drop through and complete the cell drawing using the base class' method
513 BOOL bResult = CGridCell::Draw(pDC, nRow, nCol, rect, bEraseBkgnd);
514
515 if (IsEditing())
516 {
517 SetText(strTempText);
518 }
519
520 return bResult;
521#endif
522}

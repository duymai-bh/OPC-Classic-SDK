47void SOAeCEvent::copyData(IN const ONEVENTSTRUCT& src)
48{
49 USES_CONVERSION;
50 // Data valid for all events
51 m_source = W2T(src.szSource);
52 m_ftTime = src.ftTime;
53 m_message = W2T(src.szMessage);
54 m_eventType = src.dwEventType;
55 m_eventCategory = src.dwEventCategory;
56 m_severity = src.dwSeverity;
57 m_dwCookie = src.dwCookie;
58 m_wReserved = src.wReserved;
59 m_dwNumEventAttrs = src.dwNumEventAttrs;
60 SOCmnList<SOCmnVariant> list(m_listEventAttrs);
61 list.allocateBlock(src.dwNumEventAttrs);
62
63 for (DWORD i = 0 ; i < src.dwNumEventAttrs ; i++)
64 {
65 list.add((SOCmnVariant*)(src.pEventAttributes + i));
66 }
67
68 // Following members are valid for conditional
69 // events only
70
71 if (src.dwEventType == OPC_CONDITION_EVENT)
72 {
73 m_changeMask = src.wChangeMask;
74 m_newState = src.wNewState;
75 m_conditionName = W2T(src.szConditionName);
76 m_subconditionName = W2T(src.szSubconditionName);
77 m_quality = src.wQuality;
78 m_bAckRequired = src.bAckRequired;
79 m_ftActiveTime = src.ftActiveTime;
80 }
81
82 m_actorID = W2T(src.szActorID);
83}

221void SOAeCTaskAckEvent::executeSingle(void)
222{
223 HRESULT res = S_OK;
224 SOAeCServer* server = (SOAeCServer*)(SOCltElement*)m_element;
225 CComPtr<IOPCEventServer> pSrv;
226 res = server->queryInterface(IID_IOPCEventServer, (void**)&pSrv);
227
228 if (SUCCEEDED(res))
229 {
230 USES_CONVERSION;
231 LPWSTR wszAckID = m_ackID.isEmpty() ? L"" : T2W((LPTSTR)(LPCTSTR)m_ackID);
232 LPWSTR wszComment = m_comment.isEmpty() ? L"" : T2W((LPTSTR)(LPCTSTR)m_comment);
233 LPWSTR wszSrc = T2W((LPTSTR)(LPCTSTR)m_eventAE->m_source);
234 LPWSTR wszCondName = T2W((LPTSTR)(LPCTSTR)m_eventAE->m_conditionName);
235 SOCltAutoMemory<HRESULT> pErr;
236 server->beforeInterfaceCall();
237 res = pSrv->AckCondition(1, wszAckID, wszComment, &wszSrc , &wszCondName,
238 &m_eventAE->m_ftActiveTime, &m_eventAE->m_dwCookie, &pErr);
239 server->afterInterfaceCall();
240 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, (_T("IOPCEventServer::AckCondition [0x%X]"), res));
241
242 if (SUCCEEDED(res))
243 {
244 m_eventAE->m_ackRes = pErr[0];
245 }
246
247 INVOKE_CALLBACK_ON_PTR((SOAeCEvent*)m_eventAE, SOAeCEvent, onEventAcknowledged)
248 }
249
250 setResult(res);
251 return;
252}

258void SOAeCTaskAckEvent::executeMultiple(void)
259{
260 HRESULT res = S_OK;
261 // Use the first in list to find out the server
262 SOAeCServer* srv = (SOAeCServer*)(SOCltElement*)m_element;
263 CComPtr<IOPCEventServer> pSrv;
264 res = srv->queryInterface(IID_IOPCEventServer, (void**)&pSrv);
265
266 if (SUCCEEDED(res))
267 {
268 USES_CONVERSION;
269 LPWSTR wszAckID = m_ackID.isEmpty() ? L"" : T2W((LPTSTR)(LPCTSTR)m_ackID);
270 LPWSTR wszComment = m_comment.isEmpty() ? L"" : T2W((LPTSTR)(LPCTSTR)m_comment);
271 DWORD dwCount = m_listEvents.getCount();
272 LPWSTR* wszSources = (LPWSTR*)_alloca(dwCount * sizeof(LPWSTR*));
273 LPWSTR* wszConditions = (LPWSTR*)_alloca(dwCount * sizeof(LPWSTR*));
274 FILETIME* ftTimes = (FILETIME*)_alloca(dwCount * sizeof(FILETIME));
275 DWORD* dwCookies = (DWORD*)_alloca(dwCount * sizeof(DWORD));
276 SOCmnListPosition pos = m_listEvents.getStartPosition();
277 int i;
278
279 for (i = 0; pos; i++)
280 {
281 SOAeCEvent* event = m_listEvents.getNext(pos);
282 ftTimes[i] = event->m_ftActiveTime;
283 dwCookies[i] = event->m_dwCookie;
284 wszSources[i] = T2W((LPTSTR)(LPCTSTR)event->m_source);
285 wszConditions[i] = T2W((LPTSTR)(LPCTSTR)event->m_conditionName);
286 }
287
288 SOCltAutoMemory<HRESULT> pErr;
289 srv->beforeInterfaceCall();
290 res = pSrv->AckCondition(dwCount, wszAckID, wszComment, wszSources , wszConditions,
291 ftTimes, dwCookies, &pErr);
292 srv->afterInterfaceCall();
293 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OPC_CALL_C, (_T("IOPCEventServer::AckCondition [0x%X]"), res));
294
295 if (SUCCEEDED(res))
296 {
297 pos = m_listEvents.getStartPosition();
298 i = 0;
299
300 while (pos)
301 {
302 SOAeCEvent* event = m_listEvents.getNext(pos);
303 event->m_ackRes = pErr[i++];
304 INVOKE_CALLBACK_ON_PTR(event, SOAeCEvent, onEventAcknowledged)
305 }
306 }
307 }
308
309 setResult(res);
310 return;
311}

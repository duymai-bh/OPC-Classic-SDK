43// getObject

45// - get the toolkit object from it's description elements
46//
47// returns:
48// NULL - no valid object

50//
51SOCmnObject* SODaSWatchHandler::getObject(
52 IN LPCTSTR objSpace, // object space
53 IN LPCTSTR objType) // object type
54{
55 DWORD type = SOSrvWatch::convertObjectType(objType);
56
57 if (type == SOCMNOBJECT_TYPE_ROOT)
58 {
59 SOCmnPointer<SODaSEntry> entry = getSODaSEntry();
60
61 if (_tcscmp(objSpace, SOWATCH_SPACE_DA_OBJECT) == 0)
62 {
63 return entry->getObjectRoot();
64 }
65 else if (_tcscmp(objSpace, SOWATCH_SPACE_DA_NAMESPACE) == 0)
66 {
67 return entry->getNameSpaceRoot();
68 }
69 }
70
71 return NULL;
72} // getObject

115//-----------------------------------------------------------------------------
116// hasCacheObject (OPC DA)
117//
118// - checks if the item tag object has the cache object
119//
120// returns:
121// TRUE - tag has cache object
122// FALSE - tag has not cache object
123//
124BOOL SODaSWatchHandler::hasCacheObject(
125 IN SOCmnObject* tag,
126 IN SOCmnObject* cache,
127 OUT BOOL& uniqueCache)
128{
129 SOCmnPointer<SODaSCache> tagCache;
130 tagCache = ((SODaSItemTag*)tag)->getCache();
131
132 if (!tagCache)
133 {
134 return FALSE;
135 }
136
137 if (((SOCmnObject*)(SODaSCache*)cache) == ((SODaSCache*)tagCache))
138 {
139 SOCmnPointer<SODaSEntry> entry = ::getSODaSEntry();
140 SOCmnPointer<SODaSNameSpaceRoot> root = entry->getNameSpaceRoot();
141 uniqueCache = !root->getUseTagLinks();
142 return TRUE;
143 }
144
145 return FALSE;
146} // hasCacheObject

183//-----------------------------------------------------------------------------
184// getStatistics
185//
186// - get the statistics
187//
188// returns:
189// S_OK - got statistics
190// E_FAIL - error
191//
192HRESULT SODaSWatchHandler::getStatistics(
193 SOCmnStringEx& szXML)
194{
195 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRCYCREAD, m_statistics.cyclicRead);
196 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRCHANGEREPORT, m_statistics.reportRead);
197 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRACACHEREAD, m_statistics.asyncCacheRead);
198 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRADEVREAD, m_statistics.asyncDeviceRead);
199 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRSDEVREAD, m_statistics.syncDeviceRead);
200 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRSCACHEREAD, m_statistics.syncCacheRead);
201 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRAWRITE, m_statistics.asyncWrite);
202 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRSWRITE, m_statistics.syncWrite);
203 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRREADFAILED, m_statistics.failedReadOrBadQuality);
204 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_NRWRITEFAILED, m_statistics.failedWrite);
205 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_MAXREADTIME, m_statistics.maxTimeRead);
206 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_MAXWRITETIME, m_statistics.maxTimeWrite);
207 LONGLONG reads;
208 LONGLONG writes;
209 ULONG avReadTime = 0;
210 ULONG avWriteTime = 0;
211 reads = m_statistics.asyncDeviceRead + m_statistics.syncDeviceRead + m_statistics.cyclicRead;
212
213 if (reads)
214 {
215 avReadTime = (ULONG)(m_statistics.timeRead / reads);
216 }
217
218 writes = m_statistics.asyncWrite + m_statistics.syncWrite;
219
220 if (writes)
221 {
222 avWriteTime = (ULONG)(m_statistics.timeWrite / writes);
223 }
224
225 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_AVREADTIME, avReadTime);
226 SOSrvWatch::createElementText(szXML, SOWATCH_STAT_DA_AVWRITETIME, avWriteTime);
227 SOCmnPointer<SODaSEntry> entry = ::getSODaSEntry();
228 entry->getObjectCountStatistics(szXML);
229 memset(&m_statistics, 0, sizeof(SODaSWatchHandlerStatisticsData));
230 return S_OK;
231} // getStatistics

234void SODaSWatchHandler::doRequestStatistics(IN SODaSRequest* req, IN DWORD completionSpan)
235{
236 SOCmnPointer<SOSrvEntry> entry = ::getSOSrvEntry();
237 SOCmnList<SOSrvWatch> watchList(entry->getWatchList());
238 SOCmnListPosition pos = watchList.getStartPosition();
239 SOSrvWatch* watch;
240 SOCmnPointer<SODaSWatchHandler> DAHandler;
241
242 while (pos)
243 {
244 watch = watchList.getNext(pos);
245 DAHandler = (SODaSWatchHandler*)watch->getDAHandler();
246
247 if (DAHandler.isNotNull())
248 {
249 SOCmnSingleLock<SOCmnSync> lock(watch->getStatisticsLock());
250 HRESULT reqRes = req->getResult();
251 SODaSRequest::requestOperation operation = req->getOperation();
252 BOOL deviceRead = FALSE;
253
254 if ((SUCCEEDED(reqRes)) && (operation == SODaSRequest::read))
255 {
256 if (req->getType() != SODaSRequest::report)
257 {
258 SOCmnPointer<SODaSCache> cache = req->getCache();
259
260 if (cache.isNotNull())
261 {
262 if ((cache->getQuality(req) & OPC_QUALITY_GOOD) != OPC_QUALITY_GOOD)
263 {
264 reqRes = E_FAIL;
265 }
266 }
267 else
268 {
269 reqRes = E_FAIL;
270 }
271 }
272 else if ((req->getReportQuality() & OPC_QUALITY_GOOD) != OPC_QUALITY_GOOD)
273 {
274 reqRes = E_FAIL;
275 }
276 }
277
278 if (SUCCEEDED(reqRes))
279 {
280 switch (req->getType())
281 {
282 case SODaSRequest::cyclic:
283 DAHandler->m_statistics.cyclicRead++;
284 deviceRead = TRUE;
285 break;
286
287 case SODaSRequest::report:
288 DAHandler->m_statistics.reportRead++;
289 break;
290
291 case SODaSRequest::asyncDevice:
292 case SODaSRequest::asyncMaxAge:
293 if (operation == SODaSRequest::read)
294 {
295 DAHandler->m_statistics.asyncDeviceRead++;
296 deviceRead = TRUE;
297 }
298 else
299 {
300 DAHandler->m_statistics.asyncWrite++;
301 }
302
303 break;
304
305 case SODaSRequest::asyncCache:
306 DAHandler->m_statistics.asyncCacheRead++;
307 break;
308
309 case SODaSRequest::serverDevice:
310 case SODaSRequest::serverMaxAge:
311 case SODaSRequest::syncDevice:
312 case SODaSRequest::syncMaxAge:
313 if (operation == SODaSRequest::read)
314 {
315 DAHandler->m_statistics.syncDeviceRead++;
316 deviceRead = TRUE;
317 }
318 else
319 {
320 DAHandler->m_statistics.syncWrite++;
321 }
322
323 break;
324
325 case SODaSRequest::syncCache:
326 DAHandler->m_statistics.syncCacheRead++;
327 break;
328
329 default:
330 break;
331 }
332
333 if (deviceRead)
334 {
335 DAHandler->m_statistics.timeRead += completionSpan;
336
337 if (completionSpan > DAHandler->m_statistics.maxTimeRead)
338 {
339 DAHandler->m_statistics.maxTimeRead = completionSpan;
340 }
341 }
342
343 if (operation == SODaSRequest::write)
344 {
345 DAHandler->m_statistics.timeWrite += completionSpan;
346
347 if (completionSpan > DAHandler->m_statistics.maxTimeWrite)
348 {
349 DAHandler->m_statistics.maxTimeWrite = completionSpan;
350 }
351 }
352 }
353 else
354 {
355 if (operation == SODaSRequest::write)
356 {
357 DAHandler->m_statistics.failedWrite++;
358 }
359 else
360 {
361 DAHandler->m_statistics.failedReadOrBadQuality++;
362 }
363 }
364 }
365 }
366}

238void OpcServer::handleRequests(void)
239{
240 m_requestsJanitor.lock();
241 {
242 size_t count = m_requests.size();
243 DaRequest *request = NULL;
244
245 for (size_t i = 0; i < count; i++)
246 {
247 request = m_requests[i];
248
249 if ((request != NULL) && (request->progressRequestState(EnumRequestState_PENDING, EnumRequestState_INPROGRESS)))
250 {
251 MyDaAddressSpaceElement* element = (MyDaAddressSpaceElement*)request->getAddressSpaceElement();
252
253 if (element == NULL)
254 {
255 request->setResult(E_FAIL);
256 }
257 else
258 {
259 if (request->getTransactionType() == EnumTransactionType_READ)
260 {
261 if (request->getPropertyId() == 0)
262 {
263 // get address space element value take the toolkit cache value
264 ValueQT cacheValue;
265 element->getCacheValue(cacheValue);
266 request->setValue(cacheValue);
267 request->setResult(S_OK);
268 }
269 else
270 {
271 // the element's property will handle this request
272 element->getPropertyValue(request);
273 } // end if ... else
274 }
275 else
276 {
277 // EnumTransactionType_WRITE
278 ValueQT* pValue = request->getValue();
279 long result = element->valueChanged(*pValue);
280 request->setResult(result);
281 } // end if ... else
282 } // end if ... else
283 }
284 request->complete();
285 } // end for
286
287 // release all requests
288 m_requests.clear();
289 }
290 m_requestsJanitor.unlock();
291} // end handleRequests

71void BrowseAddressSpace(IN OTCObjectHandle sessionHandle, IN OTCObjectHandle addressSpaceElementHandle, IN DWORD indent)
72{
73 OTCAddressSpaceBrowseOptions browseOptions;
74 OTCAddressSpaceElementData* pAddressSpaceElement;
75 DWORD addressSpaceElementDataCount;
76 DWORD i;
77 // browse all child leaves
78 memset(&browseOptions, 0, sizeof(OTCAddressSpaceBrowseOptions));
79 browseOptions.m_type = OT_ADDRESSSPACETYPE_LEAF;
80 pAddressSpaceElement = NULL;
81
82 if (SUCCEEDED(OTCBrowseAddressSpace(sessionHandle, addressSpaceElementHandle, NULL, NULL, &browseOptions,
83 &addressSpaceElementDataCount, &pAddressSpaceElement, NULL)))
84 {
85 for (i = 0; i < addressSpaceElementDataCount; i++)
86 {
87 PrintChildren(&pAddressSpaceElement[i], indent);
88 OTCReleaseAddressSpaceElement(pAddressSpaceElement[i].m_objectHandle, FALSE);
89 }
90 }
91
92 for (i = 0; i < addressSpaceElementDataCount; i++)
93 {
94 OTFreeMemory(pAddressSpaceElement[i].m_name);
95 }
96
97 if (pAddressSpaceElement)
98 {
99 OTFreeMemory(pAddressSpaceElement);
100 }
101
102 // browse all child nodes
103 browseOptions.m_type = OT_ADDRESSSPACETYPE_BRANCH;
104 pAddressSpaceElement = NULL;
105
106 if (SUCCEEDED(OTCBrowseAddressSpace(sessionHandle, addressSpaceElementHandle, NULL, NULL, &browseOptions,
107 &addressSpaceElementDataCount, &pAddressSpaceElement, NULL)))
108 {
109 for (i = 0; i < addressSpaceElementDataCount; i++)
110 {
111 PrintChildren(&pAddressSpaceElement[i], indent);
112 BrowseAddressSpace(sessionHandle, pAddressSpaceElement[i].m_objectHandle, indent + 1);
113 OTCReleaseAddressSpaceElement(pAddressSpaceElement[i].m_objectHandle, FALSE);
114 }
115 }
116
117 for (i = 0; i < addressSpaceElementDataCount; i++)
118 {
119 OTFreeMemory(pAddressSpaceElement[i].m_name);
120 }
121
122 if (pAddressSpaceElement)
123 {
124 OTFreeMemory(pAddressSpaceElement);
125 }
126}

128void PrintChildren(OTCAddressSpaceElementData* pAddressSpaceElement, DWORD indent)
129{
130 TCHAR szNameAndIndent[41];
131 LPCTSTR szType;
132
133 switch (pAddressSpaceElement->m_type)
134 {
135 case OT_ADDRESSSPACETYPE_BRANCH:
136 szType = _T("B ");
137 break;
138
139 case OT_ADDRESSSPACETYPE_LEAF:
140 szType = _T("L");
141 break;
142 }
143
144 memset(szNameAndIndent, _T(' '), indent * sizeof(TCHAR));
145 _tcsncpy(szNameAndIndent + indent, pAddressSpaceElement->m_name, 40 - indent);
146 szNameAndIndent[40] = _T('\0');
147 _tprintf(_T("%-40.40s | %1.1s\n"), szNameAndIndent, szType);
148}

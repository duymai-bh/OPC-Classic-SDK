52// - handle console control events
53//
54// returns:
55// TRUE - handled event
56// FALSE - event not handled
57//
58BOOL WINAPI controlHandler(
59 IN DWORD crtlType) // control type
60{
61 switch (crtlType)
62 {
63 case CTRL_BREAK_EVENT: // use Ctrl+C or Ctrl+Break to end server
64 case CTRL_C_EVENT:
65 {
66 if (g_endEvent)
67 {
68 SetEvent(g_endEvent);
69 }
70
71 return TRUE;
72 }
73 break;
74 }
75
76 return FALSE;
77} // controlHandler

132int _tmain(int argc, _TCHAR* argv[])
133{
134 g_endEvent = CreateEvent(NULL, FALSE, FALSE, NULL);
135 SetConsoleCtrlHandler(controlHandler, TRUE);
136 OTCExecutionOptions executionOptions;
137 OTCObjectHandle sessionHandle;
138 OTCObjectHandle subscriptionHandle;
139 BOOL end = FALSE;
140 DWORD waitRet;
141 LONG res;
142 executionOptions.m_executionType = OTC_EXECUTIONTYPE_ASYCHRONOUS;
143 executionOptions.m_executionContext = 0;
144 OTActivate(OTC_FEATURE_AE_CLIENT, _T("SAMPLE"));
145
146 if (SUCCEEDED(OTCInitialize(&g_initData)))
147 {
148 OTCAdvise(&g_callbackFunctions);
149 OTCAddAESession(_T("opcae:///Softing.OPCToolboxDemo_ServerAE.1/{2E565243-B238-11D3-842D-0008C779D775}"), 0, &sessionHandle);
150 OTCAddAESubscription(sessionHandle, 1, &subscriptionHandle);
151 OTCChangeTargetState(sessionHandle, OTC_STATE_CONNECTED, TRUE);
152
153 if (S_OK == OTCPerformStateTransition(sessionHandle, TRUE, NULL))
154 {
155 usage();
156
157 while (!end)
158 {
159 waitRet = WaitForSingleObject(g_endEvent, 100);
160
161 if (waitRet == WAIT_OBJECT_0)
162 {
163 end = TRUE;
164 }
165 else
166 {
167 if (_kbhit())
168 {
169 int c;
170 c = _getch();
171
172 switch (c)
173 {
174 case _T('A'):
175 OTCChangeTargetState(sessionHandle, OTC_STATE_ACTIVE, TRUE);
176 OTCPerformStateTransition(sessionHandle, TRUE, &executionOptions);
177 executionOptions.m_executionContext++;
178 break;
179
180 case _T('a'):
181 OTCChangeTargetState(sessionHandle, OTC_STATE_ACTIVE, TRUE);
182 OTCPerformStateTransition(sessionHandle, TRUE, NULL);
183 break;
184
185 case _T('C'):
186 OTCChangeTargetState(sessionHandle, OTC_STATE_CONNECTED, TRUE);
187 OTCPerformStateTransition(sessionHandle, TRUE, &executionOptions);
188 executionOptions.m_executionContext++;
189 break;
190
191 case _T('c'):
192 OTCChangeTargetState(sessionHandle, OTC_STATE_CONNECTED, TRUE);
193 OTCPerformStateTransition(sessionHandle, TRUE, NULL);
194 break;
195
196 case _T('D'):
197 OTCChangeTargetState(sessionHandle, OTC_STATE_DISCONNECTED, TRUE);
198 OTCPerformStateTransition(sessionHandle, TRUE, &executionOptions);
199 executionOptions.m_executionContext++;
200 break;
201
202 case _T('d'):
203 OTCChangeTargetState(sessionHandle, OTC_STATE_DISCONNECTED, TRUE);
204 OTCPerformStateTransition(sessionHandle, TRUE, NULL);
205 break;
206
207 case _T('S'):
208 OTCGetServerStatus(sessionHandle, NULL, &executionOptions);
209 break;
210
211 case _T('s'):
212 {
213 OTCServerStatus serverStatus;
214 memset(&serverStatus, 0, sizeof(OTCServerStatus));
215
216 if (SUCCEEDED(res = OTCGetServerStatus(sessionHandle, &serverStatus, NULL)))
217 {
218 PrintServerStatus(&serverStatus);
219 }
220 else
221 {
222 _tprintf(_T("Get Status failed [0x%8.8X]\n"), res);
223 }
224
225 if (serverStatus.m_vendorInfo != NULL)
226 {
227 OTFreeMemory(serverStatus.m_vendorInfo);
228 }
229
230 if (serverStatus.m_productVersion != NULL)
231 {
232 OTFreeMemory(serverStatus.m_productVersion);
233 }
234
235 if (serverStatus.m_statusInfo != NULL)
236 {
237 OTFreeMemory(serverStatus.m_statusInfo);
238 }
239
240 if (serverStatus.m_supportedLCIDs != NULL)
241 {
242 OTFreeMemory(serverStatus.m_supportedLCIDs);
243 }
244 }
245 break;
246
247 case _T('M'):
248 case _T('m'):
249 if (SUCCEEDED(res = OTCActivateConnectionMonitor(sessionHandle, TRUE, 5000, 10, 10000, 300000)))
250 {
251 _tprintf(_T("Activated connection monitor\n"));
252 }
253 else
254 {
255 _tprintf(_T("Activate connection monitor failed [0x%8.8X]\n"), res);
256 }
257
258 break;
259
260 case _T('N'):
261 case _T('n'):
262 if (SUCCEEDED(res = OTCActivateConnectionMonitor(sessionHandle, FALSE, 0, 0, 0, 0)))
263 {
264 _tprintf(_T("Deactivated connection monitor\n"));
265 }
266 else
267 {
268 _tprintf(_T("Deactivate connection monitor failed [0x%8.8X]\n"), res);
269 }
270
271 break;
272
273 case _T('?'):
274 case _T('U'):
275 case _T('u'):
276 usage();
277 break;
278
279 case _T('E'):
280 case _T('e'):
281 case _T('Q'):
282 case _T('q'):
283 end = TRUE;
284 break;
285 }
286 }
287 }
288 }
289 }
290
291 OTCTerminate();
292 }
293
294 return 0;
295}

116//-----------------------------------------------------------------------------
117// findBranch
118//
119// - find branch below the branch
120//
121// returns:
122// pointer to the branch - branch found
123// NULL - no branch found
124//
125SOCmnElement* SODaCServer::findBranch(
126 IN LPCTSTR name, // name of branch
127 IN OPTIONAL BOOL deep) // deep search or only direct child
128{
129 SOCmnElement* branch = NULL;
130 SOCmnString path;
131 LPTSTR child = NULL;
132 LPTSTR remainingPath = NULL;
133
134 if (!name)
135 {
136 return NULL;
137 }
138
139 path = name;
140 reducePath(path, child, remainingPath, TRUE); // slip path
141
142 if ((!deep) && (remainingPath)) // no deep search and remaining path
143 {
144 return NULL; // -> name can't be a direct child of this node
145 }
146
147 SOCmnList<SOCmnElement> branchList(getBranchList());
148
149 if (!!branchList)
150 {
151 SOCmnPointer<SOCmnElement> search;
152 SOCmnElement* pSearch;
153 SOCmnListPosition pos;
154 pos = branchList.getStartPosition();
155
156 while (pos)
157 {
158 pSearch = branchList.getNext(pos);
159
160 if (pSearch->hasKey(child))
161 {
162 search = pSearch;
163 search.addRef();
164 break;
165 }
166 }
167
168 if (!!search)
169 {
170 if ((!deep) || (!remainingPath))
171 {
172 // no deep search or no remaining path
173 branch = search;
174 branch->addRef();
175 }
176 else
177 {
178 // deep search and remaining path
179 branch = search->findBranch(remainingPath, TRUE);
180 }
181 }
182 }
183
184 return branch;
185} // findBranch

188//-----------------------------------------------------------------------------
189// doStateTransition
190//
191// - do the state transition for the server if the target state differs from the
192// actual state of the server
193//
194// returns:
195// TRUE - transit to new state
196// FALSE - error during state change
197//
198HRESULT SODaCServer::doStateTransition(BOOL deep, IN BYTE stateDeep)
199{
200 HRESULT res = S_OK;
201 BYTE targetState = getTargetObjectState();
202
203 if (disconnected == targetState)
204 {
205 doRecurseChildren(getBranchList(), disconnected);
206 doRecurseChildren(getLeafList(), disconnected);
207 doRecurseChildren(getAdditionalList(), disconnected);
208 }
209
210 res = doStateTransitionImpl();
211
212 if (deep && SUCCEEDED(res) && disconnected != targetState)
213 {
214 if (S_OK != doRecurseChildren(getBranchList(), stateDeep))
215 {
216 res = S_FALSE;
217 }
218
219 if (S_OK != doRecurseChildren(getLeafList(), stateDeep))
220 {
221 res = S_FALSE;
222 }
223
224 if (S_OK != doRecurseChildren(getAdditionalList(), stateDeep))
225 {
226 res = S_FALSE;
227 }
228 }
229
230 return res;
231} // doStateTransition

239//-----------------------------------------------------------------------------
240// addGroup
241//
242// - insert group to the group list of the server. The group is connected by
243// calling AddGroup of the IOPCServer interface
244//
245// returns:
246// pointer to the new group
247// NULL - error creating the group
248//
249SODaCGroup* SODaCServer::addGroup(
250 IN LPCTSTR groupName, // group name
251 IN DWORD reqUpdateRate, // requested update rate
252 IN OPTIONAL FLOAT* deadband,// deadband
253 IN OPTIONAL LONG* timeBias, // time zone
254 IN OPTIONAL DWORD LCID) // locale id
255{
256 // create group object
257 SOCmnPointer<SODaCEntry> session = ::getSODaCEntry();
258 SOCmnPointer<SODaCCreator> creator = session->getCreator();
259 SODaCGroup* pCGrp = creator->createGroup(getProtocol(), this);
260 pCGrp->init(this);
261
262 // Set attributes
263 if (groupName)
264 {
265 pCGrp->setName(groupName);
266 }
267 else
268 {
269 SOCmnString grpName;
270 grpName.format(_T("%lu_%8.8lx_%lu"), reqUpdateRate, (DWORD)pCGrp->getOpcHandle(), GetTickCount());
271 pCGrp->setName(grpName);
272 }
273
274 pCGrp->setReqUpdateRate(reqUpdateRate);
275
276 if (timeBias)
277 {
278 pCGrp->setTimeBias(*timeBias);
279 }
280
281 if (deadband)
282 {
283 pCGrp->setDeadBand(*deadband);
284 }
285
286 if (LCID)
287 {
288 pCGrp->setLCID(LCID);
289 }
290
291 pCGrp->setConnectionType(SODaCGroup::ctAddGroup);
292
293 // Insert into the tree
294 if (!addBranch(pCGrp))
295 {
296 // group couldn't be added to the server; I delete the allocated memory of the group.
297 pCGrp->release();
298 pCGrp = NULL;
299 }
300
301 return pCGrp;
302} // addGroup

343BOOL SODaCServer::serialize(IN OUT SOCltStorage& ar, IN BOOL isStoring)
344{
345 BOOL ok = TRUE;
346 SOCmnList<SOCmnElement> list(getBranchList());
347 SODaCGroup* grp;
348
349 if (isStoring)
350 {
351 if (!getPersistence())
352 {
353 return TRUE;
354 }
355
356 // store own data
357 ar << getValidParameters();
358 ar << getServerName();
359 ar << getNodeName();
360 ar << getClientName();
361 ar << getLCID();
362 ar << getUserName();
363 ar << getPassword();
364 ar << getForcedOPCVersion();
365 serializeImpl(ar, isStoring);
366 // store children
367 DWORD dwCount = 0;
368 SOCmnListPosition pos = list.getStartPosition();
369
370 while (pos)
371 {
372 grp = (SODaCGroup*)list.getNext(pos);
373
374 if (grp->getPersistence())
375 {
376 dwCount++;
377 }
378 }
379
380 ar << dwCount;
381 pos = list.getStartPosition();
382
383 while (pos && ok)
384 {
385 grp = (SODaCGroup*)list.getNext(pos);
386
387 if (grp->getPersistence())
388 {
389 ok = grp->serialize(ar, isStoring);
390 }
391 }
392 }
393 else
394 {
395 // load own data
396 SOCmnPointer<SODaCEntry> entry = getSODaCEntry();
397#ifdef SOFEATURE_DCOM
398 CLSID clsid = CLSID_NULL;
399 DWORD clsctx = 0;
400#endif
401 SOCmnString nodeName;
402 SOCmnString clientName;
403 SOCmnString serverName;
404 LCID lcid = 0;
405 ar >> m_validParameter;
406 ar >> serverName;
407 setServerName(serverName);
408 ar >> nodeName;
409 setNodeName(nodeName);
410#ifdef SOFEATURE_DCOM
411
412 if (entry->getFileVersion() < MAKELONG(3, 4))
413 {
414 ar >> clsid;
415 setClassID(clsid);
416 ar >> clsctx;
417 setClsctx(clsctx);
418 }
419
420#endif
421 ar >> clientName;
422
423 if (m_validParameter & SOCLT_SRVPAR_CLIENT_NAME)
424 {
425 setClientName(clientName);
426 }
427
428 ar >> lcid;
429
430 if (m_validParameter & SOCLT_SRVPAR_LCID)
431 {
432 setLCID(lcid);
433 }
434
435 if (entry->getFileVersion() >= MAKELONG(1, 4))
436 {
437 SOCmnString userName;
438 SOCmnString password;
439 ar >> userName;
440 setUserName(userName);
441 ar >> password;
442 setPassword(password);
443 }
444
445 if (entry->getFileVersion() >= MAKELONG(9, 4))
446 {
447 WORD fov = 0;
448 ar >> fov;
449 forceOPCVersion(fov);
450 }
451
452 if (entry->getFileVersion() >= MAKELONG(3, 4))
453 {
454 serializeImpl(ar, isStoring);
455 }
456
457 // load childeren
458 DWORD dwCount = 0;
459 ar >> dwCount;
460
461 if (dwCount > 0)
462 {
463 while (dwCount-- && ok)
464 {
465 // create group
466 grp = addGroup(NULL, 0);
467
468 if (grp)
469 {
470 ok = grp->serialize(ar, isStoring);
471 grp->release();
472 }
473 }
474 }
475 }
476
477 return ok;
478} // serialize

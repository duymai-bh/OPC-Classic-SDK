11/*
12 * VARENUM usage key,
13 *
14 * * [V] - may appear in a VARIANT
15 * * [T] - may appear in a TYPEDESC
16 * * [P] - may appear in an OLE property set
17 * * [S] - may appear in a Safe Array
18 *
19 *
20 * VT_EMPTY [V] [P] nothing
21 * VT_NULL [V] [P] SQL style Null
22 * VT_I2 [V][T][P][S] 2 byte signed int
23 * VT_I4 [V][T][P][S] 4 byte signed int
24 * VT_R4 [V][T][P][S] 4 byte real
25 * VT_R8 [V][T][P][S] 8 byte real
26 * VT_CY [V][T][P][S] currency
27 * VT_DATE [V][T][P][S] date
28 * VT_BSTR [V][T][P][S] OLE Automation string
29 * VT_DISPATCH [V][T][P][S] IDispatch *
30 * VT_ERROR [V][T][P][S] SCODE
31 * VT_BOOL [V][T][P][S] True=-1, False=0
32 * VT_VARIANT [V][T][P][S] VARIANT *
33 * VT_UNKNOWN [V][T] [S] IUnknown *
34 * VT_DECIMAL [V][T] [S] 16 byte fixed point
35 * VT_RECORD [V] [P][S] user defined type
36 * VT_I1 [V][T][P][s] signed char
37 * VT_UI1 [V][T][P][S] unsigned char
38 * VT_UI2 [V][T][P][S] unsigned short
39 * VT_UI4 [V][T][P][S] unsigned short
40 * VT_I8 [T][P] signed 64-bit int
41 * VT_UI8 [T][P] unsigned 64-bit int
42 * VT_INT [V][T][P][S] signed machine int
43 * VT_UINT [V][T] [S] unsigned machine int
44 * VT_VOID [T] C style void
45 * VT_HRESULT [T] Standard return type
46 * VT_PTR [T] pointer type
47 * VT_SAFEARRAY [T] (use VT_ARRAY in VARIANT)
48 * VT_CARRAY [T] C style array
49 * VT_USERDEFINED [T] user defined type
50 * VT_LPSTR [T][P] null terminated string
51 * VT_LPWSTR [T][P] wide null terminated string
52 * VT_FILETIME [P] FILETIME
53 * VT_BLOB [P] Length prefixed bytes
54 * VT_STREAM [P] Name of the stream follows
55 * VT_STORAGE [P] Name of the storage follows
56 * VT_STREAMED_OBJECT [P] Stream contains an object
57 * VT_STORED_OBJECT [P] Storage contains an object
58 * VT_BLOB_OBJECT [P] Blob contains an object
59 * VT_CF [P] Clipboard format
60 * VT_CLSID [P] A Class ID
61 * VT_VECTOR [P] simple counted array
62 * VT_ARRAY [V] SAFEARRAY*
63 * VT_BYREF [V] void* for local use
64 * VT_BSTR_BLOB Reserved for system use
65 */

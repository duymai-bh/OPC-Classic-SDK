33SOCmnObject::~SOCmnObject(void)
34{
35 RELEASE_OPC_HANDLE(m_opcHandle);
36 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_OBJ_REF, (_T("free")))
37 _TRACE_DELETE_GENOBJ();
38 // The reference count (m_refCount) of an object MUST be 0 in the
39 // destuctor of the object. With a reference count greater than 0 here
40 // you will have some references pointing to an invalid object.
41 // Check for:
42 // - delete calls on objects
43 // - local created objects
44 _ASSERTION((m_refCount == 0), "The reference count of an object must be 0 in the destuctor of the object");
45 m_refCount = -1;
46 m_objectType = 0;
47}

67// - free object if no reference left
68//
69// returns:
70// reference count
71//
72LONG SOCmnObject::release(void)
73{
74 LONG refCount = interlockedDecrement(&m_refCount);
75 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OBJ_REF, (_T("release %li"), refCount))
76
77 if (refCount <= 0)
78 {
79 SOCMN_FREE_OBJECT_THIS // last reference -> free object
80 return 0;
81 }
82

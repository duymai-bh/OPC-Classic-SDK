212class SOCLT_EXPORT SOCltCallback : public SOCmnData
213{
214public:
215 SOCltCallback()
216 {
217 e = 0;
218 vtlb = 0;
219 type = 0;
220 }
221
222 virtual void execute(void) const
223 {
224 _ASSERTION(0 != vtlb || 0 != e , "Invalid callback-data");
225
226 switch (type)
227 {
228 default:
229 _ASSERTION(FALSE, "Invalid callback type");
230 break;
231
232 case 0 :
233 (vtlb->*f)();
234 break;
235
236 case 1 :
237 (vtlb->*f1.ferr)(f1.dw1, f1.dw2);
238 break;
239 }
240 }
241
242 DWORD type;
243 union // switch[type]
244 {
245 SOCLTELEMENT_CALLBACKTYPE f;
246
247 struct _f1
248 {
249 SOCLTELEMENT_CALLBACKTYPE_ERROR ferr;
250 DWORD dw1;
251 DWORD dw2;
252 } f1;
253 };
254
255 SOCmnObject* e; // pointer to object
256 SOCLTELEMENT_VTBL vtlb; // v-table, may be different from the default one
257 // if multiple inheritance is used
258 virtual ~SOCltCallback(void)
259 {
260 if (e)
261 {
262 e->release();
263 }
264 }
265}; // SOCltCallback

42//-----------------------------------------------------------------------------
43// lock
44//
45// - gain access to the synchronisation object
46//
47// returns:
48// TRUE - got access to the object
49//
50BOOL SOCmnSync::lock(IN OPTIONAL DWORD timeout)
51{
52#ifdef SOOS_LINUX
53 BOOL res;
54
55 if (timeout == INFINITE)
56 {
57 pthread_mutex_lock(&m_mutex);
58 res = TRUE;
59 }
60 else if (timeout == 0)
61 {
62 res = !pthread_mutex_trylock(&m_mutex);
63 }
64 else
65 {
66#ifndef SODEVENV_CYGWIN
67 struct timespec ts;
68 struct timeval tv;
69 gettimeofday(&tv, NULL);
70 ts.tv_sec = tv.tv_sec + timeout / 1000;
71 ts.tv_nsec = 1000 * tv.tv_usec;
72 ts.tv_nsec += 1000000 * (timeout % 1000);
73
74 if (ts.tv_nsec > 999999999)
75 {
76 ts.tv_nsec -= 1000000000;
77 ts.tv_sec ++;
78 }
79
80 res = !pthread_mutex_timedlock(&m_mutex, &ts);
81#else
82 res = !pthread_mutex_lock(&m_mutex);
83#endif
84 }
85
86 return res;
87#endif // SOOS_LINUX
88#ifdef SOOS_WINDOWS
89 DWORD curThreadId = GetCurrentThreadId();
90
91 if (InterlockedIncrement(&m_lockCount) == 0)
92 {
93 // thread now owns the SOCmnSync
94 m_ownThreadId = curThreadId;
95 m_recurseCount = 1;
96 }
97 else
98 {
99 if (m_ownThreadId == curThreadId)
100 {
101 // still owning SOCmnSync
102 m_recurseCount++;
103 }
104 else
105 {
106 // not owning SOCmnSync
107 DWORD res;
108 res = ::WaitForSingleObject(m_waitEvent, timeout);
109
110 if (res == WAIT_TIMEOUT)
111 {
112 // stop waiting for the lock
113 InterlockedDecrement(&m_lockCount);
114 return FALSE;
115 }
116
117 // thread now owns the SOCmnSync
118 m_ownThreadId = curThreadId;
119 m_recurseCount = 1;
120 }
121 }
122
123#endif
124 return TRUE;
125} // lock

128//-----------------------------------------------------------------------------
129// unlock
130//
131// - release access to the synchronisation object
132//
133// returns:
134// TRUE - released access to the object
135// FALSE - not owned the object
136//
137BOOL SOCmnSync::unlock(void)
138{
139#ifdef SOOS_LINUX
140 return !pthread_mutex_unlock(&m_mutex);
141#endif // SOOS_LINUX
142#ifdef SOOS_WINDOWS
143 DWORD curThreadId = GetCurrentThreadId();
144 BOOL unlocked = TRUE;
145
146 if (m_ownThreadId == curThreadId)
147 {
148 m_recurseCount--;
149
150 if (m_recurseCount > 0)
151 {
152 // still own the SOCmnSync
153 InterlockedDecrement(&m_lockCount);
154 }
155 else
156 {
157 // last unlock
158 m_ownThreadId = 0;
159
160 if (InterlockedDecrement(&m_lockCount) >= 0)
161 {
162 // other thread is waiting
163 ::SetEvent(m_waitEvent); // signal the event
164 }
165 }
166 }
167 else
168 {
169 unlocked = FALSE; // object not owned by the thread
170 }
171
172 return unlocked;
173#endif
174} // unlock

419SOCmnSync* SOCmnObjectLock::doGetObjectLock(IN SOCmnObject* pObj, IN void* pVoid)
420{
421 SOCmnList<SOCmnSyncCounter> syncCounterList(m_syncCounterList);
422 void* pV = (pObj != NULL) ? (void*)pObj->getOpcHandle() : pVoid;
423 // use array
424 DWORD idx;
425 idx = ((DWORD)(void*)pV) >> 4;
426 idx &= SOCMNOBJECTLOCK_ARRAY_MASK;
427 SOCmnSync* pArrayLock = m_syncCounterArray[idx]->getCounterLock();
428 pArrayLock->lock();
429
430 if (m_syncCounterArray[idx]->getRefCount() == 1)
431 {
432 // index not used
433 m_syncCounterArray[idx]->setKey((size_t)pV);
434 m_syncCounterArray[idx]->addRef();
435 interlockedIncrement(&m_countSyncArrayUsed);
436 pArrayLock->unlock();
437 return m_syncCounterArray[idx]->getSyncObject();
438 }
439 else
440 {
441 // index used
442 // check if array is used for the pV
443 if (m_syncCounterArray[idx]->getKey() == (size_t)pV)
444 {
445 m_syncCounterArray[idx]->addRef();
446 pArrayLock->unlock();
447 return m_syncCounterArray[idx]->getSyncObject();
448 }
449 }
450
451 // add reference for using list to the array
452 // the array will only be used if all list entries for this index are freed
453 m_syncCounterArray[idx]->addRef();
454 pArrayLock->unlock();
455 // use list
456 syncCounterList.autoLock();
457 SOCmnPointer<SOCmnSyncCounter> syncCounter = syncCounterList.findKey((size_t)pV);
458
459 if (!syncCounter)
460 {
461 SOCmnSync* pSync;
462 SOCmnSync* pCounterLock;
463 SOCMN_ALLOCATE_OBJECT(pSync, SOCmnSync());
464
465 if (pSync == NULL)
466 {
467 return NULL;
468 }
469
470 SOCMN_ALLOCATE_OBJECT(pCounterLock, SOCmnSync());
471
472 if (pCounterLock == NULL)
473 {
474 return NULL;
475 }
476
477 SOCMN_ALLOCATE_OBJECT_C(syncCounter, SOCmnSyncCounter(pSync, pCounterLock), (SOCmnSyncCounter*));
478 syncCounter->setKey((size_t)pV);
479 syncCounterList.add(syncCounter, (size_t)pV);
480 syncCounter.addRef();
481 interlockedIncrement(&m_countSyncCreated);
482 _TRACE_IF(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OBJECTLOG)
483 {
484 if (pObj)
485 {
486 _TRACEC(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OBJECTLOG, _T("SOCmnObjectLock"), (_T("created 0x%lX | %s | (%lu)"), pObj->getOpcHandle(), pObj->getObjTraceId(), m_countSyncCreated));
487 }
488 else
489 {
490 _TRACEC(SOCMNTRACE_L_DEB, SOCMNTRACE_G_OBJECTLOG, _T("SOCmnObjectLock"), (_T("created 0x%lX | pV | (%lu)"), (size_t)pV, m_countSyncCreated));
491 }
492 }
493 return pSync;
494 }

496 syncCounter.addRef();

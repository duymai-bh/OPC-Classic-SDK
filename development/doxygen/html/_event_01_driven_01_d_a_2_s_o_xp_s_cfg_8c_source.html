162//-----------------------------------------------------------------------------
163// SOXpSStart
164//
165// - starts additional thread of the DLL
166// - this function is called after building all spaces (name, event, area)
167//
168// returns:
169// TRUE - DLL startup OK, continue starting the OPC server
170// FALSE - error during startup, end the OPC server
171//
172DWORD SOXPSCFG_CALL SOXpSStart(void)
173{
174 DWORD tid;
175 g_reportThreadEndEvent = CreateEvent(NULL, FALSE, FALSE, NULL);
176 g_reportThread = CreateThread(NULL, 0, reportThreadMain, NULL, 0, &tid);
177 return TRUE;
178} // SOXpSStart

181//-----------------------------------------------------------------------------
182// SOXpSStop
183//
184// - stops additional thread of the DLL
185// - this function is called before termination of the OPC server
186//
187void SOXPSCFG_CALL SOXpSStop(void)
188{
189 if (g_reportThread != INVALID_HANDLE_VALUE)
190 {
191 if (g_reportThreadEndEvent != INVALID_HANDLE_VALUE)
192 {
193 SetEvent(g_reportThreadEndEvent);
194 WaitForSingleObject(g_reportThread, INFINITE);
195 }
196 CloseHandle(g_reportThread);
197 }
198
199 if (g_reportThreadEndEvent != INVALID_HANDLE_VALUE)
200 CloseHandle(g_reportThreadEndEvent);
201
202} // SOXpSStop

205//-----------------------------------------------------------------------------
206// SOXpSBuildDANameSpace
207//
208// - build the name(address) space of the OPC DA server
209// - this function is called once during the initialization of the OPC server
210//
211void SOXPSCFG_CALL SOXpSBuildDANameSpace(
212 IN SOXpSCreateNode createNode, // create function for nodes
213 IN SOXpSCreateTag createTag, // create function for tags
214 IN SOXpSCreateProperty createProperty) // create function for properties
215{
216 SOXpSNodeHandle node1;
217 SOXpSNodeHandle node2;
218 SOXpSItemHandle property;
219 VARIANT descrPropVal;
220
221 //-- time branch --
222 // the tags include the current time (local or GMT)
223
224 if (!(createNode(0, _T("time"), VT_EMPTY, 0, SOXPS_IOMODE_NOIO, 0, &node1)))
225 return;
226
227 // description of time node
228 descrPropVal.vt = VT_BSTR;
229 descrPropVal.bstrVal = SysAllocString(L"the current time");
230 if (!(createProperty((SOXpSNodeOrItemHandle)node1, 101, NULL, VT_BSTR, SOXPS_ACCESSRIGHT_READABLE,
231 SOXPS_IOMODE_CONSTANT, &descrPropVal, 0, &property)))
232 return;
233 VariantClear(&descrPropVal);
234
235 if (!(createNode(node1, _T("local"), VT_EMPTY, 0, SOXPS_IOMODE_NOIO, 0, &node2)))
236 return;
237
238 if (!(createTag(node2, _T("sec"), VT_BSTR, SOXPS_ACCESSRIGHT_READABLE, SOXPS_IOMODE_DIRECT_REPORT, 1, &g_tagSec)))
239 return;
240
241 if (!(createTag(node2, _T("min"), VT_BSTR, SOXPS_ACCESSRIGHT_READABLE, SOXPS_IOMODE_DIRECT_REPORT, 2, &g_tagMin)))
242 return;
243
244 if (!(createTag(node2, _T("hour"), VT_BSTR, SOXPS_ACCESSRIGHT_READABLE, SOXPS_IOMODE_DIRECT_REPORT, 3, &g_tagHour)))
245 return;
246} // SOXpSBuildDANameSpace

249//-----------------------------------------------------------------------------
250// SOXpSHandleDARequests
251//
252// - handle the OPC DA I/O requests
253// - this function is called for every device read or write of a OPC client
254//
255void SOXPSCFG_CALL SOXpSHandleDARequests(
256 IN DWORD ioOperation, // type of I/O operation
257 IN DWORD numRequests, // number of requests
258 IN OUT SOXpSDARequestData *requestArray)// request data
259{
260 DWORD i;
261
262 for (i = 0; i < numRequests; i++) // handle all requests
263 {
264 // SOXpSHandleDARequests is only called once to get the initial data
265 SYSTEMTIME time;
266 TCHAR type = _T(' ');
267 WCHAR wideString[10];
268
269 GetLocalTime(&time);
270
271 VariantInit(&requestArray[i].m_value);
272 requestArray[i].m_value.vt = VT_BSTR;
273
274 switch (requestArray[i].m_itemUserData)
275 {
276 case 1:
277 swprintf(wideString, L"%2.2d:%2.2d:%2.2d", time.wHour, time.wMinute, time.wSecond);
278 requestArray[i].m_value.bstrVal = SysAllocString(wideString);
279 break;
280 case 2:
281 swprintf(wideString, L"%2.2d:%2.2d", time.wHour, time.wMinute, time.wSecond);
282 requestArray[i].m_value.bstrVal = SysAllocString(wideString);
283 break;
284 case 3:
285 swprintf(wideString, L"%2.2d", time.wHour, time.wMinute, time.wSecond);
286 requestArray[i].m_value.bstrVal = SysAllocString(wideString);
287 break;
288 }
289 // set the quality to good
290 requestArray[i].m_quality = SOXPS_QUALITY_GOOD;
291 // set the request state to handled
292 requestArray[i].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
293
294 // if you don't want to handle the initial read, you can return use cache to XPress
295 // then the client will receive the quality "waiting for initial data"
296 // in this case the function will be called until the initial data was set
297// requestArray[i].m_requestState = SOXPS_REQUEST_STATE_HANDLED_USE_CACHE;
298 }
299
300} // SOXpSHandleDARequests

304//-----------------------------------------------------------------------------
305// reportThreadMain
306//
307// - thread for reporting requests
308//
309DWORD WINAPI reportThreadMain(
310 IN LPVOID context) // thread data
311{
312 BOOL end = FALSE;
313 DWORD waitR;
314 static WORD actSec = 0xFFFF; // current second
315 static WORD actMin = 0xFFFF; // current minute
316 static WORD actHour = 0xFFFF; // current hour
317 SYSTEMTIME now; // current time
318 WCHAR strnow[10]; // current time (as string)
319 SOXpSDARequestData requestArray[3];
320 DWORD reqCnt;
321
322 requestArray[0].m_itemHandle = g_tagSec;
323 requestArray[0].m_itemPath = _T("time.local.sec");
324 requestArray[0].m_requestHandle = NULL;
325 requestArray[0].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
326 requestArray[0].m_result = S_OK;
327 requestArray[0].m_quality = SOXPS_QUALITY_GOOD;
328 VariantInit(&requestArray[0].m_value);
329 requestArray[1].m_itemHandle = g_tagMin;
330 requestArray[1].m_itemPath = _T("time.local.min");
331 requestArray[1].m_requestHandle = NULL;
332 requestArray[1].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
333 requestArray[1].m_result = S_OK;
334 requestArray[1].m_quality = SOXPS_QUALITY_GOOD;
335 VariantInit(&requestArray[1].m_value);
336 requestArray[2].m_itemHandle = g_tagHour;
337 requestArray[2].m_itemPath = _T("time.local.hour");
338 requestArray[2].m_requestHandle = NULL;
339 requestArray[2].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
340 requestArray[2].m_result = S_OK;
341 requestArray[2].m_quality = SOXPS_QUALITY_GOOD;
342 VariantInit(&requestArray[2].m_value);
343
344 while (!end)
345 {
346 // wait for next second or the end
347 waitR = WaitForSingleObject(g_reportThreadEndEvent, 1000);
348 if (waitR == WAIT_TIMEOUT)
349 {
350 GetLocalTime(&now);
351
352 if (now.wSecond != actSec)
353 {
354 actSec = now.wSecond;
355
356 VariantClear(&requestArray[0].m_value);
357 swprintf(strnow, L"%02.2d:%02.2d:%02.2d", now.wHour, now.wMinute, now.wSecond);
358 requestArray[0].m_value.vt = VT_BSTR;
359 requestArray[0].m_value.bstrVal = SysAllocString(strnow);
360 memset(&requestArray[0].m_timeStamp, 0, sizeof(FILETIME));
361 reqCnt = 1;
362
363 if (now.wMinute != actMin)
364 {
365 actMin = now.wMinute;
366
367 VariantClear(&requestArray[1].m_value);
368 swprintf(strnow, L"%02.2d:%02.2d", now.wHour, now.wMinute);
369 requestArray[1].m_value.vt = VT_BSTR;
370 requestArray[1].m_value.bstrVal = SysAllocString(strnow);
371 memset(&requestArray[1].m_timeStamp, 0, sizeof(FILETIME));
372 reqCnt = 2;
373
374 if (now.wHour != actHour)
375 {
376 actHour = now.wHour;
377
378 VariantClear(&requestArray[2].m_value);
379 swprintf(strnow, L"%02.2d", now.wHour);
380 requestArray[2].m_value.vt = VT_BSTR;
381 requestArray[2].m_value.bstrVal = SysAllocString(strnow);
382 memset(&requestArray[2].m_timeStamp, 0, sizeof(FILETIME));
383 reqCnt = 3;
384 }
385 }
386
387 g_completedReq(reqCnt, requestArray);
388 }
389 }
390 else
391 end = TRUE;
392 }
393
394 return 0;
395}

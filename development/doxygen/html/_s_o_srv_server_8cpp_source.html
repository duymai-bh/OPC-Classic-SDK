211//-----------------------------------------------------------------------------
212// workThread
213//
214// - thread routine of working thread
215//
216// returns:
217// thread end state
218//
219DWORD SOSrvServer::workThread(void)
220{
221 BOOL end = FALSE;
222 DWORD ret;
223 DWORD waitTime = SOCMN_TIME_40DAYS;
224 DWORD loopStart;
225 DWORD loopTime = 0;
226 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SRV_THREAD, (_T("started work thread")));
227
228 while (!end)
229 {
230 if (waitTime != 0)
231 {
232 ret = m_workThread.waitForEvents(waitTime);
233 }
234 else
235 {
236 ret = SOCMNEVENT_INVALID_ID;
237 }
238
239 if (ret == m_workThread.getStopEventId())
240 {
241 // end
242 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SRV_THREAD, (_T("terminate work thread")));
243 end = TRUE;
244 }
245 else
246 {
247 // work
248 // addRef()
249 LONG refCount = interlockedIncrement(&m_refCount);
250
251 if (refCount <= 1)
252 {
253 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SRV_THREAD, (_T("terminate work thread (refCount==0)")));
254 // only start the work loop if not in the release phase
255 interlockedDecrement(&m_refCount);
256 return 0;
257 }
258
259 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_SRV_THREAD, (_T("loop >>>")));
260 loopStart = ::GetTickCount();
261 // do work
262 waitTime = doWork(loopStart);
263
264 // max wait time is 40 days
265 if (SOCMN_TIME_40DAYS < waitTime)
266 {
267 waitTime = SOCMN_TIME_40DAYS;
268 }
269 else
270 {
271 loopTime = getTimeSpan(loopStart, ::GetTickCount());
272
273 if (loopTime != 0)
274 {
275 if (waitTime >= loopTime)
276 {
277 waitTime -= loopTime;
278 }
279 else
280 {
281 waitTime = 0;
282 }
283 }
284
285 if (m_minRequestLoopTime < waitTime)
286 {
287 waitTime = m_minRequestLoopTime;
288 }
289 }
290
291 _TRACE(SOCMNTRACE_L_DEB, SOCMNTRACE_G_SRV_THREAD, (_T("loop next:%lu needed:%lu <<<"), waitTime, loopTime));
292 refCount = interlockedDecrement(&m_refCount);
293
294 if (refCount <= 0)
295 {
296 _TRACE(SOCMNTRACE_L_INF, SOCMNTRACE_G_SRV_THREAD, (_T("terminate work thread & free object (refCount==0)")));
297 // terminate own thread & free object
298 m_workThread.detach();
299 SOCMN_FREE_OBJECT_THIS; // last reference -> free object
300 return 0;
301 }
302 }
303 }
304
305 return 0;
306} // workThread

994// stdMatchStringFilter
995//
996// - string filter function of OPC Toolkit V1.x
997//
998// returns:
999// TRUE - string matches filter
1000// FALSE - string is filtered out
1001//
1002BOOL SOSrvServer::stdMatchStringFilter(
1003 IN LPCTSTR string, // string
1004 IN LPCTSTR filter) // filter
1005{
1006 BOOL match;
1007 int actS, actF;
1008 int prev;
1009
1010 if (!filter)
1011 {
1012 return TRUE;
1013 }
1014
1015 if (_tcslen(filter) == 0)
1016 {
1017 return TRUE;
1018 }
1019
1020 if (!string)
1021 {
1022 return FALSE;
1023 }
1024
1025 actS = 0;
1026 actF = 0;
1027 prev = -1;
1028
1029 while (((string[actS] == filter[actF]) ||
1030 (filter[actF] == _T('*')) ||
1031 (prev != -1))
1032 && (string[actS] != _T('\0')))
1033 {
1034 if (filter[actF] != _T('*'))
1035 {
1036 if (string[actS] == filter[actF])
1037 {
1038 actF++;
1039 }
1040 else
1041 {
1042 actF = prev;
1043
1044 if (string[actS] == filter[actF + 1])
1045 {
1046 // match Puchheim with *heim
1047 actF += 2;
1048 }
1049 }
1050 }
1051
1052 if (filter[actF] == _T('*'))
1053 {
1054 prev = actF;
1055
1056 if (filter[actF + 1] != _T('\0'))
1057 {
1058 if (string[actS + 1] == filter[actF + 1])
1059 {
1060 actF++;
1061 }
1062 }
1063 }
1064
1065 actS++;
1066 }
1067
1068 if (filter[actF] == _T('*'))
1069 {
1070 actF++;
1071 }
1072
1073 match = ((string[actS] == _T('\0')) && (filter[actF] == _T('\0')));
1074 return match;
1075} // stdMatchStringFilter

1078//-----------------------------------------------------------------------------
1079// opcMatchStringFilter
1080//
1081// - string filter function of the OPC Foundation
1082//
1083// returns:
1084// TRUE - string matches filter
1085// FALSE - string is filtered out
1086//
1087BOOL SOSrvServer::opcMatchStringFilter(
1088 IN LPCTSTR String, // string
1089 IN LPCTSTR Pattern, // filter
1090 IN BOOL bCaseSensitive) // case sensitive or not
1091{
1092 if (!String)
1093 {
1094 return FALSE;
1095 }
1096
1097 if (!Pattern)
1098 {
1099 return TRUE;
1100 }
1101
1102 TCHAR c, p, l;
1103
1104 for (; ;)
1105 {
1106 switch (p = (TCHAR)((bCaseSensitive) ? *Pattern++ : toupper(*Pattern++)))
1107 {
1108 case 0: // end of pattern
1109 return *String ? FALSE : TRUE; // if end of string TRUE
1110
1111 case _T('*'):
1112 while (*String)
1113 {
1114 // match zero or more char
1115 if (opcMatchStringFilter(String++, Pattern, bCaseSensitive))
1116 {
1117 return TRUE;
1118 }
1119 }
1120
1121 return opcMatchStringFilter(String, Pattern, bCaseSensitive);
1122
1123 case _T('?'):
1124 if (*String++ == 0) // match any one char
1125 {
1126 return FALSE; // not end of string
1127 }
1128
1129 break;
1130
1131 case _T('['):
1132 if ((c = (TCHAR)((bCaseSensitive) ? *String++ : toupper(*String++))) == 0) // match char set
1133 {
1134 return FALSE; // syntax
1135 }
1136
1137 l = 0;
1138
1139 if (*Pattern == _T('!')) // match a char if NOT in set []
1140 {
1141 ++Pattern;
1142
1143 while ((p = (TCHAR)((bCaseSensitive) ? *Pattern++ : toupper(*Pattern++))) != _T('\0'))
1144 {
1145 if (p == _T(']')) // if end of char set, then
1146 {
1147 break; // no match found
1148 }
1149
1150 if (p == _T('-'))
1151 {
1152 // check a range of chars?
1153 p = (TCHAR)((bCaseSensitive) ? *Pattern++ : toupper(*Pattern++)); // get high limit of range
1154
1155 if (p == 0 || p == _T(']'))
1156 {
1157 return FALSE; // syntax
1158 }
1159
1160 if (c >= l && c <= p)
1161 {
1162 return FALSE; // if in range, return FALSE
1163 }
1164 }
1165
1166 l = p;
1167
1168 if (c == p) // if char matches this element
1169 {
1170 return FALSE; // return false
1171 }
1172 }
1173 }
1174 else // match if char is in set []
1175 {
1176 while ((p = (TCHAR)((bCaseSensitive) ? *Pattern++ : toupper(*Pattern++))) != _T('\0'))
1177 {
1178 if (p == _T(']')) // if end of char set, then
1179 {
1180 return FALSE; // no match found
1181 }
1182
1183 if (p == _T('-'))
1184 {
1185 // check a range of chars?
1186 p = (TCHAR)((bCaseSensitive) ? *Pattern++ : toupper(*Pattern++)); // get high limit of range
1187
1188 if (p == 0 || p == _T(']'))
1189 {
1190 return FALSE; // syntax
1191 }
1192
1193 if (c >= l && c <= p)
1194 {
1195 break; // if in range, move on
1196 }
1197 }
1198
1199 l = p;
1200
1201 if (c == p) // if char matches this element
1202 {
1203 break; // move on
1204 }
1205 }
1206
1207 while (p && p != _T(']')) // got a match in char set
1208 {
1209 p = *Pattern++; // skip to end of set
1210 }
1211 }
1212
1213 break;
1214
1215 case _T('#'):
1216 c = *String++;
1217
1218 if (!_istdigit(c))
1219 {
1220 return FALSE; // not a digit
1221 }
1222
1223 break;
1224
1225 default:
1226 c = (TCHAR)((bCaseSensitive) ? *String++ : toupper(*String++));
1227
1228 if (c != p) // check for exact char
1229 {
1230 return FALSE; // not a match
1231 }
1232
1233 break;
1234 }
1235 }
1236} // opcMatchStringFilter

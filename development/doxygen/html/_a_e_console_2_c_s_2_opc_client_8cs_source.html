33 private string GetStateToString(EnumConditionState state)
34 {
35 string stateToString = string.Empty;
36
37 if ((state & EnumConditionState.ACTIVE) == EnumConditionState.ACTIVE)
38 {
39 stateToString += " ACT";

46 {

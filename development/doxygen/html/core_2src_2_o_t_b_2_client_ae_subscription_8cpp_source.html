66void ClientAESubscription::onReceivedEvents(DWORD count, SOAeCEvent** pEvents, BYTE refresh, BYTE lastRefresh)
67{
68 if (g_callbackFunctions.m_OTCOnReceivedEvents)
69 {
70 OTCObjectContext objectContext;
71 OTCEventData* pEventsO;
72 DWORD i, j;
73 SOCmnList<SOCmnVariant> eventAttrs;
74 SOCmnListPosition pos;
75 SOCmnVariant* pVariant;
76 getOTCObjectContext((OTCObjectHandle)this->getOpcHandle(), &objectContext);
77 pEventsO = (OTCEventData*)OTAllocateMemory(sizeof(OTCEventData) * count);
78
79 for (i = 0; i < count; i++)
80 {
81 pEventsO[i].m_eventType = pEvents[i]->m_eventType;
82 pEventsO[i].m_eventCategory = pEvents[i]->m_eventCategory;
83 pEventsO[i].m_sourcePath = pEvents[i]->m_source;
84 pEventsO[i].m_severity = pEvents[i]->m_severity;
85 pEventsO[i].m_message = pEvents[i]->m_message;
86 pEventsO[i].m_actorID = pEvents[i]->m_actorID;
87 pEventsO[i].m_occurenceTime = pEvents[i]->m_ftTime;
88 pEventsO[i].m_changeMask = pEvents[i]->m_changeMask;
89 pEventsO[i].m_newState = (BYTE)pEvents[i]->m_newState;
90 pEventsO[i].m_ackRequired = (BYTE)pEvents[i]->m_bAckRequired;
91 pEventsO[i].m_quality = pEvents[i]->m_quality;
92 pEventsO[i].m_conditionName = pEvents[i]->m_conditionName;
93 pEventsO[i].m_subConditionName = pEvents[i]->m_subconditionName;
94 pEventsO[i].m_activeTime = pEvents[i]->m_ftActiveTime;
95 pEventsO[i].m_cookie = pEvents[i]->m_dwCookie;
96 eventAttrs = pEvents[i]->m_listEventAttrs;
97 pEventsO[i].m_eventAttrCount = eventAttrs.getCount();
98
99 if (pEventsO[i].m_eventAttrCount > 0)
100 {
101 pEventsO[i].m_pEventAttrs = (OTVariant*)OTAllocateMemory(sizeof(OTVariant) * pEventsO[i].m_eventAttrCount);
102 pos = eventAttrs.getStartPosition();
103 j = 0;
104
105 while (pos)
106 {
107 pVariant = eventAttrs.getNext(pos);
108 pVariant->copyTo(&pEventsO[i].m_pEventAttrs[j]);
109 j++;
110 }
111 }
112 else
113 {
114 pEventsO[i].m_pEventAttrs = NULL;
115 }
116 }
117
118 SOCMN_TRY
119 {
120 g_callbackFunctions.m_OTCOnReceivedEvents(objectContext, count, pEventsO, refresh, lastRefresh);
121 }
122 SOCMN_CATCH(...)
123 {
124 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_CLIENT, (_T("ClientAESubscription::onReceivedEvents - Exception occurred in vendor application callback: OTCOnReceivedEvents")));
125 }
126
127 for (i = 0; i < count; i++)
128 {
129 if (pEventsO[i].m_pEventAttrs)
130 {
131 for (j = 0; j < pEventsO[i].m_eventAttrCount; j++)
132 {
133 SOCmnVariant::variantClear(&pEventsO[i].m_pEventAttrs[j]);
134 }
135
136 OTFreeMemory(pEventsO[i].m_pEventAttrs);
137 }
138 }
139
140 OTFreeMemory(pEventsO);
141 }
142}

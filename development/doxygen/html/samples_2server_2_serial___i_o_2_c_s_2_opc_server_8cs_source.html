208 public void HandleRequests()
209 {
210 lock (m_requestsJanitor)
211 {
212 foreach (MyRequest request in m_requests)
213 {
214 if (request.ProgressRequestState(EnumRequestState.PENDING, EnumRequestState.INPROGRESS))
215 {
216 MyDaAddressSpaceElement element = request.AddressSpaceElement as MyDaAddressSpaceElement;
217 if (element == null)
218 {
219 request.Result = EnumResultCode.E_FAIL;
220 }
221 else
222 {
223 switch (request.TransactionType)
224 {
225 case EnumTransactionType.READ:
226 if (request.PropertyId == 0)
227 {
228 ValueQT cacheValue = null;
229 request.Result = (EnumResultCode)element.GetCacheValue(ref cacheValue);
230 request.Value = cacheValue;
231 }
232 else
233 {
234 element.GetPropertyValue(request);
235 } // end if ... else
236 break;
237
238 case EnumTransactionType.WRITE:
239 if (request.PropertyId == 0)
240 {
241 ValueQT newValue = request.Value;
242 request.Result = (EnumResultCode)element.ValueChanged(newValue);
243 }
244 else
245 {
246 request.Result = EnumResultCode.E_BADRIGHTS;
247 } // end if ... else
248 break;
249
250 default:
251 break;
252 } // end switch
253
254 } // end if ... else
255 } // end if ... else
256 request.Complete();
257 } // end forech
258 m_requests.Clear();
259 } // end lock
260
261 } // end HandleRequests

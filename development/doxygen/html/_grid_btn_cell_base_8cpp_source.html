57BOOL CGridBtnCellBase::SetupBtns(
58 int aiWhich, // zero-based index of image to draw
59 UINT auiType, // type of frame control to draw e.g. DFC_BUTTON
60 UINT auiState, // like DrawFrameControl()'s nState e.g. DFCS_BUTTONCHECK
61 CGridBtnCellBase::CTL_ALIGN aAlign, // horizontal alignment of control image
62 int aiWidth, // fixed width of control or 0 for size-to-fit
63 BOOL abIsMbrRadioGrp, // T=btn is member of a radio group
64 const char* apszText) // Text to insert centered in button; if NULL no text
65// returns: success / failure
66{
67 if (aiWhich < 0
68 || aiWhich >= GetDrawCtlNbrMax())
69 {
70 ASSERT(FALSE);
71 return FALSE;
72 }
73
74 if (aAlign < 0
75 || aAlign >= CTL_ALIGN_FENCE)
76 {
77 ASSERT(FALSE);
78 return FALSE;
79 }
80
81 if (aAlign == CTL_ALIGN_CENTER
82 && aiWhich > 0)
83 {
84 ASSERT(FALSE); // if centered can have only 1 button
85 return FALSE;
86 }
87
88 // increase the count of controls, if necessary and possible
89 if (aiWhich >= GetDrawCtlNbr())
90 {
91 if (aiWhich >= GetDrawCtlNbrMax())
92 {
93 return FALSE; // wanted to add too many images
94 }
95
96 ASSERT(aiWhich == GetDrawCtlNbr()); // if ASSERT, then you are setting
97 // initial image 0, then image 2 -- you've
98 // skipped image 1. You can't skip an image
99 SetDrawCtlNbr((unsigned char)(aiWhich + 1));
100 }
101
102 SetDrawCtlAlign(aiWhich, aAlign);
103 SetDrawCtlWidth(aiWhich, aiWidth);
104 SetDrawCtlType(aiWhich, auiType);
105 SetDrawCtlState(aiWhich, auiState);
106 SetDrawCtlBtnText(aiWhich, apszText);
107 SetDrawCtlIsMbrRadioGrp(aiWhich, abIsMbrRadioGrp);
108 return TRUE;
109}

126BOOL CGridBtnCellBase::DrawBtnCell(CDC* pDC, int /* nRow */, int /* nCol */, CRect* prect,
127 BOOL /* bEraseBkgnd */)
128{
129 CGridCtrl* pGrid = GetGrid();
130
131 if (!pGrid || !pDC)
132 {
133 return FALSE;
134 }
135
136 if (prect->Width() <= 0
137 || prect->Height() <= 0) // prevents imagelist item from drawing even
138 {
139 return FALSE; // though cell is hidden
140 }
141
142 int nSavedDC = pDC->SaveDC();
143 // draw any button images
144 ASSERT(MAX_NBR_CTLS_INCELL > GetDrawCtlNbrMax()); // whoa!
145 const int iCtlNbr = GetDrawCtlNbr();
146 CRect RectAry[ MAX_NBR_CTLS_INCELL];
147
148 if (iCtlNbr > 0)
149 {
150 if (GetState() & GVIS_SELECTED
151 || GetState() & GVIS_DROPHILITED)
152 {
153 // draw the rectangle around the grid --
154 // we may be filling cells with controls
155 pDC->SelectStockObject(BLACK_PEN);
156 pDC->SelectStockObject(NULL_BRUSH);
157 pDC->Rectangle(*prect);
158 }
159
160 pDC->SetBkMode(TRANSPARENT);
161 prect->DeflateRect(GetMargin(), 0);
162 CFont* pOldFont = pDC->SelectObject(GetFontObject());
163
164 if (CalcDrawCtlRects(RectAry, // returns: CRects with coordinates
165 // last entry has optional leftover rect
166 // available for text, etc.
167 MAX_NBR_CTLS_INCELL,// nbr of Rects in above array
168 *prect)) // cell rectangle to work with
169 {
170 for (int i1 = 0; i1 < iCtlNbr; i1++)
171 {
172 UINT uiType = GetDrawCtlType(i1);
173 UINT uiState = GetDrawCtlState(i1);
174 pDC->DrawFrameControl(RectAry[ i1],
175 uiType,
176 uiState);
177 // if button has text, draw it, too
178 char* pszBtnText = (char*)GetDrawCtlBtnText(i1);
179
180 if (pszBtnText != NULL)
181 {
182 COLORREF ColorCurrent = pDC->GetTextColor();
183
184 if (uiState & DFCS_INACTIVE)
185 {
186 // button is grayed-out
187 // draw the text so that it matches MS's look
188 RectAry[ i1].OffsetRect(1, 1);
189 pDC->SetTextColor(RGB(255, 255, 255));
190 USES_CONVERSION;
191 pDC->DrawText(A2T(pszBtnText),
192 -1,
193 RectAry[ i1],
194 DT_CENTER | DT_VCENTER | DT_SINGLELINE | DT_END_ELLIPSIS);
195 RectAry[ i1].OffsetRect(-1, -1);
196 pDC->SetTextColor(::GetSysColor(COLOR_GRAYTEXT));
197 }
198 else
199 {
200 pDC->SetTextColor(::GetSysColor(COLOR_BTNTEXT));
201 }
202
203 USES_CONVERSION;
204 pDC->DrawText(A2T(pszBtnText),
205 -1,
206 RectAry[ i1],
207 DT_CENTER | DT_VCENTER | DT_SINGLELINE | DT_END_ELLIPSIS);
208 pDC->SetTextColor(ColorCurrent);
209 }
210 }
211
212 // allowable text area has shrunk up
213 *prect = RectAry[ iCtlNbr];
214 }
215
216 pDC->SelectObject(pOldFont);
217
218 // maybe there's nothing left to draw
219 if (prect->Width() <= 0)
220 {
221 pDC->RestoreDC(nSavedDC);
222 return TRUE;
223 }
224 }
225
226 pDC->RestoreDC(nSavedDC);
227 return TRUE;
228}

250BOOL CGridBtnCellBase::Edit(int nRow, int nCol, CRect rect, CPoint point, UINT nID, UINT nChar)
251{
252 ASSERT(m_pBtnDataBase != NULL);
253
254 if (ProcessCtlClick(WM_LBUTTONUP, // Command that invoked. e.g. WM_LBUTTONDOWN
255 point)) // point to check for hit
256 {
257 SendMessageToParent(m_iRow, m_iCol, NM_CLICK); // tell parent about it
258 return FALSE; // clicked a control rather than edited a value
259 }
260
261 const int iCtlNbr = GetDrawCtlNbr();
262
263 if (iCtlNbr > 0)
264 {
265 // if there's at least one left-aligned control and user pressed
266 // space bar, activate that control
267 if (GetDrawCtlAlign(0) == CTL_ALIGN_LEFT
268 || iCtlNbr == 1)
269 {
270 if (nChar == ' ')
271 {
272 // user hit space bar -- just like clicking a button
273 ClickedCellCtl(WM_LBUTTONDOWN, // Command that invoked. e.g. WM_LBUTTONDOWN
274 0); // zero-based index of image to draw
275 Sleep(200);
276 ClickedCellCtl(WM_LBUTTONUP, // Command that invoked. e.g. WM_LBUTTONDOWN
277 0); // zero-based index of image to draw
278 SendMessageToParent(m_iRow, m_iCol, NM_CLICK); // tell parent about it
279 return FALSE; // clicked a control rather than edited a value
280 }
281 }
282 }
283
284 // if no string text to edit, then typing a character may invoke a button
285 if (HasCellText())
286 {
287 DWORD dwStyle = ES_LEFT;
288
289 if (GetFormat() & DT_RIGHT)
290 {
291 dwStyle = ES_RIGHT;
292 }
293 else if (GetFormat() & DT_CENTER)
294 {
295 dwStyle = ES_CENTER;
296 }
297
298 m_ucEditing = TRUE;
299 // InPlaceEdit auto-deletes itself
300 CGridCtrl* pGrid = GetGrid();
301 CWnd* pWnd = new CInPlaceEdit(pGrid, rect, dwStyle, nID, nRow, nCol, GetText(), nChar);
302 m_pBtnDataBase->SetEditWnd(pWnd);
303 }
304 else
305 {
306 // since no text to edit, maybe press a button -- check for hot keys
307 int iCtl = HotKeyBtnMatch((char)nChar); // hot key character
308
309 if (iCtl >= 0)
310 {
311 // returns: index of button or -1 if no hot key matches
312 ClickedCellCtl(WM_LBUTTONDOWN, // Command that invoked. e.g. WM_LBUTTONDOWN
313 iCtl); // zero-based index of image to draw
314 Sleep(200);
315 ClickedCellCtl(WM_LBUTTONUP, // Command that invoked. e.g. WM_LBUTTONDOWN
316 iCtl); // zero-based index of image to draw
317 SendMessageToParent(m_iRow, m_iCol, NM_CLICK); // tell parent about it
318 return FALSE; // clicked a control rather than edited a value
319 }
320 }
321
322 return TRUE;
323}

446/*****************************************************************************
447Called during all the mouse events associated with clicking a control
448embedded within a cell. Override to have more elaborate handling like
449implementing radio button logic.
450
451*****************************************************************************/
452BOOL CGridBtnCellBase::ClickedCellCtl(UINT uMsg, // Command that invoked. e.g. WM_LBUTTONDOWN
453 int aiWhich) // zero-based index of image to draw
454// returns: T=redraw occurred / F=no redraw
455{
456 if (aiWhich < 0
457 || aiWhich >= GetDrawCtlNbrMax())
458 {
459 ASSERT(FALSE);
460 return FALSE;
461 }
462
463 UINT uiState = GetDrawCtlState(aiWhich);
464
465 if (uiState & DFCS_INACTIVE)
466 {
467 return FALSE; // button is inactive -- don't do anything
468 }
469
470 m_sLastCtlClicked = (short)aiWhich;
471 UINT iType = GetDrawCtlType(aiWhich);
472
473 switch (uMsg)
474 {
475 case WM_LBUTTONDOWN:
476 // appears pushed in
477 uiState |= DFCS_PUSHED;
478 SetDrawCtlState(aiWhich, uiState);
479 break;
480
481 case WM_LBUTTONUP:
482 case WM_LBUTTONDBLCLK:
483 // appears pushed out
484 uiState &= (~DFCS_PUSHED);
485
486 // auto check / uncheck controls, too
487 if (iType == DFC_BUTTON)
488 {
489 BOOL bIsMbrRadioGrp = GetDrawCtlIsMbrRadioGrp(aiWhich);
490
491 if (uiState & DFCS_BUTTONRADIO
492 || bIsMbrRadioGrp)
493 {
494 // radio buttons or any button flagged as being part
495 // of a radio group will be made to look pressed down
496 // while pushing-up / unchecking all other members
497 // of the radio group
498 const int iCtlNbr = GetDrawCtlNbr();
499 UINT uiStateRadio;
500
501 for (int i1 = 0; i1 < iCtlNbr; i1++)
502 {
503 if (i1 != aiWhich)
504 {
505 uiStateRadio = GetDrawCtlState(i1);
506 bIsMbrRadioGrp = GetDrawCtlIsMbrRadioGrp(i1);
507
508 if (uiStateRadio & DFCS_BUTTONRADIO
509 || bIsMbrRadioGrp)
510 {
511 uiStateRadio &= (~(DFCS_PUSHED | DFCS_CHECKED));
512 // push out and uncheck
513 SetDrawCtlState(i1, uiStateRadio);
514 }
515 }
516 }
517
518 uiState |= DFCS_CHECKED; // check
519
520 if (!(uiState & DFCS_BUTTONRADIO))
521 {
522 uiState |= DFCS_PUSHED; // press in if not real radio button
523 }
524 }
525 else if (!(uiState & ALL_BUT_BTN_CHK))
526 {
527 // not a pushbutton -- it's a check box
528 // (can't check for DFCS_BUTTONCHECK directly since it is bit 0)
529 if (uiState & DFCS_CHECKED)
530 {
531 uiState &= (~DFCS_CHECKED); // uncheck
532 }
533 else
534 {
535 uiState |= DFCS_CHECKED; // check
536 }
537 }
538 }
539
540 SetDrawCtlState(aiWhich, uiState);
541 break;
542
543 default:
544 ASSERT(FALSE); // gotta handle new message
545 return FALSE;
546 }
547
548 CGridCtrl* pGrid = GetGrid();
549 ASSERT(pGrid);
550 pGrid->RedrawCell(m_iRow, m_iCol);
551 return TRUE;
552}

576int CGridBtnCellBase::RelPointInCtl(const CPoint& arPoint) // Relative point coords
577// returns: Index of control that this point is within bounds of or -1 if no control matches
578{
579 CGridCtrl* pGrid = GetGrid();
580 ASSERT(pGrid);
581 CRect RectCell;
582
583 if (pGrid->GetCellRect(m_iRow,
584 m_iCol,
585 &RectCell))
586 {
587 ASSERT(MAX_NBR_CTLS_INCELL > GetDrawCtlNbrMax()); // whoa!
588 CRect RectAry[ MAX_NBR_CTLS_INCELL];
589
590 if (CalcDrawCtlRects(RectAry, // returns: CRects with coordinates
591 // last entry has optional leftover rect
592 // available for text, etc.
593 MAX_NBR_CTLS_INCELL,// nbr of Rects in above array
594 RectCell)) // cell rectangle to work with
595 {
596 const int iCtlNbr = GetDrawCtlNbr();
597 // make point absolute coord
598 CPoint pointAbs;
599 pointAbs.x = arPoint.x + RectCell.left;
600 pointAbs.y = arPoint.y + RectCell.top;
601
602 for (int i1 = 0; i1 < iCtlNbr; i1++)
603 {
604 if (pointAbs.x >= RectAry[i1].left
605 && pointAbs.x <= RectAry[i1].right
606 && pointAbs.y >= RectAry[i1].top
607 && pointAbs.y <= RectAry[i1].bottom)
608 {
609 return i1; // found it
610 }
611 }
612 }
613 }
614
615 return -1;
616}

619/*****************************************************************************
620Determines bounding rectangle to draw a button in a cell. Used to draw cells
621and for mouse hit testing.
622
623If a fixed-size control can't fit within the cell, the control will shrink in
624the required x and / or y dimension. I do this because I'm not using a
625clipping region when I display the cell -- that would be a better solution.
626
627*****************************************************************************/
628BOOL CGridBtnCellBase::CalcDrawCtlRects(CRect* apRect, // returns: CRects with coordinates
629 // last entry has optional leftover rect
630 // available for text, etc.
631 int aiNbrRectEntries, // nbr of Rects in above array
632 const CRect& arRectCell)// cell rectangle to work with
633// returns: success / fail
634{
635 ASSERT(apRect != NULL);
636
637 if (aiNbrRectEntries < GetDrawCtlNbrMax())
638 {
639 ASSERT(FALSE); // need to allow for leftover rect
640 return FALSE;
641 }
642
643 const int iCtlNbr = GetDrawCtlNbr();
644
645 if (iCtlNbr <= 0)
646 {
647 return FALSE;
648 }
649
650 int i1, i2;
651 int iSpinBoxDownIdx = -1; // can have zero or 1 spin box -- no more
652 // identifies placment of down arrow of the spin box
653 int iWidth = 0;
654 CTL_ALIGN CtlAlign;
655 UINT uiType;
656 UINT uiState;
657 CRect* pRectSav = apRect;
658 // calculate the width layout of buttons by examining
659 // all of them and noting important info
660 int iFixedSum = 0;
661 int iSizeToFitCount = 0;
662
663 for (i1 = 0; i1 < iCtlNbr; i1++)
664 {
665 CtlAlign = GetDrawCtlAlign(i1);
666
667 if (CtlAlign == CTL_ALIGN_CENTER)
668 {
669 // forget all calculations if any are centered, all controls
670 // just overwrite each other and expand to fit cell
671 for (i2 = 0; i2 < iCtlNbr; i2++)
672 {
673 apRect->operator=(arRectCell); // copy initial rectangle
674 apRect++;
675 }
676
677 apRect->operator=(CRect(0, 0, 0, 0)); // no text leftover
678 return TRUE;
679 }
680
681 iWidth = GetDrawCtlWidth(i1);
682
683 if (iWidth > 0)
684 {
685 iFixedSum += iWidth;
686 }
687 else
688 {
689 iSizeToFitCount++;
690 }
691
692 // spin box rectangles are stacked on top of each other
693 // thus, avoid doubling spin box width
694 if (iSpinBoxDownIdx < 0)
695 {
696 uiState = GetDrawCtlState(i1);
697
698 if (GetDrawCtlType(i1) == DFC_SCROLL
699 && !(uiState & (DFCS_SCROLLCOMBOBOX
700 | DFCS_SCROLLDOWN
701 | DFCS_SCROLLLEFT
702 | DFCS_SCROLLRIGHT
703 | DFCS_SCROLLSIZEGRIP))) // checking for DFCS_SCROLLUP
704 // but it is not a bit, it is 0x0!
705 {
706 if (i1 + 1 < iCtlNbr)
707 {
708 // at least 1 more -- see if we got spin box match
709 if (GetDrawCtlType(i1 + 1) == DFC_SCROLL
710 && (GetDrawCtlState(i1 + 1) & DFCS_SCROLLDOWN))
711 {
712 // it's a spin box
713 i1++; // skip looking at next control
714 iSpinBoxDownIdx = i1;
715 }
716 }
717 }
718 }
719 }
720
721 int iSizeToFitWidth = 0;
722 int iFitWidthsTotal = arRectCell.Width() - iFixedSum;
723
724 if (iSizeToFitCount > 0)
725 {
726 iSizeToFitWidth = iFitWidthsTotal / iSizeToFitCount;
727 }
728
729 int iSizeToFitCountWrk = iSizeToFitCount;
730 int iWidthSoFar = 0;
731
732 for (i1 = 0; i1 < iCtlNbr; i1++)
733 {
734 if (iSpinBoxDownIdx == i1)
735 {
736 // skip down arrow of spin box for calculations
737 apRect++;
738 continue;
739 }
740
741 apRect->operator=(arRectCell); // copy initial rectangle
742 apRect->left += iWidthSoFar;
743 apRect->right = apRect->left;
744 iWidth = GetDrawCtlWidth(i1);
745
746 if (iWidth > 0)
747 {
748 iWidthSoFar += iWidth; // fixed width
749 }
750 else
751 {
752 iSizeToFitCountWrk--; // found another one
753 // may shrink width if control is square -- saves screen real-estate
754 uiType = GetDrawCtlType(i1);
755 uiState = GetDrawCtlState(i1);
756 BOOL bIsRectangle = (uiType == DFC_BUTTON
757 && (uiState & DFCS_BUTTONPUSH) == DFCS_BUTTONPUSH);
758 // all other buttons are square
759
760 if (!bIsRectangle
761 && arRectCell.Height() < iSizeToFitWidth)
762 {
763 // it is square -- make width the cell height
764 iWidthSoFar += arRectCell.Height();
765 // recalulate size to fit
766 iFitWidthsTotal -= arRectCell.Height();
767
768 if (iSizeToFitCountWrk > 0)
769 {
770 iSizeToFitWidth = iFitWidthsTotal / iSizeToFitCountWrk;
771 }
772 }
773 else
774 {
775 iWidthSoFar += iSizeToFitWidth;
776 iFitWidthsTotal -= iSizeToFitWidth;
777 }
778 }
779
780 apRect->right = iWidthSoFar + arRectCell.left;
781 apRect++;
782 }
783
784 if (iFitWidthsTotal < (int)GetMargin())
785 {
786 // no leftover rectangle available
787 apRect->SetRectEmpty();
788 }
789 else
790 {
791 // calc leftover rectangle
792 apRect->operator=(arRectCell);
793 apRect->left += iWidthSoFar;
794 }
795
796 // I've been assuming that all controls are left-aligned. Programmer
797 // may have defined controls as right-aligned, too. Note that left
798 // and right controls can be declared in the array in any order
799 // Ok, here's the kludge. Since I know that the heights of each control
800 // are the same, use these unused height values in the CRect array to
801 // help me calculate the proper order of the widths
802 apRect = pRectSav;
803 int iSavedTop = arRectCell.left; // seed first result
804
805 for (i1 = 0; i1 < iCtlNbr; i1++)
806 {
807 if (iSpinBoxDownIdx == i1)
808 {
809 // skip down arrow of spin box for calculations
810 apRect++;
811 continue;
812 }
813
814 // analyzing just Left-aligned controls
815 CtlAlign = GetDrawCtlAlign(i1);
816 ASSERT(CtlAlign != CTL_ALIGN_CENTER); // should've taken care of
817
818 if (CtlAlign == CTL_ALIGN_LEFT)
819 {
820 apRect->top = iSavedTop;
821 iSavedTop += apRect->Width();
822 apRect->bottom = iSavedTop;
823 }
824
825 apRect++;
826 }
827
828 // text rectangle appears between left and right aligned controls
829 // and width has been saved in last rectangle
830 apRect = pRectSav + iCtlNbr;
831 apRect->top = iSavedTop + GetMargin(); // put some margin so doesn't overwrite
832 iSavedTop += apRect->Width();
833 apRect->bottom = iSavedTop;
834 iSavedTop += GetMargin(); // some more margin...
835 apRect = pRectSav;
836
837 for (i1 = 0; i1 < iCtlNbr; i1++)
838 {
839 if (iSpinBoxDownIdx == i1)
840 {
841 // skip down arrow of spin box for calculations
842 apRect++;
843 continue;
844 }
845
846 // finally, look at right-aligned controls
847 CtlAlign = GetDrawCtlAlign(i1);
848
849 if (CtlAlign == CTL_ALIGN_RIGHT)
850 {
851 apRect->top = iSavedTop;
852 iSavedTop += apRect->Width();
853 apRect->bottom = iSavedTop;
854 }
855
856 apRect++;
857 }
858
859 // flip everything back and we're done
860 apRect = pRectSav;
861
862 for (i1 = 0; i1 <= iCtlNbr; i1++) // note that I'll get leftover rect, too
863 {
864 apRect->left = apRect->top;
865
866 if (apRect->left < arRectCell.left)
867 {
868 apRect->left = arRectCell.left; // can't go beyond cell
869 }
870
871 if (apRect->left > arRectCell.right)
872 {
873 apRect->left = arRectCell.right;
874 }
875
876 apRect->right = apRect->bottom;
877
878 if (apRect->right > arRectCell.right)
879 {
880 apRect->right = arRectCell.right;
881 }
882
883 if (apRect->right < arRectCell.left)
884 {
885 apRect->right = arRectCell.left;
886 }
887
888 apRect->bottom = arRectCell.bottom;
889 apRect->top = arRectCell.top;
890 apRect++;
891 }
892
893 // but wait -- special calculations for the spin box
894 if (iSpinBoxDownIdx >= 0)
895 {
896 CRect* pRectSpinUp = pRectSav + iSpinBoxDownIdx - 1;
897 apRect = pRectSav + iSpinBoxDownIdx;
898 apRect->left = pRectSpinUp->left;
899 apRect->right = pRectSpinUp->right;
900 apRect->bottom = pRectSpinUp->bottom;
901 int iHalf = apRect->top + ((apRect->bottom - apRect->top) / 2);
902 apRect->top = iHalf;
903 pRectSpinUp->bottom = iHalf;
904 }
905
906 return TRUE;
907}

961 iNbrChars++;

973 }

975 pszCtlBtnText++;
976 }

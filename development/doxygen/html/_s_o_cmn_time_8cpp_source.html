94/*
95 From RFC 882
96
97 5. DATE AND TIME SPECIFICATION
98
99 5.1. SYNTAX
100
101 date-time = [ day "," ] date time ; dd mm yy
102 ; hh:mm:ss zzz
103
104 day = "Mon" / "Tue" / "Wed" / "Thu"
105 / "Fri" / "Sat" / "Sun"
106
107 date = 1*2DIGIT month 2DIGIT ; day month year
108 ; e.g. 20 Jun 82
109
110 month = "Jan" / "Feb" / "Mar" / "Apr"
111 / "May" / "Jun" / "Jul" / "Aug"
112 / "Sep" / "Oct" / "Nov" / "Dec"
113
114 time = hour zone ; ANSI and Military
115
116 hour = 2DIGIT ":" 2DIGIT [":" 2DIGIT]
117 ; 00:00:00 - 23:59:59
118
119 zone = "UT" / "GMT" ; Universal Time
120 ; North American : UT
121 / "EST" / "EDT" ; Eastern: - 5/ - 4
122 / "CST" / "CDT" ; Central: - 6/ - 5
123 / "MST" / "MDT" ; Mountain: - 7/ - 6
124 / "PST" / "PDT" ; Pacific: - 8/ - 7
125 / 1ALPHA ; Military: Z = UT;
126 ; A:-1; (J not used)
127 ; M:-12; N:+1; Y:+12
128 / ( ("+" / "-") 4DIGIT ) ; Local differential
129 ; hours+min. (HHMM)
130
131 5.2. SEMANTICS
132
133 If included, day-of-week must be the day implied by the date
134 specification.
135
136 Time zone may be indicated in several ways. "UT" is Univer-
137 sal Time (formerly called "Greenwich Mean Time"); "GMT" is per-
138 mitted as a reference to Universal Time. The military standard
139 uses a single character for each zone. "Z" is Universal Time.
140 "A" indicates one hour earlier, and "M" indicates 12 hours ear-
141 lier; "N" is one hour later, and "Y" is 12 hours later. The
142 letter "J" is not used. The other remaining two forms are taken
143 from ANSI standard X3.51-1975. One allows explicit indication of
144 the amount of offset from UT; the other uses common 3-character
145 strings for indicating time zones in North America.
146*/

183/*
184A single lexical representation, which is a subset of the lexical representations
185allowed by [ISO 8601], is allowed for dateTime. This lexical representation is the
186[ISO 8601] extended format CCYY-MM-DDThh:mm:ss where "CC" represents the century,
187"YY" the year, "MM" the month and "DD" the day, preceded by an optional leading "-"
188sign to indicate a negative number. If the sign is omitted, "+" is assumed. The
189letter "T" is the date/time separator and "hh", "mm", "ss" represent hour, minute
190 and second respectively. Additional digits can be used to increase the precision
191of fractional seconds if desired i.e the format ss.ss... with any number of digits
192after the decimal point is supported. The fractional seconds part is optional; other
193parts of the lexical form are not optional. To accommodate year values greater than
1949999 additional digits can be added to the left of this representation. Leading
195zeros are required if the year value would otherwise have fewer than four digits;
196otherwise they are forbidden. The year 0000 is prohibited.
197
198The CCYY field must have at least four digits, the MM, DD, SS, hh, mm and ss fields
199exactly two digits each (not counting fractional seconds); leading zeroes must be
200used if the field would otherwise have too few digits.
201
202This representation may be immediately followed by a "Z" to indicate Coordinated
203Universal Time (UTC) or, to indicate the time zone, i.e. the difference between
204the local time and Coordinated Universal Time, immediately followed by a sign, +
205or -, followed by the difference from UTC represented as hh:mm (note: the minutes
206part is required). See ISO 8601 Date and Time Formats for details about legal
207values in the various fields. If the time zone is included, both hours and minutes
208must be present.
209
210For example, to indicate 1:20 pm on May the 31st, 1999 for Eastern Standard Time
211which is 5 hours behind Coordinated Universal Time (UTC), one would write:
2121999-05-31T13:20:00-05:00.
213
214*/

216SOCmnString SOCmnDateTime::getXMLDateTime(void) const
217{
218 SOCmnString sz;
219 LONG timeBias;
220#ifdef SOOS_WINDOWS
221 FILETIME ftLocal;
222 SYSTEMTIME locT;
223 TIME_ZONE_INFORMATION timeZone;
224 DWORD timeZoneRet;
225 timeZoneRet = GetTimeZoneInformation(&timeZone);
226 timeBias = timeZone.Bias;
227
228 switch (timeZoneRet)
229 {
230 case TIME_ZONE_ID_STANDARD:
231 timeBias += timeZone.StandardBias;
232 break;
233
234 case TIME_ZONE_ID_DAYLIGHT:
235 timeBias += timeZone.DaylightBias;
236 break;
237 }
238
239 timeBias = -timeBias; /* WINDOWS uses different sign */
240 FileTimeToLocalFileTime(&m_filetime, &ftLocal);
241 FileTimeToSystemTime(&ftLocal, &locT);
242#endif // SOOS_WINDOWS
243#ifdef SOOS_LINUX
244 struct tm locT, sysT;
245 gmtime_r(&m_timeval.tv_sec, &sysT);
246 localtime_r(&m_timeval.tv_sec, &locT);
247 sysT.tm_isdst = locT.tm_isdst; /* <-- very important!!! */
248 timeBias = mktime(&locT) - mktime(&sysT); /* CET=3600, CEST=7200 */
249 timeBias /= 60; /* convert to minutes */
250#endif
251 sz.format(_T("%u-%2.2u-%2.2uT%2.2u:%2.2u:%2.2u.%3.3u%c%2.2u:%2.2u"),
252#ifdef SOOS_WINDOWS
253 locT.wYear, locT.wMonth, locT.wDay,
254 locT.wHour, locT.wMinute, locT.wSecond,
255 locT.wMilliseconds,
256#endif // SOOS_WINDOWS
257#ifdef SOOS_LINUX
258 locT.tm_year + 1900, locT.tm_mon + 1, locT.tm_mday,
259 locT.tm_hour, locT.tm_min, locT.tm_sec,
260 (int)(m_timeval.tv_usec / 1000),
261#endif
262 (timeBias < 0) ? _T('-') : _T('+'),
263 abs(timeBias) / 60, abs(timeBias) % 60
264 );
265 return sz;
266}

268BOOL SOCmnDateTime::setXMLDateTime(LPCTSTR xmlDateTime)
269{
270 if (_tcslen(xmlDateTime) < 19)
271 {
272 return FALSE;
273 }
274
275 if ((xmlDateTime[4] != _T('-')) || (xmlDateTime[7] != _T('-')) || (xmlDateTime[10] != _T('T')) ||
276 (xmlDateTime[13] != _T(':')) || (xmlDateTime[16] != _T(':')))
277 {
278 return FALSE;
279 }
280
281 LPCTSTR curP;
282#ifdef SOOS_WINDOWS
283 FILETIME tvT;
284 SYSTEMTIME sysT;
285 sysT.wYear = (WORD)_tcstoul(xmlDateTime + 0, NULL, 10);
286 sysT.wMonth = (WORD)_tcstoul(xmlDateTime + 5, NULL, 10);
287 sysT.wDay = (WORD)_tcstoul(xmlDateTime + 8, NULL, 10);
288 sysT.wHour = (WORD)_tcstoul(xmlDateTime + 11, NULL, 10);
289 sysT.wMinute = (WORD)_tcstoul(xmlDateTime + 14, NULL, 10);
290 sysT.wSecond = (WORD)_tcstoul(xmlDateTime + 17, NULL, 10);
291 sysT.wMilliseconds = 0;
292 SystemTimeToFileTime(&sysT, &tvT);
293#endif // SOOS_WINDOWS
294#ifdef SOOS_LINUX
295 struct timeval tvT;
296 struct tm locT, sysT;
297 locT.tm_year = (WORD)_tcstoul(xmlDateTime + 0, NULL, 10) - 1900;
298 locT.tm_mon = (WORD)_tcstoul(xmlDateTime + 5, NULL, 10) - 1;
299 locT.tm_mday = (WORD)_tcstoul(xmlDateTime + 8, NULL, 10);
300 locT.tm_hour = (WORD)_tcstoul(xmlDateTime + 11, NULL, 10);
301 locT.tm_min = (WORD)_tcstoul(xmlDateTime + 14, NULL, 10);
302 locT.tm_sec = (WORD)_tcstoul(xmlDateTime + 17, NULL, 10);
303 locT.tm_isdst = 0; /* assume local time w/o dst correction */
304 tvT.tv_sec = mktime(&locT);
305 gmtime_r(&tvT.tv_sec, &sysT);
306 /* add local time zone offset */
307 tvT.tv_sec += tvT.tv_sec - mktime(&sysT);
308 tvT.tv_usec = 0;
309#endif
310 curP = xmlDateTime + 19;
311
312 if (*curP == _T('.'))
313 {
314 DWORD m;
315 curP++;
316#ifdef SOOS_WINDOWS
317 m = 1000000; // 100 nanoseconds
318#endif // SOOS_WINDOWS
319#ifdef SOOS_LINUX
320 m = 100000; // 1 microsecod
321#endif
322
323 while (TRUE)
324 {
325 if (_istdigit(*curP) != 0)
326 {
327 DWORD dg = *curP - _T('0');
328#ifdef SOOS_WINDOWS
329 ULARGE_INTEGER temp;
330 temp.LowPart = tvT.dwLowDateTime;
331 temp.HighPart = tvT.dwHighDateTime;
332 temp.QuadPart += dg * m;;
333 tvT.dwLowDateTime = temp.LowPart;
334 tvT.dwHighDateTime = temp.HighPart;
335#endif // SOOS_WINDOWS
336#ifdef SOOS_LINUX
337 tvT.tv_usec += dg * m;
338#endif
339 m /= 10;
340 curP++;
341 }
342 else
343 {
344 break;
345 }
346 }
347 }
348
349 switch (*curP)
350 {
351 case _T('+'):
352 case _T('-'):
353 {
354 if (_tcslen(curP) >= 5)
355 {
356 DWORD loc;
357 loc = (_tcstoul(curP + 1, NULL, 10) * 60) + _tcstoul(curP + 4, NULL, 10);
358#ifdef SOOS_WINDOWS
359
360 if (*curP == _T('-'))
361 {
362 ULARGE_INTEGER temp;
363 temp.LowPart = tvT.dwLowDateTime;
364 temp.HighPart = tvT.dwHighDateTime;
365 temp.QuadPart += ((LONGLONG)loc) * 600000000;
366 tvT.dwLowDateTime = temp.LowPart;
367 tvT.dwHighDateTime = temp.HighPart;
368 }
369 else
370 {
371 ULARGE_INTEGER temp;
372 temp.LowPart = tvT.dwLowDateTime;
373 temp.HighPart = tvT.dwHighDateTime;
374 temp.QuadPart -= ((LONGLONG)loc) * 600000000;
375 tvT.dwLowDateTime = temp.LowPart;
376 tvT.dwHighDateTime = temp.HighPart;
377 }
378
379#endif // SOOS_WINDOWS
380#ifdef SOOS_LINUX
381
382 if (*curP == _T('-'))
383 {
384 tvT.tv_sec += loc * 60;
385 }
386 else
387 {
388 tvT.tv_sec -= loc * 60;
389 }
390
391#endif
392 }
393 else
394 {
395 return FALSE;
396 }
397 }
398 break;
399
400 case _T('Z'):
401 break;
402
403 case _T('\0'): // time zone _may_ be included.
404 {
405 // we use the local time zone here.
406#ifdef SOOS_WINDOWS
407 FILETIME gmT;
408 LocalFileTimeToFileTime(&tvT, &gmT);
409 tvT = gmT;
410#endif // SOOS_WINDOWS
411#ifdef SOOS_LINUX
412 gmtime_r(&tvT.tv_sec, &locT);
413 locT.tm_isdst = -1;
414 tvT.tv_sec = mktime(&locT);
415#endif
416 }
417 break;
418
419 default:
420 return FALSE;
421 break;
422 }
423
424 set(&tvT);
425 return TRUE;
426}

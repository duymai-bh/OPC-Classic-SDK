10 // A simple class that exposes two static Win32 functions.
11 // One is a delegate type and the other is an enumerated type.
12 public class MyWin32 {
13 // Declare the SetConsoleCtrlHandler function
14 // as external and receiving a delegate.
15 [DllImport("Kernel32")]
16 public static extern Boolean SetConsoleCtrlHandler(HandlerRoutine Handler,
17 Boolean Add);
18
19 // A delegate type to be used as the handler routine
20 // for SetConsoleCtrlHandler.
21 public delegate Boolean HandlerRoutine(CtrlTypes CtrlType);
22
23 // An enumerated type for the control messages
24 // sent to the handler routine.
25 public enum CtrlTypes {
26 CTRL_C_EVENT = 0,
27 CTRL_BREAK_EVENT,
28 CTRL_CLOSE_EVENT,
29 CTRL_LOGOFF_EVENT = 5,
30 CTRL_SHUTDOWN_EVENT
31 }
32
33 // A private static handler function in the MyApp class.
34 static public Boolean Handler(MyWin32.CtrlTypes CtrlType) {
35 String message = "This message should never be seen!";
36
37 // A switch to handle the event type.
38 switch(CtrlType) {
39 case MyWin32.CtrlTypes.CTRL_C_EVENT:
40 case MyWin32.CtrlTypes.CTRL_BREAK_EVENT:
41 message = "stop execution, since CTRL command!";
42 BISApplication.m_endEvent.Set();
43 break;
44 case MyWin32.CtrlTypes.CTRL_CLOSE_EVENT:
45 message = "A CTRL_CLOSE_EVENT was raised by the user.";
46 break;
47 case MyWin32.CtrlTypes.CTRL_LOGOFF_EVENT:
48 message = "A CTRL_LOGOFF_EVENT was raised by the user.";
49 break;
50 case MyWin32.CtrlTypes.CTRL_SHUTDOWN_EVENT:
51 message = "A CTRL_SHUTDOWN_EVENT was raised by the user.";
52 break;
53 }
54
55 // Use interop to display a message for the type of event.
56 Console.WriteLine(message);
57
58 return true;
59 }
60 }

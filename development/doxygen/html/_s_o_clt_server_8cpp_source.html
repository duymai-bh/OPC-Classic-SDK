220//-----------------------------------------------------------------------------
221// setParentBranch
222//
223BOOL SOCltServer::setParentBranch(IN SOCmnElement* parent)
224{
225 if (!SOCltElement::setParentBranch(parent))
226 {
227 return FALSE;
228 }
229
230 // Install a Connection Monitor task.
231 // Attention: installing the task creates
232 // a reference cycle. The best place to break it
233 // is when parent branch is set to NULL.
234
235 if (parent)
236 {
237 if (0 == m_cookieConnectMonitor)
238 {
239 addConnectionMonitor(5000, 10, 10000, 300000);
240 }
241 }
242 else
243 {
244 removeConnectionMonitor();
245 terminate();
246 m_workThread.shutdown();
247 }
248
249 return TRUE;
250} // setParentBranch

253BOOL SOCltServer::addConnectionMonitor(DWORD checkCycle, DWORD connect1Attempts, DWORD connect1Cycle, DWORD connect2Cycle)
254{
255 removeConnectionMonitor();
256 SOCmnPointer<SOCltTaskConnectionMonitor> task;
257 SOCmnPointer<SOCltEntry> entry = ::getSOCltEntry();
258 SOCmnPointer<SOCltCreator> creator = entry->getCltCreator();
259 task = creator->createTaskConnectionMonitor(this, checkCycle);
260 task->m_attempts = connect1Attempts;
261 task->m_cycle1 = connect1Cycle;
262 task->m_cycle2 = connect2Cycle;
263 return m_workThread.addCyclicTask((SOCltTaskConnectionMonitor*)task, &m_cookieConnectMonitor);
264}

266BOOL SOCltServer::removeConnectionMonitor(void)
267{
268 BOOL ret = FALSE;
269
270 if (0 != m_cookieConnectMonitor)
271 {
272 ret = m_workThread.removeCyclicTask(m_cookieConnectMonitor);
273 m_cookieConnectMonitor = 0;
274 }
275
276 return ret;

309BOOL SOCltServer::convertLCIDtoLocaleID(LCID lcid, SOCmnString& localeId)
310{
311 BOOL ret = TRUE;
312#ifdef SOOS_WINDOWS
313
314 // handle windows default lcids
315 if (lcid == MAKELCID(LANG_SYSTEM_DEFAULT, SORT_DEFAULT))
316 {
317 lcid = GetSystemDefaultLCID();
318 }
319
320 if (lcid == MAKELCID(LANG_USER_DEFAULT, SORT_DEFAULT))
321 {
322 lcid = GetUserDefaultLCID();
323 }
324
325#endif
326
327 switch (lcid)
328 {
329 case MAKELCID(MAKELANGID(LANG_ENGLISH, SUBLANG_ENGLISH_US), SORT_DEFAULT):
330 localeId = _T("en-US");
331 break;
332
333 case MAKELCID(MAKELANGID(LANG_GERMAN, SUBLANG_GERMAN), SORT_DEFAULT):
334 localeId = _T("de-DE");
335 break;
336
337 case MAKELCID(MAKELANGID(LANG_SWEDISH, SUBLANG_SWEDISH), SORT_DEFAULT):
338 localeId = _T("sv-SE");
339 break;
340
341 default:
342 {
343 WORD lang = LANGIDFROMLCID(lcid);
344
345 switch (PRIMARYLANGID(lang))
346 {
347 case LANG_ENGLISH:
348 localeId = _T("en");
349 break;
350
351 case LANG_GERMAN:
352 localeId = _T("de");
353 break;
354
355 case LANG_SWEDISH:
356 localeId = _T("sv");
357 break;
358
359 default:
360 ret = FALSE;
361 localeId = _T("en-US"); // default is enlish US
362 break;
363 }
364 }
365 break;
366 }
367
368 return ret;
369}

57// Used for Get/SetItem calls.
58typedef struct _GV_ITEM
59{
60 int row, col; // Row and Column of item
61 UINT mask; // Mask for use in getting/setting cell data
62 UINT nState; // cell state (focus/hilighted etc)
63 DWORD nFormat; // Format of cell
64 int iImage; // index of the list view item’s icon

67 LPARAM lParam; // 32-bit value to associate with item
68 LOGFONT lfFont; // Cell font
69 UINT nMargin; // Internal cell margin
70 CString strText; // Text in cell
71} GV_ITEM;

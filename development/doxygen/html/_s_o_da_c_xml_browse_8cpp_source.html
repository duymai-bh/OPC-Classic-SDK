61BOOL SODaCXmlNameSpaceBrowserImpl::expand30(IN SOCltBrowseObject* pParent, IN LPCTSTR itemID, IN LPCTSTR itemPath, IN DWORD type, IN SOCmnString& contPoint, OUT SOCmnList<SOCltBrowseObject> &objList)
62{
63 SOCmnPointer<SODaCEntry> entryDaC = ::getSODaCEntry();
64 SOCmnPointer<SODaCCreator> creator = entryDaC->getCreator();
65 SODaCNameSpaceBrowser* pBrowser = getObjectSODaCNameSpaceBrowser();
66 SOCmnPointer<SODaCServer> server = (SODaCServer*)pBrowser->getServer();
67 DWORD browseFilter = 0;
68 SOCmnString cpSZ;
69 USES_CONVERSION;
70
71 if (SOCLT_XML_CHECK_DEMO)
72 {
73 return FALSE;
74 }
75
76 if (!contPoint.isEmpty())
77 {
78 cpSZ = contPoint;
79 }
80
81 contPoint.empty();
82
83 if (!objList.isInit())
84 {
85 objList.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_OBJECT);
86 }
87
88 // calculate the browse type
89 if ((SOCMNOBJECT_TYPE_NODE | SOCMNOBJECT_TYPE_ITEM) == (type & (SOCMNOBJECT_TYPE_NODE | SOCMNOBJECT_TYPE_ITEM)))
90 {
91 browseFilter = OPC_BROWSE_FILTER_ALL;
92 }
93 else
94 {
95 if (SOCMNOBJECT_TYPE_NODE == (type & SOCMNOBJECT_TYPE_NODE))
96 {
97 browseFilter = OPC_BROWSE_FILTER_BRANCHES;
98 }
99 else if (SOCMNOBJECT_TYPE_ITEM == (type & SOCMNOBJECT_TYPE_ITEM))
100 {
101 browseFilter = OPC_BROWSE_FILTER_ITEMS;
102 }
103 }
104
105 if (browseFilter != 0)
106 {
107 SOCmnString elementNameFilter = pBrowser->getElementNameFilter();
108 SOCmnString vendorFilter = pBrowser->getVendorFilter();
109 SOCmnPointer<SOCltHTTPConnection> httpCon = server->getHTTPConnection();
110 SOCmnPointer<SODaCXmlCall> httpCall;
111 HRESULT res;
112 SOCMN_ALLOCATE_OBJECT_C(httpCall, SODaCXmlCall(httpCon), (SODaCXmlCall*));
113 httpCall->browse(itemID, itemPath, cpSZ, pBrowser->getMaxElements(), browseFilter, elementNameFilter, vendorFilter, pBrowser->getReturnProperties(), pBrowser->getRetrievePropertyValues());
114 server->beforeInterfaceCall();
115 res = httpCon->invoke(httpCall, TRUE);
116 server->afterInterfaceCall();
117
118 if (SUCCEEDED(res))
119 {
120 SOCmnPointer<SODaCXmlResponse> response;
121 res = httpCall->getBrowseResponse(&response);
122
123 if (SUCCEEDED(res))
124 {
125 SOCmnStringList xmlOptions(response->getResponseOptions());
126 SOCmnList<SODaCXmlResponseData> xmlData(response->getResponseData());
127 SOCmnStringEx convHlp;
128 SODaCXmlResponseData* pResData;
129 SOCmnListPosition pos;
130 SOCmnPointer<SOCltBrowseObject> bO;
131 SOCmnList<SODaCProperty> propList;
132 xmlOptions.findKey(_T("ContinuationPoint"), &contPoint);
133 pos = xmlData.getStartPosition();
134
135 while (pos)
136 {
137 pResData = xmlData.getNext(pos);
138
139 if (pResData->m_type == SODACXMLRESPONSEDATA_TYPE_ELEMENT)
140 {
141 if ((pResData->m_hasChildren) || // children
142 ((!pResData->m_hasChildren) && (!pResData->m_isItem))) // no children and no item
143 {
144 // node
145 SODaCNode* pNode;
146 DWORD flags = 0;
147 pNode = creator->createNode(pParent, pResData->m_name);
148 convHlp = pResData->m_itemName;
149 pNode->setItemID(convHlp, itemID);
150 convHlp = pResData->m_itemPath;
151 pNode->setItemPath(convHlp);
152
153 if (pResData->m_isItem)
154 {
155 flags |= OPC_BROWSE_ISITEM;
156 }
157
158 if (pResData->m_hasChildren)
159 {
160 flags |= OPC_BROWSE_HASCHILDREN;
161 }
162
163 pNode->setFlags(flags);
164 bO = pNode;
165 propList = pNode->getPropertyList();
166 }
167 else
168 {
169 // tag
170 SODaCTag* pTag;
171 DWORD flags = 0;
172 pTag = creator->createTag(pParent, pResData->m_name);
173 convHlp = pResData->m_itemName;
174 pTag->setItemID(convHlp, itemID);
175 convHlp = pResData->m_itemPath;
176 pTag->setItemPath(convHlp);
177
178 if (pResData->m_isItem)
179 {
180 flags |= OPC_BROWSE_ISITEM;
181 }
182
183 if (pResData->m_hasChildren)
184 {
185 flags |= OPC_BROWSE_HASCHILDREN;
186 }
187
188 pTag->setFlags(flags);
189 bO = pTag;
190 propList = pTag->getPropertyList();
191 }
192
193 objList.add(bO);
194 }
195
196 if (pResData->m_type == SODACXMLRESPONSEDATA_TYPE_PROPERTY)
197 {
198 SOCmnVariant propValue;
199 SOCmnPointer<SODaCProperty> prop;
200 SOCmnStringEx propName;
201 VARTYPE prDT = VT_EMPTY;
202 propName = pResData->m_name;
203 DWORD propID = server->getPropertyIDByName(propName);
204 SOCmnVariant::getVARTYPE(pResData->m_value.m_datatype, prDT);
205 convHlp = pResData->m_description;
206 prop = creator->createProperty(bO, convHlp, propID, prDT, bO->getItemID(), bO->getItemPath());
207 convHlp = pResData->m_itemName;
208 prop->setItemID(convHlp, itemID);
209 convHlp = pResData->m_name;
210 prop->setPropertyName(convHlp);
211
212 if (propValue.initFromXMLValue(&pResData->m_value))
213 {
214 if ((propID == 1) && (propValue.vt == VT_BSTR)) // datatype
215 {
216 SOCmnString propDT;
217 propValue.getBSTR(propDT);
218
219 if (response->includesNamespacePrefix(propDT))
220 {
221 // property name with namespace
222 response->expandNamespace(propDT);
223 response->removeOPCXMLDANamespace(propDT); // remove default OPC XML DA NS
224 propValue.setBSTR(propDT);
225 }
226 }
227
228 prop->setValue(propValue);
229 }
230
231 propList.add(prop);
232 }
233 }
234 }
235 }
236
237 _TRACEO_LEVEL_FROM_RES(res, SOCMNTRACE_G_OPC_CALL_C, server, (_T("Browse [0x%lX]"), res));
238 }
239
240 if (SOCMNOBJECT_TYPE_PROPERTY == (type & SOCMNOBJECT_TYPE_PROPERTY))
241 {
242 SOCmnList<SODaCPropertyData> propList;
243 HRESULT res;
244 res = server->getPropertiesData(itemID, itemPath, (pBrowser->getRetrievePropertyValues()) ? SODAC_PROPDATA_VALUE : 0, propList);
245
246 if (SUCCEEDED(res))
247 {
248 SOCmnPointer<SODaCProperty> prop;
249 SODaCPropertyData* propData;
250 SOCmnListPosition posP;
251 DWORD propID;
252 SOCmnStringEx convHlp;
253 SOCmnStringEx propName;
254 posP = propList.getStartPosition();
255
256 while (posP)
257 {
258 propData = propList.getNext(posP);
259 propName = propData->m_name;
260 propID = server->getPropertyIDByName(propName);
261 convHlp = propData->m_description;
262 prop = creator->createProperty(pParent, convHlp, propData->m_propertyID, propData->m_datatype, itemID, itemPath);
263 convHlp = propData->m_itemID;
264 prop->setItemID(convHlp, itemID);
265 prop->setPropertyName(propName);
266 prop->setValue(propData->m_value);
267 objList.add(prop);
268 }
269 }
270
271 propList.destroy();
272 }
273
274 return TRUE;
275}

94void ClientDASubscription::onReadAdvise(
95 IN BYTE dataChangeOrRead,
96 IN DWORD transactionId,
97 IN HRESULT /* masterQuality */,
98 IN HRESULT masterError,
99 IN DWORD itemCnt,
100 IN SODaCItem** ppItems,
101 IN SOCmnVariant* pValues,
102 IN WORD* pQualities,
103 IN SOCmnDateTime* pTimestamps,
104 IN HRESULT* pErrors)
105{
106 OTCObjectContext objectContext;
107 OTCObjectData* pItemData;
108 OTCValueData* pValueData;
109 LPCTSTR* pItemIDs = NULL;
110 LPCTSTR* pItemPaths = NULL;
111 DWORD i;
112 getOTCObjectContext((OTCObjectHandle)this->getOpcHandle(), &objectContext);
113 pItemData = (OTCObjectData*) malloc(itemCnt * sizeof(OTCObjectData));
114 pValueData = (OTCValueData*)malloc(itemCnt * sizeof(OTCValueData));
115
116 if (transactionId != 0)
117 {
118 pItemIDs = (LPCTSTR*) malloc(itemCnt * sizeof(LPCTSTR));
119 pItemPaths = (LPCTSTR*) malloc(itemCnt * sizeof(LPCTSTR));
120 }
121
122 for (i = 0; i < itemCnt; i++)
123 {
124 pItemData[i].m_objectHandle = 0;
125 pItemData[i].m_userData = (unsigned long)NULL;
126
127 if (ppItems[i])
128 {
129 pItemData[i].m_objectHandle = (OTCObjectHandle)ppItems[i]->getOpcHandle();
130 pItemData[i].m_userData = ((ClientDAItem*)ppItems[i])->getOTCUserData();
131 pValueData[i].m_pValue = pValues[i];
132 pValueData[i].m_quality = pQualities[i];
133 pTimestamps[i].get(&pValueData[i].m_timestamp);
134
135 if (transactionId != 0)
136 {
137 pItemIDs[i] = _tcsdup((LPCTSTR)ppItems[i]->getItemID());
138 pItemPaths[i] = _tcsdup((LPCTSTR)ppItems[i]->getItemPath());
139 }
140 }
141 }
142
143 if ((dataChangeOrRead) && (g_callbackFunctions.m_OTCOnDataChange))
144 {
145 SOCMN_TRY
146 {
147 g_callbackFunctions.m_OTCOnDataChange(transactionId, objectContext, itemCnt, pItemData, pValueData, (LONG*)pErrors);
148 }
149 SOCMN_CATCH(...)
150 {
151 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_CLIENT, (_T("ClientDASubscription::onReadAdvise - Exception occurred in vendor application callback: OTCOnDataChange")));
152 }
153 }
154 else if ((!dataChangeOrRead) && (g_callbackFunctions.m_OTCOnReadComplete))
155 {
156 SOCMN_TRY
157 {
158 g_callbackFunctions.m_OTCOnReadComplete(transactionId, objectContext, masterError, itemCnt, pItemData, (LPTSTR*)pItemIDs, (LPTSTR*)pItemPaths, pValueData, (LONG*)pErrors);
159 }
160 SOCMN_CATCH(...)
161 {
162 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_CLIENT, (_T("ClientDASubscription::onReadAdvise - Exception occurred in vendor application callback: OTCOnReadComplete")));
163 }
164 }
165
166 if (transactionId != 0)
167 {
168 for (i = 0; i < itemCnt; i++)
169 {
170 free((void*)pItemIDs[i]);
171 free((void*)pItemPaths[i]);
172 }
173
174 free(pItemIDs);
175 free(pItemPaths);
176 }
177
178 free(pItemData);
179 free(pValueData);
180}

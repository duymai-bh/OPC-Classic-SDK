58int _tmain(int argc, _TCHAR* argv[])
59{
60 _tprintf(_T("Usage:\n"));
61 _tprintf(_T("DABrowse <server URL> [<address space element id>]\n\n"));
62
63 if (argc < 2)
64 {
65 return -1;
66 }
67
68 OTCObjectHandle sessionHandle;
69 LPTSTR addressSpaceElementID = NULL;
70
71 if (argc == 3)
72 {
73 addressSpaceElementID = argv[2];
74 }
75
76 OTActivate(OTC_FEATURE_DA_CLIENT, _T("SAMPLE"));
77
78 if (SUCCEEDED(OTCInitialize(&g_initData)))
79 {
80 OTCAddDASession(argv[1], 0, &sessionHandle);
81 OTCChangeTargetState(sessionHandle, OTC_STATE_CONNECTED, TRUE);
82
83 if (S_OK == OTCPerformStateTransition(sessionHandle, TRUE, NULL))
84 {
85 _tprintf(_T("Connected to server: %s\n\n"), argv[1]);
86
87 if (addressSpaceElementID == NULL) // browse the whole server
88 {
89 BrowseAddressSpace(sessionHandle, 0, 0);
90 }
91 else // browse only one node (asynchronous)
92 {
93 OTCAdvise(&g_callbackFunctions);
94 BrowseAddressSpaceElement(sessionHandle, addressSpaceElementID);
95 }
96 }
97 else
98 {
99 _tprintf(_T("Can't connect to server: %s\n"), argv[1]);
100 }
101
102 OTCTerminate();
103 }
104
105 return 0;
106}

108void BrowseAddressSpace(IN OTCObjectHandle sessionHandle, IN OTCObjectHandle addressSpaceElementHandle, IN DWORD indent)
109{
110 OTCAddressSpaceBrowseOptions browseOptions;
111 OTCAddressSpaceElementData* pAddressSpaceElement;
112 DWORD addressSpaceElementDataCount;
113 DWORD i;
114 // browse all child leaves
115 memset(&browseOptions, 0, sizeof(OTCAddressSpaceBrowseOptions));
116 browseOptions.m_type = OT_ADDRESSSPACETYPE_LEAF;
117 browseOptions.m_retrieveItemID = TRUE;
118 pAddressSpaceElement = NULL;
119
120 if (SUCCEEDED(OTCBrowseAddressSpace(sessionHandle, addressSpaceElementHandle, NULL, NULL, &browseOptions,
121 &addressSpaceElementDataCount, &pAddressSpaceElement, NULL)))
122 {
123 for (i = 0; i < addressSpaceElementDataCount; i++)
124 {
125 PrintChildren(&pAddressSpaceElement[i], indent);
126 OTCReleaseAddressSpaceElement(pAddressSpaceElement[i].m_objectHandle, FALSE);
127 }
128 }
129
130 for (i = 0; i < addressSpaceElementDataCount; i++)
131 {
132 OTFreeMemory(pAddressSpaceElement[i].m_name);
133 }
134
135 if (pAddressSpaceElement)
136 {
137 OTFreeMemory(pAddressSpaceElement);
138 }
139
140 // browse all child nodes
141 browseOptions.m_type = OT_ADDRESSSPACETYPE_BRANCH;
142 browseOptions.m_retrieveItemID = TRUE;
143 pAddressSpaceElement = NULL;
144
145 if (SUCCEEDED(OTCBrowseAddressSpace(sessionHandle, addressSpaceElementHandle, NULL, NULL, &browseOptions,
146 &addressSpaceElementDataCount, &pAddressSpaceElement, NULL)))
147 {
148 for (i = 0; i < addressSpaceElementDataCount; i++)
149 {
150 PrintChildren(&pAddressSpaceElement[i], indent);
151 BrowseAddressSpace(sessionHandle, pAddressSpaceElement[i].m_objectHandle, indent + 1);
152 OTCReleaseAddressSpaceElement(pAddressSpaceElement[i].m_objectHandle, FALSE);
153 }
154 }
155
156 for (i = 0; i < addressSpaceElementDataCount; i++)
157 {
158 OTFreeMemory(pAddressSpaceElement[i].m_name);
159 }
160
161 if (pAddressSpaceElement)
162 {
163 OTFreeMemory(pAddressSpaceElement);
164 }
165}

208void PrintChildren(OTCAddressSpaceElementData* pAddressSpaceElement, DWORD indent)
209{
210 TCHAR szNameAndIndent[41];
211 LPCTSTR szType;
212
213 switch (pAddressSpaceElement->m_type)
214 {
215 case OT_ADDRESSSPACETYPE_BRANCH:
216 szType = _T("B ");
217 break;
218
219 case OT_ADDRESSSPACETYPE_BRANCH | OT_ADDRESSSPACETYPE_ITEM :
220 szType = _T("BI");
221 break;
222
223 case OT_ADDRESSSPACETYPE_LEAF | OT_ADDRESSSPACETYPE_ITEM :
224 szType = _T("LI");
225 break;
226 }
227
228 memset(szNameAndIndent, _T(' '), indent * sizeof(TCHAR));
229 _tcsncpy(szNameAndIndent + indent, pAddressSpaceElement->m_name, 40 - indent);
230 szNameAndIndent[40] = _T('\0');
231 _tprintf(_T("%-40.40s | %2.2s | %s\n"), szNameAndIndent, szType, pAddressSpaceElement->m_itemID);
232}

415{

418 // destroy name space

439}

695//-----------------------------------------------------------------------------
696// createNewNameSpaceRoot
697//
698// - release the old namespace object -> a new one is created via the next access
699//
700void SODaSEntry::createNewNameSpaceRoot(
701 IN OPTIONAL BYTE serverInstance)
702{
703 if (serverInstance >= SODAS_SERVER_INSTANCE_COUNT)
704 {
705 return;
706 }
707
708 if (m_nsRoot[serverInstance] != NULL)
709 {
710 m_nsRoot[serverInstance]->release();
711 m_nsRoot[serverInstance] = NULL;
712 }
713
714 // reset the browse position of all servers to the new namespace root
715 SOCmnList<SODaSServer> serverList((SOCmnList<SODaSServer> *)m_objectRoot->getBranchList());
716 SOCmnListPosition posSrv;
717 SODaSServer* server;
718 posSrv = serverList.getStartPosition();
719
720 while (posSrv)
721 {
722 server = serverList.getNext(posSrv);
723 server->changeBrowsePosition(SOSrvBrowser::dirTo, _T(""));
724 }
725
726 serverList.autoUnlock();
727} // createNewNameSpaceRoot

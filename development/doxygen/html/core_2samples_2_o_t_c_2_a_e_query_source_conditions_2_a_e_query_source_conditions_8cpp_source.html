65int _tmain(int argc, _TCHAR* argv[])
66{
67 _tprintf(_T("Usage:\n"));
68 _tprintf(_T("AEQuerySourceConditions [/s][/a] <server URL> <source path>\n\n"));
69
70 if (argc < 3)
71 {
72 return -1;
73 }
74
75 OTCObjectHandle sessionHandle;
76 LPTSTR url = NULL;
77 LPTSTR sourcePath = NULL;
78 OTCExecutionOptions executionOptions;
79 executionOptions.m_executionType = OTC_EXECUTIONTYPE_SYCHRONOUS;
80 executionOptions.m_executionContext = 0;
81
82 if (argc == 3)
83 {
84 url = argv[1];
85 sourcePath = argv[2];
86 }
87 else if (argc > 3)
88 {
89 if (_tcsicmp(argv[1], _T("/a")) == 0)
90 {
91 executionOptions.m_executionType = OTC_EXECUTIONTYPE_ASYCHRONOUS;
92 }
93
94 url = argv[2];
95 sourcePath = argv[3];
96 }
97
98 OTActivate(OTC_FEATURE_AE_CLIENT, _T("SAMPLE"));
99
100 if (SUCCEEDED(OTCInitialize(&g_initData)))
101 {
102 OTCAdvise(&g_callbackFunctions);
103 OTCAddAESession(url, 0, &sessionHandle);
104 OTCChangeTargetState(sessionHandle, OTC_STATE_CONNECTED, TRUE);
105
106 if (S_OK == OTCPerformStateTransition(sessionHandle, TRUE, NULL))
107 {
108 _tprintf(_T("Connected to server: %s\n\n"), url);
109 LONG result;
110 DWORD conditionCount;
111 LPTSTR* pConditionNames;
112 DWORD i;
113 _tprintf(_T("Query source conditions of: %s\n"), sourcePath);
114
115 if (SUCCEEDED(result = OTCQueryAESourceConditions(sessionHandle, sourcePath, &conditionCount, &pConditionNames, &executionOptions)))
116 {
117 if (executionOptions.m_executionType == OTC_EXECUTIONTYPE_SYCHRONOUS)
118 {
119 for (i = 0; i < conditionCount; i++)
120 {
121 printf(_T("[%3.3d] %s"), i, pConditionNames[i]);
122 OTFreeMemory(pConditionNames[i]);
123 }
124 }
125 else
126 {
127 while (!g_gotCallback)
128 {
129 Sleep(10);
130 }
131 }
132 }
133 else
134 {
135 _tprintf(_T("Failed to query the conditions of source: %s 0x%X\n"), sourcePath, result);
136 }
137 }
138 else
139 {
140 _tprintf(_T("Can't connect to server: %s\n"), argv[1]);
141 }
142
143 OTCTerminate();
144 }
145
146 return 0;
147}

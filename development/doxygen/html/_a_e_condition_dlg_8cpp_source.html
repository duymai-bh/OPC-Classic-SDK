23tstring MyAeSubscription::getStateToString(EnumConditionState state)
24{
25 tstring stateToString = _T("");
26
27 if ((state & EnumConditionState_ACTIVE) == EnumConditionState_ACTIVE)
28 {
29 stateToString += _T(" ACT");
30 }
31
32 if ((state & EnumConditionState_ENABLED) == EnumConditionState_ENABLED)
33 {
34 stateToString += _T(" ENA");
35 }
36
37 if ((state & EnumConditionState_ACKED) == EnumConditionState_ACKED)
38 {
39 stateToString += _T(" ACK");
40 }
41
42 if (state == 0)
43 {
44 stateToString = _T("DIS");
45 }
46
47 return stateToString;
48} //end getStateToString

50void MyAeSubscription::handleAeConditionsChanged(const std::vector<AeCondition*>& someAeConditions)
51{
52 unsigned long i;
53 int nItem;
54 CAEConditionDlg::conditionList.DeleteAllItems();
55 ASSERT(CAEConditionDlg::conditionList.GetItemCount() == 0);
56
57 for (i = 0; i < someAeConditions.size(); i++)
58 {
59 AeCondition* condition = someAeConditions[i];
60 tstring state = getStateToString(condition->getState());
61 LPTSTR lvItemText = _tcsdup(state.c_str());
62 LVITEM lvItem;
63 lvItem.mask = LVIF_TEXT | LVIF_PARAM;
64 lvItem.lParam = (LPARAM)condition;
65 lvItem.iItem = 0;
66 lvItem.iSubItem = 0;
67 lvItem.pszText = lvItemText;
68 nItem = CAEConditionDlg::conditionList.InsertItem(&lvItem);
69 _TCHAR severity[6];
70 _stprintf(severity, _T("%d"), condition->getSeverity());
71 tstring time = condition->getOcurrenceTime().toString();
72 CAEConditionDlg::conditionList.SetItemText(nItem, 1, condition->getSourcePath().c_str());
73 CAEConditionDlg::conditionList.SetItemText(nItem, 2, condition->getConditionName().c_str());
74 CAEConditionDlg::conditionList.SetItemText(nItem, 3, severity);
75 CAEConditionDlg::conditionList.SetItemText(nItem, 4, condition->getMessage().c_str());
76 CAEConditionDlg::conditionList.SetItemText(nItem, 5, time.c_str());
77 CAEConditionDlg::conditionList.SetItemText(nItem, 6, condition->getActorId().c_str());
78 CAEConditionDlg::conditionList.SetItemText(nItem, 7, condition->getSubConditionName().c_str());
79 free(lvItemText);
80 } //end for
81} //end handleAeConditionsChanged

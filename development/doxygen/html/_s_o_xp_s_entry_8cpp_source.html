57//-----------------------------------------------------------------------------
58// onReleasedLastOPCObject
59//
60// - handle release of last COM object
61//
62void SOXpSEventHandler::onReleasedLastOPCObject(void)
63{
64 if (g_canCloseEvent)
65 {
66 SetEvent(g_canCloseEvent);
67 }
68
69#ifndef SOXPS_SERVICE
70 if (m_startedByCom)
71 { // started via COM
72 SOCmnPointer<SOAsSEntry> entry = ::getSOAsSEntry();
73 // -> try to close application if last object was released
74 if (S_OK == entry->canUnloadNow())
75 {
76 entry->revokeClassObjects();
77 if (m_wnd)
78 {
79 PostMessage(m_wnd, WM_CLOSE, 0, 0);
80 }
81 }
82 }
83#endif
84} // onReleasedLastOPCObject

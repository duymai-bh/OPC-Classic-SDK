122// Show - Show the titletip if needed
123// rectTitle - The rectangle within which the original
124// title is constrained - in client coordinates
125// lpszTitleText - The text to be displayed
126// xoffset - Number of pixel that the text is offset from
127// left border of the cell
128void CTitleTip::Show(CRect rectTitle, LPCTSTR lpszTitleText, int xoffset /*=0*/,
129 LPRECT lpHoverRect /*=NULL*/,
130 const LOGFONT* lpLogFont /*=NULL*/,
131 COLORREF crTextClr /* CLR_DEFAULT */,
132 COLORREF crBackClr /* CLR_DEFAULT */)
133{
134 if (!IsWindow(m_hWnd))
135 {
136 Create(m_pParentWnd);
137 }
138
139 ASSERT(::IsWindow(GetSafeHwnd()));
140
141 if (rectTitle.IsRectEmpty())
142 {
143 return;
144 }
145
146 // If titletip is already displayed, don't do anything.
147 if (IsWindowVisible())
148 {
149 return;
150 }
151
152 m_rectHover = (lpHoverRect != NULL) ? lpHoverRect : rectTitle;
153 m_rectHover.right++;
154 m_rectHover.bottom++;
155 m_pParentWnd->ClientToScreen(m_rectHover);
156 ScreenToClient(m_rectHover);
157
158 // Do not display the titletip is app does not have focus
159 if (GetFocus() == NULL)
160 {
161 return;
162 }
163
164 // Define the rectangle outside which the titletip will be hidden.
165 // We add a buffer of one pixel around the rectangle
166 m_rectTitle.top = -1;
167 m_rectTitle.left = -xoffset - 1;
168 m_rectTitle.right = rectTitle.Width() - xoffset;
169 m_rectTitle.bottom = rectTitle.Height() + 1;
170 // Determine the width of the text
171 m_pParentWnd->ClientToScreen(rectTitle);
172 CClientDC dc(this);
173 CString strTitle = _T("");
174 strTitle += _T(" ");
175 strTitle += lpszTitleText;
176 strTitle += _T(" ");
177 CFont font, *pOldFont = NULL;
178
179 if (lpLogFont)
180 {
181 font.CreateFontIndirect(lpLogFont);
182 pOldFont = dc.SelectObject(&font);
183 }
184 else
185 {
186 // use same font as ctrl
187 pOldFont = dc.SelectObject(m_pParentWnd->GetFont());
188 }
189
190 CSize size = dc.GetTextExtent(strTitle);
191 TEXTMETRIC tm;
192 dc.GetTextMetrics(&tm);
193 size.cx += tm.tmOverhang;
194 CRect rectDisplay = rectTitle;
195 rectDisplay.left += xoffset;
196 rectDisplay.right = rectDisplay.left + size.cx + xoffset;
197
198 // Do not display if the text fits within available space
199 if (rectDisplay.right > rectTitle.right - xoffset)
200 {
201 // Show the titletip
202 SetWindowPos(&wndTop, rectDisplay.left, rectDisplay.top,
203 rectDisplay.Width(), rectDisplay.Height(),
204 SWP_SHOWWINDOW | SWP_NOACTIVATE);
205
206 // FNA - handle colors correctly
207 if (crBackClr != CLR_DEFAULT)
208 {
209 CBrush backBrush(crBackClr);
210 CBrush* pOldBrush = dc.SelectObject(&backBrush);
211 CRect rect;
212 dc.GetClipBox(&rect); // Erase the area needed
213 dc.PatBlt(rect.left, rect.top, rect.Width(), rect.Height(), PATCOPY);
214 dc.SelectObject(pOldBrush);
215 }
216
217 // Set color
218 if (crTextClr != CLR_DEFAULT)//FNA
219 {
220 dc.SetTextColor(crTextClr); //FA
221 }
222
223 dc.SetBkMode(TRANSPARENT);
224 dc.TextOut(0, 0, strTitle);
225 SetCapture();
226 }
227
228 dc.SelectObject(pOldFont);
229}

246void CTitleTip::OnMouseMove(UINT nFlags, CPoint point)
247{
248 if (!m_rectHover.PtInRect(point))
249 {
250 Hide();
251 // Forward the message
252 ClientToScreen(&point);
253 CWnd* pWnd = WindowFromPoint(point);
254
255 if (pWnd == this)
256 {
257 pWnd = m_pParentWnd;
258 }
259
260 int hittest = (int)pWnd->SendMessage(WM_NCHITTEST, 0, MAKELONG(point.x, point.y));
261
262 if (hittest == HTCLIENT)
263 {
264 pWnd->ScreenToClient(&point);
265 pWnd->PostMessage(WM_MOUSEMOVE, nFlags, MAKELONG(point.x, point.y));
266 }
267 else
268 {
269 pWnd->PostMessage(WM_NCMOUSEMOVE, hittest, MAKELONG(point.x, point.y));
270 }
271 }
272}

274BOOL CTitleTip::PreTranslateMessage(MSG* pMsg)
275{
276 // Used to qualify WM_LBUTTONDOWN messages as double-clicks
277 DWORD dwTick = 0;
278 BOOL bDoubleClick = FALSE;
279 CWnd* pWnd;
280 int hittest;
281
282 switch (pMsg->message)
283 {
284 case WM_LBUTTONDOWN:
285 // Get tick count since last LButtonDown
286 dwTick = GetTickCount();
287 bDoubleClick = ((dwTick - m_dwLastLButtonDown) <= m_dwDblClickMsecs);
288 m_dwLastLButtonDown = dwTick;
289 // NOTE: DO NOT ADD break; STATEMENT HERE! Let code fall through
290
291 case WM_RBUTTONDOWN:
292 case WM_MBUTTONDOWN:
293 {
294 POINTS pts = MAKEPOINTS(pMsg->lParam);
295 POINT point;
296 point.x = pts.x;
297 point.y = pts.y;
298 ClientToScreen(&point);
299 Hide();
300 pWnd = WindowFromPoint(point);
301
302 if (!pWnd)
303 {
304 return CWnd::PreTranslateMessage(pMsg);
305 }
306
307 if (pWnd->GetSafeHwnd() == GetSafeHwnd())
308 {
309 pWnd = m_pParentWnd;
310 }
311
312 hittest = (int)pWnd->SendMessage(WM_NCHITTEST, 0, MAKELONG(point.x, point.y));
313
314 if (hittest == HTCLIENT)
315 {
316 pWnd->ScreenToClient(&point);
317 pMsg->lParam = MAKELONG(point.x, point.y);
318 }
319 else
320 {
321 switch (pMsg->message)
322 {
323 case WM_LBUTTONDOWN:
324 pMsg->message = WM_NCLBUTTONDOWN;
325 break;
326
327 case WM_RBUTTONDOWN:
328 pMsg->message = WM_NCRBUTTONDOWN;
329 break;
330
331 case WM_MBUTTONDOWN:
332 pMsg->message = WM_NCMBUTTONDOWN;
333 break;
334 }
335
336 pMsg->wParam = hittest;
337 pMsg->lParam = MAKELONG(point.x, point.y);
338 }
339
340 // If this is the 2nd WM_LBUTTONDOWN in x milliseconds,
341 // post a WM_LBUTTONDBLCLK message instead of a single click.
342 pWnd->PostMessage(bDoubleClick ? WM_LBUTTONDBLCLK : pMsg->message,
343 pMsg->wParam,
344 pMsg->lParam);
345 return TRUE;
346 }
347
348 case WM_KEYDOWN:
349 case WM_SYSKEYDOWN:
350 Hide();
351 m_pParentWnd->PostMessage(pMsg->message, pMsg->wParam, pMsg->lParam);
352 return TRUE;
353 }
354
355 if (GetFocus() == NULL)
356 {
357 Hide();
358 return TRUE;
359 }
360
361 return CWnd::PreTranslateMessage(pMsg);
362}

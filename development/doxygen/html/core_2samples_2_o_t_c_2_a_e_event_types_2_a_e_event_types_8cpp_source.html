87int _tmain(int argc, _TCHAR* argv[])
88{
89 _tprintf(_T("Usage:\n"));
90 _tprintf(_T("AEEventTypes [/a] <server URL>\n"));
91 _tprintf(_T("\t/a : asynchonous execution of the OCT functions\n\n"));
92 LPTSTR url = NULL;
93 g_executionOptions.m_executionType = OTC_EXECUTIONTYPE_SYCHRONOUS;
94 g_executionOptions.m_executionContext = 0;
95
96 if (argc == 2)
97 {
98 url = argv[1];
99 }
100 else if (argc > 2)
101 {
102 if (_tcsicmp(argv[1], _T("/a")) == 0)
103 {
104 g_executionOptions.m_executionType = OTC_EXECUTIONTYPE_ASYCHRONOUS;
105 }
106
107 url = argv[2];
108 }
109
110 OTCObjectHandle sessionHandle;
111 LONG res;
112 DWORD i;
113 OTActivate(OTC_FEATURE_AE_CLIENT, _T("SAMPLE"));
114
115 if (SUCCEEDED(OTCInitialize(&g_initData)))
116 {
117 OTCAdvise(&g_callbackFunctions);
118 OTCAddAESession(url, 0, &sessionHandle);
119 OTCChangeTargetState(sessionHandle, OTC_STATE_CONNECTED, TRUE);
120
121 if (S_OK == OTCPerformStateTransition(sessionHandle, TRUE, NULL))
122 {
123 _tprintf(_T("Connected to server: %s\n\n"), url);
124 // query categories
125 res = OTCQueryAECategories(sessionHandle, &g_catCount, &g_pEventTypes, &g_pCategoryIds, &g_pCategoryDescriptions, &g_executionOptions);
126
127 if (g_executionOptions.m_executionType == OTC_EXECUTIONTYPE_ASYCHRONOUS)
128 {
129 res = WaitForExecution(g_executionOptions.m_executionContext);
130 g_executionOptions.m_executionContext++;
131 }
132
133 if (SUCCEEDED(res))
134 {
135 for (i = 0; i < g_catCount; i++)
136 {
137 HandleCategory(sessionHandle, g_pEventTypes[i], g_pCategoryIds[i], g_pCategoryDescriptions[i]);
138 OTFreeMemory(g_pCategoryDescriptions[i]);
139 }
140
141 OTFreeMemory(g_pEventTypes);
142 OTFreeMemory(g_pCategoryIds);
143 OTFreeMemory(g_pCategoryDescriptions);
144 }
145
146 // query available filters
147 res = OTCQueryAvailableAEFilters(sessionHandle, &g_availableFilter, &g_executionOptions);
148
149 if (g_executionOptions.m_executionType == OTC_EXECUTIONTYPE_ASYCHRONOUS)
150 {
151 res = WaitForExecution(g_executionOptions.m_executionContext);
152 g_executionOptions.m_executionContext++;
153 }
154
155 if (SUCCEEDED(res))
156 {
157 _tprintf(_T("Available filter: %li\n"), g_availableFilter);
158 }
159 }
160 else
161 {
162 _tprintf(_T("Can't connect to server: %s\n"), argv[1]);
163 }
164
165 OTCTerminate();
166 }
167
168 return 0;
169}

176void HandleCategory(OTCObjectHandle sessionHandle, DWORD eventType, DWORD catId, LPTSTR catDescr)
177{
178 _tprintf(_T("Category: Event type: %s Id: %lu Description: %s\n"), g_eventTypes[eventType - 1], catId, catDescr);
179 LONG res;
180 DWORD i, j;
181 // query attributes
182 res = OTCQueryAEAttributes(sessionHandle, catId, &g_attCount, &g_pAttributeIds, &g_pAttributeDescriptions, &g_pAttributeDatatypes, &g_executionOptions);
183
184 if (g_executionOptions.m_executionType == OTC_EXECUTIONTYPE_ASYCHRONOUS)
185 {
186 res = WaitForExecution(g_executionOptions.m_executionContext);
187 g_executionOptions.m_executionContext++;
188 }
189
190 if (SUCCEEDED(res))
191 {
192 for (i = 0; i < g_attCount; i++)
193 {
194 _tprintf(_T(" Attribute: Id: %lu Description: %s Datatype: %u\n"), g_pAttributeIds[i], g_pAttributeDescriptions[i], g_pAttributeDatatypes[i]);
195 OTFreeMemory(g_pAttributeDescriptions[i]);
196 }
197
198 OTFreeMemory(g_pAttributeDatatypes);
199 OTFreeMemory(g_pAttributeIds);
200 OTFreeMemory(g_pAttributeDescriptions);
201 }
202
203 if (eventType == OT_EVENTTYPE_CONDITION)
204 {
205 // query conditions
206 res = OTCQueryAEConditions(sessionHandle, catId, &g_conCount, &g_pConditionNames, &g_executionOptions);
207
208 if (g_executionOptions.m_executionType == OTC_EXECUTIONTYPE_ASYCHRONOUS)
209 {
210 res = WaitForExecution(g_executionOptions.m_executionContext);
211 g_executionOptions.m_executionContext++;
212 }
213
214 if (SUCCEEDED(res))
215 {
216 for (i = 0; i < g_conCount; i++)
217 {
218 _tprintf(_T(" Condition: Name: %s\n"), g_pConditionNames[i]);
219 // query sub conditions
220 res = OTCQueryAESubConditions(sessionHandle, g_pConditionNames[i], &g_subConCount, &g_pSubConditionNames, &g_executionOptions);
221
222 if (g_executionOptions.m_executionType == OTC_EXECUTIONTYPE_ASYCHRONOUS)
223 {
224 res = WaitForExecution(g_executionOptions.m_executionContext);
225 g_executionOptions.m_executionContext++;
226 }
227
228 if (SUCCEEDED(res))
229 {
230 for (j = 0; j < g_subConCount; j++)
231 {
232 _tprintf(_T(" Sub condition: Name: %s\n"), g_pSubConditionNames[j]);
233 OTFreeMemory(g_pSubConditionNames[j]);
234 }
235
236 OTFreeMemory(g_pSubConditionNames);
237 }
238
239 OTFreeMemory(g_pConditionNames[i]);
240 }
241
242 OTFreeMemory(g_pConditionNames);
243 }
244 }

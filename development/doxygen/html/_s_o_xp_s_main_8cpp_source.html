72//-----------------------------------------------------------------------------
73// addToMessageLog
74//
75// - write message to error log
76//
77void addToMessageLog(
78 IN DWORD id,
79 IN LPCTSTR par)
80{
81 HANDLE eventSource;
82 LPCTSTR strings[2];
83
84 eventSource = RegisterEventSource(NULL, _T("SOXpS"));
85 if (eventSource != NULL)
86 {
87 TCHAR path[512];
88 path[0] = 0;
89 GetModuleFileName(NULL, path, 512);
90
91 strings[0] = (LPCTSTR)path;
92 strings[1] = par;
93 if (par)
94 ReportEvent(eventSource, EVENTLOG_ERROR_TYPE, 0, id, NULL, 2, 0, (LPCTSTR *)strings, NULL);
95 else
96 ReportEvent(eventSource, EVENTLOG_ERROR_TYPE, 0, id, NULL, 1, 0, (LPCTSTR *)strings, NULL);
97
98 DeregisterEventSource(eventSource);
99 }
100} // addToMessageLog

328void terminateServer(HANDLE deviceThread)
329{
330 SOCmnString shutdownReason;
331 DWORD waitTime = 0;
332 g_canCloseEvent = getEventPool()->popEvent();
333
334 if (g_pShutdownReason)
335 {
336 shutdownReason = *g_pShutdownReason;
337 delete g_pShutdownReason;
338 g_pShutdownReason = NULL;
339 }
340 else
341 shutdownReason = _T("Server shutdown");
342
343 {
344 SOCmnPointer<SODaSEntry> entryDA = getSODaSEntry();
345 SOCmnPointer<SOSrvObjectRoot> rootDA = entryDA->getObjectRoot();
346 SOCmnList<SOCmnElement> listDA(rootDA->getBranchList());
347 SOCmnListPosition posDA;
348 SODaSServer *pServer;
349 SOCmnPointer<SOAeSEntry> entryAE = getSOAeSEntry();
350 SOCmnPointer<SOSrvObjectRoot> rootAE = entryAE->getObjectRoot();
351 SOCmnList<SOCmnElement> listAE(rootAE->getBranchList());
352
353 if (listAE.getCount() > 0)
354 {
355 entryAE->shutdownAllClients(shutdownReason);
356 waitTime = 10000;
357 }
358 // check for DCOM server releated objects
359 posDA = listDA.getStartPosition();
360 while (posDA)
361 {
362 pServer = (SODaSServer *)listDA.getNext(posDA);
363 if (pServer->isDCOM())
364 {
365 entryDA->shutdownAllClients(shutdownReason);
366 waitTime = 10000;
367 break;
368 }
369 }
370 }
371 if (waitTime)
372 WaitForSingleObject(g_canCloseEvent, waitTime); // give the clients time to disconnect
373
374 getEventPool()->pushEvent(g_canCloseEvent);
375 g_canCloseEvent = NULL;
376
377
378 SOCmnPointer<SOAsSEntry> entry = getSOAsSEntry();
379 entry->stop();
380
381 // stop DA request thread
382 if ((deviceThread != INVALID_HANDLE_VALUE) && (g_requestThreadEndEvent != INVALID_HANDLE_VALUE))
383 {
384 ::SetEvent(g_requestThreadEndEvent);
385 ::WaitForSingleObject(deviceThread, INFINITE);
386 ::CloseHandle(deviceThread);
387 }
388
389 if (g_fpSOXpSStop)
390 g_fpSOXpSStop();
391}

440//-----------------------------------------------------------------------------
441// cmdRegisterService
442//
443// - create service entries in registry
444// - create OPC entries in registry
445//
446BOOL registerService()
447{
448 SC_HANDLE service;
449 SC_HANDLE SCManager;
450 TCHAR path[512];
451 BOOL ret = FALSE;
452
453 if (GetModuleFileName(NULL, path, 512) == 0)
454 {
455 return FALSE;
456 }
457
458 SCManager = OpenSCManager(NULL, NULL, SC_MANAGER_ALL_ACCESS);
459
460 if (SCManager)
461 {
462 service = CreateService(SCManager, g_initData->m_serviceName, g_initData->m_serviceDescription,
463 SERVICE_ALL_ACCESS, SERVICE_WIN32_OWN_PROCESS, SERVICE_DEMAND_START,
464 SERVICE_ERROR_NORMAL, path, NULL, NULL, g_initData->m_serviceDependencies, NULL, NULL);
465 if (service)
466 {
467 _tprintf(_T("%s installed.\n"), g_initData->m_serviceDescription);
468 CloseServiceHandle(service);
469 ret = TRUE;
470 }
471 else
472 {
473 DWORD lastE = GetLastError();
474 if (lastE == ERROR_SERVICE_EXISTS)
475 {
476 service = OpenService(SCManager, g_initData->m_serviceName, SERVICE_ALL_ACCESS);
477 if (service)
478 {
479 BYTE buffer[1000];
480 QUERY_SERVICE_CONFIG *pServiceConfig = (QUERY_SERVICE_CONFIG *)buffer;
481 DWORD bn;
482 if (QueryServiceConfig(service, pServiceConfig, 1000, &bn))
483 {
484 if (_tcscmp(pServiceConfig->lpDisplayName, g_initData->m_serviceDescription) == 0)
485 { // same service
486 stopService(service);
487 if (ChangeServiceConfig(service, SERVICE_WIN32_OWN_PROCESS, SERVICE_DEMAND_START,
488 SERVICE_ERROR_NORMAL, path, NULL, NULL, g_initData->m_serviceDependencies,
489 NULL, NULL, g_initData->m_serviceDescription))
490 {
491 _tprintf(_T("%s reinstalled.\n"), g_initData->m_serviceDescription);
492 ret = TRUE;
493 }
494 }
495 }
496 CloseServiceHandle(service);
497 }
498 }
499 }
500 CloseServiceHandle(SCManager);
501 }
502
503 return ret;
504} // cmdRegisterService

548BOOL processCommandLine(IN LPCTSTR cmdLine, OUT BOOL *startedByCOM)
549{
550 BOOL end;
551
552 SOCmnPointer<SOAsSEntry> entry = getSOAsSEntry();
553 SOCmnPointer<SOAeSEntry> entryAE = entry->getAEEntry();
554 SOCmnPointer<SODaSEntry> entryDA = entry->getDAEntry();
555 end = FALSE;
556 *startedByCOM = FALSE;
557
558 if ( ( _tcsicmp( _T("-RegServer"), cmdLine ) == 0 ) ||
559 ( _tcsicmp( _T("/RegServer"), cmdLine ) == 0 ) )
560 { // register OPC Server
561#ifdef SOXPS_SERVICE
562 registerService();
563#endif
564 entry->registerServer();
565 entryDA->registerCategories();
566 entryAE->registerCategories();
567
568 // register event viewer source
569 CRegKey keyEvS;
570 DWORD ret;
571 ret = keyEvS.Create(HKEY_LOCAL_MACHINE, _T("System\\CurrentControlSet\\Services\\EventLog\\Application\\SOXpS"));
572 if (ret == ERROR_SUCCESS)
573 {
574 TCHAR path[512];
575 path[0] = _T('\0');
576 GetModuleFileName(NULL, path, 512);
577
578 keyEvS.SetValue(path, _T("EventMessageFile"));
579 keyEvS.SetValue(7, _T("TypesSupported"));
580 }
581
582 end = TRUE;
583 }
584 else if ( ( _tcsicmp( _T("-UnregServer"), cmdLine ) == 0 ) ||
585 ( _tcsicmp( _T("/UnregServer"), cmdLine ) == 0 ) )
586 { // unregister OPC Server
587 entry->unregisterServer();
588 entryDA->unregisterCategories();
589 entryAE->unregisterCategories();
590#ifdef SOXPS_SERVICE
591 unregisterService();
592#endif
593
594 // unregister event viewer source
595 CRegKey keyEvS;
596 DWORD ret;
597 ret = keyEvS.Open(HKEY_LOCAL_MACHINE, _T("System\\CurrentControlSet\\Services\\EventLog\\Application"));
598 if (ret == ERROR_SUCCESS)
599 {
600 keyEvS.RecurseDeleteKey(_T("SOXpS"));
601 }
602
603 end = TRUE;
604 }
605 else if ( ( _tcsicmp( _T("-Embedding"), cmdLine ) == 0 ) ||
606 ( _tcsicmp( _T("/Embedding"), cmdLine ) == 0 ) ||
607 ( _tcsicmp( _T("-Embedded"), cmdLine ) == 0 ) ||
608 ( _tcsicmp( _T("/Embedded"), cmdLine ) == 0 ) )
609 {
610 *startedByCOM = TRUE;
611 }
612
613 return end;
614}

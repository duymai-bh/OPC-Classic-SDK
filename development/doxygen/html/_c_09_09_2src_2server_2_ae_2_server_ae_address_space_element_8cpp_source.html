41BOOL AeAddressSpaceElement::addChild(AddressSpaceElement* aChild)
42{
43 if (!m_hasChildren)
44 {
45 Application::Instance()->trace(EnumTraceLevel_WRN, EnumTraceGroup_OPCSERVER,
46 _T("AeAddressSpaceElement::addChild"),
47 _T("this element does not allow children"));
48 return FALSE;
49 } // end if
50
51 AeAddressSpaceElement* pChild = (AeAddressSpaceElement*)aChild;
52
53 if (pChild == NULL)
54 {
55 Application::Instance()->trace(EnumTraceLevel_WRN, EnumTraceGroup_OPCSERVER,
56 _T("AeAddressSpaceElement::addChild"), _T("Invalid child provided"));
57 return FALSE;
58 } // end if
59
60 AeAddressSpaceRoot* aeRoot = Application::Instance()->getAeAddressSpaceRoot();
61
62 // only add the child if no handle assigned
63 if (pChild->getObjectHandle() == 0)
64 {
65 long result = E_FAIL;
66 OTSAddressSpaceElementData data;
67 data.m_name = (OTChar*)OTAllocateMemory((unsigned long)(pChild->getName().size() + 1) * sizeof(OTChar));
68
69 if (data.m_name != NULL)
70 {
71#ifdef TBC_OS_WINDOWS
72 _tcscpy(data.m_name, pChild->getName().c_str());
73#endif
74#ifdef TBC_OS_LINUX
75 strcpy(data.m_name, pChild->getName().c_str());
76#endif
77 }
78
79 tstring itemId = m_name + Application::Instance()->getAddressSpaceDelimiter() + pChild->getName();
80 data.m_itemID = (OTChar*)OTAllocateMemory((unsigned long)(itemId.size() + 1) * sizeof(OTChar));
81
82 if (data.m_itemID != NULL)
83 {
84#ifdef TBC_OS_WINDOWS
85 _tcscpy(data.m_itemID, itemId.c_str());
86#endif
87#ifdef TBC_OS_LINUX
88 strcpy(data.m_itemID, itemId.c_str());
89#endif
90 }
91
92 data.m_hasChildren = (unsigned char)pChild->getHasChildren();
93 data.m_isBrowsable = (unsigned char)pChild->getIsBrowsable();
94 data.m_elementType = (unsigned char)EnumAddressSpaceElementType_AE;
95 data.m_userData = pChild->getUserData();
96
97 if (aeRoot->addElementToArray(pChild))
98 {
99 unsigned long childObjectHandle = 0;
100 result = OTSAddAddressSpaceElement(m_objectHandle, &data, &childObjectHandle);
101 pChild->setObjectHandle(childObjectHandle);
102 } // end if
103
104 // release the memory allocated
105 if (data.m_name != NULL)
106 {
107 OTFreeMemory(data.m_name);
108 data.m_name = NULL;
109 } // end if
110
111 if (data.m_itemID != NULL)
112 {
113 OTFreeMemory(data.m_itemID);
114 data.m_itemID = NULL;
115 } // end if
116
117 if (!SUCCEEDED(result))
118 {
119 result = aeRoot->removeElementFromArray(aChild);
120
121 if (!SUCCEEDED(result))
122 {
123 // Odd situation, just log an error
124 Application::Instance()->trace(EnumTraceLevel_ERR, EnumTraceGroup_OPCSERVER,
125 _T("AeAddressSpaceElement::addChild"), _T("RemoveElementFromArray has failed"));
126 } // end if
127
128 delete aChild;
129 return FALSE;
130 } // end if ... else
131 }
132 else
133 {
134 aeRoot->addElementToArray(pChild);
135 } // end if ... else
136
137 return AddressSpaceElement::addChild(aChild);
138} // end addChild

141//-----------------------------------------------------------------------------
142// Removes Child and if there, its children too
143//
144// returns:
145// true - Child removed
146// false - Child was not removed
147BOOL AeAddressSpaceElement::removeChild(AddressSpaceElement* aChild)
148{
149 if (!m_hasChildren)
150 {
151 Application::Instance()->trace(EnumTraceLevel_WRN, EnumTraceGroup_OPCSERVER,
152 _T("AeAddressSpaceElement::removeChild"),
153 _T("this element does not allow children"));
154 return FALSE;
155 } // end if
156
157 AeAddressSpaceElement* pAeChild = (AeAddressSpaceElement*)(aChild);
158
159 if (pAeChild == NULL)
160 {
161 Application::Instance()->trace(EnumTraceLevel_WRN, EnumTraceGroup_OPCSERVER,
162 _T("AeAddressSpaceElement::removeChild"), _T("Invalid child provided"));
163 return FALSE;
164 } // end if
165
166 if (pAeChild->getHasChildren())
167 {
168 // Remove the clildren of the child
169 std::vector<AddressSpaceElement*> childrenList = pAeChild->getChildren();
170 std::vector<AddressSpaceElement*>::iterator pos;
171
172 for (pos = childrenList.begin(); pos != childrenList.end(); pos++)
173 {
174 AddressSpaceElement* element = *pos;
175
176 if (element != NULL)
177 {
178 pAeChild->removeChild(element);
179 } // end if
180 } // end for
181 } // end if
182
183 long result = S_OK;
184 AddressSpaceElement::removeChild(aChild);
185 unsigned long childHandle = pAeChild->getObjectHandle();
186 AeAddressSpaceRoot* aeRoot = Application::Instance()->getAeAddressSpaceRoot();
187 result = aeRoot->removeElementFromArray(pAeChild);
188
189 if (!SUCCEEDED(result))
190 {
191 Application::Instance()->trace(EnumTraceLevel_WRN, EnumTraceGroup_OPCSERVER,
192 _T("AeAddressSpaceElement::removeChild"), _T("removeElementFromArray has failed"));
193 } // end if
194
195 delete pAeChild;
196 pAeChild = NULL;
197
198 if (childHandle != 0)
199 {
200 result = OTSRemoveAddressSpaceElement(childHandle);
201 } // end if
202
203 if (!SUCCEEDED(result))
204 {
205 Application::Instance()->trace(EnumTraceLevel_WRN, EnumTraceGroup_OPCSERVER, _T("AeAddressSpaceElement::removeChild"),
206 _T("OTSRemoveAddressSpaceElement has failed"));
207 return FALSE;
208 } // end if
209
210 return TRUE;
211} // end removeChild

419{
420 ServerApp* pApp = ::getServerApp();
421 SOCmnList<SOAeSEvent> eventList(refresh->getEventList());
422 SOCmnList<SOAeSCondition> alarmList(pApp->getAlarmList());
423 SOAeSCondition* cond;
424 SOCmnListPosition posC;
425 SOCmnPointer<SOAeSEvent> event;
426 WORD condState;
427 posC = alarmList.getStartPosition();
428
429 while (posC)
430 {
431 cond = alarmList.getNext(posC);
432 condState = cond->getObjectState();
433

439 eventList.add((SOAeSEvent*)event);
440 }
441 }
442

576HRESULT ServerAEServer::ackConditionByName(
577 IN LPCTSTR sourceName, // source name
578 IN LPCTSTR conditionName, // condition name
579 IN LPCTSTR ackID, // acknowledger ID
580 IN LPCTSTR ackComment, // acknowledger comment
581 LPFILETIME activeTime, // active time
582 DWORD cookie) // notification cookie
583{
584 ServerApp* pApp = ::getServerApp();
585
586 if (!pApp->m_callbackFunctions.m_OTSAcknowledgeCondition)
587 {
588 return E_INVALIDARG;
589 }
590
591 SOCmnList<SOAeSCondition> alarmList(pApp->getAlarmList());
592 SOCmnPointer<SOAeSCondition> cond;
593 cond = alarmList.findKey(cookie);
594
595 if (cond.isNotNull())
596 {
597 if (cond->getName() != conditionName)
598 {
599 return E_INVALIDARG;
600 }
601
602 if (cond->getSourceName() != sourceName)
603 {
604 return E_INVALIDARG;
605 }
606
607 // check active time
608 FILETIME subActTime;
609 cond->getActSubConditionTime(&subActTime);
610
611 if ((activeTime->dwHighDateTime != subActTime.dwHighDateTime) ||
612 (activeTime->dwLowDateTime != subActTime.dwLowDateTime))
613 {
614 return OPC_E_INVALIDTIME;
615 }
616
617 if (!(cond->getObjectState() & OPC_CONDITION_ACKED))
618 {
619 HRESULT res;
620 res = cond->acknowledge(ackID, ackComment, activeTime, cookie);
621
622 if (SUCCEEDED(res))
623 {
624 return S_OK;
625 }
626 else
627 {
628 return E_INVALIDARG;
629 }
630 }
631 else
632 {
633 return OPC_S_ALREADYACKED; // condition is already acknowledged
634 }
635 }
636
637 return E_INVALIDARG;
638} // ackConditionByName

563void ServerDAServer::initItemTag(IN SODaSItemTag* itemTag)
564{
565 SOCMN_TRY
566 {
567 ServerApp* pApp = ::getServerApp();
568 SOCmnPointer<SODaSEntry> entry = ::getSODaSEntry();
569 bool handledEx = false;
570 SOCmnString itemID = itemTag->getItemID();
571 unsigned short cachedElementDataFilter = OTS_ELEMENTDATA_ANY;
572 unsigned short requiredElementDataFilter = OTS_ELEMENTDATA_ACCESSRIGHTS | OTS_ELEMENTDATA_IOMODE | OTS_ELEMENTDATA_ISBROWSABLE | OTS_ELEMENTDATA_USERDATA | OTS_ELEMENTDATA_DATATYPE;
573 OTSAddressSpaceElementData *pElementData = NULL;
574 OTSAddressSpaceElementData *pExistingElementData = NULL;
575 long result = S_FALSE;
576
577 if (pApp->m_callbackFunctions.m_OTSQueryAddressSpaceElementDataEx)
578 {
579 // get any existing, cached element data
580 entry->getStringBasedObjectDataEx(itemID, &pExistingElementData, cachedElementDataFilter);
581
582 if ((requiredElementDataFilter & cachedElementDataFilter) == requiredElementDataFilter)
583 {
584 handledEx = true;
585 }
586 else
587 {
588 unsigned char pass = 0;
589
590 while ((pass < 2) && (!handledEx))
591 {
592 if (!pElementData)
593 {
594 pElementData = (OTSAddressSpaceElementData*)OTAllocateMemory(sizeof(OTSAddressSpaceElementData));
595 }
596
597 ++pass;
598 // negate current cache value with required mask and see what needs to be asked for in the callback
599 unsigned short requestedElementDataFilter = requiredElementDataFilter & ~cachedElementDataFilter;
600
601 SOCMN_TRY
602 {
603 result = pApp->m_callbackFunctions.m_OTSQueryAddressSpaceElementDataEx(itemID, OT_ADDRESSSPACEELEMENT_TYPE_DA,
604 requestedElementDataFilter, pElementData);

614 {
615 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_SERVER, (
616 _T("ServerDAServer::initItemTag/QueryAddressSpaceElementDataEx - Vendor application callback did not provide all data (requested 0x%03X, provided 0x%03X) for tag %s"),
617 requiredElementDataFilter & ~cachedElementDataFilter, requestedElementDataFilter, (LPCTSTR)itemID));
618 // vendor callback was not able to provide required data once, don`t call again
619 ++pass;
620
621 if (requestedElementDataFilter != OTS_ELEMENTDATA_NONE)
622 {
623 // if vendor app callback did provide something however, store in cache
624 entry->setStringBasedObjectDataEx(itemID, pElementData, requestedElementDataFilter);

626 }

628 {
629 // merge provided data with existing data
630 if (entry->setStringBasedObjectDataEx(itemID, pElementData, requestedElementDataFilter))

637 }
638 // otherwise try once more with the *Ex callback (see pass + while), request only data that expired
639 }
640 else
641 {
642 // surely cache was disabled to begin with, so don`t try again, no need
643 ++pass;
644 pExistingElementData = pElementData;
645 }
646 }
647 }
648 else
649 {
650 // callback failed, don`t call again
651 ++pass;
652 }
653 }
654 }
655 }
656
657 if ((handledEx == false) && (pApp->m_callbackFunctions.m_OTSQueryAddressSpaceElementData))
658 {
659 cachedElementDataFilter = requiredElementDataFilter;
660
661 if (!entry->getStringBasedObjectDataEx(itemID, &pExistingElementData, cachedElementDataFilter))
662 {
663 if (!pElementData)
664 {
665 pElementData = (OTSAddressSpaceElementData*)OTAllocateMemory(sizeof(OTSAddressSpaceElementData));

681 }

686 if (entry->setStringBasedObjectDataEx(itemID, pElementData))

688 // if cache is enabled

694 }
695 }

854HRESULT ServerDAServer::getChildrenData(
855 IN LPCTSTR itemId,
856 IN enum browserFilterType fType,
857 IN LPCTSTR elementNameFilter,
858 IN LPCTSTR /* vendorFilter */,
859 IN DWORD maxElementsReturned,
860 IN OUT SOCmnString& contPoint,
861 OUT BOOL& moreElements,
862 IN OUT SOCmnList<SODaSBrowseElementData> *pListBE)
863{
864 ServerApp* pApp = ::getServerApp();
865 HRESULT res = S_OK;
866 DWORD inCnt = pListBE->getCount();
867 moreElements = FALSE;
868
869 SOCMN_TRY
870 {
871 if (pApp->m_callbackFunctions.m_OTSQueryAddressSpaceElementChildren)
872 {
873 SOCmnListPosition posC, posP, posN, posSBstart;
874 SODaSBrowseElementData* pBE = NULL;
875 BOOL removeFromList = FALSE;
876 DWORD cnt = 0;
877 BOOL continuationPointFound = FALSE;
878 addStringBasedChildren(itemId, pListBE, FALSE);
879 posN = pListBE->getStartPosition();
880 posC = NULL;
881 posSBstart = NULL;
882
883 while (posN)
884 {
885 posP = posC;
886 posC = posN;
887 pBE = pListBE->getNext(posN);
888
889 if (cnt < inCnt)
890 {
891 // skip the object based entries
892 cnt++;
893 continue;
894 }
895
896 if (!posSBstart)
897 {
898 posSBstart = posC; // save start of string based entries
899 }
900
901 removeFromList = FALSE;
902
903 // apply filters
904 if (cnt < maxElementsReturned + inCnt)
905 {
906 if ((!contPoint.isEmpty()) && (!moreElements) && (!continuationPointFound))
907 {
908 // search for continuation point
909 if (_tcscmp(contPoint, pBE->m_itemID) != 0)
910 {
911 removeFromList = TRUE;
912 }
913 else
914 {
915 continuationPointFound = TRUE;
916 }
917 }
918
919 if (!removeFromList)
920 {
921 if ((fType == SOSrvBrowser::typeAll) ||
922 ((fType == SOSrvBrowser::typeItem) && (pBE->m_flags & OPC_BROWSE_ISITEM)) ||
923 ((fType == SOSrvBrowser::typeBranch) && (pBE->m_flags & OPC_BROWSE_HASCHILDREN)))
924 {
925 if ((elementNameFilter) && (_tcslen(elementNameFilter) != 0))
926 {
927 if (!matchStringFilter(pBE->m_name, elementNameFilter))
928 {
929 removeFromList = TRUE;
930 }
931 }
932 }
933 else
934 {
935 removeFromList = TRUE;
936 }
937 }
938 }
939 else
940 {
941 removeFromList = TRUE;
942
943 if (!moreElements)
944 {
945 // set continuation point
946 contPoint = pBE->m_itemID;
947 moreElements = TRUE;
948 }
949 }
950
951 if (removeFromList)
952 {
953 pListBE->removeAt(posC, posP);
954
955 if (posSBstart == posC)
956 {
957 posSBstart = posP;
958 }
959
960 posC = posP;
961 }
962 else
963 {
964 cnt++;
965 }
966 }
967
968 if (posSBstart != NULL)
969 {
970 // remove duplicates from list
971 SOCmnListPosition posL;
972 SODaSBrowseElementData* pBEL;
973 posC = NULL;
974 posN = pListBE->getStartPosition();
975
976 while (posN != posSBstart)
977 {
978 // check if element which was already in the list was added again
979 posP = posC;
980 posC = posN;
981 pBE = pListBE->getNext(posN);
982 posL = posSBstart;
983
984 while (posL)
985 {
986 pBEL = pListBE->getNext(posL);
987
988 if (_tcscmp(pBE->m_itemID, pBEL->m_itemID) == 0)
989 {
990 pListBE->removeAt(posC, posP);
991 posP = posC;
992 break;
993 }
994 }
995 }
996 }
997
998 if (!moreElements)
999 {
1000 if (!contPoint.isEmpty())
1001 {
1002 if (continuationPointFound == FALSE)
1003 {
1004 res = OPC_E_INVALIDCONTINUATIONPOINT;
1005 }
1006 else
1007 {
1008 contPoint.empty();
1009 }
1010 }
1011 }
1012 }
1013 }
1014 SOCMN_CATCH(...)
1015 {
1016 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_SERVER, (_T("ServerDAServer::getChildrenData - Exception occurred")));
1017 }
1018
1019 return res;

1133void ServerDAServer::queryAvailablePropertiesByItemID(
1134 IN LPCTSTR itemId,
1135 IN BOOL isSOAP,
1136 IN BOOL returnPropertyValues,
1137 OUT SOCmnList<SODaSPropertyData> &propertyList)
1138{
1139 SOCMN_TRY
1140 {
1141 ServerApp* pApp = ::getServerApp();
1142 SOCmnPointer<SODaSEntry> entry = ::getSODaSEntry();
1143 SOCmnString key(itemId);
1144 bool handledEx = false;
1145 OTSAddressSpaceElementData *pElementData = NULL;
1146 OTSAddressSpaceElementData *pExistingElementData = NULL;
1147 unsigned short requiredElementDataFilter = OTS_ELEMENTDATA_IOMODE;
1148 unsigned short cachedElementDataFilter = OTS_ELEMENTDATA_ANY;
1149
1150 if (pApp->m_callbackFunctions.m_OTSQueryAddressSpaceElementDataEx)
1151 {
1152 if (returnPropertyValues)
1153 {
1154 requiredElementDataFilter |= OTS_ELEMENTDATA_DATATYPE;
1155 }
1156
1157 // get any existing, cached element data
1158 entry->getStringBasedObjectDataEx(key, &pExistingElementData, cachedElementDataFilter);
1159
1160 if ((requiredElementDataFilter & cachedElementDataFilter) == requiredElementDataFilter)
1161 {
1162 // cache has everything
1163 handledEx = true;
1164 }
1165 else
1166 {
1167 unsigned char pass = 0;
1168
1169 while ((pass < 2) & (!handledEx))
1170 {
1171 if (!pElementData)
1172 {
1173 pElementData = (OTSAddressSpaceElementData*)OTAllocateMemory(sizeof(OTSAddressSpaceElementData));
1174 }
1175
1176 ++pass;
1177 // negate current cache value with required mask and see what needs to be asked for in the callback
1178 unsigned short requestedElementDataFilter = requiredElementDataFilter & ~cachedElementDataFilter;
1179 long result = S_FALSE;
1180
1181 SOCMN_TRY
1182 {
1183 result = pApp->m_callbackFunctions.m_OTSQueryAddressSpaceElementDataEx((LPTSTR)itemId, OT_ADDRESSSPACEELEMENT_TYPE_DA,
1184 requestedElementDataFilter, pElementData);
1185 }
1186 SOCMN_CATCH(...)
1187 {
1188 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_SERVER, (_T("ServerDAServer::queryAvailablePropertiesByItemID - Exception occurred in vendor application callback: OTSQueryAddressSpaceElementDataEx")));
1189 }
1190
1191 if (S_OK == result)
1192 {
1193 if (((cachedElementDataFilter | requestedElementDataFilter) & requiredElementDataFilter) != requiredElementDataFilter)
1194 {
1195 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_SERVER, (
1196 _T("ServerDAServer::queryAvailablePropertiesByItemID/QueryAddressSpaceElementDataEx - Vendor application callback did not provide all data (requested 0x%03X, provided 0x%03X) for tag %s"),
1197 requiredElementDataFilter & ~cachedElementDataFilter, requestedElementDataFilter, (LPCTSTR)itemId));
1198 // skip requesting once more, vendor callback was not able to provide data once, won`t stress again
1199 ++pass;
1200
1201 if (requestedElementDataFilter != 0)
1202 {
1203 entry->setStringBasedObjectDataEx(itemId, pElementData, requestedElementDataFilter);
1204 }
1205 }
1206 else
1207 {
1208 if (entry->setStringBasedObjectDataEx(itemId, pElementData, requestedElementDataFilter))
1209 {
1210 // if cache is enabled
1211 if (entry->getStringBasedObjectDataEx(itemId, &pExistingElementData, requiredElementDataFilter))
1212 {
1213 // if the part from cache did not expire until now (e.g. callback might had taken a while)..
1214 handledEx = true;
1215 }
1216 // otherwise give it one more try with the *Ex callback (see pass + while), request only data that expired
1217 }
1218 else
1219 {
1220 // if cache is disabled, all data was requested (none was found in cache). skip any remaining while cycles
1221 ++pass;
1222 pExistingElementData = pElementData;
1223 }
1224 }
1225 }
1226 else
1227 {
1228 // callback failed, don`t call again
1229 ++pass;
1230 }
1231 }
1232 }
1233 }
1234
1235 if ((handledEx == false) && (pApp->m_callbackFunctions.m_OTSQueryAddressSpaceElementData))
1236 {
1237 requiredElementDataFilter = OTS_ELEMENTDATA_IOMODE;
1238
1239 if (returnPropertyValues)
1240 {
1241 requiredElementDataFilter |= OTS_ELEMENTDATA_DATATYPE;
1242 }
1243
1244 if (!entry->getStringBasedObjectDataEx(key, &pElementData, requiredElementDataFilter))
1245 {
1246 if (!pElementData)
1247 {
1248 pElementData = (OTSAddressSpaceElementData*)OTAllocateMemory(sizeof(OTSAddressSpaceElementData));
1249 }
1250 else
1251 {
1252 FreeOTSAddressSpaceElementData(pElementData);
1253 }
1254
1255 long result = S_FALSE;
1256
1257 SOCMN_TRY
1258 {
1259 result = pApp->m_callbackFunctions.m_OTSQueryAddressSpaceElementData((LPTSTR)itemId, OT_ADDRESSSPACEELEMENT_TYPE_DA, pElementData);
1260 }
1261 SOCMN_CATCH(...)
1262 {
1263 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_SERVER, (_T("ServerDAServer::queryAvailablePropertiesByItemID - Exception occurred in vendor application callback: OTSQueryAddressSpaceElementData")));
1264 }
1265
1266 if (S_OK == result)
1267 {
1268 if (entry->setStringBasedObjectDataEx(key, pElementData))
1269 {
1270 // cache enabled?
1271 entry->getStringBasedObjectDataEx(key, &pExistingElementData, requiredElementDataFilter);
1272 }
1273 else
1274 {
1275 pExistingElementData = pElementData;
1276 }
1277
1278 handledEx = TRUE; // mark success in order to continue
1279 }
1280 }
1281 }
1282
1283 if (handledEx)
1284 {
1285 if (pExistingElementData->m_ioMode != OTS_IOMODE_NONE)
1286 {
1287 getStandardTagProperties((VARTYPE)pExistingElementData->m_datatype, isSOAP, propertyList);
1288 }
1289 }
1290
1291 if (pApp->m_callbackFunctions.m_OTSQueryProperties)
1292 {
1293 DWORD propCountO = 0;

1295 DWORD i;
1296 SODaSPropertyData* pPropData;
1297
1298 SOCMN_TRY
1299 {

1301 }

1305 }
1306
1307 for (i = 0; i < propCountO; i++)
1308 {
1309 SOCMN_ALLOCATE_OBJECT(pPropData, SODaSPropertyData());
1310 ServerDAAddressSpaceElement::clonePropertyData(NULL, &pPropDataO[i], pPropData);
1311 propertyList.add(pPropData);

1315 {
1316 FreeOTSPropertyData(&pPropDataO[i]);
1317 }
1318
1319 OTFreeMemory(pPropDataO);
1320 }

1324 }

1328 }

215//-----------------------------------------------------------------------------
216// OnCreate
217//
218// - initialize the main frame
219//
220// returns:
221// 0 - all ok
222// -1 - error
223//
224int SODmCMainFrame::OnCreate(
225 IN LPCREATESTRUCT lpCreateStruct) // create struct
226{
227 if (CFrameWnd::OnCreate(lpCreateStruct) == -1)
228 {
229 return -1;
230 }
231
232 if (getComCtlVersion() >= COMCTL32_VERSION_IE4)
233 {
234 if (!m_wndToolBar.CreateEx(this) ||
235 !m_wndToolBar.LoadToolBar(IDR_MAINFRAME))
236 {
237 return -1; // fail to create
238 }
239
240 if (!m_wndDAWriteBar.Create(this, IDR_DAWRITE,
241 CBRS_ALIGN_TOP, AFX_IDW_DIALOGBAR))
242 {
243 return -1; // fail to create
244 }
245
246 m_wndDAWriteBar.EnableToolTips();
247
248 if (!m_wndReBar.Create(this) ||
249 !m_wndReBar.AddBar(&m_wndToolBar) ||
250 !m_wndReBar.AddBar(&m_wndDAWriteBar))
251 {
252 return -1; // fail to create
253 }
254 }
255 else
256 {
257 EnableDocking(CBRS_ALIGN_ANY);
258
259 if (!m_wndToolBar.Create(this) ||
260 !m_wndToolBar.LoadToolBar(IDR_MAINFRAME))
261 {
262 return -1; // fail to create
263 }
264
265 m_wndToolBar.EnableDocking(CBRS_ALIGN_TOP);
266 DockControlBar(&m_wndToolBar, AFX_IDW_DOCKBAR_TOP);
267
268 if (!m_wndDAWriteBar.Create(this, IDR_DAWRITE,
269 CBRS_ALIGN_TOP, AFX_IDW_DIALOGBAR))
270 {
271 return -1; // fail to create
272 }
273
274 m_wndDAWriteBar.EnableDocking(CBRS_ALIGN_TOP);
275 m_wndDAWriteBar.EnableToolTips();
276 DockControlBar(&m_wndDAWriteBar, AFX_IDW_DOCKBAR_TOP);
277 }
278
279 if (!m_wndStatusBar.Create(this) ||
280 !m_wndStatusBar.SetIndicators(indicators,
281 sizeof(indicators) / sizeof(UINT)))
282 {
283 return -1; // fail to create
284 }
285
286 m_wndDAWriteBar.ShowWindow(SW_HIDE);
287 m_wndToolBar.SetBarStyle(m_wndToolBar.GetBarStyle() |
288 CBRS_TOOLTIPS | CBRS_FLYBY);
289 HICON hIcon = AfxGetApp()->LoadIcon(IDR_MAINFRAME);
290 SetIcon(hIcon, TRUE); // Set big icon
291 SetIcon(hIcon, FALSE); // Set small icon
292 // DA write dialog bar
293 HWND hWndCtrl;
294 m_wndDAWriteBar.GetDlgItem(IDC_DAITEMWRITE, &hWndCtrl);
295 m_writeDAItemButton.SubclassWindow(hWndCtrl);
296 m_wndDAWriteBar.GetDlgItem(IDC_DAITEMVALUE, &hWndCtrl);
297 m_writeDAItemValue.SubclassWindow(hWndCtrl);
298 m_writeDAItemValue.setNotificationWnd(this);
299 onViewDAWriteBar();
300 return 0;
301} // OnCreate

328//-----------------------------------------------------------------------------
329// OnCreateClient
330//
331// - create splitter windows
332//
333// returns:
334// return of base class
335//
336BOOL SODmCMainFrame::OnCreateClient(IN LPCREATESTRUCT lpcs, IN CCreateContext* pContext)
337{
338 // calculate sizes
339 CSize szLeft(205, 500);
340 CSize szRight(585, 500);
341
342 // create a splitter with 1 row, 2 columns
343 if (!m_wndSplitter.CreateStatic(this, 1, 2))
344 {
345 return FALSE;
346 }
347
348 m_wndSplitter.SetColumnInfo(0, 205, 10);
349 m_wndSplitter.SetColumnInfo(1, 585, 10);
350
351 // add the first splitter pane
352 if (!m_wndSplitter.CreateView(0, 0, RUNTIME_CLASS(SODmCObjectView), szLeft, pContext))
353 {
354 return FALSE;
355 }
356
357 // add the second splitter pane
358 if (!m_wndSplitter.CreateView(0, 1, RUNTIME_CLASS(SODmCMainView), szRight, pContext))
359 {
360 return FALSE;
361 }
362
363 // activate the object view
364 SetActiveView((CView*)m_wndSplitter.GetPane(0, 0));
365 SODmCPropertyFrame* pPropFrame = createPropertyFrame();
366 pPropFrame->ShowWindow(SW_HIDE);
367 return CFrameWnd::OnCreateClient(lpcs, pContext);
368} // OnCreateClient

377{

379
380 if (m_pPropFrame)
381 {
382 if ((nState == WA_ACTIVE) || (nState == WA_CLICKACTIVE))
383 {
384 // bring property frame below window if visible
385 if ((m_pPropFrame->IsWindowVisible()) || (m_propFrameMinimized))
386 {
387 m_propFrameMinimized = FALSE;
388 m_pPropFrame->SetWindowPos(&wndTop, 0, 0, 0, 0, SWP_NOACTIVATE | SWP_NOMOVE | SWP_NOSIZE | SWP_SHOWWINDOW);
389 }
390 }
391 else if ((nState == WA_INACTIVE) && (bMinimized))
392 {
393 // bring property frame below window if visible
394 if (m_pPropFrame->IsWindowVisible())
395 {
396 m_pPropFrame->ShowWindow(SW_HIDE);
397 m_propFrameMinimized = TRUE;
398 }
399 }
400 }
401} // OnActivate

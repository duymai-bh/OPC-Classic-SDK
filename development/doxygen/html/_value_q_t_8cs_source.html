108 public override string ToString()
109 {
110 if (Data == null)
111 {
112 return string.Empty;
113 } // end if
114
115 IntPtr pData = Marshal.AllocCoTaskMem(ValueQT.VARIANT_SIZE);
116 Marshal.GetNativeVariantForObject(Data, pData);
117
118 IntPtr pInput = Marshal.AllocCoTaskMem(Marshal.SystemDefaultCharSize*(int) MAX_LENGTH_OF_VARIANT_TO_STRING);
119
120 IntPtr pOutput = OTBFunctions.OTVariantToString(
121 pData,
122 MAX_LENGTH_OF_VARIANT_TO_STRING,
123 pInput);
124
125 string output = Marshal.PtrToStringUni(pOutput);
126
127 Marshal.FreeCoTaskMem(pInput);
128 OTBFunctions.OTVariantClear(pData);
129 Marshal.FreeCoTaskMem(pData);
130
131 return output;
132 } // end ToString()

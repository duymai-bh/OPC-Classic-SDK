125BOOL SOSrvTPDiscovery::initCommunication(
126 const SOCmnString& address,
127 USHORT port,
128 USHORT localPort,
129 DWORD timeout)
130{
131 USES_CONVERSION;
132
133 if (address.isEmpty() != TRUE)
134 {
135 m_udpCommunicationHandlerSnd->setBroadcastAddress(inet_addr(T2CA((LPCTSTR)address)));
136 } // end if
137
138 m_udpCommunicationHandlerSnd->setBroadCastPort(port);
139 m_udpCommunicationHandlerSnd->setLocalPort(localPort);
140 m_udpCommunicationHandlerSnd->setBroadcastReplyTimeOut(timeout);
141 m_udpCommunicationHandlerSnd->init(FALSE);
142
143 if (address.isEmpty() != TRUE)
144 {
145 m_udpCommunicationHandlerRcv->setBroadcastAddress(inet_addr(T2CA((LPCTSTR)address)));
146 } // end if
147
148 m_udpCommunicationHandlerRcv->setBroadCastPort(port + 1);
149 m_udpCommunicationHandlerRcv->setLocalPort(localPort);
150 m_udpCommunicationHandlerRcv->setBroadcastReplyTimeOut(timeout);
151 m_udpCommunicationHandlerRcv->init(TRUE);
152 _TRACEC(SOCMNTRACE_L_DEB, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "),
153 (_T("The communication to devices was establish.")));
154 return TRUE;
155} // end initCommunication

158BOOL SOSrvTPDiscovery::init(
159 const DiscoveryDeviceType& discoveryDeviceType,
160 const DiscoveryRequestType& discoveryRequestType,
161 const SOCmnString& ipAddress)
162{
163 m_discoveryDeviceType = discoveryDeviceType;
164 m_discoveryRequestType = discoveryRequestType;
165
166 if (m_discoveryRequestType == TP_DISCOVERY_REQUEST_TYPE_UNICAST)
167 {
168 m_ipAddress = ipAddress;
169 } // end if
170
171 if (m_ipAddress != _T("255.255.255.255"))
172 {
173 USES_CONVERSION;
174 m_udpCommunicationHandlerRcv->setBroadcastAddress(inet_addr(T2CA((LPCTSTR)ipAddress)));
175 m_udpCommunicationHandlerSnd->setBroadcastAddress(inet_addr(T2CA((LPCTSTR)ipAddress)));
176 }
177
178 m_listenThread->start(listenThreadProc, this);
179 setObjectState(SOCMNSOCKET_STATE_LISTEN);
180 return TRUE;
181} // end startSend

213BOOL SOSrvTPDiscovery::marshalRequest(
214 UCHAR** message,
215 ULONG& size)
216{
217 T_TP_REQ_HEADER request;
218 request.Length = TP_DISCOVERY_REQUEST_HEADER_LENGTH;
219 request.ResponseType = (unsigned short)m_discoveryRequestType;
220 request.Version = TP_DISCOVERY_VERSION_NUMBER;
221 request.Service = TP_DISCOVERY_SERVICE;
222 request.DeviceType = (unsigned short)m_discoveryDeviceType;;
223 m_marshaler->BeginMarshal(sizeof(T_TP_REQ_HEADER));
224 getMarshaler()->MarshalString(m_ipAddress);
225 getMarshaler()->MarshalShort(request.Length);
226 getMarshaler()->MarshalShort(request.ResponseType);
227 getMarshaler()->MarshalShort(request.Version);
228 getMarshaler()->MarshalShort(request.Service);
229 getMarshaler()->MarshalShort(request.DeviceType);
230
231 if (getMarshaler()->GetMarshalDataLen() <= 0 ||
232 getMarshaler()->GetMarshalData() == 0)
233 {
234 _TRACEC(SOCMNTRACE_L_ERR, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "), (_T("The request message could not be marshaled.")));
235 return FALSE;
236 } // end if
237
238 // else allocate and copy the message
239 size = getMarshaler()->GetMarshalDataLen();
240 *message = (UCHAR*)malloc(size);
241 memcpy(*message, getMarshaler()->GetMarshalData(), size);
242 _TRACEC(SOCMNTRACE_L_DEB, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "),
243 (_T("The request message contains: IP = %s, Length = %d, ResponseType = %d, Version = %d, Service = %d, DeviceType = %d"),
244 request.Ip, request.Length, request.ResponseType, request.Version, request.Service, request.DeviceType));
245 return TRUE;
246} // end marshalRequest

249DWORD _stdcall SOSrvTPDiscovery::listenThreadProc(
250 void* pContext)
251{
252 if (pContext == NULL)
253 {
254 return S_FALSE;
255 } // end if
256
257 SOSrvTPDiscovery* pT = reinterpret_cast<SOSrvTPDiscovery*>(pContext);
258 pT->addRef();
259 SOCmnThread* pMe = pT->getListenThread();
260 DWORD waitR;
261 BOOL end = FALSE;
262 DWORD dwWait = 500;
263 UCHAR* pSentMessage = NULL;
264 ULONG sizeSentMessage = 0;
265 SOCmnPointer<SOCmnUDPSocket> pCommunicationHandlerSnd = pT->getCommunicationHandleSnd();
266 SOCmnPointer<SOCmnUDPSocket> pCommunicationHandlerRcv = pT->getCommunicationHandleRcv();
267
268 while (!end)
269 {
270 waitR = pMe->waitForEvents(dwWait);
271
272 if (waitR == pMe->getStopEventId())
273 {
274 end = TRUE;
275 }
276 else if (waitR == pT->getListenSendEvent())
277 {
278 pMe->getEvents()->reset(pT->getListenSendEvent());
279 bool sentOk = false;
280
281 if (pT->m_ipAddress == _T("255.255.255.255"))
282 {
283 USES_CONVERSION;
284 UINT n = pCommunicationHandlerSnd->getNoOfIfaces();
285
286 for (UINT i = 0; i < n; ++i)
287 {
288 if (pCommunicationHandlerSnd->getIfaceAddress(i) != 0x0100007F) // != 127.0.0.1
289 {
290 IN_ADDR inetBcastAddr = {0};
291 inetBcastAddr.S_un.S_addr = pCommunicationHandlerSnd->getIfaceBcastAddr(i);
292 pCommunicationHandlerSnd->setBroadcastAddress(inetBcastAddr.S_un.S_addr);
293 pT->m_ipAddress.format(_T("%s"), A2W(inet_ntoa(inetBcastAddr)));
294 pT->marshalRequest(&pSentMessage, sizeSentMessage);
295
296 if (pCommunicationHandlerSnd->sendData((char*)pSentMessage, sizeSentMessage) == S_OK)
297 {
298 sentOk = true;
299 }
300 }
301 }
302 }
303 else
304 {
305 pT->marshalRequest(&pSentMessage, sizeSentMessage);
306
307 if (pCommunicationHandlerSnd->sendData((char*)pSentMessage, sizeSentMessage) == S_OK)
308 {
309 sentOk = true;
310 }
311 }
312
313 if (sentOk)
314 {
315 pMe->getEvents()->signal(pT->getListenReceiveEvent());
316 pMe->getEvents()->reset(pT->getListenSendEvent());
317 _TRACEC(SOCMNTRACE_L_INF, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "),
318 (_T("The request message was sent.")));
319 }
320 else
321 {
322 pMe->getEvents()->signal(pT->getListenSendEvent());
323 _TRACEC(SOCMNTRACE_L_ERR, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "),
324 (_T("The request message was not sent, it will be resend")));
325 }
326
327 delete pSentMessage;
328 pSentMessage = NULL;
329 }//end if..else
330 else if (waitR == pT->getListenReceiveEvent())
331 {
332 _TRACEC(SOCMNTRACE_L_DEB, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "),
333 (_T("I is waiting for response.")));
334 UCHAR* pReceivedMessage = NULL;
335 ULONG sizeReceivedMessage = 0;
336 int counter = 0;
337
338 while (TRUE)
339 {
340 if (pCommunicationHandlerRcv->hasPendingData() == TRUE)
341 {
342 if (pCommunicationHandlerRcv->receiveData((char**)&pReceivedMessage, (ULONG)sizeReceivedMessage) == S_OK)
343 {
344 pT->handleResponse(pReceivedMessage, sizeReceivedMessage);
345 counter++;
346 }
347 else
348 {
349 break;
350 }
351 }
352 else
353 {
354 break;
355 }
356 }
357
358 if (counter == 0)
359 {
360 _TRACEC(SOCMNTRACE_L_INF, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "),
361 (_T("No device had response to the request.")));
362 }
363 else
364 {
365 _TRACEC(SOCMNTRACE_L_INF, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "),
366 (_T("%d devices had response to the request."), counter));
367 }//end if..else
368
369 pMe->getEvents()->reset(pT->getListenReceiveEvent());
370 }//end if..else
371 } // end while
372
373 pT->m_isPendingData = FALSE;
374 pT->release();
375 return S_OK;
376} // end listenThreadProc

379BOOL SOSrvTPDiscovery::handleResponse(
380 UCHAR* pMessage,
381 ULONG size)
382{
383 if (pMessage == NULL || size == 0)
384 {
385 return FALSE;
386 } // end if
387
388 // unmarshal the message
389 m_unMarshaler->SetMarshalData(pMessage, size);
390 T_TP_DEVICE_RES response;
391 m_unMarshaler->UnMarshalShort(&response.Header.Length);
392
393 if (response.Header.Length == TP_DISCOVERY_RESPONSE_HEADER_LENGTH)
394 {
395 m_unMarshaler->UnMarshalShort(&response.Header.Version);
396 m_unMarshaler->UnMarshalShort(&response.Header.Service);
397 m_unMarshaler->UnMarshalShort(&response.Header.Status);
398 m_unMarshaler->UnMarshalShort(&response.Header.DeviceType);
399 SOCmnString url;
400 m_unMarshaler->UnMarshalString(&url);
401 SOCmnString serverName;
402 m_unMarshaler->UnMarshalString(&serverName);
403 SOCmnString vendorInfo;
404 m_unMarshaler->UnMarshalString(&vendorInfo);
405 m_unMarshaler->UnMarshalShort(&response.MajorVersion);
406 m_unMarshaler->UnMarshalShort(&response.MinorVersion);
407 m_unMarshaler->UnMarshalShort(&response.PatchVersion);
408 m_unMarshaler->UnMarshalLong(&response.BuildNumber);
409 SOCmnString hostName;
410 m_unMarshaler->UnMarshalString(&hostName);
411
412 if ((response.Header.DeviceType != TP_DISCOVERY_RESPONSE_DEVICE_TYPE) &&
413 (response.Header.Service != TP_DISCOVERY_RESPONSE_SERVICE))
414 {
415 return FALSE;
416 } // end if
417
418 _TRACEC(SOCMNTRACE_L_INF, SOCMNTRACE_G_DISCOVERY_SRV, _T("Tunnel Discovery: "),
419 (_T("Device with IP address: %s had response. It was added to the device list"), hostName));
420 BOOL exists = FALSE;
421 SOCmnListPosition pos = m_deviceList.getStartPosition();
422
423 while (pos)
424 {
425 DeviceIdentity* devIdent = m_deviceList.getNext(pos);
426
427 // current configurations should be visible under remote node with all of the machine`s ip addresses
428 // therefore no exclusion is done by host and configuration name
429 if (devIdent->getUrl() == url)
430 {
431 exists = TRUE;
432 }
433 } // end while
434
435 if (exists == FALSE)
436 {
437 DeviceIdentity* pDeviceIdentity = NULL;
438 SOCMN_ALLOCATE_OBJECT_C(pDeviceIdentity, DeviceIdentity(), (DeviceIdentity*));
439 pDeviceIdentity->setUrl(&url);
440 pDeviceIdentity->setServerName(&serverName);
441 pDeviceIdentity->setVendorInfo(&vendorInfo);
442 pDeviceIdentity->setHostName(&hostName);
443 pDeviceIdentity->setMajorVersion(response.MajorVersion);
444 pDeviceIdentity->setMinorVersion(response.MinorVersion);
445 pDeviceIdentity->setPatchVersion(response.PatchVersion);
446 pDeviceIdentity->setBuildNumber(response.BuildNumber);
447 m_deviceList.add(pDeviceIdentity);
448 } // end if
449 }//end if
450
451 return TRUE;
452} // end handleResponse

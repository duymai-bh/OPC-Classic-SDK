314 private string GetState(EnumConditionState state)
315 {
316 string stateToString = string.Empty;
317 if ((state & EnumConditionState.ACTIVE) == EnumConditionState.ACTIVE)
318 {
319 stateToString += " ACT";
320 }
321 if ((state & EnumConditionState.ENABLED) == EnumConditionState.ENABLED)
322 {
323 stateToString += " ENA";
324 }
325 if ((state & EnumConditionState.ACKED) == EnumConditionState.ACKED)
326 {
327 stateToString += " ACK";
328 }
329 if (state == 0)
330 {
331 stateToString += " DIS";
332 }
333 return stateToString;

37// NotQuiteNullDacl.h: interface for the NotQuiteNullDacl class.
38//
39// This class allows the user to simply create a Not-Quite-Null
40// Discretionary Access Control List (Dacl). This Dacl has
41// many of the attributes of the Null Dacl that is often used
42// to open up access to file and kernel objects but denies the
43// user access to the objects security attributes. Therefore
44// it is less subject to abuse than the simple Null Dacl.
45//
46// This class is based heavily on the concepts and sample code
47// found in Chapter 10 of Programming Server-Side Applications
48// for Microsoft Windows 2000 by Richter and Clark. It also
49// used ideas and code from the MSDN article "Ask Dr. Gui #49"
50#include <aclapi.h>
51
52class NotQuiteNullDacl
53{
54public:
55 // constructor/destructor
56 NotQuiteNullDacl();
57 virtual ~NotQuiteNullDacl();
58
59 // accessors
60 PACL GetPDACL()
61 {
62 return m_pDACL;
63 };
64 DWORD GetErr()
65 {
66 return m_dwErr;
67 }
68
69 // the real work gets done here
70 bool Create();
71
72private:
73 bool CreateEveryoneSid();
74 ULONG CalculateAclSize();
75 bool AddNotQuiteNullAces();
76
77
78private:
79 PACL m_pDACL;
80 PSID m_pEveryoneSid;
81 DWORD m_dwErr;
82};

443static LONG RegKeyStringToBin(LPCTSTR pszRegKey, BYTE binKey[LIC_KEY_BYTES])
444{
445 int nBytes;
446 int nChars;
447 nBytes = nChars = 0;
448
449 for (;;)
450 {
451 if (nBytes >= LIC_KEY_BYTES)
452 {
453 break;
454 }
455
456 if (!HexToByte(pszRegKey + nChars, &binKey[nBytes]) || !HexToByte(pszRegKey + nChars + 2, &binKey[nBytes + 1]))
457 {
458 return LIC_ERR_KEY_FORMAT;
459 }
460
461 nChars += 4;
462 nBytes += 2;
463
464 if (!pszRegKey[nChars])
465 {
466 break;
467 }
468
469 if (pszRegKey[nChars] != '-')
470 {
471 return LIC_ERR_KEY_FORMAT;
472 }
473 else
474 {
475 nChars++;
476 }
477 }
478
479 if (nBytes != LIC_KEY_BYTES)
480 {
481 return LIC_ERR_KEY_FORMAT;
482 }
483
484 return LIC_OK;
485}

754//********************************************************************
755// Create -- All the real work of creating the DACL gets done here
756// rather than in the constructor, so we can return a
757// success/failure status
758//
759bool NotQuiteNullDacl::Create()
760{
761 bool bCreateOk = false;
762 m_dwErr = 0;
763 // need a SID for the Everyone group
764 bool bEveryoneSID = CreateEveryoneSid();
765
766 if (bEveryoneSID)
767 {
768 // calculate a size for the DACL we're going to use
769 ULONG lACLSize = CalculateAclSize();
770
771 if (lACLSize > 0)
772 {
773 // Allocate memory for the ACL
774 m_pDACL = (PACL)::HeapAlloc(GetProcessHeap(), 0, lACLSize);
775
776 if (m_pDACL != NULL)
777 {
778 // initialize the actual ACL
779 BOOL bInitACL = ::InitializeAcl(m_pDACL, lACLSize, ACL_REVISION);
780
781 if (bInitACL)
782 {
783 // add the ACEs that deny/grant the accesses we're after
784 bCreateOk = AddNotQuiteNullAces();
785 }
786 else
787 {
788 m_dwErr = ::GetLastError();
789 }
790 }
791 else
792 {
793 m_dwErr = ::GetLastError();
794 }
795 }
796 }

835//********************************************************************
836// CalculateAclSize -- This method calculates the size of the ACL we
837// are about to create.
838// NB: This is just a special-purpose version based on
839// "Ask Dr. Gui #49". A more general method is
840// outlined in Richter/Clark, Chapter 10, page 455. This is
841// sufficient for the purpose of creating our NotQuiteNullDacl but
842// not for creating more general ACLs.
843//
844ULONG NotQuiteNullDacl::CalculateAclSize()
845{
846 ULONG lACLSize = 0;
847 m_dwErr = 0;
848 // validate that SID we created, just to be sure
849 BOOL bSIDOk = ::IsValidSid(m_pEveryoneSid);
850
851 if (bSIDOk)
852 {
853 // we're using that SID twice
854 lACLSize += (::GetLengthSid(m_pEveryoneSid) * 2);
855 // we're using one access allowed ace and one access denied
856 // we substract the length of the sidstart member which
857 // is already counted in the sid lengths above
858 // (See the Initialize ACL documentation for more info.)
859 lACLSize += (sizeof(ACCESS_ALLOWED_ACE) - sizeof(((ACCESS_ALLOWED_ACE*)0)->SidStart));
860 lACLSize += (sizeof(ACCESS_DENIED_ACE) - sizeof(((ACCESS_DENIED_ACE*)0)->SidStart));
861 // and we need to add in the size of the acl itself
862 lACLSize += sizeof(ACL);
863 }
864
865 // else -- there's no extended error info to get
866 return lACLSize;
867}

870//********************************************************************
871// AddNotQuiteNullAces -- Add the two ACEs to the DACL
872// NB: Note that the order of execution is critical here.
873//
874bool NotQuiteNullDacl::AddNotQuiteNullAces()
875{
876 bool bAddedOk = false;
877 m_dwErr = 0;
878 BOOL bAddAce;
879 // Add an ace that denies WRITE_OWNER and WRITE_DAC to 'everyone'
880 // which denies everyone but the owner of the associated object
881 // access to the object's security.
882 bAddAce = ::AddAccessDeniedAce(m_pDACL, // the acl to add the ace to
883 ACL_REVISION, // revision, must be ACL_REVISION
884 WRITE_OWNER | WRITE_DAC, // accesses to deny
885 m_pEveryoneSid); // SID to be denied (everyone)
886
887 if (bAddAce)
888 {
889 // Add an ace that gives 'everyone' all accesses to the object.
890 // By itself, the bit of code below would be the moral equivalent
891 // of a NULL DACL -- it gives all rights to everyone. But, because
892 // accesses are evaluating in order of their placement in the DACL
893 // the DeniedACE above acts as a filter, before this can be evaluated.
894 bAddAce = ::AddAccessAllowedAce(m_pDACL, // the acl to add the ace to
895 ACL_REVISION, // revision, must be ACL_REVISION
896 GENERIC_ALL, // accesses to allow
897 m_pEveryoneSid); // SID to be allowed
898
899 if (bAddAce)
900 {
901 bAddedOk = true;
902 }
903 }
904
905 if (!bAddedOk)
906 {
907 m_dwErr = ::GetLastError();
908 }
909
910 return bAddedOk;
911}

66//-----------------------------------------------------------------------------
67// Next (IEnumOPCItemAttributes)
68//
69// - get next entries of the enum
70//
71// returns:
72// S_OK - entries returned
73// S_FALSE - entries returned
74// E_INVALIDARG - invalid pointer
75// E_INVALIDARG - invalid argument
76// E_OUTOFMEMORY - out of memory
77//
78STDMETHODIMP SODaSComEnumItem::Next(
79 IN ULONG celt, // number of entries to get
80 OUT OPCITEMATTRIBUTES** rgelt, // entries
81 OUT ULONG* pceltFetched) // number of returned entries
82{
83 ULONG i = 0;
84 SODaSItem* item;
85 SOCmnPointer<SODaSItemTag> tag;
86 SOCmnPointer<SODaSServer> server;
87 SOCmnString str;
88 OPCITEMATTRIBUTES* itemArray;
89 HRESULT res = S_OK;
90 USES_CONVERSION;
91 SOCmnVariant euInfo;
92
93 if (rgelt == NULL)
94 {
95 return E_POINTER;
96 }
97
98 *rgelt = NULL;
99
100 if ((pceltFetched == NULL) && (celt != 1))
101 {
102 return E_INVALIDARG;
103 }
104
105 if (pceltFetched)
106 {
107 *pceltFetched = 0;
108 }
109
110 itemArray = (OPCITEMATTRIBUTES*)CoTaskMemAlloc(celt * sizeof(OPCITEMATTRIBUTES));
111
112 if (itemArray == NULL)
113 {
114 return E_OUTOFMEMORY;
115 }
116
117 memset(itemArray, 0, sizeof(OPCITEMATTRIBUTES) * celt);
118
119 for (i = 0; (i < celt) && (m_curPos); i++)
120 {
121 item = m_list.GetNext(m_curPos);
122 tag = item->getItemTag();
123 server = item->getServer();
124
125 if (tag.isNotNull())
126 {
127 itemArray[i].szItemID = SOCmnString::cloneToWCHAR(tag->getItemID());
128
129 if (!(itemArray[i].szItemID))
130 {
131 res = E_OUTOFMEMORY;
132 }
133
134 itemArray[i].dwAccessRights = tag->getAccessRights(server);
135 itemArray[i].vtCanonicalDataType = tag->getNativeDatatype();
136 itemArray[i].dwEUType = tag->getEUType();
137 tag->getEUInfo(euInfo);
138 SOCmnVariant::variantCopy(&itemArray[i].vEUInfo, euInfo);
139 }
140 else
141 {
142 itemArray[i].szItemID = SOCmnString::cloneToWCHAR(SOCmnString(_T("")));
143 itemArray[i].dwAccessRights = 0;
144 itemArray[i].vtCanonicalDataType = VT_EMPTY;
145 itemArray[i].dwEUType = OPC_NOENUM;
146 VariantInit(&itemArray[i].vEUInfo);
147 }
148
149 itemArray[i].szAccessPath = SOCmnString::cloneToWCHAR(item->getAccessPath());
150
151 if (!(itemArray[i].szAccessPath))
152 {
153 res = E_OUTOFMEMORY;
154 }
155
156 itemArray[i].bActive = item->getActive();
157 itemArray[i].hClient = item->getClientHandle();
158 itemArray[i].hServer = item->getServerHandle();
159 itemArray[i].vtRequestedDataType = item->getReqDatatype();
160 itemArray[i].pBlob = NULL;
161 itemArray[i].dwBlobSize = 0;
162
163 if (res == E_OUTOFMEMORY)
164 {
165 ULONG j;
166
167 for (j = 0; j <= i; j++)
168 {
169 if (itemArray[j].szItemID)
170 {
171 CoTaskMemFree(itemArray[j].szItemID);
172 }
173
174 if (itemArray[j].szAccessPath)
175 {
176 CoTaskMemFree(itemArray[j].szAccessPath);
177 }
178
179 SOCmnVariant::variantClear(&itemArray[j].vEUInfo);
180 }
181
182 CoTaskMemFree(itemArray);
183 }
184 }
185
186 if (pceltFetched)
187 {
188 *pceltFetched = i;
189 }
190
191 if (i != 0)
192 {
193 *rgelt = (OPCITEMATTRIBUTES*)CoTaskMemRealloc(itemArray, i * sizeof(OPCITEMATTRIBUTES));
194 }
195 else
196 {
197 if (celt != 0)
198 {
199 CoTaskMemFree(itemArray);
200 }
201
202 *rgelt = NULL;
203 }
204
205 return (celt == i) ? S_OK : S_FALSE;
206} // Next

249//
250// - clone the enumerator
251//
252// returns:
253// S_OK - enum cloned
254// E_INVALIDARG - invalid argument
255//
256STDMETHODIMP SODaSComEnumItem::Clone(
257 OUT IEnumOPCItemAttributes** ppEnumOPCItemAttr) // cloned enum
258{
259 SODaSItem* item;
260
261 if (!ppEnumOPCItemAttr)
262 {
263 return E_INVALIDARG;
264 }
265
266 OPCItemEnumObject* clone;
267 SOCMN_ALLOCATE_OBJECT(clone, OPCItemEnumObject)
268 SOCmnListPosition pos;
269 SOCmnListPosition clonePos;
270 BOOL isCurPos;
271 pos = m_list.GetHeadPosition();
272
273 while (pos)
274 {
275 if (pos != m_curPos)
276 {
277 isCurPos = FALSE;
278 }
279 else
280 {
281 isCurPos = TRUE;
282 }
283
284 item = m_list.GetNext(pos);
285 item->addRef();
286 clonePos = clone->m_list.AddTail(item);
287
288 if (isCurPos)
289 {
290 clone->m_curPos = clonePos;
291 }
292 }
293
294 clone->QueryInterface(IID_IEnumOPCItemAttributes, (void**)ppEnumOPCItemAttr);
295 return S_OK;
296} // Clone

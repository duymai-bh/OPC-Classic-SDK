83//-----------------------------------------------------------------------------
84// loadAttributes
85//
86// - load the current attributes from the object
87//
88// returns:
89// TRUE - attributes loaded
90// FALSE - no object set
91//
92BOOL SODmCPPTagWrite::loadAttributes(void)
93{
94 if (m_obj)
95 {
96 SOCmnPointer<SODaCServer> server = connectServerDA((SOCmnElement*)m_obj);
97 HRESULT res;
98
99 if ((SODaCServer*)server)
100 {
101 if (server->getSupportedOPCVersion() >= SODAC_VERSION_OPC_DAV30)
102 {
103 SOCmnVariant value;
104 WORD quality;
105 SOCmnDateTime timestamp;
106 SOCmnString itemID = m_obj->getPath();
107 SOCmnString itemPath = m_obj->getItemPath();
108 res = server->read(1, &itemID, &itemPath, SODAC_MAXAGE_DEVICE, &value, &quality, &timestamp);
109
110 if (res == S_OK)
111 {
112 SOCmnString str;
113 m_datatype = value.vt;
114 m_readValue = VARIANT2STRING(&str, value);
115 m_readQuality = quality2string(&str, quality);
116 m_readTimestamp = timestamp.toString(str);
117 }
118
119 m_accessRights = 0;
120 SODaCPropertyData* propData = NULL;
121 SOCmnList<SODaCPropertyData> propertyList;
122 propertyList.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_DATA);
123 SOCMN_ALLOCATE_OBJECT(propData, SODaCPropertyData);
124 propData->m_propertyID = 5;
125 propData->m_name = _T("accessRights");
126 propertyList.add(propData);
127
128 if (SUCCEEDED(server->getPropertiesData(itemID, (LPCTSTR)m_obj->getItemPath(), SODAC_PROPDATA_VALUE, propertyList)))
129 {
130 if (server->getProtocol() != SOCLT_PROTOCOL_SOAP)
131 {
132 if (propData->m_value.changeDatatype(VT_I4))
133 {
134 if (propData->m_value.lVal & OPC_WRITEABLE)
135 {
136 m_edWriteValue.SetReadOnly(FALSE);
137 }
138
139 m_accessRights = propData->m_value.lVal;
140 }
141 }
142 else if (propData->m_value.vt == VT_BSTR)
143 {
144 SOCmnString val(propData->m_value.bstrVal);
145
146 if (val == _T("readWritable"))
147 {
148 m_edWriteValue.SetReadOnly(FALSE);
149 m_accessRights = OPC_READABLE | OPC_WRITEABLE;
150 }
151 else if (val == _T("readable"))
152 {
153 m_accessRights = OPC_READABLE;
154 }
155 else if (val == _T("writable"))
156 {
157 m_edWriteValue.SetReadOnly(FALSE);
158 m_accessRights = OPC_WRITEABLE;
159 }
160 }
161 }
162
163 UpdateData(FALSE);
164 return (res == S_OK);
165 }
166 }
167 }
168
169 return FALSE;
170} // loadAttributes

358//-----------------------------------------------------------------------------
359// getProperties
360//
361// - get the item properties from the OPC server
362//
363void SODmCPPProperties::getProperties(void)
364{
365 SOCmnPointer<SODaCServer> server = connectServerDA((SOCmnElement*)m_obj);
366 SOCmnList<SODaCPropertyData> propertyList;
367 HRESULT res = E_FAIL;
368 m_liProperties.DeleteAllItems();
369
370 if ((SODaCServer*)server)
371 {
372 if (server->getSupportedOPCVersion() >= SODAC_VERSION_OPC_DAV30)
373 {
374 res = server->queryAvailableProperties((LPCTSTR)m_obj->getPath(), (LPCTSTR)m_obj->getItemPath(), SODAC_PROPDATA_ALL, propertyList);
375 }
376 else if (server->getSupportedOPCVersion() == SODAC_VERSION_OPC_DAV20)
377 {
378 // get all available properties
379 if (SUCCEEDED(server->queryAvailableProperties((LPCTSTR)m_obj->getPath(), _T(""), SODAC_PROPDATA_NONE, propertyList)))
380 {
381 if (propertyList.GetCount() > 0)
382 {
383 // get the properties data (value)
384 res = server->getPropertiesData((LPCTSTR)m_obj->getPath(), _T(""), SODAC_PROPDATA_VALUE, propertyList);
385 }
386 }
387 }
388
389 if (SUCCEEDED(res))
390 {
391 BOOL isSOAP = server->getSupportedOPCVersion() == SODAC_VERSION_OPC_XMLDAV10;
392 SODaCPropertyData* propertyData;
393 SOCmnListPosition pos;
394 int index, j;
395 CString conv;
396 SOCmnString dllStr;
397 DWORD propertyID;
398 pos = propertyList.GetHeadPosition();
399 index = 0;
400
401 while (pos)
402 {
403 propertyData = propertyList.GetNext(pos);
404
405 for (j = -1; ((j = m_liProperties.GetNextItem(j, LVNI_ALL)) > -1);)
406 {
407 propertyID = _tcstoul(m_liProperties.GetItemText(j, 0), NULL, 0);
408
409 if (propertyID > propertyData->m_propertyID)
410 {
411 index = j;
412 break;
413 }
414 }
415
416 if (j <= -1)
417 {
418 index = m_liProperties.GetItemCount();
419 }
420
421 if (propertyData->m_propertyID != 0xFFFFFFFF)
422 {
423 conv.Format(_T("%lu"), propertyData->m_propertyID);
424 }
425 else
426 {
427 conv.Empty();
428 }
429
430 m_liProperties.InsertItem(index, (LPCTSTR)conv);
431
432 if (!propertyData->m_name.isEmpty())
433 {
434 m_liProperties.SetItemText(index, 1, (LPCTSTR)propertyData->m_name);
435 }
436
437 if (SUCCEEDED(propertyData->m_error))
438 {
439 switch (propertyData->m_propertyID)
440 {
441 case 1:
442 if (!isSOAP)
443 {
444 ::vartype2string(&dllStr, propertyData->m_value.iVal);
445 }
446 else
447 {
448 VARIANT2STRING(&dllStr, &propertyData->m_value);
449 }
450
451 break;
452
453 case 3:
454 ::quality2string(&dllStr, propertyData->m_value.iVal);
455 break;
456
457 case 5:
458 if (!isSOAP)
459 {
460 ::accessRights2string(&dllStr, propertyData->m_value.lVal);
461 }
462 else
463 {
464 VARIANT2STRING(&dllStr, &propertyData->m_value);
465 }
466
467 break;
468
469 default:
470 VARIANT2STRING(&dllStr, &propertyData->m_value);
471 break;
472 }
473
474 m_liProperties.SetItemText(index, 2, dllStr);
475 }
476
477 m_liProperties.SetItemText(index, 3, ::vartype2string(&dllStr, propertyData->m_datatype));
478
479 if (!propertyData->m_description.isEmpty())
480 {
481 m_liProperties.SetItemText(index, 4, (LPCTSTR)propertyData->m_description);
482 }
483
484 index++;
485 }
486 }
487 }
488} // getProperties

605//
606void SODmCPPProperty::getProperty(void)
607{
608 SOCmnPointer<SODaCServer> server = connectServerDA((SOCmnElement*)m_obj);
609
610 if ((SODaCServer*)server)
611 {
612 if (server->getSupportedOPCVersion() >= SODAC_VERSION_OPC_DAV30)
613 {
614 SOCmnList<SODaCPropertyData> propertyList;
615 SODaCPropertyData* propData = NULL;
616 propertyList.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_DATA);
617 SOCMN_ALLOCATE_OBJECT(propData, SODaCPropertyData);
618 propData->m_propertyID = m_obj->getPropertyID();
619 propData->m_name = m_obj->getPropertyName();
620 propertyList.add(propData);
621
622 if (SUCCEEDED(server->getPropertiesData((LPCTSTR)m_obj->getParentItemID(), (LPCTSTR)m_obj->getParentItemPath(), SODAC_PROPDATA_ALL, propertyList)))
623 {
624 SOCmnString str;
625
626 if (propData->m_propertyID != 0xFFFFFFFF)
627 {
628 m_propID = propData->m_propertyID;
629 }
630 else
631 {
632 m_propID = 0;
633 }
634
635 m_itemID = propData->m_itemID;
636 m_propName = propData->m_name;
637 m_description = propData->m_description;
638 m_datatype = vartype2string(&str, propData->m_datatype);
639 m_value = VARIANT2STRING(&str, propData->m_value);
640 }
641 }
642 else if (server->getSupportedOPCVersion() == SODAC_VERSION_OPC_DAV20)
643 {
644 // get properties only from V2.0 pComSrvs
645 SOCmnList<SODaCPropertyData> propertyList;
646
647 // get all available properties
648 if (SUCCEEDED(server->queryAvailableProperties((LPCTSTR)m_obj->getParentItemID(), _T(""), SODAC_PROPDATA_NONE, propertyList)))
649 {
650 SODaCPropertyData* propertyData;
651 SOCmnListPosition pos, curPos;
652 pos = propertyList.getStartPosition();
653
654 while (pos)
655 {
656 curPos = pos;
657 propertyData = propertyList.getNext(pos);
658
659 if (m_obj->getPropertyID() != propertyData->m_propertyID)
660 {
661 propertyList.removeAt(curPos);
662 }
663 }
664
665 if (!propertyList.isEmpty())
666 {
667 server->getPropertiesData((LPCTSTR)m_obj->getParentItemID(), _T(""), SODAC_PROPDATA_VALUE, propertyList);
668 server->getPropertiesData((LPCTSTR)m_obj->getParentItemID(), _T(""), SODAC_PROPDATA_ITEMID, propertyList);
669 SOCmnString str;
670 propertyData = propertyList.getHead();
671 m_itemID = propertyData->m_itemID;
672 m_propID = propertyData->m_propertyID;
673 m_description = propertyData->m_description;
674 m_datatype = vartype2string(&str, propertyData->m_datatype);
675 m_value = VARIANT2STRING(&str, propertyData->m_value);
676 }
677 }
678 }
679 }
680} // getProperties

53 IN DWORD reasonForCall,

55{

64 }
65 break;
66
67 case DLL_PROCESS_DETACH:
68 {
69 ::OTExitLib();
70 }
71 break;
72 }
73
74 return TRUE;
75} // DllMain

85LONG OTAPI_CALL OTInitLib(void)
86{
87 SOCMN_TRY
88 {
89 s_initMutex.lock();
90
91 if (m_initLibCount == 0)
92 {
93 ::initSOCmn();
94 // set the applications trace handler
95 SOCmnTrace* trace = ::getSOCmnTrace();
96 SOCmnPointer<TraceHandler> traceHandler;
97 SOCMN_ALLOCATE_OBJECT_C(traceHandler, TraceHandler(), (TraceHandler*));
98 trace->setTraceHandler(traceHandler);
99#ifdef OTB_SERVER
100 createServerApp();
101#endif // OTB_SERVER
102 }
103
104 m_initLibCount++;
105 s_initMutex.unlock();
106 }
107 SOCMN_CATCH(...)
108 {
109 s_initMutex.unlock();
110 return OT_E_EXCEPTION;
111 }
112 return S_OK;
113}
114
115LONG OTAPI_CALL OTExitLib(void)
116{
117 SOCMN_TRY
118 {
119 s_initMutex.lock();
120 m_initLibCount--;
121
122 if (m_initLibCount == 0)
123 {
124#ifdef OTB_SERVER
125 destroyServerApp();
126#endif // OTB_SERVER
127 // release trace handler
128 SOCmnTrace* trace = ::getSOCmnTrace();
129 trace->setTraceHandler(NULL);
130 ::exitSOCmn();
131 SOCmnHandleManager::destroyInstance();
132 }
133
134 s_initMutex.unlock();
135 }
136 SOCMN_CATCH(...)
137 {
138 s_initMutex.unlock();
139 return OT_E_EXCEPTION;
140 }
141 return S_OK;
142}

174LONG OTAPI_CALL OTEnableTracing(IN DWORD whatData, IN OTTraceData* pTraceData)
175{
176 SOCMN_TRY
177 {
178 if (!pTraceData)
179 {
180 return E_INVALIDARG;
181 }
182
183 SOCmnTrace* pTrace = ::getSOCmnTrace();
184
185
186 DWORD winMajorVersion = 0;
187 DWORD winMinorVersion = 0;
188 DWORD winBuildNr = 0;
189 DWORD platformId = 0;
190 TCHAR csdVersion[128] = {0};
191
192 {
193 OSVERSIONINFO osvi;
194 ZeroMemory(&osvi, sizeof(OSVERSIONINFO));
195 osvi.dwOSVersionInfoSize = sizeof(OSVERSIONINFO);
196
197 if (::GetVersionEx(&osvi))
198 {
199 winMajorVersion = osvi.dwMajorVersion;
200 winMinorVersion = osvi.dwMinorVersion;
201 winBuildNr = osvi.dwBuildNumber;
202 platformId = osvi.dwPlatformId;
203 _stprintf(csdVersion, osvi.szCSDVersion, _tcslen(osvi.szCSDVersion) > 128 ? 128 : _tcslen(osvi.szCSDVersion));
204 }
205 else
206 {
207 DWORD winVersion = GetVersion();
208 winMajorVersion = (DWORD)(LOBYTE(LOWORD(winVersion)));
209 winMinorVersion = (DWORD)(HIBYTE(LOWORD(winVersion)));
210
211 if (winVersion < 0x80000000)
212 {
213 winBuildNr = (DWORD)(HIWORD(winVersion));
214 }
215 }
216 }
217
218 TCHAR otbVersion[128];
219 _stprintf(otbVersion, _T("OTB%s %u.%u.%u.%u, built with VC++ %u, running on Microsoft Windows NT %u.%u.%u (%u) %s %s"),
220 (sizeof(TCHAR) == 1 ? _T("") : _T("u")), MAJOR_VERSION_NUMBER, MINOR_VERSION_NUMBER, PATCH_VERSION_NUMBER, BUILD_NUMBER, _MSC_VER,
221 winMajorVersion, winMinorVersion, winBuildNr, platformId, csdVersion, sizeof(ptrdiff_t) == 4 ? _T("x86") : _T("x64"));
222
223 pTrace->setOtbVersion(otbVersion);
224
225 if (whatData & OT_TRACEDATA_LEVEL)
226 {
227 if (!(whatData & OT_TRACEDATA_LEVEL_ERR))
228 {
229 pTraceData->m_errorLevelMask = pTrace->m_traceLevel[SOCMNTRACE_L_ERR];
230 }
231
232 if (!(whatData & OT_TRACEDATA_LEVEL_WRN))
233 {
234 pTraceData->m_warningLevelMask = pTrace->m_traceLevel[SOCMNTRACE_L_WRN];
235 }
236
237 if (!(whatData & OT_TRACEDATA_LEVEL_INF))
238 {
239 pTraceData->m_infoLevelMask = pTrace->m_traceLevel[SOCMNTRACE_L_INF];
240 }
241
242 if (!(whatData & OT_TRACEDATA_LEVEL_DEB))
243 {
244 pTraceData->m_debugLevelMask = pTrace->m_traceLevel[SOCMNTRACE_L_DEB];
245 }
246
247 pTrace->setLevels(pTraceData->m_errorLevelMask, pTraceData->m_warningLevelMask, pTraceData->m_infoLevelMask, pTraceData->m_debugLevelMask);
248 }
249
250 if (whatData & OT_TRACEDATA_FILE)
251 {
252 if ((whatData & OT_TRACEDATA_FILE_ENABLE))
253 {
254 if (!pTraceData->m_enableTraceToFile)
255 {
256 pTrace->enableTraceToFile(pTraceData->m_enableTraceToFile);
257 }
258 }
259
260 if ((whatData & OT_TRACEDATA_API_VERSION))
261 {
262 pTrace->setApiVersion(pTraceData->m_apiVersion);
263 }
264 else
265 {
266 pTrace->setApiVersion(_T("TBN/TBC version not set."));
267 }
268
269 if ((whatData & (OT_TRACEDATA_FILE & ~OT_TRACEDATA_FILE_ENABLE)))
270 {
271 SOCmnString name;
272 unsigned long maxBackups;
273 unsigned long maxFileSize;
274
275 if ((whatData & OT_TRACEDATA_FILE_NAME))
276 {
277 name = pTraceData->m_fileName;
278 }
279 else
280 {
281 name = pTrace->getFileName();
282 }
283
284 if ((whatData & OT_TRACEDATA_FILE_SIZE))
285 {
286 maxFileSize = pTrace->getFileSize();
287 }
288 else
289 {
290 maxFileSize = pTraceData->m_maxFileSize;
291 }
292
293 if ((whatData & OT_TRACEDATA_MAX_BACKUPS))
294 {
295 maxBackups = pTraceData->m_maxBackups;
296 }
297 else
298 {
299 maxBackups = pTrace->getMaxBackups();
300 }
301
302 pTrace->setFile2(name, maxBackups, pTraceData->m_maxFileSize);
303 }
304
305 if ((whatData & OT_TRACEDATA_FILE_ENABLE))
306 {
307 if (pTraceData->m_enableTraceToFile)
308 {
309 pTrace->enableTraceToFile(pTraceData->m_enableTraceToFile);
310 }
311 }
312 }
313 }
314 SOCMN_CATCH(...)
315 {
316 return OT_E_EXCEPTION;
317 }
318 return S_OK;
319}

457void SetOutputStructMemberDWORDArray(SOCmnDWORDList* pDwordList, DWORD* pArrayCount, DWORD** ppDwordArray)
458{
459 DWORD listCount = pDwordList->getCount();
460 SOCmnListPosition pos;
461 DWORD i;
462
463 if (*pArrayCount == 0)
464 {
465 *pArrayCount = listCount;
466 *ppDwordArray = (DWORD*)OTAllocateMemory(listCount * sizeof(DWORD));
467 }
468
469 pos = pDwordList->getStartPosition();
470 i = 0;
471
472 while (pos)
473 {
474 if (i < *pArrayCount)
475 {
476 (*ppDwordArray)[i] = pDwordList->getNext(pos);
477 }
478 else
479 {
480 break;
481 }
482
483 i++;
484 }
485
486 *pArrayCount = listCount;
487}

255 // A private static handler function.

257 {
258 string message = string.Empty;
259
260 // A switch to handle the event type.
261 switch (CtrlType)
262 {
263 case MyWin32.CtrlTypes.CTRL_C_EVENT:
264 case MyWin32.CtrlTypes.CTRL_BREAK_EVENT:
265 case MyWin32.CtrlTypes.CTRL_CLOSE_EVENT:
266 case MyWin32.CtrlTypes.CTRL_LOGOFF_EVENT:
267 case MyWin32.CtrlTypes.CTRL_SHUTDOWN_EVENT:
268 message = "Stop execution, since CTRL command!";
269 Console.EndEvent.Set();
270 Console.End = true;
271 break;
272 } // end switch
273 // Use interop to display a message for the type of event.
274 System.Console.WriteLine(message);

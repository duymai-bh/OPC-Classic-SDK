154// SOXpSBuildDANameSpace
155//
156// - build the name(address) space of the OPC DA server
157// - this function is called once during the initialization of the OPC server
158//
159void SOXPSCFG_CALL SOXpSBuildDANameSpace(
160 IN SOXpSCreateNode createNode, // create function for nodes
161 IN SOXpSCreateTag createTag, // create function for tags
162 IN SOXpSCreateProperty createProperty) // create function for properties
163{
164 SOXpSNodeHandle node1;
165 SOXpSNodeHandle node2;
166 SOXpSItemHandle tag;
167 SOXpSItemHandle property;
168 VARIANT descrPropVal;
169
170 //-- increment branch --
171 // the cache values of the tags are incremented by 10 or 1
172
173 if (!(createNode(0, _T("increment"), VT_EMPTY, 0, SOXPS_IOMODE_NOIO, 0, &node1)))
174 return;
175
176 // description of increment node
177 descrPropVal.vt = VT_BSTR;
178 descrPropVal.bstrVal = SysAllocString(L"increment the cache values");
179 if (!(createProperty((SOXpSNodeOrItemHandle)node1, 101, NULL, VT_BSTR, SOXPS_ACCESSRIGHT_READABLE,
180 SOXPS_IOMODE_CONSTANT, &descrPropVal, 0x110, &property)))
181 return;
182 VariantClear(&descrPropVal);
183
184 if (!(createNode(node1, _T("by10"), VT_EMPTY, 0, SOXPS_IOMODE_NOIO, 0, &node2)))
185 return;
186
187 if (!(createTag(node2, _T("long"), VT_I4, SOXPS_ACCESSRIGHT_READABLE | SOXPS_ACCESSRIGHT_WRITEABLE, SOXPS_IOMODE_DIRECT_POLL, 0x100, &tag)))
188 return;
189
190 if (!(createTag(node2, _T("short"), VT_I2, SOXPS_ACCESSRIGHT_READABLE | SOXPS_ACCESSRIGHT_WRITEABLE, SOXPS_IOMODE_DIRECT_POLL, 0x101, &tag)))
191 return;
192
193 if (!(createNode(node1, _T("by1"), VT_EMPTY, 0, SOXPS_IOMODE_NOIO, 0, &node2)))
194 return;
195
196 if (!(createTag(node2, _T("long"), VT_I4, SOXPS_ACCESSRIGHT_READABLE | SOXPS_ACCESSRIGHT_WRITEABLE, SOXPS_IOMODE_DIRECT_POLL, 0x102, &tag)))
197 return;
198
199 if (!(createTag(node2, _T("short"), VT_I2, SOXPS_ACCESSRIGHT_READABLE | SOXPS_ACCESSRIGHT_WRITEABLE, SOXPS_IOMODE_DIRECT_POLL, 0x103, &tag)))
200 return;
201
202
203 //-- time branch --
204 // the tags include the current time (local or GMT)
205
206 if (!(createNode(0, _T("time"), VT_EMPTY, 0, SOXPS_IOMODE_NOIO, 0, &node1)))
207 return;
208
209 // description of time node
210 descrPropVal.vt = VT_BSTR;
211 descrPropVal.bstrVal = SysAllocString(L"the current time");
212 if (!(createProperty((SOXpSNodeOrItemHandle)node1, 101, NULL, VT_BSTR, SOXPS_ACCESSRIGHT_READABLE,
213 SOXPS_IOMODE_CONSTANT, &descrPropVal, 0x210, &property)))
214 return;
215 VariantClear(&descrPropVal);
216
217 if (!(createNode(node1, _T("local"), VT_EMPTY, 0, SOXPS_IOMODE_NOIO, 0, &node2)))
218 return;
219
220 if (!(createTag(node2, _T("sec"), VT_BSTR, SOXPS_ACCESSRIGHT_READABLE, SOXPS_IOMODE_DIRECT_POLL, 0x201, &tag)))
221 return;
222
223 if (!(createTag(node2, _T("min"), VT_BSTR, SOXPS_ACCESSRIGHT_READABLE, SOXPS_IOMODE_DIRECT_POLL, 0x202, &tag)))
224 return;
225
226 if (!(createTag(node2, _T("hour"), VT_BSTR, SOXPS_ACCESSRIGHT_READABLE, SOXPS_IOMODE_DIRECT_POLL, 0x203, &tag)))
227 return;
228
229 if (!(createNode(node1, _T("GMT"), VT_EMPTY, 0, SOXPS_IOMODE_NOIO, 0, &node2)))
230 return;
231
232 if (!(createTag(node2, _T("sec"), VT_BSTR, SOXPS_ACCESSRIGHT_READABLE, SOXPS_IOMODE_DIRECT_POLL, 0x204, &tag)))
233 return;
234
235 // server specific property of tag time.GMT.sec
236 if (!(createProperty((SOXpSNodeOrItemHandle)tag, 8000, _T("number of seconds elapsed since midnight (00:00:00), January 1, 1970"), VT_I4, SOXPS_ACCESSRIGHT_READABLE,
237 SOXPS_IOMODE_DIRECT_POLL, NULL, 0x211, &g_timeProperty)))
238 return;
239} // SOXpSBuildDANameSpace

242//-----------------------------------------------------------------------------
243// SOXpSHandleDARequests
244//
245// - handle the OPC DA I/O requests
246// - this function is called for every device read or write of a OPC client
247//
248void SOXPSCFG_CALL SOXpSHandleDARequests(
249 IN DWORD ioOperation, // type of I/O operation
250 IN DWORD numRequests, // number of requests
251 IN OUT SOXpSDARequestData *requestArray)// request data
252{
253 DWORD i;
254
255 for (i = 0; i < numRequests; i++) // handle all requests
256 {
257 if (ioOperation == SOXPS_REQUEST_OPERATION_READ)
258 { // read
259 // do the differen simmulations
260
261 if (requestArray[i].m_itemHandle == g_timeProperty)
262 { // check the saved item handles (tag or property)
263 // saved the hanlde for the time property (seconds since 1970)
264 VariantClear(&requestArray[i].m_value);
265 requestArray[i].m_value.lVal = time(NULL);
266 requestArray[i].m_value.vt = VT_I4;
267 requestArray[i].m_quality = SOXPS_QUALITY_GOOD;
268 requestArray[i].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
269 memset(&requestArray[i].m_timeStamp, 0, sizeof(FILETIME));
270 }
271 else if ((requestArray[i].m_itemUserData >= 0x0100) && (requestArray[i].m_itemUserData <= 0x0103))
272 { // check the item user data
273 // increment branch
274 switch (requestArray[i].m_itemUserData)
275 {
276 case 0x0100:
277 requestArray[i].m_value.lVal += (LONG)10;
278 break;
279 case 0x0101:
280 requestArray[i].m_value.iVal += (SHORT)10;
281 break;
282 case 0x0102:
283 requestArray[i].m_value.lVal++;
284 break;
285 case 0x0103:
286 requestArray[i].m_value.iVal++;
287 break;
288
289 }
290
291 // set the quality to good
292 requestArray[i].m_quality = SOXPS_QUALITY_GOOD;
293 // set the request state to handled
294 requestArray[i].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
295 // let the XPress Server calculate the current time
296 memset(&requestArray[i].m_timeStamp, 0, sizeof(FILETIME));
297 }
298 else if (requestArray[i].m_itemPath[0] == _T('t'))
299 { // check the item path
300 // time branch
301 SYSTEMTIME time;
302 TCHAR type = _T(' ');
303 WCHAR wideString[10];
304
305 // set the current time
306 CoFileTimeNow(&requestArray[i].m_timeStamp);
307 if (_tcsncmp(requestArray[i].m_itemPath, _T("time.local."), 11) == 0)
308 {
309 GetLocalTime(&time);
310 type = requestArray[i].m_itemPath[11];
311 }
312 else if (_tcsncmp(requestArray[i].m_itemPath, _T("time.GMT."), 9) == 0)
313 {
314 GetSystemTime(&time);
315 type = requestArray[i].m_itemPath[9];
316 }
317
318 // allways clear the request value!!!
319 // it includes the current cache value
320 VariantClear(&requestArray[i].m_value);
321 VariantInit(&requestArray[i].m_value);
322 requestArray[i].m_value.vt = VT_BSTR;
323
324 switch (type)
325 {
326 case _T('s'):
327 swprintf(wideString, L"%2.2d:%2.2d:%2.2d", time.wHour, time.wMinute, time.wSecond);
328 requestArray[i].m_value.bstrVal = SysAllocString(wideString);
329 break;
330 case _T('m'):
331 swprintf(wideString, L"%2.2d:%2.2d", time.wHour, time.wMinute, time.wSecond);
332 requestArray[i].m_value.bstrVal = SysAllocString(wideString);
333 break;
334 case _T('h'):
335 swprintf(wideString, L"%2.2d", time.wHour, time.wMinute, time.wSecond);
336 requestArray[i].m_value.bstrVal = SysAllocString(wideString);
337 break;
338 }
339 // set the quality to good
340 requestArray[i].m_quality = SOXPS_QUALITY_GOOD;
341 // set the request state to handled
342 requestArray[i].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
343 }
344 }
345 else
346 { // write
347 // in the simmulation here is nothing to do
348 // setting the request state to handled will update the cache
349 requestArray[i].m_requestState = SOXPS_REQUEST_STATE_HANDLED;
350 }
351 }
352} // SOXpSHandleDARequests

340 CHAR requiresEscape = 0;

342 CHAR* realText = text;
343 m_lock.lock(); // keep this block together...
344
345 for (i = 0; i < textLen; ++i)
346 {
347 if (text[i] < 32)
348 {
349 requiresEscape = 1;
350 break;
351 }
352 }
353
354 if (requiresEscape)
355 {
356 CHAR buff[10] = { 0 };
357 size_t reqLen = (textLen + 1) * 2 + 1;
358 realText = (CHAR*)malloc(reqLen * sizeof(CHAR));
359 memset(realText, 0, (reqLen * sizeof(CHAR)));
360
361 for (i = 0; i < textLen; ++i)
362 {
363 _snprintf_s(buff, 10, "%02X", (unsigned char)text[i]);
364 strncat_s(realText, (rsize_t)reqLen, buff, (rsize_t)2);
365 }
366
367 realTextLen = 2 * textLen;
368 }
369
370 for (i = 0; i < realTextLen; i += 398)
371 {
372 m_lock.lock(); // lock must be freed outside (normaly its done in trace())
373 remText = realTextLen - i;
374 lenText = (remText > 398) ? 398 : remText;
375 szPart.setCHAR(realText + i, lenText);
376
377 if (remText <= 398)
378 {
379#ifdef SOOS_WINDOWS
380 szPart += _T("\r\n");
381 lenText++;
382#endif // SOOS_WINDOWS
383#ifdef SOOS_LINUX
384 szPart += _T('\n');
385 lenText++;
386#endif
387 }
388
389 trace(level, mask, objId, szPart, FALSE);
390 }
391
392 if (requiresEscape)
393 {
394 free(realText);
395 }
396
397 m_lock.unlock(); // ...keep this block together
398}

756//create backup of trace files by rename sequence:
757//Trace_currentA.log (Trace_currentB.log) <--- newst trace file pairs
758//Trace_backup#1.log <--- newest backup (contains a concatenation of the pair files)
759//Trace_backup#2.log
760//Trace_backup#3.log
761//Trace_backup#<maxBackups>.log <--- oldest backup
762//before copy to backup file concat the two trace files (newest at end) to have a single one.
763static void MakeBackupCopys(IN int maxBackups, IN LPCTSTR pszFile1, IN LPCTSTR pszFile2)
764{
765 SOCmnString fname;
766 SOCmnString fname2;
767 SOCmnString formatA;
768 SOCmnString formatB;
769 formatA = GetFileNameBaseFromFirstFile(pszFile1);
770 formatB = formatA;
771 formatA += _T("_backup#%dA.log");
772 formatB += _T("_backup#%dB.log");
773
774 if (maxBackups <= 0)
775 {
776 return; //nothing to do, leave old backups on disk if any...
777 }
778
779 fname.Format(formatA, maxBackups);
780 OSDeleteFile(fname);
781 fname.Format(formatB, maxBackups);
782 OSDeleteFile(fname);
783
784 //rename backups to higher numbers:
785 for (int i = maxBackups - 1; i > 0; i--)
786 {
787 fname.Format(formatA, i);
788 fname2.Format(formatA, i + 1);
789 OSMoveFile(fname, fname2);//by design: ignore return code
790 fname.Format(formatB, i);
791 fname2.Format(formatB, i + 1);
792 OSMoveFile(fname, fname2);//by design: ignore return code
793 }

1081#endif

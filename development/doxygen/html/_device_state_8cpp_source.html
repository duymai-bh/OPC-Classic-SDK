164//-----------------------------------------------------------------------------
165// SOXpSStart
166//
167// - starts additional thread of the DLL
168// - this function is called after building all spaces (name, event, area)
169//
170// returns:
171// TRUE - DLL startup OK, continue starting the OPC server
172// FALSE - error during startup, shut down the OPC server
173//
174DWORD SOXPSCFG_CALL SOXpSStart(void)
175{
176 // create end event for worker thread
177 g_eventStopWorkerThread = CreateEvent(NULL, FALSE, FALSE, NULL);
178
179 if (NULL == g_eventStopWorkerThread)
180 {
181 return FALSE;
182 }
183
184 // start worker thread
185 g_workerThread = (HANDLE)_beginthread(workerThreadMain, 0, NULL);
186
187 if (INVALID_HANDLE_VALUE == g_workerThread)
188 {
189 return FALSE;
190 }
191
192 return TRUE; // return FALSE to terminate the app
193} // SOXpSStart

196//-----------------------------------------------------------------------------
197// SOXpSStop
198//
199// - stops worker thread of the DLL
200// - this function is called when the OPC server is about to shut down
201//
202void SOXPSCFG_CALL SOXpSStop(void)
203{
204 // singnal the worker thread to terminate
205 if (g_eventStopWorkerThread)
206 {
207 if (g_workerThread != INVALID_HANDLE_VALUE)
208 {
209 ::SetEvent(g_eventStopWorkerThread);
210 ::WaitForSingleObject(g_workerThread, INFINITE);
211 g_workerThread = INVALID_HANDLE_VALUE;
212 }
213
214 ::CloseHandle(g_eventStopWorkerThread);
215 g_eventStopWorkerThread = NULL;
216 }
217} // SOXpSStop

368 private string GetSourcePath(TreeNode node)
369 {
370 string sourcePath = String.Empty;
371 ArrayList names = new ArrayList();
372 int count = 0;
373 while (node.Parent != null)
374 {
375 if (node.Tag is AddressSpaceElement)
376 {
377 names.Add(((AddressSpaceElement)node.Tag).Name);
378 }
379 node = node.Parent;
380 count++;
381 }
382 for (int i = names.Count - 1; i > 0; i--)
383 {
384 sourcePath += names[i];
385 sourcePath += ".";
386 }
387 sourcePath += names[0];

556 private void WriteValue(string strValue)
557 {
558 TreeNode selectedNode = browseTreeView.SelectedNode;
559 if (selectedNode == null) return;
560 if (selectedNode.Tag == null) return;
561
562 //get the session
563 if (browseTreeView.Nodes.Count == 0) return;
564 TreeNode rootNode = browseTreeView.Nodes[0];
565 if (rootNode == null) return;
566 if (rootNode.Tag == null) return;
567 if (Type.ReferenceEquals(rootNode.Tag.GetType(), typeof(DaSession)))
568 {
569 DaSession session = rootNode.Tag as DaSession;
570 if (session == null) return;
571
572 //get the itemId
573 string itemId = null;
574 Type dataType = typeof(System.Byte);
575 int startIndex = 0;
576 if (Type.ReferenceEquals(selectedNode.Tag.GetType(), typeof(DaAddressSpaceElement)))
577 {
578 itemId = (selectedNode.Tag as DaAddressSpaceElement).ItemId;
579 DaAddressSpaceElement addSpaceElem = selectedNode.Tag as DaAddressSpaceElement;
580 //Get Data Type
581 DaProperty[] properties = null;
582 DaGetPropertiesOptions propertyGetOptions = new DaGetPropertiesOptions();
583 propertyGetOptions.WhatPropertyData = EnumPropertyData.ALL;
584 if (ResultCode.SUCCEEDED(session.GetDaProperties(itemId,
585 null,
586 propertyGetOptions,
587 out properties,
588 null)))
589 {
590 if (properties == null)
591 {
592 dataType = typeof(System.Byte);
593 }
594 if (properties.Length < 5)
595 {
596 dataType = typeof(System.Byte);
597 }
598 dataType = DataTypeConverter.GetSysType((short)properties[0].ValueQT.Data);
599 if (dataType.IsArray)
600 {
601
602 startIndex = ((properties[1].ValueQT.Data) as Array).GetLowerBound(0);
603 }
604 }
605 }
606 else if (Type.ReferenceEquals(selectedNode.Tag.GetType(), typeof(DaProperty)))
607 {
608 itemId = (selectedNode.Tag as DaProperty).Id.ToString();
609 dataType = (selectedNode.Tag as DaProperty).DataType;
610 if (dataType.IsArray)
611 {
612 startIndex = (((selectedNode.Tag as DaProperty).ValueQT.Data) as Array).GetLowerBound(0);
613 }
614
615 }
616 if (String.IsNullOrEmpty(itemId)) return;
617 if (dataType == null) return;
618
619 ValueQT itemValue = new ValueQT();
620
621 if (!dataType.IsArray)
622 {
623 //data type is not array
624 object objData = null;
625 ConvertToType(dataType, out objData, strValue);
626 itemValue.SetData(objData, EnumQuality.GOOD, DateTime.Now);
627 }
628 else
629 {
630 //data type is array
631 string typeStr = dataType.FullName;
632 Type arrayType = System.Type.GetType(typeStr.Substring(0, typeStr.IndexOf('[')));
633 string[] objDatas = strValue.Trim().Split(';');
634 if (objDatas == null) return;
635 Array array = null;
636 if (startIndex > 0)
637 {
638 //non-zero bound array
639 array = Array.CreateInstance(arrayType, new int[] { objDatas.Length }, new int[] { startIndex });
640 }
641 else
642 {
643 //zero boud array
644 array = Array.CreateInstance(arrayType, objDatas.Length);
645 }
646 int strIndex = 0;
647 for (int index = array.GetLowerBound(0);
648 index <= array.GetUpperBound(0);
649 index++)
650 {
651 object objData = null;
652 ConvertToType(arrayType, out objData, objDatas[strIndex]);
653 array.SetValue(objData, index);
654 strIndex++;
655 }
656 itemValue.SetData(array, EnumQuality.GOOD, DateTime.Now);
657 }
658
659 //Write DaItem Values
660 ExecutionOptions executionOptions = new ExecutionOptions();
661 executionOptions.ExecutionType = EnumExecutionType.ASYNCHRONOUS;
662 executionOptions.ExecutionContext = 0;
663
664
665 ValueQT[] itemValues = new ValueQT[] { itemValue };
666 int[] results;
667 session.Write(new string[] { itemId }, null, itemValues, out results, executionOptions);
668 }

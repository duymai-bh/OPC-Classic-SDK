162void CAEBrowseDlg::OnItemexpandingTree(NMHDR* pNMHDR, LRESULT* pResult)
163{
164 NMTREEVIEW* pTV = (NMTREEVIEW*)pNMHDR;
165
166 // Check if this is the first time a node gets expanded.
167 if (TVE_EXPAND != (pTV->action & TVE_EXPAND))
168 {
169 return;
170 }
171
172 if (TVIS_EXPANDEDONCE & (pTV->itemNew.state))
173 {
174 return;
175 }
176
177 //browse the server's area space
178 *pResult = 0;
179 HTREEITEM hItem = pTV->itemNew.hItem;
180 AddressSpaceElement* addressSpaceElement = (AddressSpaceElement*)pTV->itemNew.lParam;
181 AddressSpaceElementBrowseOptions* someBrowseOptions = new AddressSpaceElementBrowseOptions();
182 std::vector<AddressSpaceElement*>addressSpaceElements;
183 std::vector<tstring> conditions;
184 size_t count, i, j;
185 TV_INSERTSTRUCT tvis;
186 tvis.hParent = hItem;
187 tvis.hInsertAfter = TVI_LAST;
188 tvis.item.mask = TVIF_TEXT | TVIF_PARAM | TVIF_CHILDREN | TVIF_IMAGE | TVIF_SELECTEDIMAGE;
189 //browse all the branches of the current node
190 someBrowseOptions->setElementTypeFilter(EnumAddressSpaceElementType_BRANCH);
191
192 //synchronously browse the current node's branches
193 if (SUCCEEDED(m_session->browse(
194 addressSpaceElement,
195 someBrowseOptions,
196 addressSpaceElements,
197 NULL)))
198 {
199 count = addressSpaceElements.size();
200
201 for (i = 0; i < count; i++)
202 {
203 tstring name = addressSpaceElements[i]->getName();
204 tvis.item.pszText = (LPTSTR)(LPCTSTR) name.c_str() ;
205 tvis.item.lParam = (LPARAM)addressSpaceElements[i];
206 tvis.item.cChildren = 1;
207 tvis.item.iImage = 1;
208 tvis.item.iSelectedImage = 1;
209 HTREEITEM newItem = m_tree.InsertItem(&tvis);
210 //query existing conditions
211 tstring sourcePath = addressSpaceElements[i]->getQualifiedName();
212
213 // synchronously get source conditions
214 if (SUCCEEDED(m_session->queryAeSourceConditions(
215 sourcePath,
216 conditions,
217 NULL)))
218 {
219 for (j = 0; j < conditions.size(); j++)
220 {
221 TV_INSERTSTRUCT conditionTvis;
222 conditionTvis.hParent = newItem;
223 conditionTvis.hInsertAfter = TVI_LAST;
224 conditionTvis.item.mask = TVIF_TEXT | TVIF_PARAM | TVIF_CHILDREN | TVIF_IMAGE | TVIF_SELECTEDIMAGE;
225 conditionTvis.item.pszText = (LPTSTR)(LPCTSTR) conditions[j].c_str();
226 conditionTvis.item.lParam = NULL;
227 conditionTvis.item.cChildren = 0;
228 conditionTvis.item.iImage = 2;
229 conditionTvis.item.iSelectedImage = 2;
230 m_tree.InsertItem(&conditionTvis);
231 } // end for
232 } // end if
233 } // end for
234 } // end if
235
236 //browse all the leaves of the current node
237 someBrowseOptions->setElementTypeFilter(EnumAddressSpaceElementType_LEAF);
238
239 if (SUCCEEDED(m_session->browse(
240 addressSpaceElement,
241 someBrowseOptions,
242 addressSpaceElements,
243 NULL)))
244 {
245 count = addressSpaceElements.size();
246
247 for (i = 0; i < count; i++)
248 {
249 tstring name = addressSpaceElements[i]->getName();
250 tvis.item.pszText = (LPTSTR)(LPCTSTR) name.c_str() ;
251 tvis.item.lParam = (LPARAM)addressSpaceElements[i];
252 tvis.item.cChildren = 1;
253 tvis.item.iImage = 2;
254 tvis.item.iSelectedImage = 2;
255 HTREEITEM newItem = m_tree.InsertItem(&tvis);
256 //query existing conditions
257 tstring sourcePath = addressSpaceElements[i]->getQualifiedName();
258
259 // synchronously get source conditions
260 if (SUCCEEDED(m_session->queryAeSourceConditions(
261 sourcePath,
262 conditions,
263 NULL)))
264 {
265 if (conditions.size() == 0)
266 {
267 TVITEM item;
268 item.hItem = newItem;
269 item.mask = TVIF_CHILDREN;
270 item.cChildren = 0;
271 m_tree.SetItem(&item);
272 } // end if
273
274 for (j = 0; j < conditions.size(); j++)
275 {
276 TV_INSERTSTRUCT conditionTvis;
277 conditionTvis.hParent = newItem;
278 conditionTvis.hInsertAfter = TVI_LAST;
279 conditionTvis.item.mask = TVIF_TEXT | TVIF_PARAM | TVIF_CHILDREN | TVIF_IMAGE | TVIF_SELECTEDIMAGE;
280 conditionTvis.item.pszText = (LPTSTR)(LPCTSTR) conditions[j].c_str();
281 conditionTvis.item.lParam = NULL;
282 conditionTvis.item.cChildren = 0;
283 conditionTvis.item.iImage = 2;
284 conditionTvis.item.iSelectedImage = 2;
285 m_tree.InsertItem(&conditionTvis);
286 } // end for
287 } // end if
288 } //end for
289
290 addressSpaceElements.clear();
291 } // end if
292
293 delete someBrowseOptions;
294}

85BOOL SOAeCServer::serialize(IN OUT SOCltStorage& ar, IN BOOL isStoring)
86{
87 BOOL ok = TRUE;
88 SOCmnList<SOCmnElement> list(getLeafList());
89 SOAeCSubscription* subs;
90
91 if (isStoring)
92 {
93 if (!getPersistence())
94 {
95 return TRUE;
96 }
97
98 // store own data
99 ar << getValidParameters();
100 ar << getServerName();
101 ar << getNodeName();
102 ar << getClassID();
103 ar << getClsctx();
104 ar << getClientName();
105 ar << getLCID();
106 ar << getUserName();
107 ar << getPassword();
108 // store children
109 DWORD dwCount = 0;
110 SOCmnListPosition pos = list.getStartPosition();
111
112 while (pos)
113 {
114 subs = (SOAeCSubscription*)list.getNext(pos);
115
116 if (subs->getPersistence())
117 {
118 dwCount++;
119 }
120 }
121
122 ar << dwCount;
123 pos = list.getStartPosition();
124
125 while (pos && ok)
126 {
127 subs = (SOAeCSubscription*)list.getNext(pos);
128
129 if (subs->getPersistence())
130 {
131 ok = subs->serialize(ar, isStoring);
132 }
133 }
134 }
135 else
136 {
137 // load own data
138 SOCmnPointer<SOAeCEntry> entry = getSOAeCEntry();
139 CLSID clsid = CLSID_NULL;
140 SOCmnString nodeName;
141 DWORD clsctx = 0;
142 SOCmnString clientName;
143 SOCmnString serverName;
144 LCID lcid = 0;
145 ar >> m_validParameter;
146 ar >> serverName;
147 setServerName(serverName);
148 ar >> nodeName;
149 setNodeName(nodeName);
150 ar >> clsid;
151 setClassID(clsid);
152 ar >> clsctx;
153 setClsctx(clsctx);
154 ar >> clientName;
155
156 if (m_validParameter & SOCLT_SRVPAR_CLIENT_NAME)
157 {
158 setClientName(clientName);
159 }
160
161 ar >> lcid;
162
163 if (m_validParameter & SOCLT_SRVPAR_LCID)
164 {
165 setLCID(lcid);
166 }
167
168 if (entry->getFileVersion() >= MAKELONG(01, 4))
169 {
170 SOCmnString userName;
171 SOCmnString password;
172 ar >> userName;
173 setUserName(userName);
174 ar >> password;
175 setPassword(password);
176 }
177
178 // load childeren
179 DWORD dwCount = 0;
180 ar >> dwCount;
181
182 if (dwCount > 0)
183 {
184 while (dwCount-- && ok)
185 {
186 // create subscription
187 subs = addSubscription();
188
189 if (!!subs)
190 {
191 ok = subs->serialize(ar, isStoring);
192 subs->release();
193 }
194 }
195 }
196 }
197
198 return ok;
199} // serialize

247//-----------------------------------------------------------------------------
248// doStateTransition
249//
250// - do the state transition for the server if the target state differs from the
251// actual state of the server
252//
253// returns:
254// TRUE - transit to new state
255// FALSE - error during state change
256//
257HRESULT SOAeCServer::doStateTransition(BOOL deep, IN BYTE stateDeep)
258{
259 HRESULT res = S_OK;
260 BYTE targetState = getTargetObjectState();
261
262 if (disconnected == targetState)
263 {
264 doRecurseChildren(getBranchList(), disconnected);
265 doRecurseChildren(getLeafList(), disconnected);
266 doRecurseChildren(getAdditionalList(), disconnected);
267 }
268
269 res = doStateTransitionImpl();
270
271 if (deep && SUCCEEDED(res) && disconnected != targetState)
272 {
273 if (S_OK != doRecurseChildren(getBranchList(), stateDeep))
274 {
275 res = S_FALSE;
276 }
277
278 if (S_OK != doRecurseChildren(getLeafList(), stateDeep))
279 {
280 res = S_FALSE;
281 }
282
283 if (S_OK != doRecurseChildren(getAdditionalList(), stateDeep))
284 {
285 res = S_FALSE;
286 }
287 }
288
289 return res;
290} // doStateTransition

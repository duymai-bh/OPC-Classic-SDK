296//-----------------------------------------------------------------------------
297// static decodeBase64String2String
298//
299// - decodes a base64 string to a Variant byte array (VT_ARRAY | VT_UI1)
300//
301// returns:
302// 0 - if no array was decoded
303// >0 - number of array elements
304//
305DWORD SOCmnXmlValue::decodeBase64String2String(SOCmnString& szBase64, SOCmnString& string)
306{
307 BYTE bError = FALSE;
308 DWORD dwBase64Length = szBase64.getLength();
309 DWORD dwElements = ((dwBase64Length * 6) / 8); // calculate length of byte array
310 BYTE* pBytes = new BYTE[dwElements + 1];
311 memset(pBytes, 0, dwElements + 1);
312 LPTSTR pszChar = szBase64.getBuffer((DWORD) - 1);
313
314 for (DWORD count = 0; count < dwBase64Length && pszChar[count] != _T('='); count++)
315 {
316 SHORT sixBitVal = 0;
317 BYTE bitshift = (BYTE)(10 - (count * 6) % 8);
318 DWORD offset = (count * 6) / 8;
319
320 if (pszChar[count] >= _T('A') && pszChar[count] <= _T('Z'))
321 {
322 sixBitVal = (pszChar[count] - _T('A'));
323 }
324 else if (pszChar[count] >= _T('a') && pszChar[count] <= _T('z'))
325 {
326 sixBitVal = ((pszChar[count] - _T('a')) + 26);
327 }
328 else if (pszChar[count] >= _T('0') && pszChar[count] <= _T('9'))
329 {
330 sixBitVal = ((pszChar[count] - _T('0')) + 52);
331 }
332 else if (pszChar[count] == _T('+'))
333 {
334 sixBitVal = 62;
335 }
336 else if (pszChar[count] == _T('/'))
337 {
338 sixBitVal = 63;
339 }
340 else
341 {
342 bError = TRUE;
343 break;
344 }
345
346 sixBitVal <<= bitshift;
347
348 if (bitshift != 10)
349 {
350 _ASSERTION(offset + 1 <= dwElements, "array overrun");
351 pBytes[offset] |= HIBYTE(sixBitVal);
352 pBytes[offset + 1] |= LOBYTE(sixBitVal);
353 }
354 else
355 {
356 _ASSERTION(offset <= dwElements, "array overrun");
357 pBytes[offset] |= HIBYTE(sixBitVal);
358 }
359 }
360
361 string.setCHAR((LPCSTR)pBytes, dwElements);
362 delete [] pBytes;
363 return dwElements;
364}// decodeBase64String2Variant

367//-----------------------------------------------------------------------------
368// static decodeBase64String2Variant
369//
370// - decodes a base64 string to a Variant byte array (VT_ARRAY | VT_UI1)
371//
372// returns:
373// 0 - if no array was decoded
374// >0 - number of array elements
375//
376DWORD SOCmnXmlValue::decodeBase64String2Variant(SOCmnString& szBase64, SOCmnVariant& ui1Array)
377{
378 BYTE bError = FALSE;
379 DWORD dwBase64Length = szBase64.getLength();
380 DWORD dwElements = ((dwBase64Length * 6) / 8); // calculate length of byte array
381 BYTE* pBytes = new BYTE[dwElements + 1];
382 memset(pBytes, 0, dwElements + 1);
383 LPTSTR pszChar = szBase64.getBuffer((DWORD) - 1);
384 DWORD count;
385
386 for (count = 0; count < dwBase64Length && pszChar[count] != _T('='); count++)
387 {
388 SHORT sixBitVal = 0;
389 BYTE bitshift = (BYTE)(10 - (count * 6) % 8);
390 DWORD offset = (count * 6) / 8;
391
392 if (pszChar[count] >= _T('A') && pszChar[count] <= _T('Z'))
393 {
394 sixBitVal = (pszChar[count] - _T('A'));
395 }
396 else if (pszChar[count] >= _T('a') && pszChar[count] <= _T('z'))
397 {
398 sixBitVal = ((pszChar[count] - _T('a')) + 26);
399 }
400 else if (pszChar[count] >= _T('0') && pszChar[count] <= _T('9'))
401 {
402 sixBitVal = ((pszChar[count] - _T('0')) + 52);
403 }
404 else if (pszChar[count] == _T('+'))
405 {
406 sixBitVal = 62;
407 }
408 else if (pszChar[count] == _T('/'))
409 {
410 sixBitVal = 63;
411 }
412 else
413 {
414 bError = TRUE;
415 break;
416 }
417
418 sixBitVal <<= bitshift;
419
420 if (bitshift != 10)
421 {
422 _ASSERTION(offset + 1 <= dwElements, "array overrun");
423 pBytes[offset] |= HIBYTE(sixBitVal);
424 pBytes[offset + 1] |= LOBYTE(sixBitVal);
425 }
426 else
427 {
428 _ASSERTION(offset <= dwElements, "array overrun");
429 pBytes[offset] |= HIBYTE(sixBitVal);
430 }
431 }
432
433 // correct dwElements with padding signs =
434 if (count < dwBase64Length)
435 {
436 dwElements -= (dwBase64Length - count);
437 }
438
439 ui1Array.clear();
440
441 if (!bError)
442 {
443 SAFEARRAYBOUND dim;
444 dim.lLbound = 0;
445 dim.cElements = dwElements;
446 V_VT(&ui1Array) = (VT_ARRAY | VT_UI1);
447 V_ARRAY(&ui1Array) = SOCmnVariant::safeArrayCreate((VARTYPE)(V_VT(&ui1Array) & ~VT_ARRAY), 1, &dim);
448 BYTE* pArray;
449
450 if (SUCCEEDED(SafeArrayAccessData(V_ARRAY(&ui1Array), (void**)&pArray)))
451 {
452 memcpy(pArray, pBytes, dwElements);
453 SafeArrayUnaccessData(V_ARRAY(&ui1Array));
454 }
455 else
456 {
457 bError = TRUE;
458 }
459 }
460
461 delete pBytes;
462 return dwElements;
463}// decodeBase64String2Variant

466//-----------------------------------------------------------------------------
467// static encodeBytes2base64
468//
469// -
470//
471// returns:
472// empty - if nothing to encode or error
473// string - a wellformed base64 string
474//
475SOCmnString SOCmnXmlValue::encodeBytes2base64(SOCmnVariant& ui1Array)
476{
477 SOCmnString szBase64;
478
479 if (V_VT(&ui1Array) != (VT_ARRAY | VT_UI1))
480 {
481 return szBase64;
482 }
483
484 BYTE bError = FALSE;
485 BYTE* pArray;
486 LONG lBound, uBound;
487
488 if (SUCCEEDED(SafeArrayGetLBound(V_ARRAY(&ui1Array), 1, &lBound))
489 && SUCCEEDED(SafeArrayGetUBound(V_ARRAY(&ui1Array), 1, &uBound))
490 && SUCCEEDED(SafeArrayAccessData(V_ARRAY(&ui1Array), (void**)&pArray)))
491 {
492 DWORD dwArrayCount = uBound - lBound + 1;
493 DWORD sizeBase64 = (((dwArrayCount * 8) + 4) / 6);
494 LPTSTR pszBase64 = NULL;
495 DWORD numEquals = dwArrayCount % 3;
496 DWORD idx;
497
498 if (numEquals > 0)
499 {
500 numEquals = 3 - numEquals;
501 }
502
503 pszBase64 = (LPTSTR)malloc((sizeBase64 + numEquals + 1) * sizeof(TCHAR));
504
505 for (idx = 0; idx < sizeBase64 + numEquals; idx++)
506 {
507 pszBase64[idx] = _T('=');
508 }
509
510 pszBase64[sizeBase64 + numEquals] = 0;
511
512 for (idx = 0; (idx < sizeBase64); idx++)
513 {
514 BYTE bitshift;
515 bitshift = (BYTE)(10 - (idx * 6) % 8);
516 DWORD offset = (idx * 6) / 8;
517 SHORT number = 0;
518
519 if (offset + 1 < dwArrayCount)
520 {
521 number = MAKEWORD(pArray[offset + 1], pArray[offset]);
522 }
523 else if (offset < dwArrayCount)
524 {
525 number = MAKEWORD(0, pArray[offset]);
526 }
527
528 BYTE sixBitVal = (BYTE)(number >> bitshift);
529 sixBitVal &= 0x3f;
530
531 // add decoding to base64 here...
532 if (sixBitVal <= 25)
533 {
534 pszBase64[idx] = (_T('A') + sixBitVal);
535 }
536 else if (sixBitVal >= 26 && sixBitVal <= 51)
537 {
538 pszBase64[idx] = (_T('a') + (sixBitVal - 26));
539 }
540 else if (sixBitVal >= 52 && sixBitVal <= 61)
541 {
542 pszBase64[idx] = (_T('0') + (sixBitVal - 52));
543 }
544 else if (sixBitVal == 62)
545 {
546 pszBase64[idx] = _T('+');
547 }
548 else if (sixBitVal == 63)
549 {
550 pszBase64[idx] = _T('/');
551 }
552 else
553 {
554 bError = TRUE;
555 break;
556 }
557 }
558
559 if (bError)
560 {
561 szBase64.empty();
562 }
563 else
564 {
565 szBase64 = pszBase64;
566 }
567
568 free(pszBase64);
569 SafeArrayUnaccessData(V_ARRAY(&ui1Array));
570 }
571
572 return szBase64;
573} //encodeBytes2base64

600BOOL SOCmnXmlFile::readContents(void)
601{
602#ifdef SOOS_LINUX
603 // read contents of file to buffer
604 struct stat fileStat;
605
606 if (0 != stat(m_fileName, &fileStat))
607 {
608 return FALSE;
609 }
610
611 // open configuration file
612 FILE* pFile;
613 pFile = fopen(m_fileName, "r");
614
615 if (!pFile)
616 {
617 return FALSE;
618 }
619
620 m_pData = (BYTE*)malloc(fileStat.st_size);
621 m_lenData = fread((void*)m_pData, 1, fileStat.st_size, pFile);
622 fclose(pFile);
623
624 if (m_lenData != (DWORD)fileStat.st_size)
625 {
626 freeContents();
627 return FALSE;
628 }
629
630 return TRUE;
631#endif // SOOS_LINUX
632#ifdef SOOS_WINDOWS
633 // open configuration file
634 HANDLE hFile = CreateFile((LPCTSTR)m_fileName, GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
635
636 if (hFile == INVALID_HANDLE_VALUE)
637 {
638 return FALSE;
639 }
640
641 // calculate the size of the file
642 DWORD fileSize;
643
644 if ((fileSize = ::GetFileSize(hFile, NULL)) == INVALID_FILE_SIZE)
645 {
646 CloseHandle(hFile);
647 return FALSE;
648 }
649
650 // get the contents of the file
651 m_pData = (BYTE*)malloc(fileSize);
652
653 if (!ReadFile(hFile, m_pData, fileSize, &m_lenData, NULL))
654 {
655 CloseHandle(hFile);
656 freeContents();
657 return FALSE;
658 }
659
660 CloseHandle(hFile);
661 return TRUE;
662#endif // SOOS_WINDOWS
663}

719LPCTSTR SOCmnXmlFile::expandNamespace(IN OUT SOCmnString& szXML)
720{
721 int nsIdx;
722 nsIdx = szXML.find(_T(':'));
723
724 if (nsIdx == -1)
725 {
726 szXML.format(_T("%s|%s"), (LPCTSTR)m_defNamespace, (LPCTSTR)szXML);
727 }
728 else
729 {
730 SOCmnString prefix = szXML.left(nsIdx);
731 SOCmnString data = szXML.right(szXML.getLength() - nsIdx - 1);
732 szXML.format(_T("%s|%s"), (LPCTSTR)m_namespaceList.findKey(prefix), (LPCTSTR)data);
733 }
734
735 return (LPCTSTR)szXML;
736}

794void SOCmnXmlMethod::handleStartElement(LPCSTR nameA, LPCSTR* /* atts */)
795{
796 USES_CONVERSION;
797 LPCTSTR name = A2CT(nameA);
798
799 switch (getObjectState())
800 {
801 case SOCMNXMLMETHOD_STATE_INIT: // wait for SOAP envelope
802 if (_tcsicmp(name, _T("http://schemas.xmlsoap.org/soap/envelope/|Envelope")) == 0)
803 {
804 setObjectState(SOCMNXMLMETHOD_STATE_SOAP_ENVELOPE);
805 m_elementStack.push(name);
806 }
807 else
808 {
809 setObjectState(SOCMNXMLMETHOD_STATE_ERROR);
810 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_CALL, (_T("No SOAP envelope: instead %s"), name));
811 }
812
813 break;
814
815 case SOCMNXMLMETHOD_STATE_SOAP_ENVELOPE: // wait for SOAP body
816 if (_tcsicmp(name, _T("http://schemas.xmlsoap.org/soap/envelope/|Body")) == 0)
817 {
818 setObjectState(SOCMNXMLMETHOD_STATE_SOAP_BODY);
819 m_elementStack.push(name);
820 }
821 else if (_tcsicmp(name, _T("http://schemas.xmlsoap.org/soap/envelope/|Header")) == 0)
822 {
823 // do nothing just skip the header
824 m_elementStack.push(name);
825 }
826 else
827 {
828 setObjectState(SOCMNXMLMETHOD_STATE_ERROR);
829 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_CALL, (_T("No SOAP body: instead %s"), name));
830 }
831
832 break;
833
834 case SOCMNXMLMETHOD_STATE_END:
835 setObjectState(SOCMNXMLMETHOD_STATE_ERROR);
836 _TRACE(SOCMNTRACE_L_ERR, SOCMNTRACE_G_OPC_CALL, (_T("End of method: instead %s"), name));
837 break;
838
839 default:
840 break;
841 }

877LPCTSTR SOCmnXmlMethod::expandNamespace(IN OUT SOCmnString& szXML)
878{
879 int nsIdx;
880 nsIdx = szXML.find(_T(':'));
881
882 if (nsIdx == -1)
883 {
884 szXML.format(_T("%s|%s"), (LPCTSTR)m_defNamespace, (LPCTSTR)szXML);
885 }
886 else
887 {
888 SOCmnString prefix = szXML.left(nsIdx);
889 SOCmnString data = szXML.right(szXML.getLength() - nsIdx - 1);
890 szXML.format(_T("%s|%s"), (LPCTSTR)m_namespaceList.findKey(prefix), (LPCTSTR)data);
891 }
892
893 return (LPCTSTR)szXML;
894}

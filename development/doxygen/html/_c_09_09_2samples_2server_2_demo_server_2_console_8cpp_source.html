37void* keyListenerThread(void* context)
38{
39 _TCHAR pressedKey;
40 int i = 0;
41
42 while (1)
43 {
44 usleep(1);
45 i = kbhit();

47 if (i != 0)
48 {
49 pressedKey = fgetc(stdin);
50 OpcServer* pServer = getOpcServer();
51 DemoDaAddressSpaceElement* keyElement = pServer->getKeyElement();
52
53 if (keyElement)
54 {
55 DateTime now;
56 now.now();
57 Variant aVariant((unsigned short)pressedKey);
58 ValueQT val(aVariant, EnumQuality_GOOD, now);
59 keyElement->valueChanged(val);
60 }
61 }
62
63 bool keepListening = false;
64 pthread_mutex_lock(&g_keyListenerMutex);
65 keepListening = g_listenKey;
66 pthread_mutex_unlock(&g_keyListenerMutex);
67
68 if (!keepListening)
69 {
70 break;
71 }
72 }
73
74 pthread_mutex_destroy(&g_keyListenerMutex);
75 pthread_exit(NULL);
76 return NULL;

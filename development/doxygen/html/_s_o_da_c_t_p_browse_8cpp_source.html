67BOOL SODaCTPNameSpaceBrowserImpl::expand30(
68 IN SOCltBrowseObject* pParent,
69 IN LPCTSTR itemID,
70 IN LPCTSTR /* itemPath */,
71 IN DWORD type,
72 IN SOCmnString& contPoint,
73 OUT SOCmnList<SOCltBrowseObject> &objList)
74{
75 SODaCNameSpaceBrowser* pBrowser = getObjectSODaCNameSpaceBrowser();
76 SOCmnPointer<SODaCServer> server = (SODaCServer*)pBrowser->getServer();
77 DWORD browseFilter = 0;
78 SOCmnString cpSZ;
79
80 if (!contPoint.isEmpty())
81 {
82 cpSZ = contPoint;
83 }
84
85 contPoint.empty();
86
87 if (!objList.isInit())
88 {
89 objList.create(SOCMNLIST_TYPE_ORG_LIST | SOCMNLIST_TYPE_KEY_NO | SOCMNLIST_TYPE_OBJ_OBJECT);
90 }
91
92 // calculate the browse type
93 if ((SOCMNOBJECT_TYPE_NODE | SOCMNOBJECT_TYPE_ITEM) == (type & (SOCMNOBJECT_TYPE_NODE | SOCMNOBJECT_TYPE_ITEM)))
94 {
95 browseFilter = OPC_BROWSE_FILTER_ALL;
96 }
97 else
98 {
99 if (SOCMNOBJECT_TYPE_NODE == (type & SOCMNOBJECT_TYPE_NODE))
100 {
101 browseFilter = OPC_BROWSE_FILTER_BRANCHES;
102 }
103 else if (SOCMNOBJECT_TYPE_ITEM == (type & SOCMNOBJECT_TYPE_ITEM))
104 {
105 browseFilter = OPC_BROWSE_FILTER_ITEMS;
106 }
107 }
108
109 if (browseFilter != 0)
110 {
111 SOCmnString elementNameFilter = pBrowser->getElementNameFilter();
112 SOCmnString vendorFilter = pBrowser->getVendorFilter();
113 SOCmnPointer<SOCltTPConnection> tpCon = server->getTPConnection();
114 SOCmnPointer<SODaCTPCall> tpCall;
115 HRESULT res;
116 SOCMN_ALLOCATE_OBJECT_C(tpCall, SODaCTPCall(tpCon), (SODaCTPCall*));
117 server->beforeInterfaceCall();
118 res = tpCall->browse(pParent, itemID, cpSZ, pBrowser->getMaxElements(), browseFilter, elementNameFilter, vendorFilter, pBrowser->getReturnProperties(), pBrowser->getRetrievePropertyValues(), NULL, objList, &contPoint, NULL);
119 server->afterInterfaceCall();
120 _TRACEO_LEVEL_FROM_RES(res, SOCMNTRACE_G_OPC_CALL_C, server, (_T("TPDA::Server::Browse [0x%X]"), (unsigned int)res));
121
122 if (FAILED(res))
123 {
124 server->setLastError(res, SODAC_ERR_SRV_BROWSE);
125 }
126 }
127
128 return TRUE;
129}
